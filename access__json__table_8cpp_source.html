41  column::trigger_on_error (const JSON_DOC &input, const TP_DOMAIN_STATUS &status_cast, db_value &value_out)
42  {
43  (void) pr_clear_value (&value_out);
44  (void) db_make_null (&value_out);
45 
46  switch (m_on_error.m_behavior)
47  {
48  case JSON_TABLE_RETURN_NULL:
49  er_clear ();
50  return NO_ERROR;
51 
52  case JSON_TABLE_THROW_ERROR:
53  {
54  cubmem::private_unique_ptr<char> unique_ptr_json_body (db_json_get_raw_json_body_from_document (&input), NULL);
55 
56  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_JSON_TABLE_ON_ERROR_INCOMP_DOMAIN, 4,
57  unique_ptr_json_body.get (), m_path, m_column_name,
58  pr_type_name (TP_DOMAIN_TYPE (m_domain)));
59 
60  return ER_JSON_TABLE_ON_ERROR_INCOMP_DOMAIN;
61  }
62 
63  case JSON_TABLE_DEFAULT_VALUE:
64  assert (m_on_error.m_default_value != NULL);
65  er_clear ();
66  if (pr_clone_value (m_on_error.m_default_value, &value_out) != NO_ERROR)
67  {
68  assert (false);
69  }
70  return NO_ERROR;
71 
72  default:
73  assert (false);
74  return ER_FAILED;
75  }

126  int
127  column::evaluate_extract (const JSON_DOC &input)
128  {
129  int error_code = NO_ERROR;
130  JSON_DOC_STORE docp;
131  TP_DOMAIN_STATUS status_cast = TP_DOMAIN_STATUS::DOMAIN_COMPATIBLE;
132 
133  error_code = db_json_extract_document_from_path (&input, m_path, docp);
134  if (error_code != NO_ERROR)
135  {
136  ASSERT_ERROR ();
137  assert (db_value_is_null (m_output_value_pointer));
138  return ER_FAILED;
139  }
140 
141  if (docp.is_null ())
142  {
143  error_code = trigger_on_empty (*m_output_value_pointer);
144  if (error_code != NO_ERROR)
145  {
146  ASSERT_ERROR ();
147  }
148  return error_code;
149  }
150 
151  // clear previous output_value
152  pr_clear_value (m_output_value_pointer);
153  db_make_json_from_doc_store_and_release (*m_output_value_pointer, docp);
154 
155  status_cast = tp_value_cast (m_output_value_pointer, m_output_value_pointer, m_domain, false);
156  if (status_cast != TP_DOMAIN_STATUS::DOMAIN_COMPATIBLE)
157  {
158  error_code = trigger_on_error (input, status_cast, *m_output_value_pointer);
159  if (error_code != NO_ERROR)
160  {
161  ASSERT_ERROR ();
162  }
163  }
164 
165  return error_code;
166  }
