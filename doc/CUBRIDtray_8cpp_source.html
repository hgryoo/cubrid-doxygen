303 char* CAboutDlg::sGetReleaseString( char* sPath )
304 {
305  char* sFullName;
306  char* sValue;
307  FILE* fd_rel;
308  struct _stat buf;
309  int result;
310 
311  if( !sPath ) return NULL;
312 
313  sFullName = new char[ strlen( sPath ) + 15 ];
314  memset( sFullName, 0x00, strlen( sPath ) + 15 );
315  sprintf( sFullName, "%s\\RELEASE_STRING", sPath );
316 
317  fd_rel = fopen( sFullName, "r" );
318 
319  if( fd_rel )
320  {
321  delete sFullName;
322  return NULL;
323  }
324 
325  result = _stat( sFullName, &buf );
326 
327  /* Check if statistics are valid: */
328  if( result != 0 )
329  {
330  delete sFullName;
331  return NULL;
332  }
333 
334  sValue = new char[ buf.st_size + 1 ];
335  memset( sValue, 0x00, buf.st_size );
336 
337  fread( sValue, buf.st_size, 1, fd_rel );
338 
339  fclose( fd_rel );
340  return sValue;
341 }
