34 /*
35  * Commit 정책
36  * (implicit commit: IRowsetChange등에서 AutoCommit() 호출)
37  * (explicit commit: ITransaction::Commit 또는 Rowset이 닫힘)
38  *
39  * 1. implicit commit in manual-commit mode
40  *
41  * 아무 일도 하지 않는다.
42  *
43  * 2. implicit commit in auto-commit mode
44  *
45  * TxnCallback을 호출해서 다른 Rowset을 좀비로 만든다.
46  *
47  * 3. explicit commit in manual-commit mode
48  *
49  * 실제로 Commit 하고 TxnCallback을 호출해서 모든 Rowset을 좀비로 만든다.
50  *
51  * 4. explicit commit in auto-commit mode
52  *
53  * 실제로 Commit 한다(다른 Rowset은 이미 좀비이고, 현재 Rowset은 닫는 중)
54  *
55  */

57 HRESULT CCUBRIDSession::DoCASCCICommit(bool bCommit)
58 {
59  int hConn;
60  HRESULT hr = GetConnectionHandle(&hConn);
61  if(FAILED(hr)) return E_FAIL;
62 
63  T_CCI_ERROR err_buf;
64  int rc = cci_end_tran(hConn, bCommit?CCI_TRAN_COMMIT:CCI_TRAN_ROLLBACK, &err_buf);
65  if(rc<0) return bCommit?XACT_E_COMMITFAILED:E_FAIL;
66 
67  // TxnCallback을 호출해 좀비로 만든다.
68  POSITION pos = m_grpTxnCallbacks.GetHeadPosition();
69  while(pos)
70  m_grpTxnCallbacks.GetNext(pos)->TxnCallback(0);
71 
72  return S_OK;
73 }

108 HRESULT CCUBRIDSession::SetCASCCIIsoLevel(ISOLEVEL isoLevel, bool bCheckOnly)
109 {
110  int hConn;
111  if(!bCheckOnly)
112  {
113  HRESULT hr = GetConnectionHandle(&hConn);
114  if(FAILED(hr)) return E_FAIL;
115  }
116 
117  int cci_isolevel;
118  switch(isoLevel)
119  {
120  case ISOLATIONLEVEL_READCOMMITTED:
121  cci_isolevel = TRAN_READ_COMMITTED; break;
122  case ISOLATIONLEVEL_REPEATABLEREAD:
123  cci_isolevel = TRAN_REPEATABLE_READ; break;
124  case ISOLATIONLEVEL_SERIALIZABLE:
125  cci_isolevel = TRAN_SERIALIZABLE; break;
126  default:
127  return XACT_E_ISOLATIONLEVEL;
128  }
129 
130  if(!bCheckOnly)
131  {
132  T_CCI_ERROR err_buf;
133  int rc = cci_set_db_parameter(hConn, CCI_PARAM_ISOLATION_LEVEL, &cci_isolevel, &err_buf);
134  if(rc<0) return E_FAIL;
135  m_isoLevel = isoLevel;
136  }
137 
138  ATLTRACE(atlTraceDBProvider, 2, "Current Isolation Level:[%ld]\n", isoLevel);
139 
140  return S_OK;
141 }

158 STDMETHODIMP CCUBRIDSession::StartTransaction(ISOLEVEL isoLevel, ULONG isoFlags,
159  ITransactionOptions *pOtherOptions, ULONG *pulTransactionLevel)
160 {
161  ATLTRACE(atlTraceDBProvider, 2, "CCUBRIDSession::StartTransaction\n");
162 
163  // we do not support nested transactions
164  if(!m_bAutoCommit) return XACT_E_XTIONEXISTS;
165  if(isoFlags!=0) return XACT_E_NOISORETAIN;
166 
167  HRESULT hr = SetCASCCIIsoLevel(isoLevel);
168  if(FAILED(hr)) return hr;
169 
170  // flat transaction이므로 항상 1
171  if(pulTransactionLevel) *pulTransactionLevel = 1;
172  m_bAutoCommit = false;
173  return S_OK;
174 }

176 STDMETHODIMP CCUBRIDSession::Commit(BOOL fRetaining, DWORD grfTC, DWORD grfRM)
177 {
178  ATLTRACE(atlTraceDBProvider, 2, "CCUBRIDSession::Commit\n");
179 
180  if(grfTC==XACTTC_ASYNC_PHASEONE || grfTC==XACTTC_SYNC_PHASEONE || grfRM!=0) return XACT_E_NOTSUPPORTED;
181  if(m_bAutoCommit) return XACT_E_NOTRANSACTION;
182 
183  HRESULT hr = DoCASCCICommit(true);
184  if(FAILED(hr)) return hr;
185 
186  // fRetaining==true 면 transaction을 유지한다.
187  if(!fRetaining) m_bAutoCommit = true; //EnterAutoCommitMode();
188 
189  return S_OK;
