40  * adj_ar_concat_strings() - concatenate the specified strings
41  * return: concatenated string
42  * string1(in): the first string to be concatednated
43  * string2(in): the second string to be concatednated
44  *
45  * Note: The returned pointer must not be freed, and its contents must not be
46  * accessed after the next call to this function.
47  */
48 const char *
49 adj_ar_concat_strings (const char *string1, const char *string2, ...)
50 {
51  va_list next_arg;
52  const char *next_string;
53  static ADJ_ARRAY *string_buffer = NULL;
54 
55  if (string_buffer == NULL)
56  {
57  string_buffer = adj_ar_new (1, 0, 2.0);
58  }
59 
60  (void) adj_ar_reset (string_buffer, 1, 64, 2.0);
61 
62  if (!string1)
63  {
64  string1 = "?";
65  }
66  (void) adj_ar_append (string_buffer, string1, (int) strlen (string1));
67  if (!string2)
68  {
69  string2 = "?";
70  }
71  (void) adj_ar_append (string_buffer, string2, (int) strlen (string2));
72 
73  va_start (next_arg, string2);
74  while ((next_string = va_arg (next_arg, const char *)))
75  {
76  (void) adj_ar_append (string_buffer, next_string, (int) strlen (next_string));
77  }
78 
79  (void) adj_ar_append (string_buffer, "\0", 1);
80 
81  va_end (next_arg);
82 
83  return (const char *) adj_ar_get_buffer (string_buffer);

135 /*
136  * adj_ar_reset() - reset ADJ_ARRAY with the given values.
137  * return: ADJ_ERR_CODE
138  * adj_array_p(in/out) : the ADJ_ARRAY pointer
139  * element_size(in) : size of array element
140  * min(in) : minimum elements in array
141  * growth_rate(in) : growth rate of array

143 int
144 adj_ar_reset (ADJ_ARRAY * adj_array_p, int element_size, int min, float growth_rate)
145 {
146  assert (adj_array_p != NULL);
147 
148  if (element_size < 1)
149  {
150  return ADJ_ERR_BAD_ELEMENT;
151  }
152  if (min < 0)
153  {
154  return ADJ_ERR_BAD_MIN;
155  }
156  if (growth_rate < 1.0)
157  {
158  return ADJ_ERR_BAD_RATE;
159  }
160 
161  adj_array_p->max_length *= adj_array_p->element_size / element_size;
162  adj_array_p->cur_length = 0;
163  adj_array_p->element_size = element_size;
164  adj_array_p->min_length = min;
165  adj_array_p->rate = growth_rate;
166 
167  return ADJ_NOERROR;

179 {

221 /*
222  * adj_ar_replace() - replace the ADJ_ARRAY buffer with the given data
223  * to the specified range
224  * return: ADJ_ERR_CODE
225  * adj_array_p(in/out): the ADJ_ARRAY pointer
226  * src (in): source buffer
227  * src_length (in) : number of elements in src buffer
228  * start (in): start position in the array buffer to be replaced, inclusive
229  * end (in): end position in the array buffer to be replaced, exclusive
230  */
231 int
232 adj_ar_replace (ADJ_ARRAY * adj_array_p, const void *src, int src_length, int start, int end)
233 {
234  int new_length;
235 
236  assert (adj_array_p != NULL);
237 
238  /* Expand ADJ_AR_EOA. */
239  if (end == ADJ_AR_EOA)
240  {
241  end = adj_array_p->cur_length;
242  }
243  if (start == ADJ_AR_EOA)
244  {
245  start = adj_array_p->cur_length;
246  }
247 
248  /* Check for errors. */
249  if (start < 0)
250  {
251  return ADJ_ERR_BAD_START;
252  }
253  else if (start > adj_array_p->cur_length)
254  {
255  return ADJ_ERR_BAD_START;
256  }
257  else if (end < start)
258  {
259  return ADJ_ERR_BAD_END;
260  }
261  else if (end > adj_array_p->cur_length)
262  {
263  return ADJ_ERR_BAD_END;
264  }
265  else if (src_length < 0)
266  {
267  return ADJ_ERR_BAD_NFROM;
268  }
269 
270  new_length = adj_array_p->cur_length + src_length - (end - start);
271  if (new_length > adj_array_p->max_length)
272  {
273  /* allocate larger buffer. */
274  void *new_buffer;
275  int new_max = MAX (adj_array_p->min_length,
276  MAX ((int) (adj_array_p->max_length * adj_array_p->rate),
277  new_length));
278 
279  if (adj_array_p->buffer)
280  {
281  new_buffer = realloc (adj_array_p->buffer, new_max * adj_array_p->element_size);
282  }
283  else
284  {
285  new_buffer = malloc (new_max * adj_array_p->element_size);
286  }
287 
288  if (!new_buffer)
289  {
290  return ADJ_ERR_BAD_ALLOC;
291  }
292  adj_array_p->buffer = new_buffer;
293  adj_array_p->max_length = new_max;
294  }
295 
296  /* Shift elements following replaced subarray. */
297  memmove ((void *) ((char *) adj_array_p->buffer + (start + src_length) * adj_array_p->element_size),
298  (void *) ((char *) adj_array_p->buffer + end * adj_array_p->element_size),
299  (adj_array_p->cur_length - end) * adj_array_p->element_size);
300 
301  if (src)
302  {
303  memmove ((void *) ((char *) adj_array_p->buffer + start * adj_array_p->element_size), src,
304  src_length * adj_array_p->element_size);
305  }
306  adj_array_p->cur_length = new_length;
307 
308  return ADJ_NOERROR;
309 }
