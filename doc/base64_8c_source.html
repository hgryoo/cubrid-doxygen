139 /*
140  * base64_partition_into_chunk () - Partition base64 encoded string into
141  * multiple chunks. Partition stops when
142  * meeting the first chunk ending in
143  * padding(s).
144  *
145  * return: int(NO_ERROR if successful, otherwise other base64 error code)
146  * src(in): source string(base64 encoded)
147  * src_len(in): source string length
148  * chunk_num_out(in/out): actual number of chunks calculated
149  * dst_len_out(in/out): actual length of buffer to store decoded bytes
150  * pppchunk(in/out): array pointer of chunks to store encoded bytes
151  */
152 static int
153 base64_partition_into_chunk (const unsigned char *src, int src_len, int *chunk_num_out, int *dst_len_out,
154  BASE64_CHUNK *** pppchunk)
155 {
156  int err = NO_ERROR;
157  int i, j, chunk_count, chunk_num, dst_len;
158  unsigned char d1, d2, d3, d4;
159  bool tail_chunk_flag;
160  BASE64_CHUNK *chk = NULL;
161  BASE64_CHUNK **ppchunk = NULL;
162 
163  assert (src != NULL && src_len > 0 && chunk_num_out != NULL && dst_len_out != NULL && pppchunk != NULL);
164 
165  *chunk_num_out = 0;
166  *dst_len_out = 0;
167  *pppchunk = NULL;
168 
169  /* require 4-byte alignment for chunk */
170  if ((src_len & 0x3) != 0)
171  {
172  err = BASE64_INVALID_INPUT;
173  goto end;
174  }
175 
176  /* chunk array pointer to accommodate maximum chunk_num base64 chunks */
177  chunk_num = src_len / 4;
178  ppchunk = (BASE64_CHUNK **) db_private_alloc (NULL, chunk_num * sizeof (BASE64_CHUNK *));
179 
180  if (ppchunk == NULL)
181  {
182  err = ER_OUT_OF_VIRTUAL_MEMORY;
183  goto end;
184  }
185 
186  memset (ppchunk, 0, chunk_num * sizeof (BASE64_CHUNK *));
187 
188  tail_chunk_flag = false;
189  i = chunk_count = dst_len = 0;
190  while (chunk_count < chunk_num)
191  {
192  d1 = src[i];
193  d2 = src[i + 1];
194  d3 = src[i + 2];
195  d4 = src[i + 3];
196 
197  /* check all invalid conditions */
198  if (char_is_invalid (d1) || char_is_invalid (d2) || char_is_invalid (d3) || char_is_invalid (d4)
199  || char_is_padding (d1) || char_is_padding (d2))
200  {
201  err = BASE64_INVALID_INPUT;
202  goto end;
203  }
204 
205  if (char_is_padding (d3) && !char_is_padding (d4))
206  {
207  err = BASE64_INVALID_INPUT;
208  goto end;
209  }
210 
211  chk = (BASE64_CHUNK *) db_private_alloc (NULL, sizeof (BASE64_CHUNK));
212  if (chk == NULL)
213  {
214  err = ER_OUT_OF_VIRTUAL_MEMORY;
215  goto end;
216  }
217 
218  memset (chk, 0, sizeof (BASE64_CHUNK));
219 
220  chk->padding_len = char_is_padding (d4) ? 1 : 0;
221  chk->padding_len += char_is_padding (d3) ? 1 : 0;
222 
223  for (j = 0; j <= (3 - chk->padding_len); j++)
224  {
225  chk->base64_bytes[j] = (char) find_base64 (src[i + j]);
226  }
227 
228  ppchunk[chunk_count] = chk;
229 
230  if (char_is_padding (d4) == true)
231  {
232  /* middle chunk that has padding is invalid */
233  if ((chunk_count + 1) != chunk_num)
234  {
235  err = BASE64_INVALID_INPUT;
236  goto end;
237  }
238  else /* tail chunk that has padding */
239  {
240  /* how many decoded bytes to add in this chunk, e.g, from_base64(YWE=)='aa',from_base64(YW==)='a' */
241  dst_len += char_is_padding (d3) ? 1 : 2;
242  tail_chunk_flag = true;
243  }
244  }
245 
246  chunk_count++;
247 
248  if (tail_chunk_flag == true)
249  {
250  break;
251  }
252 
253  dst_len += 3;
254  i += 4;
255  }
256 
257 end:
258 
259  if (err == NO_ERROR)
260  {
261  *chunk_num_out = chunk_count;
262  *dst_len_out = dst_len;
263  *pppchunk = ppchunk;
264  }
265  else
266  {
267  /* free resource if error */
268  if (ppchunk != NULL)
269  {
270  free_base64_chunk (ppchunk, chunk_count);
271  }
272 
273  *chunk_num_out = 0;
274  *dst_len_out = 0;
275  *pppchunk = NULL;
276  }
277 
278  return err;
279 }

281 /*
282  * base64_remove_space () - remove space characters to facilitate base64
283  * chunk partition
284  * return: NO_ERROR
285  * src(in): input string
286  * size(in):length of input string
287  * dest(in/out): buffer to store the string that has no space characters
288  * dst_len(in/out): length of dest
289  *
290  * Note: dest is allocated in this function but freed in caller. dst_len may
291  * differ from size in case of space characters.
292  */
293 static int
294 base64_remove_space (const unsigned char *src, int size, unsigned char *dest, int *dst_len)
295 {
296  int i;
297  unsigned char *q;
298 
299  assert (src != NULL && dest != NULL && dst_len != NULL);
300 
301  *dst_len = 0;
302  q = dest;
303 
304  i = 0;
305  while (i < size)
306  {
307  /* search for the next char that is non-space character */
308  while (from_base64_table[src[i]] == CH_SPACE && i < size)
309  {
310  i++;
311  }
312 
313  if (i >= size)
314  {
315  break;
316  }
317 
318  /* be careful of memory bounds! */
319  *q++ = src[i++];
320  (*dst_len)++;
321  }
322 
323  return NO_ERROR;
324 }

326 /*
327  * base64_decode_chunk () - convert encoded bytes stored in chunks to
328  * plain-text bytes
329  * return: NO_ERROR
330  * dest(in/out): buffer to store decoded bytes
331  * chunk_num(in): number of chunks
332  * ppchunk(in): buffer to store chunks
333  *
334  * Note: each chunk has four encoded bytes and can be converted to [1,2,3]
335  * decoded bytes depending on padding.
336  */
337 static int
338 base64_decode_chunk (unsigned char *dest, int chunk_num, BASE64_CHUNK ** ppchunk)
339 {
340  int i, d, copy_num;
341  char decode_bytes[3];
342  BASE64_CHUNK *chk;
343 
344  assert (dest != NULL && ppchunk != NULL);
345 
346  i = 0;
347  d = 0;
348  while (i < chunk_num)
349  {
350  chk = ppchunk[i];
351  d += (chk->base64_bytes[0]) << 18;
352  d += (chk->base64_bytes[1]) << 12;
353  d += (chk->base64_bytes[2]) << 6;
354  d += (chk->base64_bytes[3]) << 0;
355 
356  decode_bytes[0] = (d >> 16) & 0xff;
357  decode_bytes[1] = (d >> 8) & 0xff;
358  decode_bytes[2] = (d >> 0) & 0xff;
359 
360  /* only allow for 0,1,2 */
361  assert (chk->padding_len < 3);
362  copy_num = 3 - chk->padding_len;
363 
364  memcpy (dest, decode_bytes, copy_num);
365 
366  dest += copy_num;
367  d = 0;
368  i++;
369  }
370 
371  return NO_ERROR;
372 }

374 /*
375  * base64_decode() -
376  * return: int(NO_ERROR if successful,otherwise failure error code)
377  * src(in): source plain-text string
378  * src_len(in): length of source string
379  * dest(in/out): buffer to store decoded bytes, allocated in this func
380  * dest_len(in/out): required length to store decoded buffer
381  *
382  * Note: there are some error handlings to handle invalid characters.
383  * some buffers are allocated in this function or its callees,
384  * and freed in the end.
385  *
386  */
387 int
388 base64_decode (const unsigned char *src, int src_len, unsigned char **dest, int *dest_len)

391  int len_no_space, real_dest_len, chunk_num = 0;
392  unsigned char *str_no_space = NULL;
393  unsigned char *real_dest = NULL;
394  BASE64_CHUNK **ppchunk = NULL;
395 
396  assert (src != NULL && dest != NULL && dest_len != NULL);
397 
398  *dest = NULL;
399  *dest_len = 0;
400 
401  len_no_space = 0;
402 
403  /* assume there are no space characters in source string */
404  str_no_space = (unsigned char *) db_private_alloc (NULL, src_len + 1);
405  if (str_no_space == NULL)
406  {
407  error_status = ER_OUT_OF_VIRTUAL_MEMORY;
408  goto buf_clean;
409  }
410 
411  /* remove space characters */
412  base64_remove_space (src, src_len, str_no_space, &len_no_space);
413 
414  /* ' ' has spaces. after space removal, it is empty string */
415  if (len_no_space == 0)
416  {
417  error_status = BASE64_EMPTY_INPUT;

419  }
420 
421  /* Partition encoded bytes with no space characters into multiple chunks for decoding, stop when there is a middle
422  * chunk ending in padding(s), which is considered as invalid input */
423  error_status = base64_partition_into_chunk (str_no_space, len_no_space, &chunk_num, &real_dest_len, &ppchunk);
424 
425  if (error_status != NO_ERROR)
426  {
427  goto buf_clean;
428  }
429 
430  real_dest = (unsigned char *) db_private_alloc (NULL, real_dest_len + 1);
431  if (real_dest == NULL)
432  {
433  error_status = ER_OUT_OF_VIRTUAL_MEMORY;
434  goto buf_clean;
435  }
436 
437  real_dest[real_dest_len] = '\0';
438 
439  assert (chunk_num > 0);
440  base64_decode_chunk (real_dest, chunk_num, ppchunk);
441 
442  *dest = real_dest;
443  *dest_len = real_dest_len;
444 
445  error_status = NO_ERROR;
446 
447 buf_clean:
448 
449  if (str_no_space != NULL)
450  {
451  db_private_free_and_init (NULL, str_no_space);
452  }
453 
454  if (error_status != NO_ERROR && real_dest != NULL)
455  {
456  db_private_free_and_init (NULL, real_dest);
457  }
458 
459  if (ppchunk != NULL)
460  {
461  free_base64_chunk (ppchunk, chunk_num);
462  }
463 
464  return error_status;
465 }

545 /*
546  * base64_encode_local () - convert a plain-text string with base64,
547  * invoked by base64_encode
548  * return: NO_ERROR
549  * src(in): source plain-text string
550  * src_len(in): length of source string
551  * dest(in): base64 encoded string
552  *
553  * Note: it's the caller's responsibility to claim and reclaim(if needed)
554  * memory space for dest
555  */
556 static int
557 base64_encode_local (const unsigned char *src, int src_len, unsigned char *dest)
558 {
559  const unsigned char *p;
560  unsigned int d;
561  int i, encoded_len, fill, line_break_count;
562 
563  assert (src != NULL && src_len >= 0 && dest != NULL);
564 
565  encoded_len = 0;
566  line_break_count = 0;
567  p = src;
568 
569  i = 0;
570  while (i < src_len)
571  {
572  /* require maximum char length per line to be 76 */
573  if (line_break_count == MAX_BASE64_LINE_LENGTH)
574  {
575  dest[encoded_len++] = '\n';
576  line_break_count = 0;
577  }
578 
579  /* move forward the source string every three bytes and translate it into four 6-bit bytes */
580  d = (p[i++] << 16); /* the most significant(3rd) bytes */
581  fill = 2; /* assuming 2 paddings needed */
582 
583  if (src_len >= (i + 1))
584  {
585  d += p[i++] << 8; /* the second byte, one padding needed */
586  fill = 1;
587  }
588 
589  if (src_len >= (i + 1)) /* least significant byte, no padding needed */
590  {
591  d += p[i++] << 0;
592  fill = 0;
593  }
594 
595  /* convert 24bit stream into four 6-bit bytes */
596  dest[encoded_len++] = base64_map[(d >> 18) & 0x3f];
597  dest[encoded_len++] = base64_map[(d >> 12) & 0x3f];
598  dest[encoded_len++] = (fill > 1) ? '=' : base64_map[(d >> 6) & 0x3f];
599  dest[encoded_len++] = (fill > 0) ? '=' : base64_map[(d >> 0) & 0x3f];
600 
601  line_break_count += 4;
602  }
603 
604  return NO_ERROR;
605 }
