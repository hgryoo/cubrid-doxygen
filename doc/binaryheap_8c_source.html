50 /*
51  * bh_up_heap () - push an element up the heap to the correct position
52  * return : void
53  * heap (in) : heap
54  * index (in) : index of the element to
55  */
56 static void
57 bh_up_heap (BINARY_HEAP * heap, int index)
58 {
59  if (index == 0)
60  {
61  return;
62  }
63 
64  if (BH_CMP (heap, BH_PARENT (index), index) == BH_LT)
65  {
66  /* swap element with parent */
67  BH_SWAP (heap, index, BH_PARENT (index));
68  /* shift parent */
69  bh_up_heap (heap, BH_PARENT (index));
70  }
71 }

73 /*
74  * bh_down_heap () - push an element down the heap to its correct position
75  * return : void
76  * heap (in) : heap
77  * index (in) : element index
78  */
79 void
80 bh_down_heap (BINARY_HEAP * heap, int index)
81 {
82  int left = BH_LEFT (index);
83  int right = BH_RIGHT (index);
84  int largest = index;
85 
86  if (left <= heap->element_count - 1 && BH_CMP (heap, left, largest) == BH_GT)
87  {
88  largest = left;
89  }
90 
91  if (right <= heap->element_count - 1 && BH_CMP (heap, right, largest) == BH_GT)
92  {
93  largest = right;
94  }
95 
96  if (largest != index)
97  {
98  BH_SWAP (heap, index, largest);
99  bh_down_heap (heap, largest);
100  }
101 }

103 /*
104  * bh_create () - create an empty heap
105  * return : heap or NULL
106  * max_capacity (in): maximum capacity of the heap
107  * elem_size (in) : the size of one heap element.
108  * cmp_func (in) : pointer to the comparison function
109  * cmp_arg (in) : argument to be passed to the comparison function
110  *
111  * Note: This implementation considers the heap to be a MAX heap (i.e.: the root of the heap is the "largest" element).
112  * To use the heap as a MIN heap, callers can use the comparison function to inverse the comparison
113  */
114 BINARY_HEAP *
115 bh_create (THREAD_ENTRY * thread_p, int max_capacity, int elem_size, bh_key_comparator cmp_func, BH_CMP_ARG cmp_arg)
116 {
117  BINARY_HEAP *heap = NULL;
118 
119  heap = (BINARY_HEAP *) db_private_alloc (thread_p, sizeof (BINARY_HEAP));
120  if (heap == NULL)
121  {
122  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (BINARY_HEAP));
123  return NULL;
124  }
125 
126  heap->members = db_private_alloc (thread_p, max_capacity * elem_size);
127  if (heap->members == NULL)
128  {
129  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, max_capacity * elem_size);
130  db_private_free (thread_p, heap);
131  return NULL;
132  }
133  heap->swap_buf = db_private_alloc (thread_p, elem_size);
134  if (heap->swap_buf == NULL)
135  {
136  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, elem_size);
137  db_private_free (thread_p, heap->members);
138  db_private_free (thread_p, heap);
139  return NULL;
140  }
141  memset (heap->members, 0, max_capacity * elem_size);
142  heap->max_capacity = max_capacity;
143  heap->elem_size = elem_size;
144  heap->cmp_func = cmp_func;
145  heap->cmp_arg = cmp_arg;
146  heap->element_count = 0;
147  heap->state = BH_HEAP_CONSISTENT;
148 
149  return heap;
150 }

174 /*
175  * bh_add () - add an element to the heap without preserving the heap property
176  * return : error code if capacity is exceeded or NO_ERROR
177  * heap (in) : heap
178  * elem (in) : new element
179  *
180  * Note: This function is provided as a convenience to speed up heap creation.
181  * It is more efficient to load about half the heap with unordered elements and call bh_build_heap to restore the
182  * heap property.
183  */
184 int
185 bh_add (BINARY_HEAP * heap, void *elem)
186 {
187  assert (heap != NULL);
188  assert (elem != NULL);
189 
190  heap->state = BH_HEAP_INCONSISTENT;
191 
192  if (heap->element_count >= heap->max_capacity)
193  {
194  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_BINARY_HEAP_OUT_OF_RANGE, 0);
195  return ER_BINARY_HEAP_OUT_OF_RANGE;
196  }
197  BH_ELEMENT_COPY (heap, BH_ELEMENT (heap, heap->element_count), elem);
198  heap->element_count++;
199 
200  return NO_ERROR;
201 }

203 /*
204  * bh_insert () - insert an element in the correct position in the heap
205  * return : error code or NO_ERROR
206  * heap (in) : heap
207  * elem (in) : new element
208  */
209 int
210 bh_insert (BINARY_HEAP * heap, void *elem)
211 {
212  assert (heap != NULL);
213  assert (elem != NULL);
214 
215  if (heap->element_count != 0)
216  {
217  assert_release (heap->state == BH_HEAP_CONSISTENT);
218  }
219  else
220  {
221  /* This is the first element, set the consistent property */
222  heap->state = BH_HEAP_CONSISTENT;
223  }
224 
225  if (heap->element_count >= heap->max_capacity)
226  {
227  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_BINARY_HEAP_OUT_OF_RANGE, 0);
228  return ER_BINARY_HEAP_OUT_OF_RANGE;
229  }
230 
231  BH_ELEMENT_COPY (heap, BH_ELEMENT (heap, heap->element_count), elem);
232  heap->element_count++;
233  bh_up_heap (heap, heap->element_count - 1);
234 
235  return NO_ERROR;
236 }

238 /*
239  * bh_try_insert () - insert an element into the heap if heap hasn't reached the full capacity or if new element is
240  * smaller than the top element
241  * return : BH_TRY_INSERT_RESULT
242  * - BH_TRY_INSERT_ACCEPTED if the heap was not full
243  * - BH_TRY_INSERT_REJECTED if the heap was full and new element was rejected
244  * - BH_TRY_INSERT_REPLACED if the heap was full and new element replaced old root.
245  * heap (in) : heap
246  * elem (in) : new element
247  * replaced (out) : value of replaced element (if replaced).
248  */
249 BH_TRY_INSERT_RESULT
250 bh_try_insert (BINARY_HEAP * heap, void *elem, void *replaced)
251 {
252  assert (heap != NULL);
253  assert (elem != NULL);
254 
255  if (heap->element_count < heap->max_capacity)
256  {
257  bh_insert (heap, elem);
258  return BH_TRY_INSERT_ACCEPTED;
259  }
260  /* if root is larger than new element, replace root */
261  if (heap->cmp_func (BH_ROOT (heap), elem, heap->cmp_arg) == BH_GT)
262  {
263  if (replaced != NULL)
264  {
265  BH_ELEMENT_COPY (heap, replaced, BH_ROOT (heap));
266  }
267  bh_replace_max (heap, elem);
268  return BH_TRY_INSERT_REPLACED;
269  }
270  return BH_TRY_INSERT_REJECTED;
271 }

273 /*
274  * bh_build_heap () - restore the heap property in an unordered heap
275  * return : void
276  * heap (in) : heap
277  */
278 void
279 bh_build_heap (BINARY_HEAP * heap)
280 {
281  int i;
282 
283  assert (heap != NULL);
284 
285  if (heap->state == BH_HEAP_CONSISTENT)
286  {
287  /* already consistent, nothing to build */
288  return;
289  }
290 
291  for (i = BH_PARENT (heap->element_count - 1); i >= 0; i--)
292  {
293  bh_down_heap (heap, i);
294  }
295  heap->state = BH_HEAP_CONSISTENT;
296 }

298 /*
299  * bh_extract_max () - pop the largest element from a heap
300  * return : true if heap is not empty and element is extracted, false if heap is empty.
301  * heap (in) : heap
302  * extract_elem (out) : extracted element.
303  */
304 bool
305 bh_extract_max (BINARY_HEAP * heap, void *extract_elem)
306 {
307  assert (heap != NULL);
308  assert (extract_elem != NULL);
309 
310  if (heap->element_count <= 0)
311  {
312  /* empty */
313  return false;
314  }
315  /* extract algorithm is:
316  * 1. replace root with last element on the last level.
317  * 2. recursive down-swap new root with children until the order is correct.
318  */
319  /* decrement element count; this also becomes the index of last element in heap */
320  heap->element_count--;
321  /* Copy root. */
322  BH_ELEMENT_COPY (heap, extract_elem, BH_ROOT (heap));
323  if (heap->element_count > 0)
324  {
325  /* replace root with last element. */
326  BH_ELEMENT_COPY (heap, BH_ROOT (heap), BH_ELEMENT (heap, heap->element_count));
327  /* down-swap until order is correct again */
328  bh_down_heap (heap, 0);
329  }
330  return true;
331 }

371 /*
372  * bh_to_sorted_array () - transform a binary heap into a sorted array
373  * return : void
374  * heap (in) : heap
375  *
376  * Note: the array is sorted smallest to largest (smallest element on the first position)
377  */
378 void
379 bh_to_sorted_array (BINARY_HEAP * heap)
380 {
381  int element_count;
382 
383  assert (heap != NULL);
384 
385  element_count = heap->element_count;
386  /* while has elements, extract max */
387  while (heap->element_count > 1)
388  {
389  BH_SWAP (heap, 0, heap->element_count - 1);
390  heap->element_count--;
391  bh_down_heap (heap, 0);
392  }
393  /* no longer consistent */
394  heap->state = BH_SORTED_ARRAY;
395 
396  /* reset element count */
397  heap->element_count = element_count;
398 }
