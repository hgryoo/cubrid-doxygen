395 static bool
396 broker_drop_one_cas_by_time_to_kill (void)
397 {
398  int cur_appl_server_num, wait_job_cnt;
399  int drop_as_index;
400  T_APPL_SERVER_INFO *drop_as_info;
401 
402  /* DROP UTS */
403  cur_appl_server_num = shm_br->br_info[br_index].appl_server_num;
404  wait_job_cnt = shm_appl->job_queue[0].id + hold_job;
405  wait_job_cnt -= (cur_appl_server_num - num_busy_uts);
406 
407  if (cur_appl_server_num <= shm_br->br_info[br_index].appl_server_min_num || wait_job_cnt > 0)
408  {
409  return false;
410  }
411 
412  drop_as_index = find_drop_as_index ();
413  if (drop_as_index < 0)
414  {
415  return false;
416  }
417 
418  pthread_mutex_lock (&broker_shm_mutex);
419  current_dropping_as_index = drop_as_index;
420  drop_as_info = &shm_appl->as_info[drop_as_index];
421  drop_as_info->service_flag = SERVICE_OFF_ACK;
422  pthread_mutex_unlock (&broker_shm_mutex);
423 
424  CON_STATUS_LOCK (drop_as_info, CON_STATUS_LOCK_BROKER);
425  if (drop_as_info->uts_status == UTS_STATUS_IDLE)
426  {
427  /* do nothing */
428  }
429  else if (drop_as_info->cur_keep_con == KEEP_CON_AUTO && drop_as_info->uts_status == UTS_STATUS_BUSY
430  && drop_as_info->con_status == CON_STATUS_OUT_TRAN
431  && time (NULL) - drop_as_info->last_access_time > shm_br->br_info[br_index].time_to_kill)
432  {
433  drop_as_info->con_status = CON_STATUS_CLOSE;
434  }
435  else
436  {
437  drop_as_info->service_flag = SERVICE_ON;
438  drop_as_index = -1;
439  }
440  CON_STATUS_UNLOCK (drop_as_info, CON_STATUS_LOCK_BROKER);
441 
442  if (drop_as_index >= 0)
443  {
444  pthread_mutex_lock (&broker_shm_mutex);
445  (shm_br->br_info[br_index].appl_server_num)--;
446  (shm_appl->num_appl_server)--;
447  pthread_mutex_unlock (&broker_shm_mutex);
448 
449  stop_appl_server (drop_as_info, br_index, drop_as_index);
450  }
451  current_dropping_as_index = -1;
452 
453  return true;
454 }
