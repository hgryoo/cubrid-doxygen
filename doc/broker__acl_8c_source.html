116  FILE *fd_access_list;

123 #endif
124 
125  fd_access_list = fopen (filename, "r");
126 
127  if (fd_access_list == NULL)
128  {

130  return -1;
131  }
132 
133  is_current_broker_section = false;
134 
135  memset (new_access_info, '\0', sizeof (new_access_info));
136 
137  while (fgets (buf, 1024, fd_access_list))
138  {
139  char *dbname, *dbuser, *ip_file, *p;
140 
141  line++;
142  p = strchr (buf, '#');
143  if (p != NULL)
144  {
145  *p = '\0';
146  }
147 
148  trim (buf);
149 
150  if (buf[0] == '\0')
151  {
152  continue;
153  }
154 
155  if (is_current_broker_section == false && strncmp (buf, "[%", 2) == 0 && buf[strlen (buf) - 1] == ']')
156  {
157  buf[strlen (buf) - 1] = '\0';
158  if (strcasecmp (shm_appl->broker_name, buf + 2) == 0)
159  {

161  continue;

163  }
164  if (is_current_broker_section == false)
165  {
166  continue;
167  }
168 
169  if (strncmp (buf, "[%", 2) == 0 && buf[strlen (buf) - 1] == ']')
170  {
171  buf[strlen (buf) - 1] = '\0';
172  if (strcasecmp (shm_appl->broker_name, buf + 2) != 0)
173  {
174  break;
175  }
176  }
177 
178  if (num_access_list >= ACL_MAX_ITEM_COUNT)
179  {
180  sprintf (admin_err_msg, "%s: error while loading access control file(%s)" " - max item count(%d) exceeded.",
181  shm_appl->broker_name, filename, ACL_MAX_ITEM_COUNT);
182  goto error;
183  }
184 
185  dbname = strtok_r (buf, ACCESS_FILE_DELIMITER, &p);
186  if (dbname == NULL || strlen (dbname) > (ACL_MAX_DBNAME_LENGTH - 1))
187  {
188  sprintf (admin_err_msg,
189  "%s: error while loading access control file(%s:%d)" " - Database name is empty or too long.",
190  shm_appl->broker_name, filename, line);
191  goto error;
192  }
193 
194  dbuser = strtok_r (NULL, ACCESS_FILE_DELIMITER, &p);
195  if (dbuser == NULL || strlen (dbuser) > (ACL_MAX_DBUSER_LENGTH - 1))

200  goto error;
201  }
202 
203  ip_file = p;
204  if (ip_file == NULL)
205  {
206  sprintf (admin_err_msg,

208  shm_appl->broker_name, filename, line);

210  }
211 
212  access_info = access_control_find_access_info (new_access_info, num_access_list, dbname, dbuser);

220

226  for (files = ip_file;; files = NULL)
227  {
228  token = strtok_r (files, IP_FILE_DELIMITER, &save);
229  if (token == NULL)
230  {
231  break;
232  }
233 
234  if (strlen (token) > BROKER_PATH_MAX - 1)
235  {
236  snprintf (admin_err_msg, ADMIN_ERR_MSG_SIZE,
237  "%s: error while loading access control file(%s)" " - a IP file path(%s) is too long",
238  shm_appl->broker_name, filename, token);
239  goto error;
240  }
241 
242  strncpy (path_buf, token, BROKER_PATH_MAX);
243  access_control_repath_file (path_buf);
244  if (access_control_read_ip_info (&(access_info->ip_info), path_buf, admin_err_msg) < 0)

246  goto error;
247  }
248  }
249  }
250 
251  fclose (fd_access_list);
252 
253 #if defined (WINDOWS)
254  MAKE_ACL_SEM_NAME (acl_sem_name, shm_appl->broker_name);
255  uw_sem_wait (acl_sem_name);

259 
260  memcpy (shm_appl->access_info, new_access_info, sizeof (new_access_info));
261  shm_appl->num_access_info = num_access_list;
262  shm_appl->acl_chn++;
263 
264 #if defined(WINDOWS)
265  uw_sem_post (acl_sem_name);
266 #else
267  uw_sem_post (&shm_appl->acl_sem);
268 #endif
269 
270  return 0;
271 
272 error:
273  fclose (fd_access_list);
274 
275  return -1;

302 static int
303 access_control_read_ip_info (IP_INFO * ip_info, char *filename, char *admin_err_msg)
304 {
305  char buf[LINE_MAX];
306  char *save;
307  FILE *fd_ip_list;
308  unsigned char i;
309  unsigned int ln = 0;
310 
311  fd_ip_list = fopen (filename, "r");
312 
313  if (fd_ip_list == NULL)
314  {
315  sprintf (admin_err_msg, "Could not open ip info file(%s)", filename);
316  return -1;
317  }
318 
319  buf[LINE_MAX - 2] = 0;
320  while (fgets (buf, LINE_MAX, fd_ip_list))
321  {
322  char *token, *p;
323  int address_index;
324 
325  ln++;
326  if (buf[LINE_MAX - 2] != 0 && buf[LINE_MAX - 2] != '\n')
327  {
328  sprintf (admin_err_msg, "Error while loading ip info file(%s)" " - %d line is too long", filename, ln);
329  goto error;
330  }
331 
332  p = strchr (buf, '#');
333  if (p != NULL)
334  {
335  *p = '\0';
336  }
337 
338  trim (buf);
339  if (buf[0] == '\0')
340  {
341  continue;
342  }
343 
344  if (ip_info->num_list >= ACL_MAX_IP_COUNT)
345  {
346  sprintf (admin_err_msg, "Error while loading ip info file(%s) line(%d)" " - max ip count(%d) exceeded.",
347  filename, ln, ACL_MAX_IP_COUNT);
348  goto error;
349  }
350 
351  token = strtok_r (buf, ".", &save);
352 
353  address_index = ip_info->num_list * IP_BYTE_COUNT;
354  for (i = 0; i < 4; i++)
355  {
356  if (token == NULL)
357  {
358  sprintf (admin_err_msg, "Error while loading ip info file(%s) line(%d)", filename, ln);
359  goto error;
360  }
361 
362  if (strcmp (token, "*") == 0)
363  {
364  break;
365  }
366  else
367  {
368  int adr = 0, result;
369 
370  result = parse_int (&adr, token, 10);
371 
372  if (result != 0 || adr > 255 || adr < 0)
373  {
374  sprintf (admin_err_msg, "Error while loading ip info file(%s) line(%d)", filename, ln);
375  goto error;
376  }
377 
378  ip_info->address_list[address_index + 1 + i] = (unsigned char) adr;
379  }
380 
381  token = strtok_r (NULL, ".", &save);
382  if (i == 3 && token != NULL)
383  {
384  sprintf (admin_err_msg, "Error while loading ip info file(%s) line(%d)", filename, ln);
385  goto error;
386  }
387  }
388  ip_info->address_list[address_index] = i;
389  ip_info->last_access_time[ip_info->num_list] = 0;
390  ip_info->num_list++;
391  }
392 
393  fclose (fd_ip_list);
394  return 0;
395 
396 error:
397  fclose (fd_ip_list);
398  return -1;
399 }

428 access_control_check_right_internal (T_SHM_APPL_SERVER * shm_as_p, char *dbname, char *dbuser, unsigned char *address)
429 {
430  int i;
431  char *address_ptr;
432  int ret_val = -1;
433  bool local_ip_flag = false;
434 
435  if (address[0] == 127 && address[1] == 0 && address[2] == 0 && address[3] == 1)
436  {
437  local_ip_flag = true;
438  }
439 
440  address_ptr = strchr (dbname, '@');
441  if (address_ptr != NULL)
442  {
443  *address_ptr = '\0';
444  }
445 
446  for (i = 0; i < num_access_info; i++)
447  {
448  if ((strcmp (access_info[i].dbname, "*") == 0

450  && (strcmp (access_info[i].dbuser, "*") == 0
451  || strncasecmp (access_info[i].dbuser, dbuser, ACL_MAX_DBUSER_LENGTH) == 0))
452  {
453  if (access_control_check_ip (shm_as_p, &access_info[i].ip_info, address, i) == 0)
454  {
455  ret_val = 0;
456  break;
457  }
458 
459  if (local_ip_flag == true
460  && access_control_check_ip (shm_as_p, &access_info[i].ip_info, shm_as_p->local_ip_addr, i) == 0)
461  {
462  break;
463  }
464  }
465  }
466 
467  if (address_ptr != NULL)
468  {
469  *address_ptr = '@';
470  }
471 
472  if (local_ip_flag == true)
473  {
474  return 0;
475  }
476 
477  return ret_val;
