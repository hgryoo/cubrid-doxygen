53 static void
54 admin_log_write (const char *log_file, const char *msg)
55 {
56  FILE *fp;
57  struct tm ct;
58  time_t ts;
59 
60  fp = fopen (log_file, "a");
61  if (fp != NULL)
62  {
63  ts = time (NULL);
64  memset (&ct, 0x0, sizeof (struct tm));
65  localtime_r (&ts, &ct);
66  fprintf (fp, "%d/%02d/%02d %02d:%02d:%02d %s\n", ct.tm_year + 1900, ct.tm_mon + 1, ct.tm_mday, ct.tm_hour,
67  ct.tm_min, ct.tm_sec, msg);
68  fclose (fp);
69  }
70  else
71  {
72  printf ("cannot open admin log file [%s]\n", log_file);
73  }
74 }
