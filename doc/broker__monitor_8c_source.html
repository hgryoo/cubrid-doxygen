450 int
451 main (int argc, char **argv)
452 {
453  T_BROKER_INFO br_info[MAX_BROKER_NUM];
454  int num_broker, master_shm_id;
455  int err, i;
456  char *br_vector;
457 #if defined(WINDOWS)
458 #else
459  WINDOW *win;
460 #endif
461  time_t time_old, time_cur;
462  double elapsed_time;
463 
464  if (argc == 2 && strcmp (argv[1], "--version") == 0)
465  {
466  fprintf (stderr, "VERSION %s\n", makestring (BUILD_NUMBER));
467  return 3;
468  }
469 
470  err = broker_config_read (NULL, br_info, &num_broker, &master_shm_id, NULL, 0, NULL, NULL, NULL);
471  if (err < 0)
472  {
473  return 2;
474  }
475 
476  ut_cd_work_dir ();
477 
478  shm_br = (T_SHM_BROKER *) uw_shm_open (master_shm_id, SHM_BROKER, SHM_MODE_MONITOR);
479  if (shm_br == NULL)
480  {
481  /* This means we have to launch broker */
482  fprintf (stdout, "master shared memory open error[0x%x]\r\n", master_shm_id);
483  return 1;
484  }
485  if (shm_br->num_broker < 1 || shm_br->num_broker > MAX_BROKER_NUM)
486  {
487  PRINT_AND_LOG_ERR_MSG ("broker configuration error\r\n");
488  return 3;
489  }
490 
491  br_vector = (char *) malloc (shm_br->num_broker);
492  if (br_vector == NULL)
493  {
494  PRINT_AND_LOG_ERR_MSG ("memory allocation error\r\n");
495  return 3;
496  }
497  for (i = 0; i < shm_br->num_broker; i++)
498  {
499  br_vector[i] = 0;
500  }
501 
502  if (get_args (argc, argv, br_vector) < 0)
503  {
504  free (br_vector);
505  return 3;
506  }
507 
508  if (refresh_sec > 0 && !tty_mode)
509  {
510 #if defined(WINDOWS)
511  h_console = GetStdHandle (STD_OUTPUT_HANDLE);
512  if (h_console == NULL)
513  {
514  refresh_sec = 0;
515  }
516  if (!GetConsoleScreenBufferInfo (h_console, &scr_info))
517  {
518  scr_info.dwSize.X = 80;
519  scr_info.dwSize.Y = 50;
520  }
521 // FillConsoleOutputCharacter(h_console, ' ', scr_info.dwSize.X * scr_info.dwSize.Y, top_left_pos, &size);
522 #else
523  win = initscr ();
524  timeout (refresh_sec * 1000);
525  noecho ();
526 #endif
527  }
528 
529  (void) time (&time_old);
530  time_old--;
531 
532  while (1)
533  {
534  (void) time (&time_cur);
535  elapsed_time = difftime (time_cur, time_old);
536 
537  if (refresh_sec > 0 && !tty_mode)
538  {
539  move (0, 0);
540  refresh ();
541  }
542 
543  if (shm_br == NULL || shm_br->magic == 0)
544  {
545  if (shm_br)
546  {
547  uw_shm_detach (shm_br);
548  }
549 
550  shm_br = (T_SHM_BROKER *) uw_shm_open (master_shm_id, SHM_BROKER, SHM_MODE_MONITOR);
551  }
552  else
553  {
554  if (monitor_flag & BROKER_MONITOR_FLAG_MASK)
555  {
556  if ((monitor_flag & ~BROKER_MONITOR_FLAG_MASK) != 0)
557  {
558  print_newline ();
559  }
560  brief_monitor (br_vector, MONITOR_T_BROKER, elapsed_time);
561  }
562 
563  if (monitor_flag & SHARDDB_MONITOR_FLAG_MASK)
564  {
565  if ((monitor_flag & ~SHARDDB_MONITOR_FLAG_MASK) != 0)
566  {
567  print_newline ();
568  str_out ("<SHARD INFO>");
569  print_newline ();
570  }
571  brief_monitor (br_vector, MONITOR_T_SHARDDB, elapsed_time);
572  }
573 
574  if (monitor_flag & PROXY_MONITOR_FLAG_MASK)
575  {
576  if ((monitor_flag & ~PROXY_MONITOR_FLAG_MASK) != 0)
577  {
578  print_newline ();
579  str_out ("<PROXY INFO>");
580  print_newline ();
581  }
582  brief_monitor (br_vector, MONITOR_T_PROXY, elapsed_time);
583  }
584 
585  if (monitor_flag & METADATA_MONITOR_FLAG_MASK)
586  {
587  if ((monitor_flag & ~METADATA_MONITOR_FLAG_MASK) != 0)
588  {
589  print_newline ();
590  str_out ("<SHARD METADATA INFO>");
591  print_newline ();
592  }
593  metadata_monitor (elapsed_time);
594  }
595 
596  if (monitor_flag & CLIENT_MONITOR_FLAG_MASK)
597  {
598  if ((monitor_flag & ~CLIENT_MONITOR_FLAG_MASK) != 0)
599  {
600  print_newline ();
601  str_out ("<CLIENT INFO>");
602  print_newline ();
603  }
604  client_monitor ();
605  }
606 
607  if (monitor_flag & UNUSABLE_DATABASES_FLAG_MASK)
608  {
609  if ((monitor_flag & ~UNUSABLE_DATABASES_FLAG_MASK) != 0)
610  {
611  print_newline ();
612  }
613  unusable_databases_monitor ();
614  }
615 
616  if (monitor_flag == 0)
617  {
618  appl_monitor (br_vector, elapsed_time);
619  }
620  }
621 
622  if (refresh_sec > 0 && !tty_mode)
623  {
624  int in_ch = 0;
625 
626  refresh ();
627  clrtobot ();
628  move (0, 0);
629  refresh ();
630  in_ch = get_char ();
631 
632 #if defined (WINDOWS)
633  if (shm_br != NULL && shm_br->magic == 0)
634  {
635  uw_shm_detach (shm_br);
636  shm_br = NULL;
637  }
638 #endif
639  if (in_ch == 'q')
640  {
641  break;
642  }
643  else if (in_ch == ' ' || in_ch == '\r' || in_ch == '\n' || in_ch == ' ')
644  {
645  clear ();
646  refresh ();
