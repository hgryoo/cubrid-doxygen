72 /* con_status lock/unlock */
73 #define CON_STATUS_LOCK_BROKER 0
74 #define CON_STATUS_LOCK_CAS 1
75 #if defined(WINDOWS)
76 #define CON_STATUS_LOCK_INIT(AS_INFO) \
77  do { \
78  (AS_INFO)->con_status_lock[CON_STATUS_LOCK_BROKER] = FALSE; \
79  (AS_INFO)->con_status_lock[CON_STATUS_LOCK_CAS] = FALSE; \
80  (AS_INFO)->con_status_lock_turn = CON_STATUS_LOCK_BROKER; \
81  } \
82  while (0)
83 
84 #define CON_STATUS_LOCK_DESTROY(AS_INFO)
85 
86 #define CON_STATUS_LOCK(AS_INFO, LOCK_OWNER) \
87  do { \
88  int LOCK_WAITER = (LOCK_OWNER == CON_STATUS_LOCK_BROKER) ? CON_STATUS_LOCK_CAS : CON_STATUS_LOCK_BROKER; \
89  (AS_INFO)->con_status_lock[LOCK_OWNER] = TRUE; \
90  (AS_INFO)->con_status_lock_turn = LOCK_WAITER; \
91  while (((AS_INFO)->con_status_lock[LOCK_WAITER] == TRUE) && \
92  ((AS_INFO)->con_status_lock_turn == LOCK_WAITER)) \
93  { \
94  SLEEP_MILISEC(0, 10); \
95  } \
96  } while (0)
97 
98 #define CON_STATUS_UNLOCK(AS_INFO, LOCK_OWNER) \
99  (AS_INFO)->con_status_lock[LOCK_OWNER] = FALSE
100 #else /* WINDOWS */
101 #define CON_STATUS_LOCK_INIT(AS_INFO) \
102  uw_sem_init (&((AS_INFO)->con_status_sem));
103 
104 #define CON_STATUS_LOCK_DESTROY(AS_INFO) \
105  uw_sem_destroy (&((AS_INFO)->con_status_sem));
