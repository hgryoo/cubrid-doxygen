404 /* Add new item to b-tree node queue */
405 #define BTREE_NODE_SCAN_ADD_PAGE_TO_QUEUE(bns, node) \
406  if ((bns)->queue_tail == NULL) \
407  { \
408  (bns)->queue_head = (bns)->queue_tail = node; \
409  } \
410  else \
411  { \
412  (bns)->queue_tail->next = node; \
413  (bns)->queue_tail = node; \
414  }
415 
416 /* Pop first item from b-tree node queue */
417 #define BTREE_NODE_SCAN_POP_PAGE_FROM_QUEUE(bns, node) \
418  if ((bns)->queue_head == NULL) \
419  { \
420  node = NULL; \
421  } \
422  else \
423  { \
424  if ((bns)->queue_head == (bns)->queue_tail) \
425  { \
426  /* Only one item in queue */ \
427  node = (bns)->queue_head; \
428  (bns)->queue_tail = NULL; \
429  (bns)->queue_head = NULL; \
430  } \
431  else \
432  { \
433  node = (bns)->queue_head; \
434  (bns)->queue_head = node->next; \

456 /* When MVCC is enabled, btree_delete/btree_insert functionality are extended

461 enum btree_op_purpose
462 {
463  BTREE_OP_NO_OP, /* No op. */
464 
465  BTREE_OP_INSERT_NEW_OBJECT, /* Insert a new object into b-tree along with its insert MVCCID. */
466  BTREE_OP_INSERT_MVCC_DELID, /* Insert delete MVCCID for object when deleted. */
467  BTREE_OP_INSERT_MARK_DELETED, /* Mark object as deleted. This is used on a unique index of a non-MVCC class. It is
468  * very similar to BTREE_OP_INSERT_MVCC_DELID. The differences are: 1. The context they
469  * are used for. MVCC delete is used to delete from MVCC-enabled classes. Mark deleted
470  * is used for unique indexes of MVCC-disabled classes like db_serial. 2. Mark deleted
471  * is followed by a postpone operation which removes the object after commit. 3. Mark
472  * deleted is not vacuumed. Object will be "cleaned" on commit/rollback. */
473  BTREE_OP_INSERT_UNDO_PHYSICAL_DELETE, /* Undo of physical delete. */
474 
475  BTREE_OP_DELETE_OBJECT_PHYSICAL, /* Physically delete an object from b-tree when MVCC is enabled. */
476  BTREE_OP_DELETE_OBJECT_PHYSICAL_POSTPONED, /* Physical delete was postponed. */
477  BTREE_OP_DELETE_UNDO_INSERT, /* Undo insert */
478  BTREE_OP_DELETE_UNDO_INSERT_UNQ_MULTIUPD, /* Undo insert into unique index, when multi-update exception to unique
479  * constraint violation is applied. Previous visible object must be
480  * returned to first position in record. */
481  BTREE_OP_DELETE_UNDO_INSERT_DELID, /* Remove only delete MVCCID for an object in b-tree. It is called when object
482  * deletion is roll-backed. */
483  BTREE_OP_DELETE_VACUUM_OBJECT, /* All object info is removed from b-tree. It is called by vacuum when the
484  * object becomes completely invisible. */
485  BTREE_OP_DELETE_VACUUM_INSID, /* Remove only insert MVCCID for an object in b-tree. It is called by vacuum when the
486  * object becomes visible to all running transactions. */
487 
488  BTREE_OP_NOTIFY_VACUUM, /* Notify vacuum of an object in need of cleanup. */
489 
490  /* Below purposes are used during online index loading. */
491  BTREE_OP_ONLINE_INDEX_IB_INSERT, /* Insert done by the Index Builder. */
492  BTREE_OP_ONLINE_INDEX_IB_DELETE, /* Delete done by the Index Builder. */
