73 cfg_get_dbinfo (char *alias, char *dbinfo)
74 {
75  FILE *file;
76  char *save, *token;
77  char delim[] = "|";
78  char filename[BROKER_PATH_MAX];
79  char line[DBINFO_MAX_LENGTH];
80 
81  if (shm_appl->db_connection_file[0] == '\0')
82  {
83 #if defined(CAS_FOR_ORACLE)
84  get_cubrid_file (FID_CAS_FOR_ORACLE_DBINFO, filename, BROKER_PATH_MAX);
85 #elif defined(CAS_FOR_MYSQL)
86  get_cubrid_file (FID_CAS_FOR_MYSQL_DBINFO, filename, BROKER_PATH_MAX);
87 #endif
88  }
89  else
90  {
91  if (IS_ABS_PATH (shm_appl->db_connection_file))
92  {
93  strncpy (filename, shm_appl->db_connection_file, BROKER_PATH_MAX - 1);
94  }
95  else
96  {
97  envvar_confdir_file (filename, BROKER_PATH_MAX, shm_appl->db_connection_file);
98  }
99  }
100 
101  file = fopen (filename, "r");
102  if (file == NULL)
103  {
104  return ERROR_INFO_SET (CAS_ER_OPEN_FILE, CAS_ERROR_INDICATOR);
105  }
106 
107  while (fgets (line, DBINFO_MAX_LENGTH - 1, file) != NULL)
108  {
109  if (line[0] == '\0' || line[0] == '#')
110  {
111  continue;
112  }
113  token = strtok_r (line, delim, &save);
114  if (token == NULL || strcmp (token, alias) != 0)
115  {
116  continue;
117  }
118  token = strtok_r (NULL, delim, &save);
119  if (token == NULL)
120  {
121  continue;
122  }
123 
124  strcpy (dbinfo, token);
125  fclose (file);
126  return 0;
127  }
128 
129  fclose (file);
130  return ERROR_INFO_SET (CAS_ER_ARGS, CAS_ERROR_INDICATOR);
131 }

133 int
134 cfg_read_dbinfo (DB_INFO ** db_info_p)
135 {
136  FILE *file;
137  char filename[BROKER_PATH_MAX];
138  char line[DBINFO_MAX_LENGTH];
139  char *str = NULL;
140  DB_INFO *databases, *db, *last;
141 
142  databases = last = NULL;
143 
144  if (shm_appl->db_connection_file[0] == '\0')
145  {
146 #if defined(CAS_FOR_ORACLE)
147  get_cubrid_file (FID_CAS_FOR_ORACLE_DBINFO, filename, BROKER_PATH_MAX);
148 #elif defined(CAS_FOR_MYSQL)
149  get_cubrid_file (FID_CAS_FOR_MYSQL_DBINFO, filename, BROKER_PATH_MAX);
150 #endif
151  }
152  else
153  {
154  if (IS_ABS_PATH (shm_appl->db_connection_file))
155  {
156  strncpy (filename, shm_appl->db_connection_file, BROKER_PATH_MAX);
157  filename[BROKER_PATH_MAX - 1] = 0;
158  }
159  else
160  {
161  envvar_confdir_file (filename, BROKER_PATH_MAX, shm_appl->db_connection_file);
162  }
163  }
164 
165  file = fopen (filename, "r");
166  if (file == NULL)
167  {
168  return ERROR_INFO_SET (CAS_ER_OPEN_FILE, CAS_ERROR_INDICATOR);
169  }
170 
171  while (fgets (line, DBINFO_MAX_LENGTH - 1, file) != NULL)
172  {
173  str = char_get_next (line);
174  if (*str != '\0' && *str != '#')
175  {
176  db = (DB_INFO *) malloc (sizeof (DB_INFO));
177  if (db == NULL)
178  {
179  if (databases != NULL)
180  {
181  cfg_free_dbinfo_all (databases);
182  }
183  *db_info_p = NULL;
184  fclose (file);
185  return ERROR_INFO_SET (CAS_ER_NO_MORE_MEMORY, CAS_ERROR_INDICATOR);
186  }
187  db->next = NULL;
188  str = cfg_pop_token (str, &db->alias);
189 #if defined(CAS_FOR_ORACLE)
190  str = cfg_pop_token (str, &db->dbinfo);
191 #elif defined(CAS_FOR_MYSQL)
192  str = cfg_pop_token (str, &db->dbinfo);
193 #endif
194  if (databases == NULL)
195  {
196  databases = db;
197  }
198  else
199  {
200  last->next = db;
201  }
202  last = db;
203 #if defined(CAS_FOR_ORACLE)
204  if (db->alias == NULL || db->dbinfo == NULL)
205 #elif defined(CAS_FOR_MYSQL)
206  if (db->alias == NULL || db->dbinfo == NULL)
207 #endif
208  {
209  if (databases != NULL)
210  {
211  cfg_free_dbinfo_all (databases);
212  }
213  *db_info_p = NULL;
214  fclose (file);
215  return ERROR_INFO_SET (CAS_ER_ARGS, CAS_ERROR_INDICATOR);
216  }
217  }
218  }
219  fclose (file);
220 
221  *db_info_p = databases;
222  return 0;
223 }

283 /*
284  * cfg_pop_token() - This looks in the buffer for the next token which is define as
285  * a string of characters started by |
286  * return: char
287  * str_p(in): buffer with tokens
288  * token_p(in/out): returned next token string
289  *
290  * Note : When found the token characters are copied into a new string
291  * and returned.
292  * The pointer to the first character following the new token in
293  * the buffer is returned.
294  */
295 static char *
296 cfg_pop_token (char *str_p, char **token_p)
297 {
298  char *p, *end, *token = NULL;
299  char delim[] = "|";
300  int length;
301 
302  token = NULL;
303  p = str_p;
304  while (char_is_delim ((int) *p, (int) *delim) && *p != '\0')
305  {
306  p++;
307  }
308  end = p;
309  while (!char_is_delim ((int) *end, (int) *delim) && *end != '\0')
310  {
311  end++;
312  }
313 
314  length = (int) (end - p);
315 
316  token = (char *) malloc (length + 1);
317 
318  if (length > 0 && token != NULL)
319  {
320  strncpy (token, p, length);
321  token[length] = '\0';
322  }
323 
324  *token_p = token;
325  return (end);
326 }
