149 FN_RETURN
150 fn_end_tran (SOCKET sock_fd, int argc, void **argv, T_NET_BUF * net_buf, T_REQ_INFO * req_info)
151 {
152  int tran_type;
153  int err_code;
154  int elapsed_sec = 0, elapsed_msec = 0;
155  struct timeval end_tran_begin, end_tran_end;
156 #ifndef LIBCAS_FOR_JSP
157  int timeout;
158 #endif /* !LIBCAS_FOR_JSP */
159 
160  if (argc < 1)
161  {
162  ERROR_INFO_SET (CAS_ER_ARGS, CAS_ERROR_INDICATOR);
163  NET_BUF_ERR_SET (net_buf);
164  return FN_KEEP_CONN;
165  }
166 
167  net_arg_get_char (tran_type, argv[0]);
168  if (tran_type != CCI_TRAN_COMMIT && tran_type != CCI_TRAN_ROLLBACK)
169  {
170  ERROR_INFO_SET (CAS_ER_TRAN_TYPE, CAS_ERROR_INDICATOR);
171  NET_BUF_ERR_SET (net_buf);
172  return FN_KEEP_CONN;
173  }
174 
175  cas_log_write (0, false, "end_tran %s", get_tran_type_str (tran_type));
176 
177  gettimeofday (&end_tran_begin, NULL);
178 
179  err_code = ux_end_tran ((char) tran_type, false);
180 
181  if ((tran_type == CCI_TRAN_ROLLBACK) && (req_info->client_version < CAS_MAKE_VER (8, 2, 0)))
182  {
183  /* For backward compatibility */
184  cas_send_result_flag = FALSE;
185  }
186 
187  gettimeofday (&end_tran_end, NULL);
188  ut_timeval_diff (&end_tran_begin, &end_tran_end, &elapsed_sec, &elapsed_msec);
189 
190  cas_log_write (0, false, "end_tran %s%d time %d.%03d%s", err_code < 0 ? "error:" : "", err_info.err_number,
191  elapsed_sec, elapsed_msec, get_error_log_eids (err_info.err_number));
192 
193  if (err_code < 0)
194  {
195  NET_BUF_ERR_SET (net_buf);
196  req_info->need_rollback = TRUE;
197  }
198  else
199  {
200  net_buf_cp_int (net_buf, 0, NULL);
201  req_info->need_rollback = FALSE;
202  }
203 
204 #ifndef LIBCAS_FOR_JSP
205  timeout =
206  ut_check_timeout (&tran_start_time, &end_tran_end, shm_appl->long_transaction_time, &elapsed_sec, &elapsed_msec);
207  if (timeout >= 0)
208  {
209  as_info->num_long_transactions %= MAX_DIAG_DATA_VALUE;
210  as_info->num_long_transactions++;
211  }
212  if (err_code < 0 || errors_in_transaction > 0)
213  {
214  cas_log_end (SQL_LOG_MODE_ERROR, elapsed_sec, elapsed_msec);
215  errors_in_transaction = 0;
216  }
217  else
218  {
219  if (timeout >= 0 || query_timeout >= 0)
220  {
221  cas_log_end (SQL_LOG_MODE_TIMEOUT, elapsed_sec, elapsed_msec);
222  }
223  else
224  {
225  cas_log_end (SQL_LOG_MODE_NONE, elapsed_sec, elapsed_msec);
226  }
227  }
228  gettimeofday (&tran_start_time, NULL);
229  gettimeofday (&query_start_time, NULL);
230  tran_timeout = 0;
231  query_timeout = 0;
232 
233  assert (as_info->con_status == CON_STATUS_IN_TRAN);
234  as_info->con_status = CON_STATUS_OUT_TRAN;
235  as_info->transaction_start_time = (time_t) 0;
236  if (as_info->cas_log_reset)
237  {
238  cas_log_reset (broker_name);
239  }
240  if (as_info->cas_slow_log_reset)
241  {
242  cas_slow_log_reset (broker_name);
243  }
244  if (shm_appl->sql_log2 != as_info->cur_sql_log2)
245  {
246  sql_log2_end (false);
247  as_info->cur_sql_log2 = shm_appl->sql_log2;
248  sql_log2_init (broker_name, shm_as_index, as_info->cur_sql_log2, true);
249  }
250 
251  if (!ux_is_database_connected ())
252  {
253  cas_log_debug (ARG_FILE_LINE, "fn_end_tran: !ux_is_database_connected()");
254  return FN_CLOSE_CONN;
255  }
256  else if (restart_is_needed () || as_info->reset_flag == TRUE)
257  {
258  cas_log_debug (ARG_FILE_LINE, "fn_end_tran: restart_is_needed() || reset_flag");
259  return FN_KEEP_SESS;
260  }
261  return FN_KEEP_CONN;
262 #endif /* !LIBCAS_FOR_JSP */
263 
264  return FN_CLOSE_CONN;
265 }

299 static FN_RETURN
300 fn_prepare_internal (SOCKET sock_fd, int argc, void **argv, T_NET_BUF * net_buf, T_REQ_INFO * req_info,
301  int *ret_srv_h_id)
302 {
303  char *sql_stmt;
304  char flag;
305  char auto_commit_mode;
306  int sql_size;
307  int srv_h_id;
308  T_SRV_HANDLE *srv_handle;
309  int i;
310 
311  if (argc < 2)
312  {
313  ERROR_INFO_SET (CAS_ER_ARGS, CAS_ERROR_INDICATOR);
314  NET_BUF_ERR_SET (net_buf);
315  return FN_KEEP_CONN;
316  }
317 
318  net_arg_get_str (&sql_stmt, &sql_size, argv[0]);
319  net_arg_get_char (flag, argv[1]);
320  if (argc > 2)
321  {
322  net_arg_get_char (auto_commit_mode, argv[2]);
323  if (cas_shard_flag == OFF)
324  {
325  for (i = 3; i < argc; i++)
326  {
327  int deferred_close_handle;
328  net_arg_get_int (&deferred_close_handle, argv[i]);
329  cas_log_write (0, true, "close_req_handle srv_h_id %d", deferred_close_handle);
330  hm_srv_handle_free (deferred_close_handle);
331  }
332  }
333  }
334  else
335  {
336  auto_commit_mode = FALSE;
337  }
338 
339 #if 0
340  ut_trim (sql_stmt);
341 #endif
342 
343 #ifndef LIBCAS_FOR_JSP
344  gettimeofday (&query_start_time, NULL);
345  query_timeout = 0;
346 #endif /* !LIBCAS_FOR_JSP */
347 
348  cas_log_write_nonl (query_seq_num_next_value (), false, "prepare %d ", flag);
349  cas_log_write_query_string (sql_stmt, sql_size - 1);
350 
351 #ifndef LIBCAS_FOR_JSP
352  SQL_LOG2_COMPILE_BEGIN (as_info->cur_sql_log2, ((const char *) sql_stmt));
353 
354  /* append query string to as_info->log_msg */
355  if (sql_stmt)
356  {
357  char *s, *t;
358  size_t l;
359 
360  for (s = as_info->log_msg, l = 0; *s && l < SHM_LOG_MSG_SIZE - 1; s++, l++)
361  {
362  /* empty body */
363  }
364  *s++ = ' ';
365  l++;
366  for (t = sql_stmt; *t && l < SHM_LOG_MSG_SIZE - 1; s++, t++, l++)
367  {
368  *s = *t;
369  }
370  *s = '\0';
371  }
372 #endif
373 
374  srv_h_id = ux_prepare (sql_stmt, flag, auto_commit_mode, net_buf, req_info, query_seq_num_current_value ());
375 
376  if (ret_srv_h_id != NULL)
377  {
378  /* this ret_srv_h_id used by following fn_execute_internal() function */
379  *ret_srv_h_id = srv_h_id;
380  }
381 
382  srv_handle = hm_find_srv_handle (srv_h_id);
383 
384  cas_log_write (query_seq_num_current_value (), false, "prepare srv_h_id %s%d%s%s", (srv_h_id < 0) ? "error:" : "",
385  (srv_h_id < 0) ? err_info.err_number : srv_h_id, (srv_handle != NULL
386  && srv_handle->use_plan_cache) ? " (PC)" : "",
387  get_error_log_eids (err_info.err_number));
388 
389 #ifndef LIBCAS_FOR_JSP
390  if (srv_h_id < 0)
391  {
392  update_error_query_count (as_info, &err_info);
393  }
394 #endif /* !LIBCAS_FOR_JSP */
395 
396  return FN_KEEP_CONN;
397 }

407 static FN_RETURN
408 fn_execute_internal (SOCKET sock_fd, int argc, void **argv, T_NET_BUF * net_buf, T_REQ_INFO * req_info,
409  int *prepared_srv_h_id)
410 {
411  int srv_h_id;
412  char flag;
413  int max_col_size;
414  int bind_value_index;
415  int max_row = 0;
416  int ret_code;
417  int param_mode_size = 0;
418  char forward_only_cursor = 0;
419  char auto_commit_mode = 0;
420  char *param_mode = NULL;
421  T_SRV_HANDLE *srv_handle;
422  const char *exec_func_name;
423  int argc_mod_2;
424  int (*ux_exec_func) (T_SRV_HANDLE *, char, int, int, int, void **, T_NET_BUF *, T_REQ_INFO *, CACHE_TIME *, int *);
425  char fetch_flag = 0;
426  CACHE_TIME clt_cache_time, *clt_cache_time_ptr;
427  int client_cache_reusable = FALSE;
428  int elapsed_sec = 0, elapsed_msec = 0;
429  struct timeval exec_begin, exec_end;
430  int app_query_timeout;
431  bool client_supports_query_timeout = false;
432  char *eid_string;
433  int err_number_execute;
434  int arg_idx = 0;
435 
436 #ifndef LIBCAS_FOR_JSP
437 #if !defined(CAS_FOR_ORACLE) && !defined(CAS_FOR_MYSQL)
438  char *plan = NULL;
439 #endif
440 #endif
441 
442  bind_value_index = 9;
443  /*
444  * query timeout is transferred from a driver only if protocol version 1
445  * or above.
446  */
447  if (req_info->client_version >= CAS_PROTO_MAKE_VER (PROTOCOL_V1))
448  {
449  client_supports_query_timeout = true;
450  bind_value_index++;
451  }
452 
453  argc_mod_2 = bind_value_index % 2;
454 
455  if ((argc < bind_value_index) || (argc % 2 != argc_mod_2))
456  {
457  ERROR_INFO_SET (CAS_ER_ARGS, CAS_ERROR_INDICATOR);
458  NET_BUF_ERR_SET (net_buf);
459  return FN_KEEP_CONN;
460  }
461 
462  if (prepared_srv_h_id != NULL)
463  {
464  srv_h_id = *prepared_srv_h_id;
465  }
466  else
467  {
468  net_arg_get_int (&srv_h_id, argv[arg_idx++]);
469  }
470  srv_handle = hm_find_srv_handle (srv_h_id);
471 
472 #if defined(CAS_FOR_ORACLE) || defined(CAS_FOR_MYSQL)
473  if (srv_handle == NULL)
474 #else /* CAS_FOR_ORACLE || CAS_FOR_MYSQL */
475  if (srv_handle == NULL || srv_handle->schema_type >= CCI_SCH_FIRST)
476 #endif /* CAS_FOR_ORACLE || CAS_FOR_MYSQL */
477  {
478  ERROR_INFO_SET (CAS_ER_SRV_HANDLE, CAS_ERROR_INDICATOR);
479  NET_BUF_ERR_SET (net_buf);
480  return FN_KEEP_CONN;
481  }
482 
483 #if defined(CAS_FOR_ORACLE) || defined(CAS_FOR_MYSQL)
484  srv_handle->next_cursor_pos = 1;
485 #endif
486 
487  net_arg_get_char (flag, argv[arg_idx++]);
488  net_arg_get_int (&max_col_size, argv[arg_idx++]);
489  net_arg_get_int (&max_row, argv[arg_idx++]);
490  net_arg_get_str (&param_mode, &param_mode_size, argv[arg_idx++]);
491  if (prepared_srv_h_id != NULL)
492  {
493 #if defined(CAS_FOR_ORACLE) || defined(CAS_FOR_MYSQL)
494  if (srv_handle->stmt_type == CUBRID_STMT_SELECT)
495 #else
496  if (srv_handle->q_result->stmt_type == CUBRID_STMT_SELECT)
497 #endif
498  {
499  fetch_flag = 1;
500  }
501  else
502  {
503  fetch_flag = 0;
504  }
505 
506  if (srv_handle->auto_commit_mode != 0)
507  {
508  auto_commit_mode = true;
509  forward_only_cursor = true;
510  }
511  else
512  {
513  auto_commit_mode = false;
514  forward_only_cursor = false;
515  }
516  }
517  else
518  {
519  /* PROTOCOL_V2 is used only 9.0.0 */
520  if (DOES_CLIENT_MATCH_THE_PROTOCOL (req_info->client_version, PROTOCOL_V2))
521  {
522 #if defined(CAS_FOR_ORACLE)
523  if (srv_handle->stmt_type == CUBRID_STMT_SELECT)
524 #elif defined(CAS_FOR_MYSQL)
525  if (srv_handle->stmt_type == CUBRID_STMT_SELECT || srv_handle->stmt_type == CUBRID_STMT_CALL_SP)
526 #else
527  if (srv_handle->q_result->stmt_type == CUBRID_STMT_SELECT)
528 #endif
529  {
530  fetch_flag = 1;
531  }
532  else
533  {
534  fetch_flag = 0;
535  }
536 
537  arg_idx++; /* skip fetch_flag from driver */
538  }
539  else
540  {
541  net_arg_get_char (fetch_flag, argv[arg_idx++]);
542  }
543 
544  net_arg_get_char (auto_commit_mode, argv[arg_idx++]);
545  net_arg_get_char (forward_only_cursor, argv[arg_idx++]);
546  }
547 
548  clt_cache_time_ptr = &clt_cache_time;
549  net_arg_get_cache_time (clt_cache_time_ptr, argv[arg_idx++]);
550 
551  if (client_supports_query_timeout == true)
552  {
553  net_arg_get_int (&app_query_timeout, argv[arg_idx++]);
554 
555  if (!DOES_CLIENT_UNDERSTAND_THE_PROTOCOL (req_info->client_version, PROTOCOL_V2))
556  {
557  /* protocol version v1 driver send query timeout in second */
558  app_query_timeout *= 1000;
559  }
560  }
561  else
562  {
563  app_query_timeout = 0;
564  }
565 
566 #ifndef LIBCAS_FOR_JSP
567  if (shm_appl->max_string_length >= 0)
568  {
569  if (max_col_size <= 0 || max_col_size > shm_appl->max_string_length)
570  max_col_size = shm_appl->max_string_length;
571  }
572 #endif /* LIBCAS_FOR_JSP */
573 
574 #if !defined(CAS_FOR_ORACLE) && !defined(CAS_FOR_MYSQL)
575  set_query_timeout (srv_handle, app_query_timeout);
576 #endif /* !CAS_FOR_ORACLE && !CAS_FOR_MYSQL */
577 
578  srv_handle->auto_commit_mode = auto_commit_mode;
579  srv_handle->forward_only_cursor = forward_only_cursor;
580 
581  if (srv_handle->prepare_flag & CCI_PREPARE_CALL)
582  {
583  exec_func_name = "execute_call";
584  ux_exec_func = ux_execute_call;
585 #if !defined(CAS_FOR_MYSQL)
586  if (param_mode)
587  {
588  ux_call_info_cp_param_mode (srv_handle, param_mode, param_mode_size);
589  }
590 #endif /* !CAS_FOR_MYSQL */
591  }
592  else if (flag & CCI_EXEC_QUERY_ALL)
593  {
594  exec_func_name = "execute_all";
595  ux_exec_func = ux_execute_all;
596  }
597  else
598  {
599  exec_func_name = "execute";
600  ux_exec_func = ux_execute;
601  }
602 
603 #ifndef LIBCAS_FOR_JSP
604  if (srv_handle->is_pooled)
605  {
606  gettimeofday (&query_start_time, NULL);
607  query_timeout = 0;
608  }
