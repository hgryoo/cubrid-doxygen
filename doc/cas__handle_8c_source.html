62 int
63 hm_new_srv_handle (T_SRV_HANDLE ** new_handle, unsigned int seq_num)
64 {
65  int i;
66  int new_max_srv_handle;
67  int new_handle_id = 0;
68  T_SRV_HANDLE **new_srv_handle_table = NULL;
69  T_SRV_HANDLE *srv_handle;
70 
71 #if !defined(LIBCAS_FOR_JSP)
72  if (cas_shard_flag == OFF && current_handle_count >= shm_appl->max_prepared_stmt_count)
73  {
74  return ERROR_INFO_SET (CAS_ER_MAX_PREPARED_STMT_COUNT_EXCEEDED, CAS_ERROR_INDICATOR);
75  }
76 #endif /* !LIBCAS_FOR_JSP */
77 
78  for (i = 0; i < max_srv_handle; i++)
79  {
80  if (srv_handle_table[i] == NULL)
81  {
82  *new_handle = srv_handle_table[i];
83  new_handle_id = i + 1;
84  break;
85  }
86  }
87 
88  if (new_handle_id == 0)
89  {
90  new_max_srv_handle = max_srv_handle + SRV_HANDLE_ALLOC_SIZE;
91  new_srv_handle_table = (T_SRV_HANDLE **) REALLOC (srv_handle_table, sizeof (T_SRV_HANDLE *) * new_max_srv_handle);
92  if (new_srv_handle_table == NULL)
93  {
94  return ERROR_INFO_SET (CAS_ER_NO_MORE_MEMORY, CAS_ERROR_INDICATOR);
95  }
96 
97  new_handle_id = max_srv_handle + 1;
98  memset (new_srv_handle_table + max_srv_handle, 0, sizeof (T_SRV_HANDLE *) * SRV_HANDLE_ALLOC_SIZE);
99  max_srv_handle = new_max_srv_handle;
100  srv_handle_table = new_srv_handle_table;
101  }
102 
103  srv_handle = (T_SRV_HANDLE *) MALLOC (sizeof (T_SRV_HANDLE));
104  if (srv_handle == NULL)
105  {
106  return ERROR_INFO_SET (CAS_ER_NO_MORE_MEMORY, CAS_ERROR_INDICATOR);
107  }
108  memset (srv_handle, 0, sizeof (T_SRV_HANDLE));
109  srv_handle->id = new_handle_id;
110  srv_handle->query_seq_num = seq_num;
111  srv_handle->use_plan_cache = false;
112  srv_handle->use_query_cache = false;
113  srv_handle->is_holdable = false;
114  srv_handle->is_from_current_transaction = true;
115 #if defined(CAS_FOR_ORACLE)
116  srv_handle->has_out_result = false;
117 #endif /* defined(CAS_FOR_ORACLE) */
118 #if defined(CAS_FOR_ORACLE) || defined(CAS_FOR_MYSQL)
119  srv_handle->send_metadata_before_execute = false;
120  srv_handle->next_cursor_pos = 1;
121 #endif
122 #if !defined(LIBCAS_FOR_JSP)
123  srv_handle->is_pooled = as_info->cur_statement_pooling;
124 #endif
125 
126 #if defined(CAS_FOR_MYSQL)
127  srv_handle->has_mysql_last_insert_id = false;
128 #endif /* CAS_FOR_MYSQL */
129 
130  *new_handle = srv_handle;
131  srv_handle_table[new_handle_id - 1] = srv_handle;
132  if (new_handle_id > max_handle_id)
133  {
134  max_handle_id = new_handle_id;
135  }
136 
137 #if !defined(LIBCAS_FOR_JSP)
138  current_handle_count++;
139 #endif
140 
141  return new_handle_id;
142 }

272  {

275  }
276 
277  if (srv_handle->schema_type < 0 || srv_handle->schema_type == CCI_SCH_CLASS

314 void
315 hm_qresult_end (T_SRV_HANDLE * srv_handle, char free_flag)
316 {
317  T_QUERY_RESULT *q_result;
318  int i;
319 
320  q_result = srv_handle->q_result;
321 #if defined(CAS_FOR_ORACLE) || defined(CAS_FOR_MYSQL)
322 
323 #if defined(CAS_FOR_MYSQL)
324  if (srv_handle->session)
325  {
326  cas_mysql_stmt_free_result (srv_handle->session);
327  }
328 #endif /* CAS_FOR_MYSQL */
329 
330  if (free_flag == TRUE)
331  {
332  ux_free_result (q_result);
333  FREE_MEM (q_result);
334  srv_handle->q_result = NULL;
335  }
336 #else /* CAS_FOR_ORACLE || CAS_FOR_MYSQL */
337  if (q_result)
338  {
339  for (i = 0; i < srv_handle->num_q_result; i++)
340  {
341  if (q_result[i].copied != TRUE && q_result[i].result)
342  {
343  ux_free_result (q_result[i].result);
344 
345  if (q_result[i].is_holdable == true)
346  {
347  q_result[i].is_holdable = false;
348 #if !defined(LIBCAS_FOR_JSP)
349  as_info->num_holdable_results--;
350 #endif
351  }
352  }
353  q_result[i].result = NULL;
354 
355  if (q_result[i].column_info)
356  {
357  db_query_format_free ((DB_QUERY_TYPE *) q_result[i].column_info);
358  }
359 
360  q_result[i].column_info = NULL;
361  if (free_flag == TRUE)
362  {
363  col_update_info_free (&(q_result[i]));
364  FREE_MEM (q_result[i].null_type_column);
365  }
366  }
367 
368  if (free_flag == TRUE)
369  {
370  FREE_MEM (q_result);
371  }
372  }
373 
374  if (free_flag == TRUE)
375  {
376  srv_handle->q_result = NULL;
377  srv_handle->num_q_result = 0;
378  srv_handle->cur_result_index = 0;
379  }
380 
381  srv_handle->cur_result = NULL;
382  srv_handle->has_result_set = false;
383 #endif /* CAS_FOR_ORACLE || CAS_FOR_MYSQL */
384 }

429  || srv_handle->schema_type == CCI_SCH_SUPERCLASS || srv_handle->schema_type == CCI_SCH_SUBCLASS)

454 #endif /* CAS_FOR_ORACLE || CAS_FOR_MYSQL */
