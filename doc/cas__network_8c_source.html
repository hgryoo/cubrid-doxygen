383 int
384 net_decode_str (char *msg, int msg_size, char *func_code, void ***ret_argv)
385 {
386  int remain_size = msg_size;
387  char *cur_p = msg;
388  char *argp;
389  int i_val;
390  void **argv = NULL;
391  int argc = 0;
392 
393  *ret_argv = (void **) NULL;
394 
395  if (remain_size < 1)
396  return CAS_ER_COMMUNICATION;
397 
398  *func_code = *cur_p;
399  cur_p += 1;
400  remain_size -= 1;
401 
402  while (remain_size > 0)
403  {
404  if (remain_size < 4)
405  {
406  FREE_MEM (argv);
407  return CAS_ER_COMMUNICATION;
408  }
409  argp = cur_p;
410  memcpy ((char *) &i_val, cur_p, 4);
411  i_val = ntohl (i_val);
412  remain_size -= 4;
413  cur_p += 4;
414 
415  if (remain_size < i_val)
416  {
417  FREE_MEM (argv);
418  return CAS_ER_COMMUNICATION;
419  }
420 
421  argc++;
422  argv = (void **) REALLOC (argv, sizeof (void *) * argc);
423  if (argv == NULL)
424  return CAS_ER_NO_MORE_MEMORY;
425 
426  argv[argc - 1] = argp;
427 
428  cur_p += i_val;
429  remain_size -= i_val;
430  }
431 
432  *ret_argv = argv;
433  return argc;
434 }

518 read_buffer (SOCKET sock_fd, char *buf, int size)
519 {
520  int read_len = -1;
521 #if defined(ASYNC_MODE)
522  struct pollfd po[2] = { {0, 0, 0}, {0, 0, 0} };
523  int timeout, po_size, n;
524 #endif /* ASYNC_MODE */
525 
526  unset_net_timeout_flag ();
527  if (net_error_flag)
528  {
529  return -1;
530  }
531 
532 #if defined(ASYNC_MODE)
533  timeout = net_timeout < 0 ? -1 : net_timeout * 1000;
534 
535  po[0].fd = sock_fd;
536  po[0].events = POLLIN;
537  po_size = 1;
538 
539  if (cas_shard_flag == OFF)
540  {
541  if (!IS_INVALID_SOCKET (new_req_sock_fd))
542  {
543  po[1].fd = new_req_sock_fd;
544  po[1].events = POLLIN;
545  po_size = 2;
546  }
547  }
548 
549 retry_poll:
550  n = poll (po, po_size, timeout);
551  if (n < 0)
552  {
553  if (errno == EINTR)
554  {
555  goto retry_poll;
556  }
557  else
558  {
559  net_error_flag = 1;
560  return -1;
561  }
562  }
563  else if (n == 0)
564  {
565  /* TIMEOUT */
566  set_net_timeout_flag ();
567  return -1;
568  }
569  else
570  {
571  if (cas_shard_flag == OFF && !IS_INVALID_SOCKET (new_req_sock_fd) && (po[1].revents & POLLIN))
572  {
573  /* CHANGE CLIENT */
574  return -1;
575  }
576  if (po[0].revents & POLLERR || po[0].revents & POLLHUP)
577  {
578  read_len = -1;
579  }
580  else if (po[0].revents & POLLIN)
581  {
582 #endif /* ASYNC_MODE */
583  /* RECEIVE NEW REQUEST */
584  read_len = READ_FROM_SOCKET (sock_fd, buf, size);
585 #if defined(ASYNC_MODE)
586  }
587  }
588 #endif /* ASYNC_MODE */
589 
590  if (read_len <= 0)
591  {
592  net_error_flag = 1;
593  }
594  return read_len;
595 }
