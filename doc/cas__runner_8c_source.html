433 static void
434 calc_min_max (double *t, int count, double *min, double *max)
435 {
436  int i;
437  if (count <= 0)
438  {
439  *min = 0;
440  *max = 0;
441  return;
442  }
443  *min = t[0];
444  *max = t[0];
445  for (i = 1; i < count; i++)
446  {
447  if (*min > t[i])
448  *min = t[i];
449  if (*max < t[i])
450  *max = t[i];
451  }
452 }

454 static double
455 calc_avg (double *t, int count)
456 {
457  double sum = 0;
458  int i;
459  for (i = 0; i < count; i++)
460  {
461  sum += t[i];
462  }
463  return (sum / count);
464 }

466 static double
467 calc_stddev (double *t, double avg, int count)
468 {
469  double sum = 0;
470  int i;
471  for (i = 0; i < count; i++)
472  {
473  sum += ((t[i] - avg) * (t[i] - avg));
474  }
475  sum /= count;
476  return (sqrt (sum));
477 }

1138 static int
1139 process_execute (char *linebuf, int *req_h, int num_bind, T_BIND_INFO * bind_info, FILE * result_fp,
1140  double *sum_execute_time)
1141 {
1142  int req_id, exec_flag;

1148  {

1153  {

1157 
1158  if (num_replica > 1)
1159  {
1160  exec_flag |= CCI_EXEC_QUERY_ALL;

1164  {
1165  int res;
1166  if (num_bind > 0)

1190  {

1208  }
1209 
1210  if (result_fp)
1211  {
1212  fprintf (result_fp, "cci_execute elapsed_time : %.3f \n", elapsed_time);
1213  }
1214 
1215  if (res < 0)
1216  {
1217  fprintf (cas_error_fp, "execute error\n%s\nrequest id %d\n", linebuf, req_id);
1218  PRINT_CCI_ERROR (res, &cci_error, result_fp);
1219  }

1222  print_result (res, req_h[req_id], result_fp);
