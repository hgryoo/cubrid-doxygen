218 static int
219 net_arg_get_xid (XID * xid, char *buf)
220 {
221  int data_size;
222  int i, id[3];
223 
224  memset (xid, 0, sizeof (XID));
225 
226  memcpy (&data_size, buf, 4);
227  data_size = ntohl (data_size);
228  buf += 4;
229 
230  for (i = 0; i < 3; i++)
231  {
232  if (data_size < 4)
233  return -1;
234  memcpy (&id[i], buf, 4);
235  id[i] = ntohl (id[i]);
236  buf += 4;
237  data_size -= 4;
238  }
239 
240  if (data_size < id[1] + id[2])
241  return -1;
242 
243  xid->formatID = id[0];
244  xid->gtrid_length = id[1];
245  xid->bqual_length = id[2];
246  memcpy (xid->data, buf, id[1] + id[2]);
247  return 0;
248 }

260 static int
261 compare_xid (XID * xid1, XID * xid2)
262 {
263  if (xid1->formatID != xid2->formatID)
264  return -1;
265  if (xid1->gtrid_length != xid2->gtrid_length)
266  return -1;
267  if (xid1->bqual_length != xid2->bqual_length)
268  return -1;
269  return (memcmp (xid1->data, xid2->data, xid1->gtrid_length + xid1->bqual_length));
270 }
