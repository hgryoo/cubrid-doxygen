291 static int
292 hm_pool_drop_node_from_list (T_REQ_HANDLE ** head, T_REQ_HANDLE ** tail, T_REQ_HANDLE * target)
293 {
294  T_REQ_HANDLE *prev_target, *next_target;
295 
296  assert (*head != NULL && *tail != NULL);
297 
298  if (*head == NULL || *tail == NULL)
299  {
300  return CCI_ER_REQ_HANDLE;
301  }
302 
303  prev_target = (T_REQ_HANDLE *) target->prev;
304  next_target = (T_REQ_HANDLE *) target->next;
305  if (prev_target != NULL)
306  {
307  prev_target->next = next_target;
308  }
309  if (next_target != NULL)
310  {
311  next_target->prev = prev_target;
312  }
313 
314  if (target == *head)
315  {
316  *head = next_target;
317  }
318  if (target == *tail)
319  {
320  *tail = prev_target;
321  }
322 
323  return CCI_ER_NO_ERROR;
324 }

462 int
463 hm_req_add_to_pool (T_CON_HANDLE * con, char *sql, int mapped_statement_id, T_REQ_HANDLE * statement)
464 {
465  char *key;
466  int *data;
467 
468  if (sql == NULL)
469  {
470  assert (sql != NULL);
471  return CCI_ER_REQ_HANDLE;
472  }
473 
474  data = (int *) cci_mht_get (con->stmt_pool, sql);
475  if (data != NULL)
476  {
477  hm_pool_drop_node_from_list (&con->pool_use_head, &con->pool_use_tail, statement);
478  return CCI_ER_REQ_HANDLE;
479  }
480 
481  if (HAS_REACHED_LIMIT_OPEN_STATEMENT (con))
482  {
483  T_REQ_HANDLE *victim = hm_pool_victimize_last_node_from_lru (con);
484  if (victim == NULL)
485  {
486  return CCI_ER_REQ_HANDLE;
487  }
488 
489  if (victim->handle_type == HANDLE_PREPARE || victim->handle_type == HANDLE_SCHEMA_INFO)
490  {
491  /* because the statement will be terminated by restarting cas all errors of qe_close_req_handle() are ignored */
492  qe_close_req_handle (victim, con);
493  }
494  cci_mht_rem (con->stmt_pool, victim->sql_text, true, true);
495  hm_req_handle_free (con, victim);
496  victim = NULL;
497  }
498 
499  key = strdup (sql);
500  if (key == NULL)
501  {
502  return CCI_ER_NO_MORE_MEMORY;
503  }
504  data = (int *) MALLOC (sizeof (int));
505  if (data == NULL)
506  {
507  FREE (key);
508  return CCI_ER_NO_MORE_MEMORY;
509  }
510 
511  if (map_get_ots_value (mapped_statement_id, data, true) != CCI_ER_NO_ERROR)
512  {
513  FREE (key);
514  FREE (data);
515 
516  return CCI_ER_REQ_HANDLE;
517  }
518 
519  if (!cci_mht_put_data (con->stmt_pool, key, data))
520  {
521  FREE (key);
522  FREE (data);
523  return CCI_ER_NO_MORE_MEMORY;
524  }
525 
526  hm_pool_move_node_from_use_to_lru (con, *data);
527 
528  return CCI_ER_NO_ERROR;
529 }
