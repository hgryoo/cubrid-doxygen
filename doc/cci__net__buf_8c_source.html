276 static int
277 net_buf_realloc (T_NET_BUF * net_buf, int size)
278 {
279  int new_alloc_size;
280 
281  if (size + net_buf->data_size > net_buf->alloc_size)
282  {
283  new_alloc_size = net_buf->alloc_size + 1024;
284  if (size + net_buf->data_size > new_alloc_size)
285  {
286  new_alloc_size = size + net_buf->data_size;
287  }
288  net_buf->data = (char *) REALLOC (net_buf->data, new_alloc_size);
289  if (net_buf->data == NULL)
290  {
291  net_buf->alloc_size = 0;
292  net_buf->err_code = CCI_ER_NO_MORE_MEMORY;
293  return -1;
294  }
295 
296  net_buf->alloc_size = new_alloc_size;
297  }
298 
299  return 0;
300 }
