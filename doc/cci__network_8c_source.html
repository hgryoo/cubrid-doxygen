133 int
134 net_connect_srv (T_CON_HANDLE * con_handle, int host_id, T_CCI_ERROR * err_buf, int login_timeout)
135 {
136  SOCKET srv_sock_fd;
137  char client_info[SRV_CON_CLIENT_INFO_SIZE];
138  char db_info[SRV_CON_DB_INFO_SIZE];
139  char ver_str[SRV_CON_VER_STR_MAX_SIZE];
140  MSG_HEADER msg_header;
141  int err_code, ret_value;
142  int err_indicator;
143  int new_port;
144  char *msg_buf, *info, *p, *ver_ptr;
145  unsigned char *ip_addr;
146  int port;
147  int body_len;
148  T_BROKER_VERSION broker_ver;
149 
150  init_msg_header (&msg_header);
151 
152  memset (client_info, 0, sizeof (client_info));
153  memset (db_info, 0, sizeof (db_info));
154 
155  strncpy (client_info, SRV_CON_CLIENT_MAGIC_STR, SRV_CON_CLIENT_MAGIC_LEN);
156  client_info[SRV_CON_MSG_IDX_CLIENT_TYPE] = cci_client_type;
157  client_info[SRV_CON_MSG_IDX_PROTO_VERSION] = CAS_PROTO_PACK_CURRENT_NET_VER;
158  client_info[SRV_CON_MSG_IDX_FUNCTION_FLAG] = BROKER_RENEWED_ERROR_CODE | BROKER_SUPPORT_HOLDABLE_RESULT;
159  client_info[SRV_CON_MSG_IDX_RESERVED2] = 0;
160 
161  info = db_info;
162  if (con_handle->db_name)
163  {
164  strncpy (info, con_handle->db_name, SRV_CON_DBNAME_SIZE);
165  }
166  info += SRV_CON_DBNAME_SIZE;
167 
168  if (con_handle->db_user)
169  {
170  strncpy (info, con_handle->db_user, SRV_CON_DBUSER_SIZE);
171  }
172  info += SRV_CON_DBUSER_SIZE;
173 
174  if (con_handle->db_passwd)
175  {
176  strncpy (info, con_handle->db_passwd, SRV_CON_DBPASSWD_SIZE);
177  }
178  info += SRV_CON_DBPASSWD_SIZE;
179 
180  strncpy (info, con_handle->url, SRV_CON_URL_SIZE - 1);
181  strncpy (ver_str, MAKE_STR (BUILD_NUMBER), SRV_CON_VER_STR_MAX_SIZE);
182 
183  ver_ptr = info + strlen (con_handle->url) + 1;
184  if (strlen (con_handle->url) + strlen (ver_str) + 3 <= SRV_CON_URL_SIZE)
185  {
186  ver_ptr[0] = (char) strlen (ver_str) + 1;
187  memcpy (ver_ptr + 1, ver_str, strlen (ver_str) + 1);
188  }
189  else
190  {
191  ver_ptr[0] = (char) 0;
192  }
193  info += SRV_CON_URL_SIZE;
194 
195  broker_ver = hm_get_broker_version (con_handle);
196  if (broker_ver == 0)
197  {
198  /* Interpretable session information supporting version later than PROTOCOL_V3 as well as version earlier than
199  * PROTOCOL_V3 should be delivered since no broker information is provided at the time of initial connection. */
200  snprintf (info, DRIVER_SESSION_SIZE, "%u", 0);
201  }
202  else if (hm_broker_understand_the_protocol (broker_ver, PROTOCOL_V3))
203  {
204  memcpy (info, con_handle->session_id.id, DRIVER_SESSION_SIZE);
205  }
206  else
207  {
208  unsigned int v;
209 
210  v = *(unsigned int *) con_handle->session_id.id;
211  snprintf (info, DRIVER_SESSION_SIZE, "%u", v);
212  }
213 
214  if (host_id < 0)
215  {
216  ip_addr = con_handle->ip_addr;
217  port = con_handle->port;
218  }
219  else
220  {
221  ip_addr = con_handle->alter_hosts[host_id].ip_addr;
222  port = con_handle->alter_hosts[host_id].port;
223  }
224  ret_value = connect_srv (ip_addr, port, con_handle->is_retry, &srv_sock_fd, login_timeout);
225  if (ret_value < 0)
226  {
227  return ret_value;
228  }
229 
230  if (net_send_stream (srv_sock_fd, client_info, SRV_CON_CLIENT_INFO_SIZE) < 0)
231  {
232  err_code = CCI_ER_COMMUNICATION;
233  goto connect_srv_error;
234  }
235 
236  ret_value = net_recv_stream (srv_sock_fd, port, (char *) &err_code, 4, login_timeout);
237  if (ret_value < 0)
238  {
239  err_code = ret_value;
240  goto connect_srv_error;
241  }
242 
243  err_code = ntohl (err_code);
244  if (err_code < 0)
245  {
246  /* in here, all errors are sent by only a broker the error greater than -10000 is sent by old broker */
247  if (err_code > -10000)
248  {
249  err_code -= 9000;
250  }
251  goto connect_srv_error;
252  }
253 
254  new_port = err_code;
255 
256  if (new_port != port && new_port > 0)
257  {
258  CLOSE_SOCKET (srv_sock_fd);
259 
260  ret_value = connect_srv (ip_addr, new_port, con_handle->is_retry, &srv_sock_fd, login_timeout);
261  if (ret_value < 0)
262  {
263  return ret_value;
264  }
265  }
266 
267  if (net_send_stream (srv_sock_fd, db_info, SRV_CON_DB_INFO_SIZE) < 0)
268  {
269  err_code = CCI_ER_COMMUNICATION;
270  goto connect_srv_error;
271  }
272 
273  ret_value = net_recv_msg_header (srv_sock_fd, port, &msg_header, login_timeout);
274  if (ret_value < 0)
275  {
276  err_code = ret_value;
277  goto connect_srv_error;
278  }
279 
280  memcpy (con_handle->cas_info, msg_header.info_ptr, MSG_HEADER_INFO_SIZE);
281 
282  msg_buf = (char *) MALLOC (*(msg_header.msg_body_size_ptr));
283  if (msg_buf == NULL)
284  {
285  err_code = CCI_ER_NO_MORE_MEMORY;
286  goto connect_srv_error;
287  }
288 
289  ret_value = net_recv_stream (srv_sock_fd, port, msg_buf, *(msg_header.msg_body_size_ptr), login_timeout);
290  if (ret_value < 0)
291  {
292  FREE_MEM (msg_buf);
293  err_code = ret_value;
294  goto connect_srv_error;
295  }
296  memcpy (&err_indicator, msg_buf + CAS_PROTOCOL_ERR_INDICATOR_INDEX, CAS_PROTOCOL_ERR_INDICATOR_SIZE);
297  err_indicator = ntohl (err_indicator);
298  if (err_indicator < 0)
299  {
300  memcpy (&err_code, msg_buf + CAS_PROTOCOL_ERR_CODE_INDEX, CAS_PROTOCOL_ERR_CODE_SIZE);
301  err_code = ntohl (err_code);
302  /* the error greater than -10000 with CAS_ERROR_INDICATOR is sent by old broker -1018
303  * (CAS_ER_NOT_AUTHORIZED_CLIENT) is especial case */
304  if ((err_indicator == CAS_ERROR_INDICATOR && err_code > -10000) || err_code == -1018)
305  {
306  err_code -= 9000;
307  }
308  if (err_indicator == DBMS_ERROR_INDICATOR)
309  {
310  if (err_buf)
311  {
312  memcpy (err_buf->err_msg, msg_buf + CAS_PROTOCOL_ERR_MSG_INDEX,
313  *(msg_header.msg_body_size_ptr) - (CAS_PROTOCOL_ERR_INDICATOR_SIZE + CAS_PROTOCOL_ERR_CODE_SIZE));
314  err_buf->err_code = err_code;
315  }
316  err_code = CCI_ER_DBMS;
317  }
318  FREE_MEM (msg_buf);
319  goto connect_srv_error;
320  }
321 
322  /* connection success */
323  con_handle->cas_pid = err_indicator;
324  p = msg_buf + CAS_PID_SIZE;
325  memcpy (con_handle->broker_info, p, BROKER_INFO_SIZE);
326  p += BROKER_INFO_SIZE;
327 
328  body_len = *(msg_header.msg_body_size_ptr);
329  broker_ver = hm_get_broker_version (con_handle);
330  if (hm_broker_understand_the_protocol (broker_ver, PROTOCOL_V4))
331  {
332  if (body_len != CAS_CONNECTION_REPLY_SIZE)
333  {
334  err_code = CCI_ER_COMMUNICATION;
335  goto connect_srv_error;
336  }
337  }
338  else if (hm_broker_understand_the_protocol (broker_ver, PROTOCOL_V3))
339  {
340  if (body_len != CAS_CONNECTION_REPLY_SIZE_V3)
341  {
342  err_code = CCI_ER_COMMUNICATION;
343  goto connect_srv_error;
344  }
345  }
346  else
347  {
348  if (body_len != CAS_CONNECTION_REPLY_SIZE_PRIOR_PROTOCOL_V3)
349  {
350  err_code = CCI_ER_COMMUNICATION;
351  goto connect_srv_error;
352  }
353  }
354 
355  if (hm_broker_understand_the_protocol (broker_ver, PROTOCOL_V4))
356  {
357  con_handle->cas_id = ntohl (*(int *) p);
358  p += CAS_PID_SIZE;
359  }
360  else
361  {
362  con_handle->cas_id = -1;
363  }
364 
365  if (hm_broker_understand_the_protocol (broker_ver, PROTOCOL_V3))
366  {
367  memcpy (con_handle->session_id.id, p, DRIVER_SESSION_SIZE);
368  }
369  else
370  {
371  memcpy (con_handle->session_id.id, p, SESSION_ID_SIZE);
372  *(unsigned int *) con_handle->session_id.id = ntohl (*(unsigned int *) con_handle->session_id.id);
373  }
374 
375  FREE_MEM (msg_buf);
376 
377  con_handle->sock_fd = srv_sock_fd;
378  con_handle->alter_host_id = host_id;
379 
380  if (con_handle->alter_host_count > 0)
381  {
382  con_handle->is_retry = 0;
383  }
384  else
385  {
386  con_handle->is_retry = 1;
387  }
388 
389  if (con_handle->isolation_level > TRAN_UNKNOWN_ISOLATION)
390  {
391  qe_set_db_parameter (con_handle, CCI_PARAM_ISOLATION_LEVEL, &(con_handle->isolation_level), err_buf);
392  }
393 
394  if (con_handle->lock_timeout != CCI_LOCK_TIMEOUT_DEFAULT)
395  {
396  qe_set_db_parameter (con_handle, CCI_PARAM_LOCK_TIMEOUT, &(con_handle->lock_timeout), err_buf);
397  }
398 
399  return CCI_ER_NO_ERROR;
400 
401 connect_srv_error:
402  CLOSE_SOCKET (srv_sock_fd);
403  return err_code;
404 }

406 static int
407 net_cancel_request_internal (unsigned char *ip_addr, int port, char *msg, int msglen)
408 {
409  SOCKET srv_sock_fd;
410  int err_code;
411 
412  if (connect_srv (ip_addr, port, 0, &srv_sock_fd, 0) < 0)
413  {
414  return CCI_ER_CONNECT;
415  }
416 
417  if (net_send_stream (srv_sock_fd, msg, msglen) < 0)
418  {
419  err_code = CCI_ER_COMMUNICATION;
420  goto cancel_error;
421  }
422 
423  if (net_recv_stream (srv_sock_fd, port, (char *) &err_code, 4, 0) < 0)
424  {
425  err_code = CCI_ER_COMMUNICATION;
426  goto cancel_error;
427  }
428 
429  err_code = ntohl (err_code);
430  if (err_code < 0)
431  goto cancel_error;
432 
433  CLOSE_SOCKET (srv_sock_fd);
434  return CCI_ER_NO_ERROR;
435 
436 cancel_error:
437  CLOSE_SOCKET (srv_sock_fd);
438  return err_code;
439 }

547 int
548 net_send_msg (T_CON_HANDLE * con_handle, char *msg, int size)
549 {
550  MSG_HEADER send_msg_header;
551  int err;
552  struct timeval ts, te;
553 
554  init_msg_header (&send_msg_header);
555 
556  *(send_msg_header.msg_body_size_ptr) = size;
557  memcpy (send_msg_header.info_ptr, con_handle->cas_info, MSG_HEADER_INFO_SIZE);
558 
559  /* send msg header */
560  if (con_handle->log_trace_network)
561  {
562  gettimeofday (&ts, NULL);
563  }
564  err = net_send_msg_header (con_handle->sock_fd, &send_msg_header);
565  if (con_handle->log_trace_network)
566  {
567  long elapsed;
568 
569  gettimeofday (&te, NULL);
570  elapsed = ut_timeval_diff_msec (&ts, &te);
571  CCI_LOGF_DEBUG (con_handle->logger, "[NET][W][H][S:%d][E:%d][T:%d]", MSG_HEADER_SIZE, err, elapsed);
572  }
573  if (err < 0)
574  {
575  return CCI_ER_COMMUNICATION;
576  }
577 
578  if (con_handle->log_trace_network)
579  {
580  gettimeofday (&ts, NULL);
581  }
582  err = net_send_stream (con_handle->sock_fd, msg, size);
583  if (con_handle->log_trace_network)
584  {
585  long elapsed;
586 
587  gettimeofday (&te, NULL);
588  elapsed = ut_timeval_diff_msec (&ts, &te);
589  CCI_LOGF_DEBUG (con_handle->logger, "[NET][W][B][S:%d][E:%d][T:%d]", size, err, elapsed);
590  }
591  if (err < 0)
592  {
593  return CCI_ER_COMMUNICATION;
594  }
595 
596  return 0;
597 }

616 int
617 net_recv_msg_timeout (T_CON_HANDLE * con_handle, char **msg, int *msg_size, T_CCI_ERROR * err_buf, int timeout)
618 {
619  char *tmp_p = NULL;
620  MSG_HEADER recv_msg_header;
621  int result_code = 0;
622  struct timeval ts, te;
623  int broker_port;
624 
625  if (con_handle->alter_host_id < 0)
626  {
627  broker_port = con_handle->port;
628  }
629  else
630  {
631  broker_port = con_handle->alter_hosts[con_handle->alter_host_id].port;
632  }
633 
634  init_msg_header (&recv_msg_header);
635 
636  if (msg)
637  {
638  *msg = NULL;
639  }
640  if (msg_size)
641  {
642  *msg_size = 0;
643  }
644 
645  if (con_handle->log_trace_network)
646  {
647  gettimeofday (&ts, NULL);
648  }
649  result_code = net_recv_msg_header (con_handle->sock_fd, broker_port, &recv_msg_header, timeout);

1010
