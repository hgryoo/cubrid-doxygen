120 int
121 t_set_get (T_SET * set, int index, T_CCI_A_TYPE a_type, void *value, int *indicator)
122 {
123  char *ele_value_p;
124  char u_type;
125  int data_size;
126  int err_code;
127 
128  if (index <= 0 || index > set->num_element)
129  return CCI_ER_SET_INDEX;
130 
131  ele_value_p = (char *) set->element[index - 1];
132  u_type = t_set_element_type (set);
133 
134  NET_STR_TO_INT (data_size, ele_value_p);
135  if (data_size <= 0)
136  {
137  *indicator = -1;
138  if (a_type == CCI_A_TYPE_STR)
139  {
140  *((void **) value) = NULL;
141  }
142  return 0;
143  }
144 
145  ele_value_p += 4;
146  *indicator = 0;
147 
148  switch (a_type)
149  {
150  case CCI_A_TYPE_STR:
151  err_code =
152  qe_get_data_str (&(set->conv_value_buffer), (T_CCI_U_TYPE) u_type, ele_value_p, data_size, value, indicator);
153  break;
154  case CCI_A_TYPE_INT:
155  err_code = qe_get_data_int ((T_CCI_U_TYPE) u_type, ele_value_p, value);
156  break;
157  case CCI_A_TYPE_BIGINT:
158  err_code = qe_get_data_bigint ((T_CCI_U_TYPE) u_type, ele_value_p, value);
159  break;
160  case CCI_A_TYPE_FLOAT:
161  err_code = qe_get_data_float ((T_CCI_U_TYPE) u_type, ele_value_p, value);
162  break;
163  case CCI_A_TYPE_DOUBLE:
164  err_code = qe_get_data_double ((T_CCI_U_TYPE) u_type, ele_value_p, value);
165  break;
166  case CCI_A_TYPE_BIT:
167  err_code = qe_get_data_bit ((T_CCI_U_TYPE) u_type, ele_value_p, data_size, value);
168  break;
169  case CCI_A_TYPE_DATE:
170  err_code = qe_get_data_date ((T_CCI_U_TYPE) u_type, ele_value_p, value);
171  break;
172  case CCI_A_TYPE_UINT:
173  case CCI_A_TYPE_UBIGINT:
174  default:
175  return CCI_ER_TYPE_CONVERSION;
176  }
177 
178  return err_code;
179 }

196 int
197 t_set_make (T_SET * set, char ele_type, int size, void *value, int *indicator)
198 {
199  int i;
200  T_NET_BUF net_buf;
201  char u_type;
202  int err_code;
203  int *ele_index = NULL;
204 
205  net_buf_init (&net_buf);
206 
207  ele_index = (int *) MALLOC (sizeof (int) * size);
208  if (ele_index == NULL)
209  {
210  return CCI_ER_NO_MORE_MEMORY;
211  }
212 
213  if (ele_type == CCI_U_TYPE_SHORT)
214  {
215  ele_type = CCI_U_TYPE_INT;
216  }
217 
218  for (i = 0; i < size; i++)
219  {
220  ele_index[i] = net_buf.data_size;
221 
222  u_type = ele_type;
223  if (indicator[i])
224  {
225  u_type = CCI_U_TYPE_NULL;
226  }
227 
228  switch (u_type)
229  {
230  case CCI_U_TYPE_NULL:
231  {
232  ADD_ARG_BYTES (&net_buf, NULL, 0);
233  }
234  break;
235  case CCI_U_TYPE_CHAR:
236  case CCI_U_TYPE_STRING:
237  case CCI_U_TYPE_NCHAR:
238  case CCI_U_TYPE_VARNCHAR:
239  case CCI_U_TYPE_NUMERIC:
240  case CCI_U_TYPE_ENUM:
241  {
242  char *ele_value;
243  ele_value = ((char **) value)[i];
244  ADD_ARG_STR (&net_buf, ele_value, strlen (ele_value) + 1, NULL);
245  }
246  break;
247  case CCI_U_TYPE_BIT:
248  case CCI_U_TYPE_VARBIT:
249  {
250  T_CCI_BIT ele_value;
251  ele_value = ((T_CCI_BIT *) value)[i];
252  ADD_ARG_BYTES (&net_buf, ele_value.buf, ele_value.size);
253  }
254  break;
255  case CCI_U_TYPE_BIGINT:
256  {
257  INT64 ele_value;
258  ele_value = ((INT64 *) value)[i];
259  ADD_ARG_BIGINT (&net_buf, ele_value);
260  }
261  break;
262  case CCI_U_TYPE_INT:
263  case CCI_U_TYPE_SHORT:
264  {
265  int ele_value;
266  ele_value = ((int *) value)[i];
267  ADD_ARG_INT (&net_buf, ele_value);
268  }
269  break;
270  case CCI_U_TYPE_FLOAT:
271  {
272  float ele_value;
273  ele_value = ((float *) value)[i];
274  ADD_ARG_FLOAT (&net_buf, ele_value);
275  }
276  break;
277  case CCI_U_TYPE_MONETARY:
278  case CCI_U_TYPE_DOUBLE:
279  {
280  double ele_value;
281  ele_value = ((double *) value)[i];
282  ADD_ARG_DOUBLE (&net_buf, ele_value);
283  }
284  break;
285  case CCI_U_TYPE_DATE:
286  case CCI_U_TYPE_TIME:
287  case CCI_U_TYPE_TIMESTAMP:
288  case CCI_U_TYPE_DATETIME:
289  {
290  T_CCI_DATE ele_value;
291  ele_value = ((T_CCI_DATE *) value)[i];
292  ADD_ARG_DATETIME (&net_buf, &ele_value);
293  }
294  break;
295  case CCI_U_TYPE_TIMESTAMPTZ:
296  case CCI_U_TYPE_TIMESTAMPLTZ:
297  case CCI_U_TYPE_DATETIMETZ:
298  case CCI_U_TYPE_DATETIMELTZ:
299  {
300  T_CCI_DATE_TZ ele_value;
301  ele_value = ((T_CCI_DATE_TZ *) value)[i];
302  ADD_ARG_DATETIMETZ (&net_buf, &ele_value);
303  }
304  break;
305  case CCI_U_TYPE_OBJECT:
306  {
307  char *tmp_str;
308  T_OBJECT ele_value;
309 
310  tmp_str = ((char **) value)[i];
311  err_code = ut_str_to_oid (tmp_str, &ele_value);
312  if (err_code < 0)
313  {
314  goto set_make_error;
315  }
316  ADD_ARG_OBJECT (&net_buf, &ele_value);
317  }
318  break;
319  case CCI_U_TYPE_USHORT:
320  case CCI_U_TYPE_UINT:
321  case CCI_U_TYPE_UBIGINT:
322  case CCI_U_TYPE_BLOB:
323  case CCI_U_TYPE_CLOB:
324  default:
325  err_code = CCI_ER_TYPE_CONVERSION;
326  goto set_make_error;
327  }
328  }
329 
330  if (net_buf.err_code < 0)
331  {
332  err_code = net_buf.err_code;
333  goto set_make_error;
334  }
335 
336  set->element = (void **) MALLOC (sizeof (void *) * size);
337  if (set->element == NULL)
338  {
339  err_code = CCI_ER_NO_MORE_MEMORY;
340  goto set_make_error;
341  }
342 
343  set->data_buf = net_buf.data;
344  set->data_size = net_buf.data_size;
345  for (i = 0; i < size; i++)
346  {
347  set->element[i] = (char *) (set->data_buf) + ele_index[i];
348  }
349  set->num_element = size;
350  set->type = ele_type;
351 
352  FREE_MEM (ele_index);
353  return 0;
354 
355 set_make_error:
356  net_buf_clear (&net_buf);
357  FREE_MEM (ele_index);
358  return err_code;
359 }

361 int
362 t_set_decode (T_SET * set)
363 {
364  void **element;
365  char *cur_p;
366  int i;
367  int remain_size;
368  int ele_size;
369 
370  if (set->num_element < 0 || set->data_size < 0)
371  return CCI_ER_COMMUNICATION;
372 
373  if (set->num_element == 0 || set->data_size == 0)
374  return 0;
375 
376  element = (void **) MALLOC (sizeof (void *) * set->num_element);
377  if (element == NULL)
378  return CCI_ER_NO_MORE_MEMORY;
379  memset (element, 0, sizeof (void *) * set->num_element);
380 
381  remain_size = set->data_size;
382  cur_p = (char *) set->data_buf;
383 
384  for (i = 0; i < set->num_element; i++)
385  {
386  element[i] = cur_p;
387  if (remain_size < 4)
388  {
389  FREE_MEM (element);
390  return CCI_ER_COMMUNICATION;
391  }
392  NET_STR_TO_INT (ele_size, cur_p);
393  cur_p += 4;
394  remain_size -= 4;
395 
396  if (ele_size <= 0)
397  continue;
398 
399  if (remain_size < ele_size)
400  {
401  FREE_MEM (element);
402  return CCI_ER_COMMUNICATION;
403  }
404  cur_p += ele_size;
405  remain_size -= ele_size;
406  }
407 
408  set->element = element;
409  return 0;
410 }

413 t_set_to_str (T_SET * set, T_VALUE_BUF * conv_val)
414 {

426  {

429  net_buf_cp_str (&net_buf, ", ", 2);
430  }
431 
432  buf = NULL;
433  err_code = t_set_get (set, i, CCI_A_TYPE_STR, (void *) &buf, &indicator);
434  if (err_code < 0)
435  {
436  net_buf_clear (&net_buf);
437  return err_code;
438  }
439 
440  if (indicator < 0 || buf == NULL)
441  {
442  buf = (char *) "NULL";
443  }
444 
445  net_buf_cp_str (&net_buf, buf, strlen (buf));
446  }
447 
448  net_buf_cp_str (&net_buf, "}", 2);
449 
450  hm_conv_value_buf_alloc (conv_val, net_buf.data_size);
451  strcpy ((char *) conv_val->data, net_buf.data);
452 
453  net_buf_clear (&net_buf);
454 
455  return 0;
