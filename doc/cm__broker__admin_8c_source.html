144 int
145 cm_get_cas_info (const char *br_name, T_CM_CAS_INFO_ALL * cas_info_all, T_CM_JOB_INFO_ALL * job_info_all,
146  T_CM_ERROR * err_buf)
147 {
148  T_AS_INFO *as_info = NULL;
149  T_JOB_INFO *job_info = NULL;
150  int res, i, num_as_info, num_job = 0;
151 
152  cm_err_buf_reset (err_buf);
153 
154  memset (cas_info_all, 0, sizeof (T_CM_CAS_INFO_ALL));
155  if (job_info_all == NULL)
156  {
157  num_as_info = uc_as_info (br_name, &as_info, NULL, NULL, err_buf->err_msg);
158  }
159  else
160  {
161  memset (job_info_all, 0, sizeof (T_CM_JOB_INFO_ALL));
162  num_as_info = uc_as_info (br_name, &as_info, &job_info, &num_job, err_buf->err_msg);
163  }
164  if (num_as_info < 0)
165  {
166  err_buf->err_code = CM_GENERAL_ERROR;
167  res = -1;
168  goto as_info_finale;
169  }
170 
171  cas_info_all->as_info = (T_CM_CAS_INFO *) malloc (sizeof (T_CM_CAS_INFO) * num_as_info);
172  if (cas_info_all->as_info == NULL)
173  {
174  err_buf->err_code = CM_GENERAL_ERROR;
175  strcpy (err_buf->err_msg, strerror (errno));
176  res = -1;
177  goto as_info_finale;
178  }
179  memset (cas_info_all->as_info, 0, sizeof (T_CM_CAS_INFO) * num_as_info);
180  if (job_info_all != NULL)
181  {
182  job_info_all->job_info = (T_CM_JOB_INFO *) malloc (sizeof (T_CM_JOB_INFO) * num_job);
183  if (job_info_all->job_info == NULL)
184  num_job = 0;
185  else
186  memset (job_info_all->job_info, 0, sizeof (T_CM_JOB_INFO) * num_job);
187  for (i = 0; i < num_job; i++)
188  job_info_copy (&(job_info_all->job_info[i]), &(job_info[i]));
189  job_info_all->num_info = num_job;
190  }
191  for (i = 0; i < num_as_info; i++)
192  {
193  cas_info_all->as_info[i].id = i + 1;
194  as_info_copy (&(cas_info_all->as_info[i]), &(as_info[i]));
195  }
196  cas_info_all->num_info = num_as_info;
197  res = num_as_info;
198 
199 as_info_finale:
200  uc_info_free (as_info);
201  uc_info_free (job_info);
202  return res;
203 }
