169 static void
170 write_class_info (FILE * fp, DB_OBJECT * classobj)
171 {
172  DB_OBJLIST *objlist, *temp;
173  DB_OBJECT *obj;
174  DB_VALUE v;
175 
176  fprintf (fp, MSGFMT, "open", "class");
177  fprintf (fp, MSGFMT, "classname", db_get_class_name (classobj));
178 
179  obj = db_get_owner (classobj);
180  db_get (obj, "name", &v);
181  fprintf (fp, MSGFMT, "owner", db_get_string (&v));
182 
183  objlist = db_get_superclasses (classobj);
184  for (temp = objlist; temp != NULL; temp = temp->next)
185  {
186  fprintf (fp, MSGFMT, "superclass", db_get_class_name (temp->op));
187  }
188  if (objlist != NULL)
189  db_objlist_free (objlist);
190 
191  if (db_is_vclass (classobj) > 0)
192  fprintf (fp, MSGFMT, "virtual", "view");
193  else
194  fprintf (fp, MSGFMT, "virtual", "normal");
195 
196  fprintf (fp, MSGFMT, "close", "class");
197 }

200 dbmt_user_login (int argc, char *argv[], int opt_begin)
201 {
202  const char *outfile, *errfile, *dbname, *dbuser, *dbpasswd;
203  FILE *outfp = NULL, *errfp = NULL;
204  bool isdba = false;
205 
206  if (argc - opt_begin < 5)
207  return -1;
208 
209  outfile = argv[opt_begin++];
210  outfile = (outfile) ? outfile : "";
211 
212  errfile = argv[opt_begin++];
213  errfile = (errfile) ? errfile : "";
214 
215  dbname = argv[opt_begin++];
216  dbname = (dbname) ? dbname : "";
217 
218  dbuser = argv[opt_begin++];
219  dbuser = (dbuser) ? dbuser : "";
220 
221  dbpasswd = argv[opt_begin++];
222  dbpasswd = (dbpasswd) ? dbpasswd : "";
223 
224  outfp = fopen (outfile, "w");
225  errfp = fopen (errfile, "w");
226  if (outfp == NULL || errfp == NULL)
227  {
228  goto login_err;
229  }
230 
231  db_login (dbuser, dbpasswd);
232  if (db_restart (argv[0], 0, dbname) < 0)
233  {
234  fprintf (errfp, "%s", db_error_string (1));
235  goto login_err;
236  }
237 
238  if (strcasecmp (dbuser, "DBA") == 0)
239  {
240  isdba = true;
241  }
242  else
243  {
244  DB_OBJECT *user, *obj;
245  DB_VALUE v;
246  DB_COLLECTION *col;
247  int i;
248  char *username;
249 
250  user = db_find_user (dbuser);
251  if (user == NULL)
252  {
253  fprintf (errfp, "%s", db_error_string (1));
254  goto login_err;
255  }
256  db_get (user, "groups", &v);
257  col = db_get_set (&v);
258  for (i = 0; i < db_set_size (col); i++)
259  {
260  db_set_get (col, i, &v);
261  obj = db_get_object (&v);
262  db_get (obj, "name", &v);
263  username = db_get_string (&v);
264  if (username != NULL && strcasecmp (username, "DBA") == 0)
265  {
266  isdba = true;
267  break;
268  }
269  }
270  }
271 
272  if (isdba == true)
273  fprintf (outfp, "isdba\n");
274  else
275  fprintf (outfp, "isnotdba\n");
276 
277  db_shutdown ();
278  fclose (outfp);
279  fclose (errfp);
280  return 0;
281 
282 login_err:
283  db_shutdown ();
284  if (outfp != NULL)
285  fclose (outfp);
286  if (errfp != NULL)
287  fclose (errfp);
288  return -1;
289 }
