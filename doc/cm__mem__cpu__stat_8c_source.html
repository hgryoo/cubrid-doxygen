371 /*
372  * 0 - init_state.
373  * 1 - load GetSystemTime.
374  * 2 - load NtQuerySystemInformation.
375  * 3 - error state.
376  */
377 static volatile int s_symbol_loaded = 0;
378 static volatile GET_SYSTEM_TIMES s_pfnGetSystemTimes = NULL;
379 static volatile NT_QUERY_SYSTEM_INFORMATION s_pfnNtQuerySystemInformation = NULL;
380 
381 static int
382 get_cpu_time (__int64 * kernel, __int64 * user, __int64 * idle)
383 {
384  /* this logic allow multi thread init multiple times */
385  if (s_symbol_loaded == 0)
386  {
387  /*
388  * kernel32.dll and ntdll.dll is essential DLL about user process.
389  * when a process started, that means kernel32.dll and ntdll.dll
390  * already load in process memory space.
391  * so call LoadLibrary() and FreeLibrary() function once, only
392  * increase and decrease dll reference counter. this behavior does
393  * not cause kernel32.dll or ntdll.dll unload from current process.
394  */
395 
396  /*
397  * first try find function GetSystemTimes(). Windows OS suport this
398  * function since Windows XP SP1, Vista, Server 2003 or Server 2008.
399  */
400  HMODULE module = LoadLibraryA ("kernel32.dll");
401  s_pfnGetSystemTimes = (GET_SYSTEM_TIMES) GetProcAddress (module, "GetSystemTimes");
402  FreeLibrary (module);
403 
404  if (s_pfnGetSystemTimes != NULL)
405  {
406  s_symbol_loaded = 1;
407  }
408  else
409  {
410  /*
411  * OS may be is Windows 2000 or Windows XP. (does not support Windows 9x/NT)
412  * try find function NtQuerySystemInformation()
413  */
414  module = LoadLibraryA ("ntdll.dll");
415  s_pfnNtQuerySystemInformation =
416  (NT_QUERY_SYSTEM_INFORMATION) GetProcAddress (module, "NtQuerySystemInformation");
417  FreeLibrary (module);
418 
419  if (s_pfnNtQuerySystemInformation == NULL)
420  {
421  s_symbol_loaded = 3;
422  }
423  else
424  {
425  s_symbol_loaded = 2;
426  }
427  }
428  }
429 
430  if (s_symbol_loaded == 1)
431  {
432  FILETIME kernel_time, user_time, idle_time;
433  ULARGE_INTEGER lk, lu, li;
434 
435  s_pfnGetSystemTimes (&idle_time, &kernel_time, &user_time);
436 
437  lk.HighPart = kernel_time.dwHighDateTime;
438  lk.LowPart = kernel_time.dwLowDateTime;
439  lu.HighPart = user_time.dwHighDateTime;
440  lu.LowPart = user_time.dwLowDateTime;
441  li.HighPart = idle_time.dwHighDateTime;
442  li.LowPart = idle_time.dwLowDateTime;
443 
444  /* In win32 system, lk includes "System Idle Process" time, so we should exclude it */
445  *kernel = lk.QuadPart - li.QuadPart;
446  *user = lu.QuadPart;
447  *idle = li.QuadPart;
448 
449  return 0;
450  }
451  else if (s_symbol_loaded == 2)
452  {
453  SYSTEM_PROCESSOR_PERFORMANCE_INFORMATION sppi;
454  ULONG len;
455 
456  s_pfnNtQuerySystemInformation (SystemProcessorPerformanceInformation, &sppi, sizeof (sppi), &len);
457 
458  /* In win32 system, sppi.KernelTime includes "System Idle Process" time, so we should exclude it */
459  *kernel = sppi.KernelTime.QuadPart - sppi.IdleTime.QuadPart;
460  *user = sppi.UserTime.QuadPart;
461  *idle = sppi.IdleTime.QuadPart;
462 
463  return 0;
464  }
465 
466  return -1;
467 }

505 int
506 cm_get_proc_stat (T_CM_PROC_STAT * stat, int pid)
507 {
508  ULARGE_INTEGER lk, lu;
509  FILETIME dummy1, dummy2, kt, ut;
510  PROCESS_MEMORY_COUNTERS pmc;
511  MEMORYSTATUSEX ms;
512  HANDLE hProcess = NULL, hToken = NULL;
513  int ret = 0;
514 
515  stat->pid = pid;
516  if (!OpenThreadToken (GetCurrentThread (), (TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY), FALSE, &hToken))
517  {
518  if (GetLastError () == ERROR_NO_TOKEN)
519  {
520  if (!ImpersonateSelf (SecurityImpersonation))
521  {
522  return -1;
523  }
524 
525  if (!OpenThreadToken (GetCurrentThread (), (TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY), FALSE, &hToken))
526  {
527  return -1;
528  }
529  }
530  else
531  {
532  return -1;
533  }
534  }
535 
536  /* enable SeDebugPrivilege */
537  if (!SetPrivilege (hToken, SE_DEBUG_NAME, TRUE))
538  {
539  ret = -1;
540  goto error_exit;
541  }
542 
543  hProcess = OpenProcess (PROCESS_ALL_ACCESS, FALSE, pid);
544  if (hProcess == NULL)
545  {
546  ret = -1;
547  goto error_exit;
548  }
549 
550  if (!GetProcessTimes (hProcess, &dummy1, &dummy2, &kt, &ut))
551  {
552  ret = -1;
553  goto error_exit;
554  }
555 
556  lk.HighPart = kt.dwHighDateTime;
557  lk.LowPart = kt.dwLowDateTime;
558  lu.HighPart = ut.dwHighDateTime;
559  lu.LowPart = ut.dwLowDateTime;
560 
561  stat->cpu_kernel = lk.QuadPart;
562  stat->cpu_user = lu.QuadPart;
563 
564  memset (&pmc, 0, sizeof (pmc));
565  pmc.cb = sizeof (pmc);
566 
567  if (!GetProcessMemoryInfo (hProcess, &pmc, sizeof (pmc)))
568  {
569  ret = -1;
570  goto error_exit;
571  }
572 
573  stat->mem_physical = pmc.WorkingSetSize;
574 
575  memset (&ms, 0, sizeof (ms));
576  ms.dwLength = sizeof (ms);
577 
578  if (!GlobalMemoryStatusEx (&ms))
579  {
580  ret = -1;
581  goto error_exit;
582  }
583 
584  stat->mem_virtual = ms.ullTotalVirtual - ms.ullAvailVirtual;
585 
586 error_exit:
587  if (hProcess != NULL)
588  {
589  CloseHandle (hProcess);
590  }
591  if (hToken != NULL)
592  {
593  CloseHandle (hToken);
594  }
595  return ret;
596 }

856  if (memfp == NULL)

861  fclose (cpufp);

863  }
864

866  {
867  sscanf (linebuf, "%49s", prefix);

871  &stat->cpu_iowait);

875  }

880  }
881

883  {
884  sscanf (linebuf, "%49s", prefix);
885  if (!strcmp (prefix, "MemTotal:"))
886  {
887  sscanf (linebuf, "%*s%lu", &stat->mem_physical_total);
888  n_memitem++;
889  }

892  sscanf (linebuf, "%*s%lu", &stat->mem_physical_free);
893  n_memitem++;
894  }
895  if (!strcmp (prefix, "Buffers:"))
896  {
897  sscanf (linebuf, "%*s%lu", &buffers);
898  }
899  if (!strcmp (prefix, "Cached:"))
900  {
901  sscanf (linebuf, "%*s%lu", &cached);
902  }
903  if (!strcmp (prefix, "SwapTotal:"))
904  {
905  sscanf (linebuf, "%*s%lu", &stat->mem_swap_total);
906  n_memitem++;
907  }
908  if (!strcmp (prefix, "SwapFree:"))
909  {
910  sscanf (linebuf, "%*s%lu", &stat->mem_swap_free);
911  n_memitem++;
912  }
913  }
914  if (n_memitem != 4)
915  {
916  goto error_handle;
917  }
918 
919  stat->mem_physical_free += (buffers + cached);
920 
921  stat->mem_physical_total *= 1024;
922  stat->mem_physical_free *= 1024;
923  stat->mem_swap_total *= 1024;
924  stat->mem_swap_free *= 1024;

926  fclose (cpufp);
927  fclose (memfp);

929 
930 error_handle:

934  fclose (memfp);
