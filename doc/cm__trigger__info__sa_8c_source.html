126 get_trigger_information (FILE * fp, DB_OBJECT * triggerobj)
127 {
128  char *trigger_name, *action, *attr, *condition, *comment;
129  DB_OBJECT *target_class;
130  DB_TRIGGER_EVENT event;
131  DB_TRIGGER_TIME eventtime, actiontime;

134  double priority;
135  char pri_string[10];
136 
137  trigger_name = action = comment = NULL;
138 
139  /* trigger name */
140  db_trigger_name (triggerobj, &trigger_name);
141  fprintf (fp, MSGFMT, "name", trigger_name);
142  if (trigger_name != NULL)
143  db_string_free ((char *) trigger_name);
144 
145  /* eventtime */
146  db_trigger_condition_time (triggerobj, &eventtime);
147  switch (eventtime)
148  {
149  case TR_TIME_BEFORE:
150  fprintf (fp, MSGFMT, "conditiontime", "BEFORE");
151  break;
152  case TR_TIME_AFTER:
153  fprintf (fp, MSGFMT, "conditiontime", "AFTER");
154  break;
155  case TR_TIME_DEFERRED:
156  fprintf (fp, MSGFMT, "conditiontime", "DEFERRED");
157  break;
158  case TR_TIME_NULL:
159  break;
160  }
161 
162  /* eventtype */
163  db_trigger_event (triggerobj, &event);
164  switch (event)
165  {
166  case TR_EVENT_UPDATE:
167  fprintf (fp, MSGFMT, "eventtype", "UPDATE");
168  break;
169  case TR_EVENT_STATEMENT_UPDATE:
170  fprintf (fp, MSGFMT, "eventtype", "STATEMENT UPDATE");
171  break;
172  case TR_EVENT_DELETE:
173  fprintf (fp, MSGFMT, "eventtype", "DELETE");
174  break;
175  case TR_EVENT_STATEMENT_DELETE:
176  fprintf (fp, MSGFMT, "eventtype", "STATEMENT DELETE");
177  break;
178  case TR_EVENT_INSERT:
179  fprintf (fp, MSGFMT, "eventtype", "INSERT");
180  break;
181  case TR_EVENT_STATEMENT_INSERT:
182  fprintf (fp, MSGFMT, "eventtype", "STATEMENT INSERT");
183  break;
184 
185  case TR_EVENT_COMMIT:
186  fprintf (fp, MSGFMT, "eventtype", "COMMIT");
187  break;
188  case TR_EVENT_ROLLBACK:
189  fprintf (fp, MSGFMT, "eventtype", "ROLLBACK");
190  break;
191  default:
192  break;
193  }
194 
195  /* trigger action */
196  db_trigger_action_type (triggerobj, &action_type);
197  switch (action_type)
198  {
199  case TR_ACT_EXPRESSION: /* act like TR_ACT_PRINT */
200  case TR_ACT_PRINT:
201  db_trigger_action (triggerobj, &action);
202  fprintf (fp, MSGFMT, "action", action);
203  if (action != NULL)
204  {
205  db_string_free ((char *) action);
206  }
207  break;
208 
209  case TR_ACT_REJECT:
210  fprintf (fp, MSGFMT, "action", "REJECT");
211  break;
212  case TR_ACT_INVALIDATE:
213  fprintf (fp, MSGFMT, "action", "INVALIDATE TRANSACTION");
214  break;
215  case TR_ACT_NULL:
216  break;
217  }
218 
219  /* target class & att */
220  db_trigger_class (triggerobj, &target_class);
221  if (target_class != NULL)
222  {
223  fprintf (fp, MSGFMT, "target_class", db_get_class_name (target_class));
224  }
225 
226  db_trigger_attribute (triggerobj, &attr);
227  if (attr != NULL)
228  {
229  fprintf (fp, MSGFMT, "target_att", attr);
230  db_string_free ((char *) attr);
231  }
232 
233  /* condition */
234  db_trigger_condition (triggerobj, &condition);
235  if (condition != NULL)
236  {
237  fprintf (fp, MSGFMT, "condition", condition);
238  db_string_free ((char *) condition);
239  }
240 
241  /* actiontime */
242  db_trigger_action_time (triggerobj, &actiontime);
243  switch (actiontime)
244  {
245  case TR_TIME_BEFORE:
246  fprintf (fp, MSGFMT, "actiontime", "BEFORE");
247  break;
248  case TR_TIME_AFTER:
249  fprintf (fp, MSGFMT, "actiontime", "AFTER");
250  break;
251  case TR_TIME_DEFERRED:
252  fprintf (fp, MSGFMT, "actiontime", "DEFERRED");
253  break;
254  case TR_TIME_NULL:
255  break;
256  }
257 
258  /* status */
259  db_trigger_status (triggerobj, &status);
260  switch (status)
261  {
262  case TR_STATUS_ACTIVE:
263  fprintf (fp, MSGFMT, "status", "ACTIVE");
264  break;
265  case TR_STATUS_INACTIVE:
266  fprintf (fp, MSGFMT, "status", "INACTIVE");
267  break;
268  case TR_STATUS_INVALID:
269  fprintf (fp, MSGFMT, "status", "INVALID");
270  }
271 
272  /* priority */
273  db_trigger_priority (triggerobj, &priority);
274  sprintf (pri_string, "%4.5f", priority);
275  fprintf (fp, MSGFMT, "priority", pri_string);
276 
277  /* comment */

279  if (comment != NULL)
280  {
281  fprintf (fp, MSGFMT, "comment", comment);
282  db_string_free ((char *) comment);
283  }
