485  if (fp == NULL)

492  info = (T_SERVER_STATUS_INFO *) malloc (sizeof (T_SERVER_STATUS_INFO) * num_alloc);
493  if (info == NULL)
494  {
495  fclose (fp);
496  return;
497  }
498 
499  while (fgets (str_buf, sizeof (str_buf), fp))
500  {
501  char *tmp_p;
502 
503  if (sscanf (str_buf, "%63s %63s", tmp_str, db_name) < 2)
504  continue;
505  if (strcmp (tmp_str, "@") == 0 || (strcmp (tmp_str, "Server") != 0 && strcmp (tmp_str, "HA-Server") != 0))
506  continue;
507 
508  tmp_p = strchr (db_name, ',');
509  if (tmp_p != NULL)
510  *tmp_p = '\0';
511 
512  num_info++;
513  if (num_info > num_alloc)
514  {
515  num_alloc += 5;
516  info = (T_SERVER_STATUS_INFO *) realloc (info, sizeof (T_SERVER_STATUS_INFO) * num_alloc);
517  if (info == NULL)
518  {
519  fclose (fp);
520  return;
521  }
522  }
523  strcpy (info[num_info - 1].db_name, db_name);
524  info[num_info - 1].ha_mode = (strcmp (tmp_str, "HA-Server") == 0) ? 1 : 0;
525  }
526  fclose (fp);
527 
528  res->num_result = num_info;
529  res->result = info;
530 }
