269 /*
270  * css_net_send_no_block () - Sends blocks of zero length packets
271  * return:
272  * fd(in): the socket fd
273  * buffer(in): the buffer full of zero data
274  * size(in): amout of data to send
275  *
276  * Note: This allows the server to either detect one of the following cases:
277  * The client has gone down (the send fails)
278  * The client is alive, but not waiting for server input (EWOULDBLOCK)
279  * The client is waiting for a server request to complete, but is
280  * still consuming blocks of zero length data. (This could be the
281  * case when a client is waiting for a lock or query result).
282  */
283 int
284 css_net_send_no_block (SOCKET fd, const char *buffer, int size)
285 {
286 #if defined(WINDOWS)
287  int rc, total = 0;
288  unsigned long noblock = 1, block = 0;
289  int winsock_error;
290 
291  rc = ioctlsocket (fd, FIONBIO, &noblock);
292  if (rc == SOCKET_ERROR)
293  {
294  return ERROR_ON_WRITE;
295  }
296 
297  for (total = 0; total < 2 * size; total += rc)
298  {
299  rc = send (fd, buffer, size, 0);
300  if (rc != size)
301  {
302  winsock_error = WSAGetLastError ();
303  if (rc < 0 && winsock_error != WSAEWOULDBLOCK && winsock_error != WSAEINTR)
304  {
305  return ERROR_ON_WRITE;
306  }
307  else
308  {
309  break;
310  }
311  }
312  }
313 
314  rc = ioctlsocket (fd, FIONBIO, &block);
315  if (rc != 0)
316  {
317  return ERROR_ON_WRITE;
318  }
319 
320  return NO_ERRORS;
321 #else /* WINDOWS */
322  int rc, noblock = 1, block = 0, total = 0;
323 
324  rc = ioctl (fd, FIONBIO, (caddr_t) (&noblock));
325  if (rc < 0)
326  {
327  return ERROR_ON_WRITE;
328  }
329 
330  for (total = 0; total < 2 * size; total += rc)
331  {
332  errno = 0;
333  rc = send (fd, buffer, size, 0);
334  if (rc != size)
335  {
336  if (rc <= 0 && errno != EWOULDBLOCK && errno != EINTR && errno != EAGAIN && errno != EACCES)
337  {
338  return (ERROR_ON_WRITE);
339  }
340  else
341  {
342  break;
343  }
344  }
345  }
346 
347  rc = ioctl (fd, FIONBIO, (caddr_t) (&block));
348  if (rc < 0)
349  {

354 #endif /* WINDOWS */

357 /*
358  * css_readn() - read "n" bytes from a descriptor.
359  * return: count of bytes actually read
360  * fd(in): sockert descripter
361  * ptr(out): buffer
362  * nbytes(in): count of bytes will be read
363  * timeout(in): timeout in milli-second
364  */
365 int
366 css_readn (SOCKET fd, char *ptr, int nbytes, int timeout)
367 {
368  int nleft, n;
369 
370 #if defined (WINDOWS)
371  int winsock_error;
372 #else
373  struct pollfd po[1] = { {0, 0, 0} };
374 #endif /* WINDOWS */
375 
376  if (fd < 0)
377  {
378  er_log_debug (ARG_FILE_LINE, "css_readn: fd < 0");
379  errno = EINVAL;
380  return -1;
381  }
382 
383  if (nbytes <= 0)
384  {
385  return 0;
386  }
387 
388  nleft = nbytes;
389  do
390  {
391 #if !defined (WINDOWS)
392  po[0].fd = fd;
393  po[0].events = POLLIN;
394  po[0].revents = 0;
395  n = poll (po, 1, timeout);
396  if (n == 0)
397  {
398  /* 0 means it timed out and no fd is changed. */
399  errno = ETIMEDOUT;
400  return -1;
401  }
402  else if (n < 0)
403  {
404  if (errno == EINTR)
405  {
406 #if !defined (SERVER_MODE)
407  if (css_server_timeout_fn != NULL)
408  {
409  css_server_timeout_fn ();
410  }
411 #endif /* !SERVER_MODE */
412  continue;
413  }
414  er_log_debug (ARG_FILE_LINE, "css_readn: %s", strerror (errno));
415  return -1;
416  }
417  else
418  {
419  if (po[0].revents & POLLERR || po[0].revents & POLLHUP)
420  {
421  errno = EINVAL;
422  er_log_debug (ARG_FILE_LINE, "css_readn: %s %s", (po[0].revents & POLLERR ? "POLLERR" : "POLLHUP"),
423  strerror (errno));
424  return -1;
425  }
426  }
427 #endif /* !WINDOWS */
428 
429  read_again:
430  n = recv (fd, ptr, nleft, 0);
431 
432  if (n == 0)
433  {
434  break;
435  }
436 
437  if (n < 0)
438  {
439 #if !defined(WINDOWS)
440  if (errno == EAGAIN)
441  {
442  continue;
443  }
444  if (errno == EINTR)
445  {
446  goto read_again;
447  }
448 #else
449  winsock_error = WSAGetLastError ();
450 
451  /* In Windows 2003, pass large length (such as 120MB) to recv() will temporary unavailable by error number
452  * WSAENOBUFS (10055) */
453  if (winsock_error == WSAENOBUFS)
454  {
455  goto read_again;
456  }
457 
458  if (winsock_error == WSAEINTR)
459  {
460  goto read_again;
461  }
462 #endif
463 #if !defined (SERVER_MODE)
464  css_set_networking_error (fd);
465 #endif /* !SERVER_MODE */
466 
467  er_log_debug (ARG_FILE_LINE, "css_readn: returning error n %d, errno %s\n", n, strerror (errno));
468 
469  return n; /* error, return < 0 */
470  }
471  nleft -= n;
472  ptr += n;
473  }
474  while (nleft > 0);
475 
476  return (nbytes - nleft); /* return >= 0 */
477 }

479 /*
480  * css_read_remaining_bytes() - read remaining data
481  * return: void
482  * fd(in): socket descripter
483  * len(in): count of bytes
484  *
485  * Note: This routine will "use up" any remaining data that may be on the
486  * socket, but for which no space has been allocated.
487  * This will happen if the client provides a data buffer for a request
488  * that is too small for the data sent by the server.
489  */
490 void
491 css_read_remaining_bytes (SOCKET fd, int len)
492 {
493  char temp_buffer[CSS_TRUNCATE_BUFFER_SIZE];
494  int nbytes, buf_size;
495 
496  while (len > 0)
497  {
498  if (len <= SSIZEOF (temp_buffer))
499  {
500  buf_size = len;
501  }
502  else
503  {
504  buf_size = SSIZEOF (temp_buffer);
505  }
506 
507  nbytes = css_readn (fd, temp_buffer, buf_size, -1);
508  /*
509  * nbytes will be less than the size of the buffer if any of the
510  * following hold:
511  * a) the socket has been closed for some reason (e.g., the client
512  * was killed);
513  * b) there was some other kind of read error;
514  * c) we have simply read all of the bytes that were asked for.
515  */
516  if (nbytes < buf_size)
517  {
518  break;
519  }
520  len -= buf_size;
521  }
522  /* TODO: return error or length */
523 }

525 /*
526  * css_net_recv() - reading a "packet" from the socket.
527  * return: 0 if success, or error code
528  * fd(in): socket descripter
529  * buffer(out): buffer for date be read
530  * maxlen(out): count of bytes was read
531  * timeout(in): timeout value in milli-second
532  */
533 int
534 css_net_recv (SOCKET fd, char *buffer, int *maxlen, int timeout)
535 {
536  int nbytes;
537  int templen;
538  int length_to_read;
539  int time_unit;
540  int elapsed;
541 
542  if (timeout < 0)
543  {
544  timeout = INT_MAX;
545  }
546  time_unit = timeout > 5000 ? 5000 : timeout;
547  elapsed = time_unit;
548 
549  /* read data length */
550  while (true)
551  {
552  nbytes = css_readn (fd, (char *) &templen, sizeof (int), time_unit);
553  if (nbytes < 0)
554  {
555  if (errno == ETIMEDOUT && timeout > elapsed)
556  {
557 #if defined (CS_MODE) && !defined (WINDOWS)
558  if (CHECK_SERVER_IS_ALIVE ())
559  {
560  if (css_peer_alive (fd, time_unit) == false)
561  {
562  return ERROR_WHEN_READING_SIZE;
563  }
564  if (css_check_server_alive_fn != NULL)
565  {
566  if (css_check_server_alive_fn (NULL, NULL) == false)
567  {
568  return ERROR_WHEN_READING_SIZE;
569  }
570  }
571  }
572 #endif /* CS_MODE && !WINDOWS */
573  elapsed += time_unit;
574  continue;
575  }
576  return ERROR_WHEN_READING_SIZE;
577  }
578  if (nbytes != sizeof (int))
579  {
580 #ifdef CUBRID_DEBUG
581  er_log_debug (ARG_FILE_LINE, "css_net_recv: returning ERROR_WHEN_READING_SIZE bytes %d \n", nbytes);
582 #endif
583  return ERROR_WHEN_READING_SIZE;
584  }
585  else
586  {
587  break;
588  }
589  }
590 
591  templen = ntohl (templen);
592  if (templen > *maxlen)
593  {
594  length_to_read = *maxlen;
595  }
596  else
597  {
598  length_to_read = templen;
599  }
600 
601  /* read data */
602  nbytes = css_readn (fd, buffer, length_to_read, timeout);
603  if (nbytes < length_to_read)
604  {
605 #ifdef CUBRID_DEBUG
606  er_log_debug (ARG_FILE_LINE, "css_net_recv: returning ERROR_ON_READ bytes %d\n", nbytes);
607 #endif
608  return ERROR_ON_READ;
609  }
610 
611  /*
612  * This is possible if the data buffer provided by the client is smaller
613  * than the number of bytes sent by the server
614  */
615 
616  if (nbytes && (templen > nbytes))
617  {
618  css_read_remaining_bytes (fd, templen - nbytes);
619  return RECORD_TRUNCATED;
620  }
621 
622  if (nbytes != templen)
623  {
624 #ifdef CUBRID_DEBUG
625  er_log_debug (ARG_FILE_LINE, "css_net_recv: returning READ_LENGTH_MISMATCH bytes %d\n", nbytes);
626 #endif
627  return READ_LENGTH_MISMATCH;
628  }
629 
630  *maxlen = nbytes;
631  return NO_ERRORS;
632 }

768 /*
769  * css_vector_send() - Winsock simulation of css_vector_send.
770  * return: size of sent if success, or error code
771  * fd(in): socket descripter
772  * vec(in): vector buffer
773  * len(in): vector length
774  * bytes_written(in):
775  * timeout(in): timeout value in milli-seconds
776  *
777  * Note: Does not support the "byte_written" argument for retries, we'll
778  * internally keep retrying the operation until all the data is written.
779  * That's what all the callers do anyway.
780  */
781 int
782 css_vector_send (SOCKET fd, struct iovec *vec[], int *len, int bytes_written, int timeout)
783 {
784  int i, total_size, available, amount, rc;
785  char *src, *dest;
786  int handle_os_error;
787 #if defined(SERVER_MODE)
788  int vb_index;
789 #endif
790  /* don't support this, we'll write everything supplied with our own internal retry loop */
791  handle_os_error = 1;
792  if (bytes_written)
793  {
794  rc = -1;
795  handle_os_error = 0;
796  goto error;
797  }
798 
799  /* calculate the total size of the stuff we need to send */
800  total_size = 0;
801  for (i = 0; i < *len; i++)
802  {
803  total_size += (*vec)[i].iov_len;
804  }
805 
806 #if defined(SERVER_MODE)
807  vb_index = alloc_vector_buffer ();
808  dest = css_Vector_buffer_piece[vb_index];
809 #else
810  dest = css_Vector_buffer;
811 #endif
812  available = CSS_VECTOR_SIZE;
813 
814  for (i = 0; i < *len; i++)
815  {
816  src = (*vec)[i].iov_base;
817  amount = (*vec)[i].iov_len;
818 
819  /* if we've got more than we have room for, fill and send */
820 
821  while (amount > available)
822  {
823  memcpy (dest, src, available);
824 
825 #if defined(SERVER_MODE)
826  rc = css_writen (fd, css_Vector_buffer_piece[vb_index], CSS_VECTOR_SIZE);
827 #else
828  rc = css_writen (fd, css_Vector_buffer, CSS_VECTOR_SIZE);
829 #endif
830  if (rc != CSS_VECTOR_SIZE)
831  {
832  goto error;
833  }
834 
835  src += available;
836  amount -= available;
837 #if defined(SERVER_MODE)
838  dest = css_Vector_buffer_piece[vb_index];
839 #else
840  dest = css_Vector_buffer;
841 #endif
842  available = CSS_VECTOR_SIZE;
843  }
844 
845  /* if we have some amount that fits within the buffer, store it and move on */
846  if (amount)
847  {
848  memcpy (dest, src, amount);
849  dest += amount;
850  available -= amount;
851  }
852  }
853 
854  /* see if we have any residual bytes left to be sent */
855  if (available < CSS_VECTOR_SIZE)
856  {
857  amount = CSS_VECTOR_SIZE - available;
858 #if defined(SERVER_MODE)
859  rc = css_writen (fd, css_Vector_buffer_piece[vb_index], amount);
860 #else
861  rc = css_writen (fd, css_Vector_buffer, amount);
862 #endif
863  if (rc != amount)
864  {
865  goto error;
866  }
867  }
868 
869 #if defined(SERVER_MODE)
870  free_vector_buffer (vb_index);
871 #endif
872  return total_size;
873 
874 error:
875  /*
876  * We end up with an error. The error has already been set in css_writen
877  */
878 #if defined(SERVER_MODE)
879  free_vector_buffer (vb_index);
880 #endif
881  return rc;
882 }

886  * css_vector_send() -
887  * return: size of sent if success, or error code
888  * fd(in): socket descripter
889  * vec(in): vector buffer
890  * len(in): vector length
891  * bytes_written(in):
892  * timeout(in): timeout value in milli-seconds
893  */
894 int
895 css_vector_send (SOCKET fd, struct iovec *vec[], int *len, int bytes_written, int timeout)
896 {
897  int i, n;
898  struct pollfd po[1] = { {0, 0, 0} };
899 
900  if (fd < 0)
901  {
902  er_log_debug (ARG_FILE_LINE, "css_vector_send: fd < 0");
903  errno = EINVAL;
904  return -1;
905  }
906 
907  if (bytes_written > 0)
908  {
909  er_log_debug (ARG_FILE_LINE, "css_vector_send: retry called for %d\n", bytes_written);
910 
911  for (i = 0; i < *len; i++)
912  {
913  if ((*vec)[i].iov_len <= (size_t) bytes_written)
914  {
915  bytes_written -= (*vec)[i].iov_len;
916  }
917  else
918  {
919  break;
920  }
921  }
922  (*vec)[i].iov_len -= bytes_written;
923  (*vec)[i].iov_base = ((char *) ((*vec)[i].iov_base)) + bytes_written;
924 
925  (*vec) += i;
926  *len -= i;
927  }
928 
929  while (true)
930  {
931  po[0].fd = fd;
932  po[0].events = POLLOUT;
933  po[0].revents = 0;
934  n = poll (po, 1, timeout);
935  if (n < 0)
936  {
937  if (errno == EINTR)
938  {
939  continue;
940  }
941 
942  er_log_debug (ARG_FILE_LINE, "css_vector_send: EINTR %s\n", strerror (errno));
943  return -1;
944  }
945  else if (n == 0)
946  {
947  /* 0 means it timed out and no fd is changed. */
948  errno = ETIMEDOUT;
949  return -1;
950  }
951  else
952  {
953  if (po[0].revents & POLLERR || po[0].revents & POLLHUP)
954  {
955  errno = EINVAL;
956  er_log_debug (ARG_FILE_LINE, "css_vector_send: %s %s\n",
957  (po[0].revents & POLLERR ? "POLLERR" : "POLLHUP"), strerror (errno));
958  return -1;
959  }
960  }
961 
962  write_again:
963  n = writev (fd, *vec, *len);
964  if (n > 0)
965  {
966  return n;
967  }
968  else if (n == 0)
969  {
970  return 0; /* ??? */
971  }
972  else
973  {
974  if (errno == EINTR)
975  {
976  goto write_again;
977  }
978  if (errno == EAGAIN)
979  {
980  continue;
981  }
982 #if !defined (SERVER_MODE)
983  css_set_networking_error (fd);
984 #endif /* !SERVER_MODE */
985 
986  er_log_debug (ARG_FILE_LINE, "css_vector_send: returning error n %d, errno %s\n", n, strerror (errno));
987  return n; /* error, return < 0 */
988  }
989  }
990 
991  return -1;
992 }
