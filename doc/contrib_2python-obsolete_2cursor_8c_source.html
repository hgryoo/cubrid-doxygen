98 static int
99 cursor_object_set_description(Cursor_Object *self)
100 {
101  PyObject *desc;
102  int i;
103  char colName[128];
104  int datatype, precision, scale, nullable;
105 
106  if (self->col_count == 0) {
107  Py_XDECREF(self->description);
108  self->description = PyTuple_New(0);
109  return 1;
110  }
111 
112  desc = (PyObject *) PyTuple_New(self->col_count);
113 
114  for ( i = 0 ; i < self->col_count ; i++ ) {
115  PyObject *item;
116 
117  item = PyTuple_New(7);
118 
119  strcpy(colName, CCI_GET_RESULT_INFO_NAME(self->col_info, i+1));
120  precision = CCI_GET_RESULT_INFO_PRECISION(self->col_info, i+1);
121  scale = CCI_GET_RESULT_INFO_SCALE(self->col_info, i+1);
122  nullable = CCI_GET_RESULT_INFO_IS_NON_NULL(self->col_info, i+1);
123 
124  datatype = CCI_GET_RESULT_INFO_TYPE(self->col_info, i+1);
125  /*
126  * char 1
127  * string,varchar 2
128  * nchar 3
129  * varnchar 4
130  * bit 5
131  * varbit 6
132  * numeric 7
133  * int 8
134  * short 9
135  * monetary 10
136  * float 11
137  * double 12
138  * date 13
139  * time 14
140  * timestamp 15
141  * object 19
142  * set 32
143  * multiset 64
144  * sequence 96
145  */
146 
147  PyTuple_SetItem(item, 0, PyString_FromString(colName));
148  PyTuple_SetItem(item, 1, PyInt_FromLong(datatype));
149  PyTuple_SetItem(item, 2, PyInt_FromLong(0));
150  PyTuple_SetItem(item, 3, PyInt_FromLong(0));
151  PyTuple_SetItem(item, 4, PyInt_FromLong(precision));
152  PyTuple_SetItem(item, 5, PyInt_FromLong(scale));
153  PyTuple_SetItem(item, 6, PyInt_FromLong(nullable));
154 
155  PyTuple_SetItem(desc, i, item);
156  }
157 
158  Py_XDECREF(self->description);
159  self->description = desc;
160 
161  return 1;
162 }

192 static PyObject *
193 cursor_object_execute(Cursor_Object *self, PyObject *args)
194 {
195  T_CCI_ERROR error;
196  T_CCI_COL_INFO *res_col_info;
197  T_CCI_SQLX_CMD res_sql_type;
198  T_CCI_PARAM_INFO *res_param_info = NULL;
199  char *stmt = "";
200  PyObject *params = NULL;
201  int res, res_col_count;
202 
203  if (!PyArg_ParseTuple(args, "s|O", &stmt, &params)) {
204  return NULL;
205  }
206 
207  if (!self->last_stmt || strcmp(self->last_stmt, stmt) != 0) {
208  cursor_object_reset(self);
209 
210  res = cci_prepare(self->connection, stmt, 0 /*CCI_PREPARE_INCLUDE_OID*/, &error);
211  if (res < 0) {
212  return handle_error(res, &error);
213  }
214  self->handle = res;
215  if (self->last_stmt) {
216  PyMem_Free(self->last_stmt);
217  self->last_stmt = NULL;
218  }
219  self->last_stmt = PyMem_Malloc(strlen(stmt) + 1);
220  strcpy(self->last_stmt, stmt);
221  }
222 
223  if (params) {
224  res = cursor_object_bind_param(self, params, res_param_info);
225  if (res < 0) {
226  return handle_error(res, NULL);
227  }
228  }
229 
230  res = cci_execute(self->handle, 0/*CCI_EXEC_ASYNC*/, 0, &error);
231  if (res < 0) {
232  return handle_error(res, &error);
233  }
234 
235  res_col_info = cci_get_result_info(self->handle, &res_sql_type, &res_col_count);
236  if (res_sql_type == SQLX_CMD_SELECT && !res_col_info) {
237  return handle_error(CUBRID_ER_CANNOT_GET_COLUMN_INFO, NULL);
238  }
239 
240  self->col_info = res_col_info;
241  self->col_count = res_col_count;
242  self->async_mode = 2;
243 
244  switch(res_sql_type) {
245  case SQLX_CMD_SELECT:
246  self->rowCount = res;
247  break;
248  case SQLX_CMD_INSERT:
249  case SQLX_CMD_UPDATE:
250  case SQLX_CMD_DELETE:
251  self->affected_rows = res;
252  break;
253  case SQLX_CMD_CALL:
254  self->rowCount = res;
255 
256  default:
257  break;
258  }
259 
260  if (res_sql_type == SQLX_CMD_SELECT) {
261  cursor_object_set_description(self);
262 
263  res = cci_cursor(self->handle, 1, CCI_CURSOR_CURRENT, &error);
264  if (res < 0 && res != CCI_ER_NO_MORE_DATA) {
265  return handle_error(res, &error);
266  }
267  }
268 
269  return Py_BuildValue("i", res);
270 }

278 static PyObject *
279 cursor_object_dbval_to_pyvalue(Cursor_Object *self, int type, int index)
280 {
281  int res, ind;
282  PyObject *val, *tmpval;
283  char *res_buf;
284  int int_val;
285 
286  switch (type) {
287  case CCI_U_TYPE_INT:
288  case CCI_U_TYPE_SHORT:
289  res = cci_get_data(self->handle, index, CCI_A_TYPE_INT, &int_val, &ind);
290  if (res < 0) {
291  return handle_error(res, NULL);
292  }
293  if (ind < 0) {
294  Py_INCREF(Py_None);
295  val = Py_None;
296  } else {
297  val = PyInt_FromLong(int_val);
298  }
299  break;
300 
301  case CCI_U_TYPE_FLOAT:
302  case CCI_U_TYPE_DOUBLE:
303  case CCI_U_TYPE_NUMERIC:
304  res = cci_get_data(self->handle, index, CCI_A_TYPE_STR, &res_buf, &ind);
305  if (res < 0) {
306  return handle_error(res, NULL);
307  }
308  if (ind < 0) {
309  Py_INCREF(Py_None);
310  val = Py_None;
311  } else {
312  tmpval = PyString_FromString(res_buf);
313 #if PY_MAJOR_VERSION >= 3
314  val = PyFloat_FromString(tmpval);
315 #else
316  val = PyFloat_FromString(tmpval, NULL);
317 #endif
318  Py_DECREF(tmpval);
319  }
320  break;
321 
322  default:
323  res = cci_get_data(self->handle, index, CCI_A_TYPE_STR, &res_buf, &ind);
324  if (res < 0) {
325  return handle_error(res, NULL);
326  }
327  if (ind < 0) {
328  Py_INCREF(Py_None);
329  val = Py_None;
330  } else {
331  val = PyString_FromString(res_buf);
332  }
333  break;
334  }
335 
336  return val;
337 }

343 cursor_object_dbset_to_pyvalue(Cursor_Object *self, int index)
344 {

373  cci_set_free(set);

375  return val;
