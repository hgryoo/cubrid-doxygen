58  void
59  critical_section_tracker::on_enter_as_reader (int cs_index)
60  {
61  if (!m_enabled || !is_started ())
62  {
63  return;
64  }
65 
66  check_csect_interdependencies (cs_index);
67 
68  cstrack_entry &cs_entry = m_cstrack_array[cs_index];
69 
70  // is first enter?
71  if (cs_entry.m_enter_count == 0)
72  {
73  cstrack_assert (!cs_entry.m_is_writer && !cs_entry.m_is_demoted);
74  cs_entry.m_enter_count++;
75  return;
76  }
77  // is not first enter.
78  assert (cs_entry.m_enter_count < MAX_REENTERS);
79 
80  if (cs_entry.m_is_writer)
81  {
82  // already entered as writer
83  // this case is allowed. if I am a writer, I will not be blocked as readers
84  cs_entry.m_enter_count++;
85  }
86  else if (cs_entry.m_is_demoted)
87  {
88  // I am not entirely sure why we allow when lock is demoted.
89  cs_entry.m_enter_count++;
90  }
91  else
92  {
93  // re-enter is not accepted, since it may cause deadlocks.
94  cs_entry.m_enter_count++;
95  cstrack_assert (false);
96  }
97  }

99  void
100  critical_section_tracker::on_enter_as_writer (int cs_index)
101  {
102  if (!m_enabled || !is_started ())
103  {
104  return;
105  }
106 
107  cstrack_entry &cs_entry = m_cstrack_array[cs_index];
108 
109  check_csect_interdependencies (cs_index);
110 
111  // is first enter?
112  if (cs_entry.m_enter_count == 0)
113  {
114  cstrack_assert (!cs_entry.m_is_writer && !cs_entry.m_is_demoted);
115  cs_entry.m_is_writer = true;
116  cs_entry.m_enter_count++;
117  return;
118  }
119  // is not first enter.
120 
121  // re-enter is only accepted if it was already writer
122  cstrack_assert (cs_entry.m_is_writer);
123  cs_entry.m_enter_count++;
124  }

235  void
236  critical_section_tracker::check (void)
237  {
238  bool m_printed_header = false;
239  std::ostream &os = std::cerr;
240 
241  for (int cs_index = 0; cs_index < CRITICAL_SECTION_COUNT; cs_index++)
242  {
243  if (m_cstrack_array[cs_index].m_enter_count > 0)
244  {
245  if (!m_printed_header)
246  {
247  os << " +--- Critical Sections" << std::endl;
248  m_printed_header = true;
249  }
250  os << " +--- " << csect_name_at (cs_index) << std::endl;
251  os << " +--- enter count = " << m_cstrack_array[cs_index].m_enter_count << std::endl;
252  os << " +--- is writer = " << m_cstrack_array[cs_index].m_is_writer << std::endl;
253  os << " +--- is demoted = " << m_cstrack_array[cs_index].m_is_demoted << std::endl;
254 
255  cstrack_assert (false);
256  }
257  }
