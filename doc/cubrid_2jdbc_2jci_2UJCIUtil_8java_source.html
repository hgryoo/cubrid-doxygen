188  final static String format_1 = "\\d\\d\\d\\d\\-\\d\\d\\-\\d\\d\\s++\\d\\d\\:\\d\\d\\:\\d\\d(\\.\\d*)?";
189  /* HH:MI:SS[.msec] [AM|PM] YYYY-MM-DD */
190  final static String format_2 = "\\d\\d\\:\\d\\d\\:\\d\\d(\\.\\d*)?\\s++\\d\\d\\d\\d\\-\\d\\d\\-\\d\\d";
191  /* MM/DD/YYYY HH:MI:SS[.msec] [AM|PM] */
192  final static String format_3 = "\\d\\d\\/\\d\\d\\/\\d\\d\\d\\d\\s++\\d\\d\\:\\d\\d\\:\\d\\d(\\.\\d*)?";
193  /* HH:MI:SS[.msec] [AM|PM] MM/DD/YYYY */
194  final static String format_4 = "\\d\\d\\:\\d\\d\\:\\d\\d(\\.\\d*)?\\s++\\d\\d\\/\\d\\d\\/\\d\\d\\d\\d";
195  /* HH:MI:SS [AM|PM] - time format */
196  final static String format_5 = "\\d\\d\\:\\d\\d\\:\\d\\d";

200  public final static Pattern pattern_millis = Pattern.compile("[.]");

203  public static TimeInfo parseStringTime(String str_time) throws CUBRIDException {
204  TimeInfo timeinfo = new TimeInfo();
205  String str_timestamp = "", str_timezone = "";
206  int timestamp_count = 0;
207  boolean isDateTime = false, isPM = false;
208 
209  Matcher matcher = TimePattern.pattern_ampm.matcher(str_time);
210  if (matcher.find()) {
211  String found = matcher.group();
212  str_time = str_time.replace(found, "");
213  str_time = str_time.trim();
214  if ((found.charAt(0) == 'p') || (found.charAt(0) == 'P')){
215  isPM = true;
216  }
217  }
218 
219  matcher = TimePattern.pattern_time.matcher(str_time);
220  while (matcher.find()) {
221  str_timestamp = matcher.group().trim();
222  str_timezone = str_time.substring(str_timestamp.length(), str_time.length()).trim();
223  timestamp_count++;
224  if (timestamp_count > 1) {
225  throw new CUBRIDException(CUBRIDJDBCErrorCode.invalid_value);
226  }
227  }
228 
229  if (timestamp_count == 0) {
230  throw new CUBRIDException(CUBRIDJDBCErrorCode.invalid_value);
231  }
232 
233  matcher = TimePattern.pattern_millis.matcher(str_timestamp);
234  if (matcher.find()) {
235  isDateTime = true;
236  }
237 
238  timeinfo.time = str_timestamp;
239  timeinfo.timezone = str_timezone;
240  timeinfo.isDatetime = isDateTime;
241  timeinfo.isPM = isPM;
242  return timeinfo;
243  };
