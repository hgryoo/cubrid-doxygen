281  public static CUBRIDOID getNewInstance(CUBRIDConnection con, String oidStr)
282  throws SQLException {
283  if (con == null || oidStr == null) {
284  throw new IllegalArgumentException();
285  }
286  if (oidStr.charAt(0) != '@')
287  throw new IllegalArgumentException();
288  StringTokenizer oidStringArray = new StringTokenizer(oidStr, "|");
289  try {
290  int page = Integer
291  .parseInt(oidStringArray.nextToken().substring(1));
292  short slot = Short.parseShort(oidStringArray.nextToken());
293  short vol = Short.parseShort(oidStringArray.nextToken());
294 
295  byte[] bOID = new byte[UConnection.OID_BYTE_SIZE];
296  bOID[0] = ((byte) ((page >>> 24) & 0xFF));
297  bOID[1] = ((byte) ((page >>> 16) & 0xFF));
298  bOID[2] = ((byte) ((page >>> 8) & 0xFF));
299  bOID[3] = ((byte) ((page >>> 0) & 0xFF));
300  bOID[4] = ((byte) ((slot >>> 8) & 0xFF));
301  bOID[5] = ((byte) ((slot >>> 0) & 0xFF));
302  bOID[6] = ((byte) ((vol >>> 8) & 0xFF));
303  bOID[7] = ((byte) ((vol >>> 0) & 0xFF));
304 
305  return new CUBRIDOID(con, bOID);
306  } catch (NoSuchElementException e) {
307  throw new IllegalArgumentException();
308  }
309  }
