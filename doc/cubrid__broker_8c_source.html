98 static int
99 cubrid_broker_read (void)
100 {
101  T_SHM_BROKER *shm_br = NULL;
102  T_SHM_APPL_SERVER *shm_appl = NULL;
103  int i, j;
104  int total_trs = 0;
105  int total_qps = 0;
106  value_t trs[1];
107  value_t trs_for_qps[1];
108  value_t cas_status[5];
109  time_t cur_time;
110 
111  shm_br =
112  (T_SHM_BROKER *) uw_shm_open (master_shm_id, SHM_BROKER,
113  SHM_MODE_MONITOR);
114  if (shm_br == NULL)
115  {
116  return 0;
117  }
118  time (&cur_time);
119  for (i = 0; i < shm_br->num_broker; i++)
120  {
121  if (shm_br->br_info[i].service_flag != ON)
122  continue;
123 
124  memset (trs, 0, sizeof (trs));
125  memset (trs_for_qps, 0, sizeof (trs_for_qps));
126  memset (cas_status, 0, sizeof (cas_status));
127 
128  shm_appl = (T_SHM_APPL_SERVER *)
129  uw_shm_open (shm_br->br_info[i].appl_server_shm_id,
130  SHM_APPL_SERVER, SHM_MODE_MONITOR);
131 
132  if (shm_appl == NULL)
133  {
134  return 0;
135  }
136  for (j = 0; j < shm_br->br_info[i].appl_server_max_num; j++)
137  {
138  total_trs += shm_appl->as_info[j].num_transactions_processed;
139  total_qps += shm_appl->as_info[j].num_queries_processed;
140 
141  if (shm_appl->as_info[j].service_flag == SERVICE_ON)
142  {
143  cas_status[TOTAL].gauge++;
144  if (shm_appl->as_info[j].uts_status == UTS_STATUS_BUSY)
145  {
146  if (shm_br->br_info[i].appl_server == APPL_SERVER_CAS)
147  {
148  if (shm_appl->as_info[j].con_status ==
149  CON_STATUS_OUT_TRAN)
150  cas_status[CLOSE_WAIT].gauge++;
151  else if (shm_appl->as_info[j].log_msg[0] == '\0')
152  cas_status[CLIENT_WAIT].gauge++;
153  else
154  cas_status[BUSY].gauge++;
155  }
156  else
157  {
158  cas_status[BUSY].gauge++;
159  }
160  }
161  else
162  {
163  cas_status[IDLE].gauge++;
164  }
165  }
166 
167  }
168  uw_shm_detach (shm_appl);
169 
170  if (old_transactions_processed[i] == -1)
171  {
172  trs[0].gauge = 0;
173  }
174  else
175  {
176  trs[0].gauge =
177  ((total_trs - old_transactions_processed[i]) / difftime (cur_time,
178  old_time));
179  }
180  if (old_queries_processed[i] == -1)
181  {
182  trs_for_qps[0].gauge = 0;
183  }
184  else
185  {
186  trs_for_qps[0].gauge =
187  ((total_qps - old_queries_processed[i]) / difftime (cur_time,
188  old_time));
189  }
190  old_transactions_processed[i] = total_trs;
191  old_queries_processed[i] = total_qps;
192 
193  total_trs = 0;
194  total_qps = 0;
195 
196  submit ("cubrid_broker_trs", shm_br->br_info[i].name, trs, 1);
197  submit ("cubrid_broker_qps", shm_br->br_info[i].name, trs_for_qps, 1);
198  submit ("cubrid_broker_status", shm_br->br_info[i].name, cas_status, 5);
199  }
200  old_time = cur_time;
201  uw_shm_detach (shm_br);
202  return 0;
203 }
