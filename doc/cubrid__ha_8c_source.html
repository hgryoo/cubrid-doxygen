354  else

356  delay_time[0].gauge = 0;

359  submit ("cubrid_ha_delay_time", submit_name, delay_time, 1);
360  submit ("cubrid_ha_insert", submit_name, insert_count, 1);
361  submit ("cubrid_ha_update", submit_name, update_count, 1);

367 #if defined (C_API_MODE)
