301  * fill_collection -
302  * return:
303  * co():
304  * set():
305  */
306 static int
307 fill_collection (COLLECTION_ * co, DB_SET * set)
308 {
309  int set_size;
310  int i, res;
311  DB_VALUE *val;
312 
313  assert (co != NULL);
314  if (set == NULL)
315  return ER_INTERFACE_INVALID_ARGUMENT;
316 
317  set_size = db_col_size (set);
318  for (i = 0; i < set_size; i++)
319  {
320  int li_idx = i - 1;
321 
322  res = co->indexer->ifs->check (co->indexer, li_idx, CHECK_FOR_INSERT);
323  if (res != NO_ERROR)
324  return res;
325  val = db_value_create ();
326  if (!val)
327  return ER_INTERFACE_NO_MORE_MEMORY;
328  res = db_col_get (set, i, val);
329  if (res != NO_ERROR)
330  {
331  db_value_free (val);
332  return ER_INTERFACE_GENERIC;
333  }
334  res = co->indexer->ifs->insert (co->indexer, li_idx, NULL, (API_VALUE *) val);
335  if (res != NO_ERROR)
336  {
337  db_value_free (val);
338  }
339  }
340  return NO_ERROR;
341 }
