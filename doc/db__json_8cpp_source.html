20 /*
21  * db_json.cpp - functions related to json
22  * The json feature is made as a black box to not make the whole project
23  * depend on the rapidjson library. We might change this library in the future,
24  * and this is easier when it's made this way.
25  *
26  * Rapidjson allocator usage:
27  * We made the library use our own allocator, db_private_alloc. To achieve this,
28  * we created new types, like JSON_DOC and JSON_VALUE. JSON_DOC uses inheritance
29  * and not typedef as its creator in order for it to support forward declaration.
30  * Also, we made JSON_PRIVATE_ALLOCATOR class which gets passed as template arguments to these
31  * new types. This class implements malloc, realloc and free.
32  *
33  * JSON_VALUE does not use its own allocator, but rather it uses a JSON_DOC allocator.
34  * With this we can control the JSON_VALUE's scope to match the scope of the said JSON_DOC.
35  * For example, consider this piece of code:
36  *
37  * JSON_DOC *func(const char *str)
38  * {
39  * JSON_VALUE value;
40  * JSON_DOC *doc;
41  * doc = db_json_allocate_doc ();
42  *
43  * value.SetString (str, strlen (str), doc->GetAllocator ());
44  * doc.PushBack (value, doc->GetAllocator ());
45  *
46  * return doc;
47  * }
48  *
49  * Because PushBack doesn't perform a copy, one might think that this function
50  * contains a major bug, because when value gets out of scope, the JSON_VALUE's
51  * destructor gets called and the str's internal copy in value gets destroyed,
52  * leading to memory corruptions.
53  * This isn't the case, str's internal copy gets destroyed only when doc is deleted
54  * because we used doc's allocator when calling SetString.
55  */

516 int JSON_DUPLICATE_KEYS_CHECKER::CallBefore (JSON_VALUE &value)
517 {
518  std::vector<const char *> inserted_keys;
519 
520  if (value.IsObject ())
521  {
522  for (auto it = value.MemberBegin (); it != value.MemberEnd (); ++it)
523  {
524  const char *current_key = it->name.GetString ();
525 
526  for (unsigned int i = 0; i < inserted_keys.size (); i++)
527  {
528  if (strcmp (current_key, inserted_keys[i]) == 0)
529  {
530  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_JSON_DUPLICATE_KEY, 1, current_key);
531  return ER_JSON_DUPLICATE_KEY;
532  }
533  }
534 
535  inserted_keys.push_back (current_key);
536  }
537  }
538 
539  return NO_ERROR;
540 }

2020  {

2081
