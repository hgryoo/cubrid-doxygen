133 /*
134  * rm_api_get_info -
135  * return:
136  * impl():
137  * index():
138  * type():
139  * arg():
140  * size():
141  */
142 static int
143 rm_api_get_info (API_RESULTSET_META * impl, int index, CI_RMETA_INFO_TYPE type, void *arg, size_t size)
144 {
145  OBJECT_RESULTSET *or;
146  DB_ATTRIBUTE *attr;
147  DB_DOMAIN *domain;
148  int res;
149 
150  assert (impl != NULL);
151 
152  /* convert to zero based index */
153  index--;
154 
155  or = ((OBJECT_RM_BIND *) impl)->or;
156  if (index < 0 || index >= or->nattrs)
157  {
158  return ER_INTERFACE_INVALID_ARGUMENT; /* index out of range */
159  }
160  if (arg == NULL || size <= 0)
161  {
162  return ER_INTERFACE_INVALID_ARGUMENT;
163  }
164  attr = or->attr_index[index];
165 
166  assert (attr != NULL);
167 
168  switch (type)
169  {
170  case CI_RMETA_INFO_COL_LABEL:
171  /* col label is not defined. return null string */
172  *(char *) arg = '\0';
173  return NO_ERROR;
174 
175  case CI_RMETA_INFO_COL_NAME:
176  {
177  size_t namelen;
178  const char *attr_name;
179 
180  attr_name = db_attribute_name (attr);
181 
182  assert (attr_name != NULL);
183 
184  namelen = strlen (attr_name) + 1;
185  if (namelen > size)
186  {
187  return ER_INTERFACE_INVALID_ARGUMENT; /* size insufficient */
188  }
189  strcpy ((char *) arg, attr_name);
190  return NO_ERROR;
191  }
192  case CI_RMETA_INFO_COL_TYPE:
193  if (size < sizeof (CI_TYPE))
194  {
195  return ER_INTERFACE_INVALID_ARGUMENT;
196  }
197 
198  domain = db_attribute_domain (attr);
199  if (domain == NULL)
200  {
201  return ER_INTERFACE_GENERIC;
202  }
203 
204  res = db_type_to_type (TP_DOMAIN_TYPE (domain), (CI_TYPE *) arg);
205  return res;
206 
207  case CI_RMETA_INFO_PRECISION:
208  if (size < sizeof (int))
209  {
210  return ER_INTERFACE_INVALID_ARGUMENT;
211  }
212 
213  domain = db_attribute_domain (attr);
214  if (domain == NULL)
215  {
216  return ER_INTERFACE_GENERIC;
217  }
218 
219  *(int *) arg = db_domain_precision (domain);
220  return NO_ERROR;
221 
222  case CI_RMETA_INFO_SCALE:
223  if (size < sizeof (int))
224  {
225  return ER_INTERFACE_INVALID_ARGUMENT;
226  }
227 
228  domain = db_attribute_domain (attr);
229  if (domain == NULL)
230  {
231  return ER_INTERFACE_GENERIC;
232  }
233 
234  *(int *) arg = db_domain_scale (domain);
235  return NO_ERROR;
236 
237  case CI_RMETA_INFO_TABLE_NAME:
238  {
239  const char *tbl_name;
240  size_t sz;
241 
242  assert (or->clz != NULL);
243 
244  tbl_name = db_get_class_name (or->clz);
245  if (tbl_name == NULL)
246  {
247  return ER_INTERFACE_GENERIC;
248  }
249 
250  sz = strlen (tbl_name) + 1;
251  if (sz > size)
252  {
253  return ER_INTERFACE_INVALID_ARGUMENT;
254  }
255  strcpy ((char *) arg, tbl_name);
256  return NO_ERROR;
257  }
258  case CI_RMETA_INFO_IS_AUTO_INCREMENT:
259  if (size < sizeof (int))
260  {
261  return ER_INTERFACE_INVALID_ARGUMENT;
262  }
263 
264  *(int *) arg = db_attribute_is_auto_increment (attr);
265  return NO_ERROR;
266 
267  case CI_RMETA_INFO_IS_NULLABLE:
268  if (size < sizeof (int))
269  {
270  return ER_INTERFACE_INVALID_ARGUMENT;
271  }
272 
273  *(int *) arg = db_attribute_is_non_null (attr) ? 0 : 1;
274  return NO_ERROR;
275 
276  case CI_RMETA_INFO_IS_WRITABLE:
277  if (size < sizeof (int))
278  {
279  return ER_INTERFACE_INVALID_ARGUMENT;
280  }
281 
282  if (or->obj->lock == X_LOCK || or->obj->lock == U_LOCK)
283  {
284  *(int *) arg = 1;
285  }
286  else
287  {
288  *(int *) arg = 0;
289  }
290  return NO_ERROR;
291 
292  default:
293  return ER_INTERFACE_INVALID_ARGUMENT;
294  }
295 
296  assert (0);
297  return ER_INTERFACE_GENERIC;
298 }

381  */
382 static int
383 res_api_fetch (API_RESULTSET * impl, int offset, CI_FETCH_POSITION pos)
384 {
385  OBJECT_RESULTSET *or;
386  int idx;
387  assert (impl != NULL);
388  or = ((OBJECT_RES_BIND *) impl)->or;
389  if (pos == CI_FETCH_POSITION_CURRENT || pos == CI_FETCH_POSITION_FIRST)
390  idx = offset;
391  else if (pos == CI_FETCH_POSITION_LAST)
392  idx = or->nattrs + offset;
393  else
394  return ER_INTERFACE_INVALID_ARGUMENT;
395  if (idx != 0)
396  return ER_INTERFACE_INVALID_ARGUMENT; /* index out of range */
397 
398  if (or->deleted == 1)
399  return ER_INTERFACE_GENERIC; /* already deleted */
400  return NO_ERROR;
401 }
