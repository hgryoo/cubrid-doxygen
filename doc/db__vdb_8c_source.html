289 /*
290  * db_parse_one_statement() -
291  * return:
292  * session(in) :
293  */
294 int
295 db_parse_one_statement (DB_SESSION * session)
296 {
297  if (session->dimension > 0)
298  {
299  /* check if this statement is skipped */
300  if (session->type_list)
301  {
302  db_free_query_format (session->type_list[0]);
303  }
304  if (session->statements[0])
305  {
306  parser_free_tree (session->parser, session->statements[0]);
307  session->statements[0] = NULL;
308  }
309 
310  session->dimension = 0;
311  session->stmt_ndx = 0;
312 
313  session->parser->stack_top = 0;
314  if (session->stage)
315  {
316  session->stage[0] = StatementInitialStage;
317  }
318  }
319 
320  if (parse_one_statement (1) == 0 && !pt_has_error (session->parser) && session->parser->stack_top > 0
321  && session->parser->node_stack != NULL)
322  {
323  session->parser->statements = (PT_NODE **) parser_alloc (session->parser, 2 * sizeof (PT_NODE *));
324  if (session->parser->statements == NULL)
325  {
326  return -1;
327  }
328 
329  session->parser->statements[0] = session->parser->node_stack[0];
330  session->parser->statements[1] = NULL;
331 
332  session->statements = session->parser->statements;
333  session->dimension = get_dimension_of (session->statements);
334 
335  return session->dimension;
336  }
337  else
338  {
339  session->parser->statements = NULL;
340  return -1;
341  }
342 }

398 /*
399  * db_calculate_current_server_time () -
400  * return:
401  * parser(in) :
402  * server_info(in) :
403  */
404 static void
405 db_calculate_current_server_time (PARSER_CONTEXT * parser)
406 {
407 
408  struct tm *c_time_struct;
409  DB_DATETIME datetime;
410 
411  struct timeb curr_server_timeb;
412  struct timeb curr_client_timeb;
413  int diff_mtime;
414  int diff_time;
415 
416  if (base_server_timeb.time == 0)
417  {
418  return;
419  }
420 
421  ftime (&curr_client_timeb);
422  diff_time = (int) (curr_client_timeb.time - base_client_timeb.time);
423  diff_mtime = curr_client_timeb.millitm - base_client_timeb.millitm;
424 
425  if (diff_time > MAX_SERVER_TIME_CACHE)
426  {
427  base_server_timeb.time = 0;
428  }
429  else
430  {
431  curr_server_timeb.time = base_server_timeb.time;
432  curr_server_timeb.millitm = base_server_timeb.millitm;
433 
434  /* timeb.millitm is unsigned short, so should prevent underflow */
435  if (diff_mtime < 0)
436  {
437  curr_server_timeb.time--;
438  curr_server_timeb.millitm += 1000;
439  }
440 
441  curr_server_timeb.time += diff_time;
442  curr_server_timeb.millitm += diff_mtime;
443 
444  if (curr_server_timeb.millitm >= 1000)
445  {
446  curr_server_timeb.time++;
447  curr_server_timeb.millitm -= 1000;
448  }
449 
450  c_time_struct = localtime (&curr_server_timeb.time);
451  if (c_time_struct == NULL)
452  {
453  base_server_timeb.time = 0;
454  }
455  else
456  {
457  db_datetime_encode (&datetime, c_time_struct->tm_mon + 1, c_time_struct->tm_mday,
458  c_time_struct->tm_year + 1900, c_time_struct->tm_hour, c_time_struct->tm_min,
459  c_time_struct->tm_sec, curr_server_timeb.millitm);
460 
461  db_make_datetime (&parser->sys_datetime, &datetime);
462  db_make_timestamp (&parser->sys_epochtime, (DB_TIMESTAMP) curr_server_timeb.time);
463  }
464  }
465 }

488 /*
489  * db_compile_statement_local() -
490  * return:
491  * session(in) :
492  */
493 int
494 db_compile_statement_local (DB_SESSION * session)
495 {
496  PARSER_CONTEXT *parser;
497  int stmt_ndx;
498  PT_NODE *statement = NULL;
499  PT_NODE *statement_result = NULL;
500  DB_QUERY_TYPE *qtype;
501  CUBRID_STMT_TYPE cmd_type;
502  int err;
503  static long seed = 0;
504 
505  /* obvious error checking - invalid parameter */
506  if (!session || !session->parser)
507  {
508  er_set (ER_WARNING_SEVERITY, ARG_FILE_LINE, ER_IT_INVALID_SESSION, 0);
509  return er_errid ();
510  }
511  /* no statement was given in the session */
512  if (session->dimension == 0 || !session->statements)
513  {
514  /* if the parser already has something wrong - syntax error */
515  if (pt_has_error (session->parser))
516  {
517  pt_report_to_ersys (session->parser, PT_SYNTAX);
518  return er_errid ();
519  }
520 
521  er_set (ER_WARNING_SEVERITY, ARG_FILE_LINE, ER_IT_EMPTY_STATEMENT, 0);
522  return er_errid ();
523  }
524  /* check the end of statements */
525  if (session->stmt_ndx == session->dimension)
526  {
527  /* return 0 if all statement were compiled */
528  return 0;
529  }
530 
531  /* allocate memory for session->type_list and session->stage if not allocated */
532  if (session->type_list == NULL)
533  {
534  size_t size = session->dimension * sizeof (DB_QUERY_TYPE *) + session->dimension * sizeof (char);
535  void *p = malloc (size);
536  if (p == NULL)
537  {
538  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, size);
539  return ER_OUT_OF_VIRTUAL_MEMORY;
540  }
541  (void) memset (p, 0, size);
542  session->type_list = (DB_QUERY_TYPE **) p;
543  session->stage = (char *) p + session->dimension * sizeof (DB_QUERY_TYPE *);
544  }
545 
546  /*
547  * Compilation Stage
548  */
549 
550  /* the statements in this session have been parsed without error and it is time to compile the next statement */
551  parser = session->parser;
552  stmt_ndx = session->stmt_ndx++;
553  statement = session->statements[stmt_ndx];
554  statement->use_plan_cache = 0;
555  statement->use_query_cache = 0;
556 
557  statement->is_system_generated_stmt = parser->is_system_generated_stmt;
558 
559  /* check if the statement is already processed */
560  if (session->stage[stmt_ndx] >= StatementPreparedStage)
561  {
562  return stmt_ndx + 1;
563  }
564 
565  /* forget about any previous parsing errors, if any */
566  pt_reset_error (parser);
567 
568  /* get type list describing the output columns titles of the given query */
569  cmd_type = pt_node_to_cmd_type (statement);
570  qtype = NULL;
571  if (cmd_type == CUBRID_STMT_SELECT)
572  {
573  qtype = pt_get_titles (parser, statement);
574  /* to prevent a memory leak, register the query type list to session */
575  session->type_list[stmt_ndx] = qtype;
576  }
577  if (cmd_type == CUBRID_STMT_EXECUTE_PREPARE)
578  {
579  /* we don't actually have the statement which will be executed and we need to get some information about it from
580  * the server */
581  int err = do_get_prepared_statement_info (session, stmt_ndx);
582  if (err != NO_ERROR)
583  {
584  return err;
585  }
586  }
587  /* prefetch and lock classes to avoid deadlock */
588  (void) pt_class_pre_fetch (parser, statement);
589  if (pt_has_error (parser))
590  {
591  pt_report_to_ersys_with_statement (parser, PT_SYNTAX, statement);
592  return er_errid ();
593  }
594 
595  if (seed == 0)
596  {
597  srand48 (seed = (long) time (NULL));
598  }
599 
600  /* do semantic check for the statement */
601  statement_result = pt_compile (parser, statement);
602 
603  if (statement_result == NULL || pt_has_error (parser))
604  {
605  pt_report_to_ersys_with_statement (parser, PT_SEMANTIC, statement);
606  return er_errid ();
607  }
608  statement = statement_result;
609 
610  /* get type list describing the output columns titles of the given query */
611  if (cmd_type == CUBRID_STMT_SELECT)
612  {
613  /* for a select-type query of the form: SELECT * FROM class c we store into type_list the nice column headers:
614  * c.attr1 attr2 c.attr3 before they get fully resolved by mq_translate(). */
615  if (!qtype)
616  {
617  qtype = pt_get_titles (parser, statement);
618  /* to prevent a memory leak, register the query type list to session */
619  session->type_list[stmt_ndx] = qtype;
620  }
621  if (qtype)
622  {
623  /* NOTE, this is here on purpose. If something is busting because it tries to continue corresponding this
624  * type information and the list file columns after having jacked with the list file, by for example adding a
625  * hidden OID column, fix the something else. This needs to give the results as user views the query, ie
626  * related to the original text. It may guess wrong about attribute/column updatability. Thats what they
627  * asked for. */
628  qtype = pt_fillin_type_size (parser, statement, qtype, DB_NO_OIDS, false, false);
629  }
630  }
631 
632  /* translate views or virtual classes into base classes */
633  statement_result = mq_translate (parser, statement);
634  if (!statement_result || pt_has_error (parser))
635  {
636  pt_report_to_ersys_with_statement (parser, PT_SEMANTIC, statement);
637  return er_errid ();
638  }
639  statement = statement_result;
640 
641  /* prefetch and lock translated real classes to avoid deadlock */
642  (void) pt_class_pre_fetch (parser, statement);
643  if (pt_has_error (parser))
644  {
645  pt_report_to_ersys_with_statement (parser, PT_SYNTAX, statement);
646  return er_errid ();
647  }
648 
649  /* validate include_oid setting in the session */
650  if (session->include_oid)
651  {
652  if (mq_updatable (parser, statement) == PT_UPDATABLE)
653  {
654  if (session->include_oid == DB_ROW_OIDS)
655  {
656  (void) pt_add_row_oid (parser, statement);
657  }
658  }
659  else
660  {
661  /* disallow OID column for non-updatable query */
662  session->include_oid = DB_NO_OIDS;
663  }
664  }
665 
666  /* so now, the statement is compiled */
667  session->statements[stmt_ndx] = statement;
668  session->stage[stmt_ndx] = StatementCompiledStage;
669 
670 
671  /*
672  * Preparation Stage
673  */
674 
675  statement->xasl_id = NULL; /* bullet proofing */
676 
677  /* New interface of do_prepare_statement()/do_execute_statment() is used only when the XASL cache is enabled. If it
678  * is disabled, old interface of do_statement() will be used instead. do_statement() makes a XASL everytime rather
679  * than using XASL cache. Also, it can be executed in the server without touching the XASL cache by calling
680  * prepare_and_execute_query(). */
681  if (prm_get_integer_value (PRM_ID_XASL_CACHE_MAX_ENTRIES) > 0 && statement->cannot_prepare == 0)
682  {
683 
684  /* now, prepare the statement by calling do_prepare_statement() */
685  err = do_prepare_statement (parser, statement);
686 #if 0
687  if (err == ER_QPROC_INVALID_XASLNODE)
688  {
689  /* There is a kind of problem in the XASL cache. It is possible when the cache entry was deleted by the
690  * other. In this case, retry to prepare once more (generate and stored the XASL again). */
691  statement->xasl_id = NULL;
692  er_clear ();
693  /* execute the statement by calling do_statement() */
694  err = do_prepare_statement (parser, statement);
695  }
696 #endif
697  if (err < 0)
698  {
699  if (pt_has_error (parser))
700  {
701  pt_report_to_ersys_with_statement (parser, PT_SEMANTIC, statement);
702  return er_errid ();
703  }
704  return err;
705  }
706  }
707 
708  /* so now, the statement is prepared */
709  session->stage[stmt_ndx] = StatementPreparedStage;
710 
711  return stmt_ndx + 1;
712 }

714 /*
715  * db_compile_statement() - This function compiles the next statement in the
716  * session. The first compilation reports any syntax errors that occurred
717  * in the entire session.
718  * return: an integer that is the relative statement ID of the next statement

1948  }

1952  pt_null_etc (statement);
1953  }
1954  else

1958  }
1959  break;
1960

1963  } /* switch (pt_node_to_cmd_type()) */
1964 
1965  } /* else */
1966

1968  } /* if (result) */
1969 
1970  /* Do not override original error id with */
1971  /* last error checking */
1972  if (er_errid () == NO_ERROR && pt_has_error (parser) && err != ER_QPROC_INVALID_XASLNODE)
1973  {
1974  pt_report_to_ersys_with_statement (parser, PT_EXECUTION, statement);
1975  err = er_errid ();

1987  }
1988

1990 
1991  /* free if the statement was duplicated for host variable binding */
1992  if (statement != session->statements[stmt_ndx])
1993  {
1994  parser_free_tree (parser, statement);
1995  }
1996
