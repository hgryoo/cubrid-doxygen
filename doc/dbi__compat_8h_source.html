253  } while (0)

264  do { \
265  if ((CACHE_TIME *) (T) != NULL) { \
266  PTR = or_pack_int(PTR, (T)->sec); \
267  PTR = or_pack_int(PTR, (T)->usec); \
268  } \
269  else { \
270  PTR = or_pack_int(PTR, 0); \
271  PTR = or_pack_int(PTR, 0); \
272  } \
273  } while (0)

276  do { \

279  PTR = or_unpack_int(PTR, &((T)->usec)); \
