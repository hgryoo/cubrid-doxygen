79 static int start_ddl_proxy_client (const char *program_name, DDL_CLIENT_ARGUMENT *args)
80 {
81  DB_SESSION *session = NULL;
82  int rc = NO_ERROR;
83 
84  rc = db_restart_ex (program_name, args->db_name, args->user_name, args->passwd, NULL, DB_CLIENT_TYPE_ADMIN_CSQL);
85 
86  if (rc != NO_ERROR)
87  {
88  ASSERT_ERROR ();
89  return rc;
90  }
91 
92  if (args->command != NULL)
93  {
94  int total_stmts, stmt_id, i, num_of_rows;
95  DB_QUERY_RESULT *result = NULL;
96 
97  session = db_open_buffer ((const char *) args->command);
98  if (session == NULL)
99  {
100  ASSERT_ERROR_AND_SET (rc);
101  goto error;
102  }
103 
104  if (db_get_errors (session) || er_errid () != NO_ERROR)
105  {
106  ASSERT_ERROR_AND_SET (rc);
107  goto error;
108  }
109 
110  total_stmts = db_statement_count (session);
111  for (i = 0; i < total_stmts; i++)
112  {
113  stmt_id = db_compile_statement (session);
114  if (stmt_id < 0)
115  {
116  ASSERT_ERROR_AND_SET (rc);
117  db_abort_transaction ();
118  goto error;
119  }
120 
121  if (stmt_id == 0)
122  {
123  /* this means that we processed all statements */
124  break;
125  }
126  num_of_rows = db_execute_statement (session, stmt_id, &result);
127  if (num_of_rows < 0)
128  {
129  ASSERT_ERROR_AND_SET (rc);
130  db_abort_transaction ();
131  goto error;
132  }
133 
134  if (result != NULL)
135  {
136  db_query_end (result);
137  result = NULL;
138  }
139  else
140  {
141  db_free_query (session);
142  }
143 
144  db_drop_statement (session, stmt_id);
145  }
146  }
147 
148 error:
149  if (session != NULL)
150  {
151  db_close_session (session);
152  }
153  db_commit_transaction ();
154  db_shutdown ();
155 
156  return rc;
157 }
