114 /*
115  * meta_create () - create ELO_META structure
116  * return: ELO_META instance if successful, NULL otherwise
117  * s(in): meta string
118  */
119 static ELO_META *
120 meta_create (const char *s)
121 {
122  ELO_META *meta;
123 
124  meta = db_private_alloc (NULL, sizeof (ELO_META));
125  if (meta == NULL)
126  {
127  return NULL;
128  }
129  meta->items = NULL;
130 
131  if (s != NULL)
132  {
133  char *end_p;
134  char *pp;
135  char *key_sp, *key_ep; /* key_ep = position of the last key char + 1 */
136  char *val_sp, *val_ep;
137 
138  pp = (char *) s;
139  end_p = (char *) s + strlen (s);
140  while (pp < end_p)
141  {
142  key_sp = pp;
143  key_ep = strchr (key_sp, '=');
144  if (key_ep == NULL)
145  {
146  goto error;
147  }
148 
149  val_sp = key_ep + 1;
150  val_ep = strchr (val_sp, ',');
151  if (val_ep == NULL)
152  {
153  val_ep = end_p;
154  }
155  /* make a new item. key should have positive length */
156  if (key_ep - key_sp > 0 && val_ep - val_sp >= 0)
157  {
158  ELO_META_ITEM *item = meta_item_create (key_sp, key_ep - key_sp, val_sp,
159  val_ep - val_sp);
160  if (item == NULL)
161  {
162  goto error;
163  }
164 
165  /* add to meta data items */
166  item->next = meta->items;
167  meta->items = item;
168  }
169  else
170  {
171  goto error;
172  }
173 
174  /* advance to the position of the next key */
175  pp = val_ep + 1;
176  }
177  }
178 
179  return meta;
180 
181 error:
182  if (meta != NULL)
183  {
184  meta_destroy_internal (meta);
185  }
186 
187  return NULL;
188 }

251  item = item->next;
252  }

255  return NULL;

258 /*
259  * meta_check_value () - check value for meta data
260  * return: NO_ERROR if check passed, ER_FAILED otherwise
261  * val(in): value
262  * permit_zero_len(in): if non zero value, permits zero length value
263  */
264 static int
265 meta_check_value (const char *val, bool permit_zero_len)
266 {
267  char *p;
268 
269  p = (char *) val;
270  if (p != NULL)
271  {
272  while (*p)
273  {
274  if (*p == ',' || *p == '=')
275  {
276  return ER_FAILED;
277  }
278  p++;
279  }
280  }
281 
282  if (!permit_zero_len)
283  {
284  return (p - val) > 0 ? NO_ERROR : ER_FAILED;
285  }
286  else
287  {
288  return NO_ERROR;
289  }
290 
291  return ER_FAILED;
292 }

294 /*
295  * meta_item_create () - create ELO_META_ITEM with key and value
296  * return: ELO_META_ITEM instance if successful, NULL otherwise
297  * key(in): key
298  * key_len(in): key length
299  * value(in): value
300  * value_len(in): value length
301  */
302 static ELO_META_ITEM *
303 meta_item_create (const char *key, int key_len, const char *value, int value_len)
304 {
305  ELO_META_ITEM *item;
306  char *k, *v;
307 
308  item = db_private_alloc (NULL, sizeof (ELO_META_ITEM));
309  if (item != NULL)
310  {
311  k = db_private_alloc (NULL, key_len + 1);
312  if (k != NULL)
313  {
314  memcpy (k, key, key_len);
315  k[key_len] = '\0';
316 
317  v = db_private_alloc (NULL, value_len + 1);
318  if (v != NULL)
319  {
320  memcpy (v, value, value_len);
321  v[value_len] = '\0';
322 
323  item->key = k;
324  item->value = v;
325 
326  return item;
327  }
328  db_private_free_and_init (NULL, k);
329  }
330  db_private_free_and_init (NULL, item);
331  }
332 
333  return NULL;
334 }

356 /*
357  * meta_set () - set key/value data into ELO_META structure
358  * return: NO_ERROR if successful, ER_FAILED otherwise
359  * meta(in): ELO_META instance
360  * key(in): key
361  * value(in): value
362  */
363 static int
364 meta_set (ELO_META * meta, const char *key, const char *val)
365 {
366  ELO_META_ITEM *item;
367  char *k, *v;
368 
369  assert (meta != NULL);
370  assert (key != NULL);
371 
372  if (meta_check_value (key, false) != NO_ERROR || meta_check_value (val, true) != NO_ERROR)
373  {
374  return ER_FAILED;
375  }
376 
377  item = meta_item_get (meta, key);
378  if (item != NULL)
379  {
380  v = db_private_strdup (NULL, val);
381  if (v == NULL)
382  {
383  return ER_FAILED;
384  }
385 
386  db_private_free_and_init (NULL, item->value);
387  item->value = v;
388 
389  return NO_ERROR;
390  }
391 
392  /* create new item */
393  item = meta_item_create (key, strlen (key), val, val == NULL ? 0 : strlen (val));
394  if (item == NULL)
395  {
396  return ER_FAILED;
397  }
398 
399  /* insert into item */
400  item->next = meta->items;
401  meta->items = item;
402 
403  return NO_ERROR;
404 }

406 /*
407  * meta_to_string () - create string from ELO_META
408  * return: meta data string if successful, NULL otherwise
409  * meta(in): ELO_META instance
410  */
411 static char *
412 meta_to_string (ELO_META * meta)
413 {
414  ELO_META_ITEM *item;
415  size_t sz;
416  char *buf, *buf_p;
417  const char *sep;
418 
419  assert (meta != NULL);
420 
421  if (meta->items == NULL)
422  {
423  /* do not return "" */
424  return NULL;
425  }
426 
427  /* calculate total size needed */
428  sz = 0;
429  item = meta->items;
430  while (item != NULL)
431  {
432  sz += strlen (item->key);
433  sz += strlen (item->value);
434  sz += 2; /* first('=' and '\0'), second~ (',' and '=') */
435  item = item->next;
436  }
437 
438  /* allocate buffer and copy items */
439  buf_p = buf = db_private_alloc (NULL, sz);
440  if (buf == NULL)
441  {
442  return NULL;
443  }
444 
445  item = meta->items;
446  sep = "";
447  while (item != NULL)
448  {
449  int r;
450 
451  r = snprintf (buf_p, sz, "%s%s=%s", sep, item->key, item->value);
452  buf_p = buf_p + r;
453  sz = sz - r;
454 
455  item = item->next;
456  if (sep[0] == 0)
457  {
458  sep = ",";
459  }
460  }
461 
462  assert (sz > 0);
463  *buf_p = '\0';
464 
465  return buf;
466 }

665 /*
666  * elo_copy () - Create new elo instance and copy the file located by elo.
667  * return:
668  * elo(in):
669  * dest(out):
670  */
671 int
672 elo_copy (DB_ELO * elo, DB_ELO * dest)
673 {
674  int ret;
675  ES_URI out_uri;
676  char *locator = NULL;
677  char *meta_data = NULL;
678 
679  assert (elo != NULL);
680  assert (dest != NULL);
681  assert (elo->type == ELO_FBO);
682  assert (elo->locator != NULL);
683 
684  /* create elo instance and copy file */
685  if (elo->meta_data != NULL)
686  {
687  meta_data = db_private_strdup (NULL, elo->meta_data);
688  if (meta_data == NULL)
689  {
690  assert (er_errid () != NO_ERROR);
691  return er_errid ();
692  }
693  }
694 
695  /* if it uses external storage, do transaction work */
696  elo->es_type = es_get_type (elo->locator);
697  if (!ELO_NEEDS_TRANSACTION (elo))
698  {
699  ret = es_copy_file (elo->locator, elo->meta_data, out_uri);
700  if (ret != NO_ERROR)
701  {
702  goto error_return;
703  }
704 
705  locator = db_private_strdup (NULL, out_uri);
706  if (locator == NULL)
707  {
708  es_delete_file (out_uri);
709  goto error_return;
710  }
711  }
712  else
713  {
714  LOB_LOCATOR_STATE state;
715  ES_URI real_locator;
716 
717  state = find_lob_locator (elo->locator, real_locator);
718  switch (state)
719  {
720  case LOB_TRANSIENT_CREATED:
721  case LOB_PERMANENT_DELETED:
722  {
723  ret = es_rename_file (real_locator, elo->meta_data, out_uri);
724  if (ret != NO_ERROR)
725  {
726  goto error_return;
727  }
728  locator = db_private_strdup (NULL, out_uri);
729  if (locator == NULL)
730  {
731  assert (er_errid () != NO_ERROR);
732  ret = er_errid ();
733  goto error_return;
734  }
735  ret = change_state_of_locator (elo->locator, locator, LOB_PERMANENT_CREATED);
736  if (ret != NO_ERROR)
737  {
738  goto error_return;
739  }
740  }
741  break;
742 
743  case LOB_TRANSIENT_DELETED:
744  {
745  locator = db_private_strdup (NULL, elo->locator);
746  if (locator == NULL)
747  {
748  assert (er_errid () != NO_ERROR);
749  ret = er_errid ();
750  goto error_return;
751  }
752  ret = drop_lob_locator (elo->locator);
753  if (ret != NO_ERROR)
754  {
755  goto error_return;
756  }
757  }
758  break;
759 
760  case LOB_PERMANENT_CREATED:
761  case LOB_NOT_FOUND:
762  {
763  ret = es_copy_file (real_locator, elo->meta_data, out_uri);
764  if (ret != NO_ERROR)
765  {
766  goto error_return;
767  }
768  locator = db_private_strdup (NULL, out_uri);
769  if (locator == NULL)
770  {
771  es_delete_file (out_uri);
772  goto error_return;
773  }
774  ret = add_lob_locator (locator, LOB_PERMANENT_CREATED);
775  if (ret != NO_ERROR)
776  {
777  goto error_return;
778  }
779  }
780  break;
781 
782  case LOB_UNKNOWN:
783  assert (er_errid () != NO_ERROR);
784  ret = er_errid ();
785  goto error_return;
786 
787  default:
788  assert (0);
789  return ER_FAILED;
790  }
791  }
792 
793  *dest = *elo;
794  dest->locator = locator;
795  dest->meta_data = meta_data;
796 
797  return NO_ERROR;
798 
799 error_return:
