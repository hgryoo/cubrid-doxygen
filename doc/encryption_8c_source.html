59 /*
60  * crypt_seed - Prepares a set of seeds for the encryption algorithm
61  * return:
62  * key(in): key string
63  *
64  * Note: If no seed key is given, the built in default seed key is used.
65  * This must be called prior to calling any of the other encryption
66  * functions. For decryption, the seed key must match the key used to
67  * encrypt the original string.
68  */
69 void
70 crypt_seed (const char *key)
71 {
72  char default_key[] = { 10, 7, 1, 9, 12, 2, 11, 2, 12, 19, 1, 12, 9, 7, 11, 15 };
73 
74  /* use default key if none supplied */
75  if (key == NULL)
76  {
77  key = default_key;
78  }
79 #if defined(WINDOWS)
80  /* key size must be large than 8 byte */
81  memcpy (des_Keyblob + 12, key, 8);
82 #elif defined (HAVE_RPC_DES_CRYPT_H)
83  memcpy (crypt_Key, key, 8);
84  des_setparity (crypt_Key);
85 #elif defined (HAVE_LIBGCRYPT)
86  char iv[] = { 1, 4, 2, 2, 3, 1, 1, 9 };
87  gcry_error_t err;
88 
89  gcry_check_version (NULL);
90  err = gcry_cipher_open (&cipher_Hd, GCRY_CIPHER_DES, GCRY_CIPHER_MODE_ECB, 0);
91  assert (err == 0);
92  err = gcry_cipher_setkey (cipher_Hd, key, 8); /* 56 bits from 8 bytes */
93  assert (err == 0);
94  err = gcry_cipher_setiv (cipher_Hd, iv, 8); /* 64 bits */
95  assert (err == 0);
96 #endif /* WINDOWS */
97 }

99 /*
100  * crypt_encrypt_printable - combines the basic crypt_line encryption with a
101  * conversion to all capitol letters for printability
102  * return: number of chars in encrypted string
103  * line(in): line to encrypt
104  * crypt(in): encrypted output buffer
105  * maxlen(in): maximum length of buffer
106  */
107 int
108 crypt_encrypt_printable (const char *line, char *crypt, int maxlen)
109 {
110 #if defined(WINDOWS)
111  HCRYPTPROV hProv = 0;
112  HCRYPTKEY hKey = 0;
113  DWORD dwLength;
114  DWORD dwCount;
115  DWORD padding = PKCS5_PADDING;
116  DWORD cipher_mode = CRYPT_MODE_ECB;
117  BYTE pbBuffer[2048];
118 
119  /* Get handle to user default provider. */
120  if (CryptAcquireContext (&hProv, NULL, MS_ENHANCED_PROV, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT))
121  {
122  /* Import PlainText Key */
123  if (!CryptImportKey (hProv, des_Keyblob, sizeof (des_Keyblob), 0, CRYPT_EXPORTABLE, &hKey))
124  {
125  return -1;
126  }
127  else
128  {
129  /* Set DES ECB MODE */
130  if (!CryptSetKeyParam (hKey, KP_MODE, (PBYTE) & cipher_mode, 0))
131  {
132  return -1;
133  }
134 
135  /* Set Padding PKCS#5 */
136  if (!CryptSetKeyParam (hKey, KP_PADDING, (PBYTE) & padding, 0))
137  {
138  return -1;
139  }
140 
141  dwLength = (int) strlen (line);
142  memcpy (pbBuffer, line, dwLength);
143  dwCount = dwLength;
144 
145  if (!CryptEncrypt (hKey, 0, TRUE, 0, pbBuffer, &dwCount, 2048))
146  {
147  return -1;
148  }
149  memcpy (crypt, pbBuffer, dwCount);
150  crypt[dwCount] = 0x00;
151  }
152 
153  CryptReleaseContext (hProv, 0);
154  }
155  else
156  {
157  return -1;
158  }
159  return (dwCount);
160 #elif defined (HAVE_LIBGCRYPT) || defined (HAVE_RPC_DES_CRYPT_H)
161  int outlen, inlen;
162  int i;
163  int padlen;
164  unsigned char inbuf[2048];
165  unsigned char padchar;
166 
167  inlen = strnlen (line, sizeof (inbuf));
168  strncpy ((char *) inbuf, line, inlen);
169 
170  /* Insert PKCS style padding */
171  padchar = 8 - (inlen % 8);
172  padlen = (int) padchar;
173 
174  for (i = 0; i < padlen; i++)
175  {
176  inbuf[inlen + i] = padchar;
177  }
178 
179  inlen += padlen;
180 
181 #if defined (HAVE_RPC_DES_CRYPT_H)
182  if (DES_FAILED (ecb_crypt (crypt_Key, (char *) inbuf, inlen, DES_ENCRYPT)))
183  {
184  return -1;
185  }
186  memcpy (crypt, inbuf, inlen);
187 #elif defined (HAVE_LIBGCRYPT)
188  if (gcry_cipher_encrypt (cipher_Hd, crypt, maxlen, inbuf, inlen))
189  {
190  return -1;
191  }
192  gcry_cipher_close (cipher_Hd);
193 #endif /* HAVE_RPC_DES_CRYPT_H */
194  outlen = inlen;
195  crypt[outlen] = '\0';
196  return (outlen);
197 #endif /* WINDOWS */
198 }

200 /*
201  * crypt_decrypt_printable - decrypts a line that was encrypted with
202  * crypt_encrypt_printable
203  * return: number of chars in decrypted string
204  * crypt(in): buffer to decrypt
205  * decrypt(out): decrypted output buffer
206  * maxlen(in): maximum length of output buffer
207  *
208  */
209 int
210 crypt_decrypt_printable (const char *crypt, char *decrypt, int maxlen)
211 {
212 #if defined (WINDOWS)
213  /* We don't need to decrypt */
214  return -1;
215 #elif defined (HAVE_LIBGCRYPT) || defined (HAVE_RPC_DES_CRYPT_H)
216  int outlen, inlen, padding_size;
217  int i;
218 
219  inlen = strlen (crypt);
220 #if defined (HAVE_RPC_DES_CRYPT_H)
221  memcpy (decrypt, crypt, inlen);
222  if (DES_FAILED (ecb_crypt (crypt_Key, decrypt, inlen, DES_DECRYPT)))
223  {
224  return -1;
225  }
226 #elif defined (HAVE_LIBGCRYPT)
227  if (gcry_cipher_decrypt (cipher_Hd, decrypt, maxlen, crypt, inlen))
228  {
229  return -1;
230  }
231  gcry_cipher_close (cipher_Hd);
232 #endif /* HAVE_RPC_DES_CRYPT_H */
233 
234  outlen = inlen;
235  /* Check PKCS style padding */
236  padding_size = decrypt[outlen - 1];
237  if ((padding_size < 1) || (padding_size > 8))
238  {
239  return -1;
240  }
241 
242  for (i = 1; i < padding_size; i++)
243  {
244  if (decrypt[outlen - 1 - i] != padding_size)
245  {
246  return -1;
247  }
248  }
249 
250  outlen -= padding_size;
251  decrypt[outlen] = '\0';
252 
253  return (outlen);
254 #endif /* WINDOWS */
255 }

257 /*
258  * crypt_encrypt_sha1_printable -
259  * return: number of chars in encrypted string
260  * line(in): line to hash
261  * crypt(in): hashed output buffer
262  * maxlen(in): maximum length of buffer
263  */
264 int
265 crypt_encrypt_sha1_printable (const char *line, char *crypt, int maxlen)
266 {
267  SHA1Context sha;
268  int i;
269 
270  SHA1Reset (&sha);
271  SHA1Input (&sha, (const unsigned char *) line, (int) strlen (line));
272 
273  if (!SHA1Result (&sha))
274  {
275  return -1;
276  }
277  else
278  {
279  /* convert to readable format */
280  for (i = 0; i < 5; i++)
281  {
282  sprintf (crypt + (i * 8), "%08X", sha.Message_Digest[i]);
283  }
284  crypt[40] = '\0';
285  }
286  return 40;
287 }
