60 /*
61  * es_posix_get_unique_name - make unique path string for external file
62  *
63  * return: none
64  * dirname1(out): first level directory name which is generated
65  * dirname2(out): second level directory name which is generated
66  * filename(out): generated file name
67  */
68 static void
69 es_get_unique_name (char *dirname1, char *dirname2, const char *metaname, char *filename)
70 {
71  UINT64 unum;
72  int hashval;
73  int r;
74 
75 #if defined(SERVER_MODE)
76  THREAD_ENTRY *thread_p;
77 
78  thread_p = thread_get_thread_entry_info ();
79  assert (thread_p != NULL);
80 
81  r = rand_r (&thread_p->rand_seed);
82 #else
83  r = rand ();
84 #endif
85 
86  /* get unique numbers */
87  unum = es_get_unique_num ();
88 
89  /* make a file name & a dir name */
90  snprintf (filename, NAME_MAX, "%s.%020llu_%04d", metaname, (unsigned long long) unum, r % 10000);
91 
92  hashval = es_name_hash_func (ES_POSIX_HASH1, filename);
93  snprintf (dirname1, NAME_MAX, "ces_%03d", hashval);
94 
95  hashval = es_name_hash_func (ES_POSIX_HASH2, filename);
96  snprintf (dirname2, NAME_MAX, "ces_%03d", hashval);
97 }

99 /*
100  * es_posix_make_dirs -
101  *
102  * return: error code, ER_ES_GENERAL or NO_ERROR
103  * dirname1(in): first level directory name
104  * dirname2(in): second level directory name
105  */
106 static int
107 es_make_dirs (const char *dirname1, const char *dirname2)
108 {
109  char dirbuf[PATH_MAX];
110  int ret;
111 
112 #if defined (CUBRID_OWFS_POSIX_TWO_DEPTH_DIRECTORY)
113 retry:
114  snprintf (dirbuf, PATH_MAX, "%s%c%s%c%s", es_base_dir, PATH_SEPARATOR, dirname1, PATH_SEPARATOR, dirname2);
115  ret = mkdir (dirbuf, 0755);
116  if (ret < 0 && errno == ENOENT)
117  {
118  snprintf (dirbuf, PATH_MAX, "%s%c%s", es_base_dir, PATH_SEPARATOR, dirname1);
119  ret = mkdir (dirbuf, 0755);
120  if (ret == 0 || errno == EEXIST)
121  {
122  goto retry;
123  }
124  }
125 #else
126  snprintf (dirbuf, PATH_MAX, "%s%c%s", es_base_dir, PATH_SEPARATOR, dirname1);
127  ret = mkdir (dirbuf, 0755);
128 #endif
129  if (ret < 0 && errno != EEXIST)
130  {
131  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", dirbuf);
132  return ER_ES_GENERAL;
133  }
134  return NO_ERROR;

137 /*
138  * es_rename_path -
139  *
140  * return:
141  * src(in): source path to be converted
142  * metaname(in): new file name hint replacing a part of the src
143  * tgt(out): target path
144  */
145 static void
146 es_rename_path (char *src, char *tgt, char *metaname)
147 {
148  char *s, *t;
149 
150  assert (metaname != NULL);
151  /*
152  * src: /.../ces_000/ces_tmp.123456789
153  * ^
154  * s
155  */
156  s = strrchr (src, PATH_SEPARATOR);
157  assert (s != NULL);
158  strcpy (tgt, src);
159  if (s == NULL)
160  {
161  return;
162  }
163  /*
164  * tgt: /.../ces_000/ces_tmp.123456789
165  * ^
166  * t
167  */
168  t = tgt + (s - src) + 1;
169 
170  /*
171  * tgt: /.../ces_000/ces_tmp.123456789
172  * ^
173  * s
174  */
175  s = strchr (s, '.');
176  assert (s != NULL);
177  if (s == NULL)
178  {
179  return;
180  }
181 
182  sprintf (t, "%s%s", metaname, s);
183 }

289 /*
290  * xes_posix_write_file - write to the external file
291  *
292  * return: error code, ER_ES_GENERAL or NO_ERRROR
293  * path(in): file path
294  */
295 ssize_t
296 xes_posix_write_file (const char *path, const void *buf, size_t count, off_t offset)
297 {
298  struct stat pstat;
299  int fd;
300  ssize_t nbytes;
301  size_t total = 0;
302 
303  er_log_debug (ARG_FILE_LINE, "xes_posix_write_file(%s, count %d offset %ld)\n", path, count, offset);
304 
305  /*
306  * TODO: This block of codes prevents partial update or writing at advanced
307  * position or something like that.
308  * This restriction is introduced due to OwFS's capability.
309  * We need to reconsider about this specification.
310  */
311  if (stat (path, &pstat) < 0)
312  {
313  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", path);
314  return ER_ES_GENERAL;
315  }
316  if (offset != pstat.st_size)
317  {
318  char buf[PATH_MAX];
319  snprintf (buf, PATH_MAX, "offset error %s", path);
320  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", buf);
321  return ER_ES_GENERAL;
322  }
323 
324 #if defined (WINDOWS)
325  fd = open (path, O_WRONLY | O_APPEND | O_BINARY, S_IRWXU);
326 #else /* WINDOWs */
327  fd = open (path, O_WRONLY | O_APPEND, S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH | O_LARGEFILE);
328 #endif /* !WINDOWS */
329  if (fd < 0)
330  {
331  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", path);
332  return ER_ES_GENERAL;
333  }
334 
335  while (count > 0)
336  {
337  if (lseek (fd, offset, SEEK_SET) != offset)
338  {
339  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", path);
340  close (fd);
341  return ER_ES_GENERAL;
342  }
343 
344  nbytes = write (fd, buf, (unsigned) count);
345  if (nbytes <= 0)
346  {
347  switch (errno)
348  {
349  case EINTR:
350  case EAGAIN:
351  continue;
352  default:
353  {
354  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", path);
355  close (fd);
356  return ER_ES_GENERAL;
357  }
358  }
359  }
360 
361  offset += nbytes;
362  count -= nbytes;
363  buf = (char *) buf + nbytes;
364  total += nbytes;
365  }
366  close (fd);
367 
368  return total;
369 }

371 /*
372  * xes_posix_read_file - read from the external file
373  *
374  * return: error code, ER_ES_GENERAL or NO_ERRROR
375  * path(in): file path
376  */
377 ssize_t
378 xes_posix_read_file (const char *path, void *buf, size_t count, off_t offset)
379 {
380  int fd;
381  ssize_t nbytes;
382  size_t total = 0;
383 
384  er_log_debug (ARG_FILE_LINE, "xes_posix_read_file(%s, count %d offset %ld)\n", path, count, offset);
385 
386 #if defined (WINDOWS)
387  fd = open (path, O_RDONLY | O_BINARY);
388 #else /* WINDOWS */
389  fd = open (path, O_RDONLY | O_LARGEFILE);
390 #endif /* !WINDOWS */
391  if (fd < 0)
392  {
393  if (errno == ENOENT)
394  {
395  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_FILE_NOT_FOUND, 1, path);
396  return ER_ES_FILE_NOT_FOUND;
397  }
398  else
399  {
400  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", path);
401  return ER_ES_GENERAL;
402  }
403  }
404 
405  while (count > 0)
406  {
407  if (lseek (fd, offset, SEEK_SET) != offset)
408  {
409  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", path);
410  close (fd);
411  return ER_ES_GENERAL;
412  }
413 
414  nbytes = read (fd, buf, (unsigned) count);
415  if (nbytes < 0)
416  {
417  switch (errno)
418  {
419  case EINTR:
420  case EAGAIN:
421  continue;
422  default:
423  {
424  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", path);
425  close (fd);
426  return ER_ES_GENERAL;
427  }
428  }
429  }
430  if (nbytes == 0)
431  {
432  break;
433  }
434 
435  offset += nbytes;
436  count -= nbytes;
437  buf = (char *) buf + nbytes;
438  total += nbytes;
439  }
440  close (fd);
441 
442  return total;
443 }

468 /*
469  * xes_posix_copy_file - copy the external file to new one
470  *
471  * return: error code, ER_ES_GENERAL or NO_ERRROR
472  * src_path(in): file path to be copied
473  * new_path(out): file path newly created
474  */
475 int
476 xes_posix_copy_file (const char *src_path, char *metaname, char *new_path)
477 {
478 #define ES_POSIX_COPY_BUFSIZE (4096 * 4) /* 16K */
479 
480  int rd_fd, wr_fd;
481  ssize_t ret;
482  char dirname1[NAME_MAX], dirname2[NAME_MAX], filename[NAME_MAX];
483  char buf[ES_POSIX_COPY_BUFSIZE];
484 
485  /* open a source file */
486 #if defined (WINDOWS)
487  rd_fd = open (src_path, O_RDONLY | O_BINARY);
488 #else /* WINDOWS */
489  rd_fd = open (src_path, O_RDONLY | O_LARGEFILE);
490 #endif /* !WINDOWS */
491  if (rd_fd < 0)
492  {
493  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", src_path);
494  return ER_ES_GENERAL;
495  }
496 
497 retry:
498  /* create a target file */
499  es_get_unique_name (dirname1, dirname2, metaname, filename);
500 #if defined (CUBRID_OWFS_POSIX_TWO_DEPTH_DIRECTORY)
501  snprintf (new_path, PATH_MAX, "%s%c%s%c%s%c%s", es_base_dir, PATH_SEPARATOR, dirname1, PATH_SEPARATOR, dirname2,
502  PATH_SEPARATOR, filename);
503 #else
504  /* default */
505  snprintf (new_path, PATH_MAX, "%s%c%s%c%s", es_base_dir, PATH_SEPARATOR, dirname1, PATH_SEPARATOR, filename);
506 #endif
507 
508  er_log_debug (ARG_FILE_LINE, "xes_posix_copy_file(%s, %s): %s\n", src_path, metaname, new_path);
509 
510 #if defined (WINDOWS)
511  wr_fd = open (new_path, O_WRONLY | O_CREAT | O_EXCL | O_BINARY, S_IRWXU);
512 #else /* WINDOWS */
513  wr_fd = open (new_path, O_WRONLY | O_CREAT | O_EXCL, S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH | O_LARGEFILE);
514 #endif /* !WINDOWS */
515  if (wr_fd < 0)
516  {
517  if (errno == ENOENT)
518  {
519  ret = es_make_dirs (dirname1, dirname2);
520  if (ret != NO_ERROR)
521  {
522  close (rd_fd);
523  return ER_ES_GENERAL;
524  }
525 #if defined (WINDOWS)
526  wr_fd = open (new_path, O_WRONLY | O_CREAT | O_EXCL | O_BINARY, S_IRWXU);
527 #else /* WINDOWS */
528  wr_fd = open (new_path, O_WRONLY | O_CREAT | O_EXCL, S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH | O_LARGEFILE);
529 #endif /* !WINDOWS */
530  }
531  }
532 
533  if (wr_fd < 0)
534  {
535  if (errno == EEXIST)
536  {
537  goto retry;
538  }
539  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", new_path);
540  close (rd_fd);
541  return ER_ES_GENERAL;
542  }
543 
544  /* copy data */
545  do
546  {
547  ret = read (rd_fd, buf, ES_POSIX_COPY_BUFSIZE);
548  if (ret == 0)
549  {
550  break; /* end of file */
551  }
552  else if (ret < 0)
553  {
554  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", src_path);
555  break;
556  }
557 
558  ret = write (wr_fd, buf, (unsigned) ret);
559  if (ret <= 0)
560  {
561  er_set_with_oserror (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_ES_GENERAL, 2, "POSIX", new_path);
562  break;
563  }
564  }
565  while (ret > 0);
566 
567  close (rd_fd);
568  close (wr_fd);
569 
570  return (ret < 0) ? ER_ES_GENERAL : NO_ERROR;
571 }
