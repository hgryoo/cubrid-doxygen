45 /*
46  * pp_new_cursor() - Allocate (and enter in pp_cursor_table) a new CURSOR
47  * struct, initializing it with the given name and initialize the static
48  * and dynamic fields accordingly.
49  * return : CURSOR *
50  * name(in): A character string.
51  * static_stmt(in): A prepared CSQL SELECT statement, or NULL.
52  * length(in) :
53  * dynamic_stmt(in): A STMT pointer, or NULL.
54  * host_refs(in):
55  */
56 CURSOR *
57 pp_new_cursor (char *name, char *static_stmt, int length, STMT * dynamic_stmt, HOST_LOD * host_refs)
58 {
59  CURSOR *cursor;
60 
61  cursor = pp_lookup_cursor (name);
62  if (cursor && cursor->level >= pp_nesting_level)
63  {
64  esql_yyverror (pp_get_msg (EX_CURSOR_SET, MSG_REDEFINITION), cursor->name);
65  return NULL;
66  }
67 
68  cursor = (CURSOR *) es_ht_alloc_new_symbol (sizeof (CURSOR));
69 
70  cursor->name = (unsigned char *) strdup (name);
71  cursor->cid = next_cid++;
72  cursor->level = pp_nesting_level;
73  cursor->next = NULL;
74  cursor->host_refs = host_refs;
75  cursor->static_stmt = (unsigned char *) (static_stmt ? strdup (static_stmt) : NULL);
76  cursor->stmtLength = length;
77  cursor->dynamic_stmt = dynamic_stmt;
78 
79  pp_cursor_table->add_symbol (pp_cursor_table, cursor);
80  pp_add_cursor_to_scope (cursor);
81 
82  return cursor;
83 }

85 /*
86  * pp_free_cursor() - Free the given cursor.
87  * return : void
88  * cur(in): The cursor object to be freed.
89  */
90 void
91 pp_free_cursor (CURSOR * cursor)
92 {
93  free_and_init (cursor->name);
94 
95  if (cursor->static_stmt)
96  {
97  free_and_init (cursor->static_stmt);
98  }
99  if (cursor->host_refs)
100  {
101  pp_free_host_lod (cursor->host_refs);
102  }
103  /*
104  * Don't free any dynamic statement associated with this cursor;
105  * those statements have lifetimes that differ from the lifetimes of
106  * cursors.
107  */
108 
109  es_ht_free_symbol (cursor);
110 }

158 /*
159  * pp_print_cursor() - Dump a printed representation of the cursor structure
160  * to the designated stream.
161  * return : nothing
162  * cursor(in): A pointer to a cursor to be printed.
163  * fp(in): The stream to print it on.
164  */
165 static void
166 pp_print_cursor (void *cp, void *fp)
167 {
168  CURSOR *cursor = (CURSOR *) cp;
169  FILE *stream = (FILE *) fp;
170 
171  fprintf (stream, " * %s (cid %d, level %d):\n", cursor->name, cursor->cid, cursor->level);
172 
173  if (cursor->static_stmt)
174  {
175  int i;
176  fprintf (stream, " *\t%s\n", cursor->static_stmt);
177  if (cursor->host_refs)
178  {
179  for (i = 0; i < cursor->host_refs->n_refs; ++i)
180  {
181  fprintf (stream, " *\thv[%d]: ", i);
182  pp_print_host_ref (&cursor->host_refs->refs[i], stream);
183  fputs ("\n", stream);
184  }
185  }
186  }
187  else
188  fprintf (stream, pp_get_msg (EX_CURSOR_SET, MSG_STMT_TITLE), cursor->dynamic_stmt->name);
189 
190  fputs (" *\n", stream);
191 }

253 /*
254  * pp_new_stmt() - Build a new STMT record for an embedded dynamic statement
255  * (e.g., the thing that arises from an esql statement such as
