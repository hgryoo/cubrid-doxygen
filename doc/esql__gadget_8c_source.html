38 /*
39  * db_gadget_create() - Look up the class and build attribute descriptors for
40  * all of the supplied attributes.
41  *
42  * return : gadget pointer
43  * class_name(in): name of class for which to create gadget
44  * attribute_names(in): NULL-terminated array containing names of attributes
45  * the user will supply values for
46  *
47  * note : Using gadgets to insert into views is not supported due to
48  * the special casing and extra checks required and our inability to
49  * support check option.
50  */
51 DB_GADGET *
52 db_gadget_create (const char *class_name, const char *attribute_names[])
53 {
54  DB_OBJECT *class_;
55  DB_GADGET *gadget = NULL;
56  const char **att_name = attribute_names;
57  int listlen = 0, i;
58 
59 
60  if (class_name == NULL || (class_ = db_find_class (class_name)) == NULL)
61  {
62  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_LC_UNKNOWN_CLASSNAME, 1, class_name);
63  goto error;
64  }
65 
66  if (db_is_vclass (class_))
67  {
68  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_GADGET_NO_VCLASSES, 0);
69  goto error;
70  }
71 
72  gadget = (DB_GADGET *) malloc (sizeof (DB_GADGET));
73  if (gadget == NULL)
74  {
75  goto error;
76  }
77  gadget->attrs = NULL;
78  gadget->class_ = class_;
79 
80  while (att_name[listlen])
81  {
82  listlen++;
83  }
84  gadget->num_attrs = listlen;
85 
86  /* If no attribute list was supplied, assume all attributes */
87  if (listlen == 0)
88  {
89  DB_ATTRIBUTE *attribute = db_get_attributes (gadget->class_);
90 
91  if (attribute == NULL)
92  {
93  goto error;
94  }
95  for (; attribute; attribute = db_attribute_next (attribute))
96  {
97  gadget->num_attrs++;
98  }
99  }
100 
101  gadget->attrs = (ATTR_VAL *) malloc ((gadget->num_attrs + 1) * sizeof (ATTR_VAL));
102  if (gadget->num_attrs > 0 && gadget->attrs == NULL)
103  {
104  goto error;
105  }
106 
107  for (i = 0; i < gadget->num_attrs; i++)
108  {
109  gadget->attrs[i].attr_desc = NULL;
110  gadget->attrs[i].value = NULL;
111  }
112 
113  if (listlen == 0)
114  {
115  DB_ATTRIBUTE *attribute = db_get_attributes (gadget->class_);
116 
117  if (attribute == NULL)
118  {
119  goto error;
120  }
121  for (i = 0; attribute; i++, attribute = db_attribute_next (attribute))
122  {
123  if (db_get_attribute_descriptor
124  (class_, db_attribute_name (attribute), false, true, &gadget->attrs[i].attr_desc))
125  {
126  goto error;
127  }
128  }
129  }
130  else
131  {
132  for (i = 0, att_name = attribute_names; i < gadget->num_attrs; i++, att_name++)
133  {
134  if (gadget_attr_index (gadget, *att_name) != NOT_FOUND)
135  {
136  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OBJ_DUPLICATE_ASSIGNMENT, 1, *att_name);
137  goto error;
138  }
139  if (db_get_attribute_descriptor (class_, *att_name, false, true, &gadget->attrs[i].attr_desc))
140  {
141  goto error;
142  }
143  }
144  }
145 
146  return gadget;
147 
148 error:
149  if (gadget)
150  {
151  db_gadget_destroy (gadget);
152  }
153  return NULL;
154 }

156 /*
157  * db_gadget_destroy() - Tear down the the given gadget. The gadget owns any
158  * DB_VALUEs that have been supplied via db_gadget_bind, so it should clear
159  * them before returning.
160  * return : void
161  * gadget(in): gadget pointer to tear down
162  */
163 void
164 db_gadget_destroy (DB_GADGET * gadget)
165 {
166  int i;
167 
168  if (gadget == NULL)
169  {
170  return;
171  }
172 
173  for (i = 0; gadget->attrs && (i < gadget->num_attrs); i++)
174  {
175  if (gadget->attrs[i].attr_desc)
176  {
177  db_free_attribute_descriptor (gadget->attrs[i].attr_desc);
178  }
179  if (gadget->attrs[i].value)
180  {
181  (void) db_value_clear (gadget->attrs[i].value);
182  free_and_init (gadget->attrs[i].value);
183  }
184  }
185 
186  if (gadget->attrs)
187  {
188  free_and_init (gadget->attrs);
189  }
190  free_and_init (gadget);
191 
192  return;
193 }

195 /*
196  * db_gadget_bind() - Associate the given value with the given attribute.
197  * A NULL DB_VALUE pointer indicates that any previous binding should be
198  * erased. The incoming DB_VALUE is coerced to the proper domain before
199  * being stored so as to avoid coercion during repeated inserts.
200  * return : error code
201  * gadget(in): gadget containing insert data
202  * attribute_name(in): number of db_values in dbvals array
203  * dbval(in): array containing values for unbound attributes
204  */
205 int
206 db_gadget_bind (DB_GADGET * gadget, const char *attribute_name, DB_VALUE * dbval)
207 {
208  int i = 0;
209 
210  if (gadget == NULL || gadget->class_ == NULL)
211  {
212  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_GADGET_INVALID, 0);
213  return ER_GADGET_INVALID;
214  }
215 
216  if (attribute_name == NULL || (i = gadget_attr_index (gadget, attribute_name)) == NOT_FOUND)
217  {
218  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_SM_UNKNOWN_ATTRIBUTE, 2, attribute_name,
219  db_get_class_name (gadget->class_));
220  return ER_SM_UNKNOWN_ATTRIBUTE;
221  }
222 
223  /* Free any previous value before rebinding */
224  if (gadget->attrs[i].value != NULL)
225  {
226  db_value_clear (gadget->attrs[i].value);
227  free_and_init (gadget->attrs[i].value);
228  }
229 
230  /* A null dbval indicates previous binding should be erased */
231  if (dbval == NULL)
232  {
233  return NO_ERROR;
234  }
235 
236  gadget->attrs[i].value = (DB_VALUE *) malloc (sizeof (DB_VALUE));
237  if (gadget->attrs[i].value == NULL)
238  {
239  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (DB_VALUE));
240  return ER_OUT_OF_VIRTUAL_MEMORY;
241  }
242 
243  return tp_value_coerce (dbval, gadget->attrs[i].value, db_attdesc_domain (gadget->attrs[i].attr_desc));
244 }

246 /*
247  * db_gadget_exec() - Creates a new instance using templates and the
248  * descriptors acquired when the DB_GADGET was initialized.
249  *
250  * return : pointer to inserted instance object
251  * gadget(in): gadget containing insert data
252  * num_dbvals(in): number of db_values in dbvals array
253  * dbvals(in): array containing values for unbound attributes
254  */
255 DB_OBJECT *
256 db_gadget_exec (DB_GADGET * gadget, int num_dbvals, DB_VALUE dbvals[])
257 {
258  int dbvals_index = 0;
259  DB_VALUE *value = NULL;
260  DB_OTMPL *otemplate = NULL;
261  DB_OBJECT *obj = NULL;
262  int num_vals = num_dbvals;
263  int i;
264 
265  if (gadget == NULL || gadget->class_ == NULL)
266  {
267  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_GADGET_INVALID, 0);
268  goto error;
269  }
270 
271  otemplate = dbt_create_object_internal (gadget->class_);
272  if (otemplate == NULL)
273  {
274  goto error;
275  }
276 
277  for (i = 0; i < gadget->num_attrs; i++)
278  {
279  value = gadget->attrs[i].value;
280  if (value == NULL)
281  {
282  if (dbvals_index < num_dbvals)
283  {
284  value = &dbvals[dbvals_index++];
285  }
286  }
287 
288  if (value == NULL)
289  {
290  for (i = 0; i < gadget->num_attrs; i++)
291  {
292  if (gadget->attrs[i].value)
293  {
294  num_vals++;
295  }
296  }
297  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_GADGET_ATTRS_VALS_NE, 2, gadget->num_attrs, num_vals);
298  goto error;
299  }
300 
301  if (dbt_dput_internal (otemplate, gadget->attrs[i].attr_desc, value) != NO_ERROR)
302  {
303  goto error;
304  }
305  }
306 
307  obj = dbt_finish_object (otemplate);
308  if (obj == NULL)
309  {
310  goto error;
311  }
312 
313  return obj;
314 
315 error:
316  if (otemplate)
317  {
318  dbt_abort_object (otemplate);
319  }
320  return NULL;
321 }

323 /*
324  * gadget_attr_index() - Perform case-insensitive search for given attribute
325  * and return its position within the gadget if found.
326  * return : attribute's index
327  * gadget(in): gadget to search
328  * attr_names(in): name of attributes to find index of
329  */
330 static int
331 gadget_attr_index (DB_GADGET * gadget, const char *attr_name)
332 {
333  int i;
334 
335  for (i = 0; i < gadget->num_attrs; i++)
336  {
337  if (gadget->attrs[i].attr_desc
338  && pt_str_compare (attr_name, gadget->attrs[i].attr_desc->name, CASE_INSENSITIVE) == 0)
339  {
340  return i;
341  }
342  }
343 
344  return NOT_FOUND;
345 }
