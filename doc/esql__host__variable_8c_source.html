161 /*
162  * pp_clear_host_refs() - Clears all HOST_LOD structures that have been
163  * allocated since the last call to pp_clear_host_refs(), placing them
164  * on the host_lod_free_list.
165  * return : void
166  */
167 void
168 pp_clear_host_refs (void)
169 {
170  if (host_lod_chain != NULL)
171  {
172  HOST_LOD *chain, *next;
173 
174  /*
175  * Run down the chain of allocated HOST_LOD structures, clearing
176  * them as we go but leaving the 'next' fields intact. When we
177  * reach the end, just tack the chain on the the free list, point
178  * the free list to the head of the chain, and be off.
179  */
180  next = host_lod_chain;
181  do
182  {
183  chain = next;
184  pp_clear_host_lod (chain);
185  next = chain->next;
186  }
187  while (next);
188 
189  chain->next = host_lod_free_list;
190  host_lod_free_list = host_lod_chain;
191  host_lod_chain = NULL;
192  }
193 
194  input_refs = NULL;
195  output_refs = NULL;
196  pp_host_refs = NULL;
197  pp_gathering = &input_refs;
198 }

260 /*
261  * pp_add_host_ref() - Add the var/indicator pair to the interface table for
262  * the translator.
263  * return : a pointer to the HOST_REF if everything is ok, NULL otherwise.
264  * var(in): The host variable proper.
265  * indicator(in): An optional indicator variable.
266  * structs_allowed(in): true if structures are permitted as host vars.
267  * n_refs(out): A pointer to an integer to be updated with the number
268  * of references actually added.
269  * note : If structures are allowed, they are expanded. that is, adding
270  * a ref to a structure is equivalent to adding refs to each of its
271  * components in the order in which they were declared.
272  */
273 HOST_REF *
274 pp_add_host_ref (HOST_VAR * var, HOST_VAR * indicator, bool structs_allowed, int *n_refs)
275 {
276  HOST_REF *ref;
277  C_TYPE uci_type;
278 
279  if (n_refs != NULL)
280  {
281  *n_refs = 0;
282  }
283 
284  if (var == NULL)
285  {
286  goto bad_news;
287  }
288 
289  if (indicator != NULL && (!(IS_INT (indicator->type) && !IS_LONG (indicator->type))))
290  {
291  esql_yyverror (pp_get_msg (EX_HOSTVAR_SET, MSG_MUST_BE_SHORT), vs_str (&indicator->expr));
292  goto bad_news;
293  }
294 
295  uci_type = pp_check (var, structs_allowed);
296  if (uci_type == BAD_C_TYPE)
297  {
298  goto bad_news;
299  }
300 
301  if (uci_type == C_TYPE_STRUCT)
302  {
303  if (indicator)
304  {
305  esql_yyverror (pp_get_msg (EX_HOSTVAR_SET, MSG_INDICATOR_NOT_ALLOWED));
306  goto bad_news;
307  }
308 
309  return pp_add_struct_field_refs (var, n_refs);
310  }
311 
312  if (pp_host_refs == NULL)
313  {
314  pp_host_refs = pp_new_host_lod ();
315  if (pp_host_refs == NULL)
316  {
317  goto bad_news;
318  }
319 
320  *pp_gathering = pp_host_refs;
321  }
322 
323  if (pp_host_refs->n_refs >= pp_host_refs->max_refs)
324  {
325  HOST_REF *new_refs = (HOST_REF *) pp_malloc ((pp_host_refs->max_refs + 4) * sizeof (HOST_REF));
326  memset (new_refs, 0, (pp_host_refs->max_refs + 4) * sizeof (HOST_REF));
327  if (pp_host_refs->real_refs != NULL)
328  {
329  memcpy ((char *) new_refs, (char *) pp_host_refs->real_refs,
330  (size_t) sizeof (HOST_REF) * (size_t) pp_host_refs->max_refs);
331  free_and_init (pp_host_refs->real_refs);
332  }
333  pp_host_refs->max_refs += 4;
334  pp_host_refs->real_refs = new_refs;
335  }
336 
337  pp_host_refs->refs = pp_host_refs->real_refs;
338 
339  ref = &pp_host_refs->refs[pp_host_refs->n_refs++];
340  pp_host_refs->n_real_refs++;
341 
342  ref->var = var;
343  ref->ind = indicator;
344  ref->uci_type = uci_type;
345  ref->precision_buf = NULL;
346  ref->input_size_buf = NULL;
347  ref->output_size_buf = NULL;
348  ref->expr_buf = NULL;
349  ref->addr_expr_buf = NULL;
350  ref->ind_expr_buf = NULL;
351  ref->ind_addr_expr_buf = NULL;
352 
353  if (n_refs != NULL)
354  {
355  (*n_refs)++;
356  }
357 
358  return ref;
359 
360 bad_news:
361  if (var != NULL)
362  {
363  pp_free_host_var (var);
364  }
365  if (indicator != NULL)
366  {
367  pp_free_host_var (indicator);
368  }
369  return NULL;
370 }

372 /*
373  * pp_add_struct_field_refs() - Add a host ref for each of the fields in
374  * the structure definition, using prefix as the root.
375  * Set *n_refs to 0 or the number of fields added.
376  * return : NULL if any of the fields causes a problem, a pointer to the last
377  * field otherwise
378  * var: A pointer to a HOST_VAR that is known to represent a struct.
379  * n_refs(out) : A pointer to an integer to be updated with the number of
380  * references actually added.
381  */
382 static HOST_REF *
383 pp_add_struct_field_refs (HOST_VAR * var, int *n_refs)
384 {
385  const char *prefix = vs_str (&var->expr);
386  SYMBOL *field = var->type->decl.s.val.v_struct->fields;
387  SYMBOL *next = field;
388  int errors = 0;
389  HOST_REF *result = NULL;
390 
391  if (field == NULL)
392  {
393  esql_yyverror (pp_get_msg (EX_HOSTVAR_SET, MSG_INCOMPLETE_DEF), prefix);
394  pp_free_host_var (var);
395  return NULL;
396  }
397 
398  for (; field; field = next)
399  {
400  HOST_VAR *var;
401 
402  /* Make a new host ref from this field, and then add it as a host variable. */
403  var = pp_new_host_var (NULL, field);
404  vs_prepend (&var->expr, ").");
405  vs_prepend (&var->expr, prefix);
406  vs_prepend (&var->expr, "(");
407  result = pp_add_host_ref (var, NULL, false, NULL);
408  if (result == NULL)
409  {
410  errors++;
411  }
412  if (n_refs != NULL)
413  {
414  (*n_refs)++;
415  }
416 
417  next = field->next;
418  }
419 
420  if (errors > 0)
421  {
422  if (n_refs)
423  {
424  *n_refs = 0;
425  }
426  result = NULL;
427  }
428 
429  /*
430  * We want to free the incoming host var regardless of whether there
431  * were errors or not. If there were errors, it is obviously
432  * unneeded. If there were no errors, it is still unneeded because
433  * only (host vars derived from) its components have been used.
434  */
435  pp_free_host_var (var);
436 
437  /*
438  * This choice of return value is kind of arbitrary: the important
439  * thing is to return some non-NULL pointer so that upper levels
440  * won't be deceived into believing that an error occurred.
441  */
442  return result;
443 }

484 /*
485  * pp_detach_host_refs() - Copy the current host refs, and detach it from
486  * the chain of HOST_LODs that will be cleared at the end of compiling
487  * the current statement. This means that the recipient of the pointer
488  * assumes responsibility for freeing the structure.
489  * return : HOST_LOD *
490  *
491  * note : the recipient of the pointer assumes responsibility for freeing
492  * the structure.
493  */
494 HOST_LOD *
495 pp_detach_host_refs (void)
496 {
497  HOST_LOD *refs = pp_copy_host_refs ();
498  HOST_LOD **link = &host_lod_chain;
499 
500  while (*link)
501  {
502  if (*link == refs)
503  {
504  *link = refs->next;
505  refs->next = NULL;
506  return refs;
507  }
508  link = &((*link)->next);
509  }
510 
511  return NULL;
512 }

628 /*
629  * pp_get_precision() - Return a C expression that will (when evaluated) yield
630  * the nominal precision (in type appropriate units) of the variable
631  * described by the reference.
632  * return : size_t
633  * ref(in): A pointer to a host variable reference.
634  */
635 char *
636 pp_get_precision (HOST_REF * ref)
637 {
638  assert (ref != NULL);
639  assert (ref->var != NULL);
640 
641  if (ref->precision_buf == NULL)
642  {
643  ref->precision_buf = vs_new (NULL);
644 
645  switch (ref->uci_type)
646  {
647  case C_TYPE_SHORT:
648  case C_TYPE_INTEGER:
649  case C_TYPE_LONG:
650  case C_TYPE_FLOAT:
651  case C_TYPE_DOUBLE:
652  case C_TYPE_TIME:
653  case C_TYPE_TIMESTAMP:
654  case C_TYPE_DATE:
655  case C_TYPE_MONETARY:
656  case C_TYPE_DB_VALUE:
657  case C_TYPE_BASICSET:
658  case C_TYPE_MULTISET:
659  case C_TYPE_SEQUENCE:
660  case C_TYPE_COLLECTION:
661  case C_TYPE_OBJECTID:
662  /* Precision is meaningless */
663  vs_strcpy (ref->precision_buf, "0");
664  break;
665 
666  case C_TYPE_CHAR_POINTER:
667  /*
668  * In this case, this means that we will determine the precision
669  * at runtime by calling strlen(). We could probably emit the
670  * strlen() code here, but it's less expensive in terms of code
671  * space to interpret it at runtime.
672  */
673  vs_strcpy (ref->precision_buf, "0");
674  break;
675 
676  case C_TYPE_CHAR_ARRAY:
677  vs_sprintf (ref->precision_buf, "sizeof(%s)-1", pp_get_expr (ref));
678  break;
679 
680  case C_TYPE_VARCHAR:
681  vs_sprintf (ref->precision_buf, "sizeof((%s).%s)-1", pp_get_expr (ref), VARCHAR_ARRAY_NAME);
682  break;
683 
684  case C_TYPE_VARBIT:
685  case C_TYPE_BIT:
686  vs_sprintf (ref->precision_buf, "(%s).%s", pp_get_expr (ref), VARCHAR_LENGTH_NAME);
687  break;
688 
689  case C_TYPE_STRING_CONST:
690  vs_sprintf (ref->precision_buf, "sizeof(%s)-1", pp_get_expr (ref));
691  break;
692 
693  default:
694  {
695  esql_yyverror (pp_get_msg (EX_HOSTVAR_SET, MSG_UNKNOWN_HV_TYPE), pp_type_str (ref->var->type));
696  vs_strcpy (ref->precision_buf, "0");
697  }
698  break;
699  }
700  }
701 
702  return vs_str (ref->precision_buf);
703 }

707  * yield the nominal size (in size_t units) of the variable described by

726  case C_TYPE_SHORT:
