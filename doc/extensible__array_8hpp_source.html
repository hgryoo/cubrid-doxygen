32 /* extensible_array -
33  *
34  * A Contiguous Container with static initial size that can be dynamically extended if need be. The purpose was to
35  * provide a tool to store data varying in size and handling memory needs automatically (which was often handled
36  * manually), with the best performance when static size is not exceeded.
37  *
38  *
39  * When to use:
40  *
41  * When data varying in size must be stored or collected preferably with no dynamic memory allocation.
42  *
43  * One example of code we often write for popular operations:
44  *
45  * const int BUF_SIZE = 128;
46  * char buffer[BUF_SIZE]; // static buffer
47  * char *buffer_p = buffer; // for now points to static buffer, but may point to dynamic memory
48  * int offset = 0; // current offset in buffer
49  * int capacity = BUF_SIZE; // current capacity
50  *
51  * memcpy (buffer_p, &header, sizeof (header));
52  * offset += sizeof (header);
53  * ...
54  * if (offset + size_of_data > capacity) // size_of_data can vary from 0 to 16k, but will usually be
55  * // small enough
56  * {
57  * int new_capacity = capacity;
58  * while (new_capacity < offset + size_of_data) // double until we can cover
59  * {
60  * new_capacity = 2 * new_capacity
61  * }
62  *
63  * buffer_p = db_private_realloc (thread_p, buffer_p, new_capacity);
64  * }
65  * memcpy (buffer_p, page_ptr, size_of_data);
66  * ...
67  *
68  * end:
69  * // delete if dynamically allocated
70  * if (buffer_p != buffer)
71  * {
72  * db_private_free (thread_p, buffer_p);
73  * }
74  *
75  *
76  * When not to use:
77  *
78  * 1. When memory requirements can be predicted and are reasonably small - use static array instead.
79  *
80  * 2. When performance is not an issue and dynamic allocation is affordable. Use std::string or std::vector instead.
81  *
82  *
83  * How to use:
84  *
85  * 1. Include extensible_array.hpp.
86  *
87  * 2. declare static variable with a default size:
88  * const int BUF_SIZE = 64;
89  * db_private_allocator<char> char_private_allocator (thread_p);
90  * extensible_array<char, BUF_SIZE, db_private_allocator<char> > buffer(char_private_allocator);
91  *
92  * 2. append the required buffer.
93  * buffer.append (REINTERPRET_CAST (char *, &header), sizeof (header));
94  * buffer.append (page_ptr, size_of_data); // size_of_data can vary from 0 to 16k, but will usually be
95  * // small enough
96  *
97  * 3. resources are freed automatically once variable scope ends.
98  *
99  *
100  * Templates:
101  *
102  * 1. T - buffer base type. in most cases a primitive is expected, but the buffer can accept any class.
103  *
104  * 2. Size - buffer static size. It is recommended to provide a value that covers most cases.
105  *
106  * 3. [Optional] Allocator - which allocator to be used for dynamic allocation. Default is standard allocator.
107  *
108  *
109  * Implementation details:
110  *
111  * A std::array stores the static buffer and a std::vector the dynamic buffer. When static buffer size limit is
112  * reached the vector is allocated. Size is always doubled to cover the new requirements. Vector is resized whenever
113  * necessary.
114  * There are two operations: append and copy (which starts over). They can accept an iterator (which is consumed
115  * from begin to end), or start/end iterators.
116  *
117  *
118  * Notes:
119  *
120  * If possible, provide a maximum buffer size to avoid uncontrolled allocations.
121  *
122  *
123  * TODO:
124  *
125  * Add alignment. With this feature, we may even replace OR_BUF.
126  *
127  * Should we handle out of memory?
128  */
129 template <typename T, size_t Size, typename Allocator = std::allocator<T> >
130 class extensible_array
131 {
132  public:
133  extensible_array (void);
134  extensible_array (Allocator &allocator); // Constructing with allocator is required
135  ~extensible_array ();
136 
137  inline void append (const T *source, size_t length); // append at the end of existing data
138  inline void copy (const T *source, size_t length); // overwrite entire array
139 
140  inline const T *get_array (void) const; // get array pointer
141  inline size_t get_size (void) const; // get current size
142  inline size_t get_memsize (void) const; // get current memory size
143 
144  private:
145  inline void reset (void); // reset array
146 
147  contiguous_memory_buffer<T, Size, Allocator> m_membuf; // memory handler
148  size_t m_size; // current size
149 };

221 /* C-style append & copy. they are declared unsafe and is recommended to avoid them (not always possible because of
222 * legacy code. */
223 template<typename T, size_t Size, typename Allocator>
224 inline void
225 extensible_array<T, Size, Allocator>::append (const T *source, size_t length)
226 {
227  // make sure memory is enough
228  T *buffer_p = m_membuf.resize (m_size + length);
229 
230  // copy data at the end of the array
231  std::memcpy (buffer_p + m_size, source, length * sizeof (T));
232  m_size += length;
233 }
