121 /*
122  * fpcache_initialize () - Initialize filter predicate cache.
123  *
124  * return : Error code.
125  * thread_p (in) : Thread entry.
126  */
127 int
128 fpcache_initialize (THREAD_ENTRY * thread_p)
129 {
130  int error_code = NO_ERROR;
131  HL_HEAPID save_heapid;
132 
133  fpcache_Enabled = false;
134 
135  fpcache_Soft_capacity = prm_get_integer_value (PRM_ID_FILTER_PRED_MAX_CACHE_ENTRIES);
136  if (fpcache_Soft_capacity <= 0)
137  {
138  /* Filter predicate cache disabled. */
139  return NO_ERROR;
140  }
141 
142  /* Initialize free list */
143  error_code =
144  lf_freelist_init (&fpcache_Ht_freelist, 1, fpcache_Soft_capacity, &fpcache_Entry_descriptor, &fpcache_Ts);
145  if (error_code != NO_ERROR)
146  {
147  return error_code;
148  }
149 
150  /* Initialize hash table. */
151  error_code = lf_hash_init (&fpcache_Ht, &fpcache_Ht_freelist, fpcache_Soft_capacity, &fpcache_Entry_descriptor);
152  if (error_code != NO_ERROR)
153  {
154  lf_freelist_destroy (&fpcache_Ht_freelist);
155  return error_code;
156  }
157  fpcache_Entry_counter = 0;
158  fpcache_Clone_counter = 0;
159 
160  fpcache_Clone_stack_size = prm_get_integer_value (PRM_ID_FILTER_PRED_MAX_CACHE_CLONES);
161 
162  /* Cleanup */
163  /* Use global heap to allocate binary heap. */
164  save_heapid = db_change_private_heap (thread_p, 0);
165  fpcache_Cleanup_flag = 0;
166  fpcache_Cleanup_bh =
167  bh_create (thread_p, (int) (FPCACHE_CLEANUP_RATIO * fpcache_Soft_capacity), sizeof (FPCACHE_CLEANUP_CANDIDATE),
168  fpcache_compare_cleanup_candidates, NULL);
169  (void) db_change_private_heap (thread_p, save_heapid);
170  if (fpcache_Cleanup_bh == NULL)
171  {
172  lf_freelist_destroy (&fpcache_Ht_freelist);
173  lf_hash_destroy (&fpcache_Ht);
174  ASSERT_ERROR_AND_SET (error_code);
175  return error_code;
176  }
177 
178  fpcache_Stat_lookup = 0;
179  fpcache_Stat_miss = 0;
180  fpcache_Stat_hit = 0;
181  fpcache_Stat_discard = 0;
182  fpcache_Stat_add = 0;
183  fpcache_Stat_lookup = 0;
184  fpcache_Stat_clone_miss = 0;
185  fpcache_Stat_clone_hit = 0;
186  fpcache_Stat_clone_discard = 0;
187  fpcache_Stat_cleanup = 0;
188  fpcache_Stat_cleanup_entry = 0;
189 
190  fpcache_Enabled = true;
191  return NO_ERROR;
192 }

197  * return : Void.
198  * thread_entry (in) : Thread entry.

201 fpcache_finalize (THREAD_ENTRY * thread_p)
202 {

204 
205  if (!fpcache_Enabled)
206  {
207  return;

209 
210  lf_freelist_destroy (&fpcache_Ht_freelist);

219  }
220  (void) db_change_private_heap (thread_p, save_heapid);
221 
222  fpcache_Enabled = false;
223 }

280 /*
281  * fpcache_entry_uninit () - Retire filter predicate cache entry.
282  *
283  * return : NO_ERROR.
284  * entry (in) : filter predicate cache entry.
285  */
286 static int
287 fpcache_entry_uninit (void *entry)
288 {
289  FPCACHE_ENTRY *fpcache_entry = FPCACHE_PTR_TO_ENTRY (entry);
290  THREAD_ENTRY *thread_p = thread_get_thread_entry_info ();
291  HL_HEAPID old_private_heap;
292  PRED_EXPR_WITH_CONTEXT *pred_expr = NULL;
293  int head;
294 
295  old_private_heap = db_change_private_heap (thread_p, 0);
296 
297  for (head = fpcache_entry->clone_stack_head; head >= 0; head--)
298  {
299  pred_expr = fpcache_entry->clone_stack[head];
300  assert (pred_expr != NULL);
301 
302  qexec_clear_pred_context (thread_p, pred_expr, true);
303  stx_free_additional_buff (thread_p, pred_expr->unpack_info);
304  stx_free_xasl_unpack_info (pred_expr->unpack_info);
305  db_private_free_and_init (thread_p, pred_expr->unpack_info);
306  }
307 
308  (void) db_change_private_heap (thread_p, old_private_heap);
309  fpcache_entry->clone_stack_head = -1;
310 
311  if (fpcache_entry->clone_stack != NULL)
312  {
313  free_and_init (fpcache_entry->clone_stack);
314  }
315 
316  return NO_ERROR;
317 }

333 /*
334  * fpcache_claim () - Claim a filter predicate expression from filter predicate cache. If no expression is available in
335  * cache, a new one is generated.
336  *
337  * return : Error code.
338  * thread_p (in) : Thread entry.
339  * btid (in) : B-tree ID.
340  * or_pred (in) : Filter predicate (string and stream).
341  * filter_pred (out) : Filter predicate expression (with context - unpack buffer).
342  */
343 int
344 fpcache_claim (THREAD_ENTRY * thread_p, BTID * btid, OR_PREDICATE * or_pred, PRED_EXPR_WITH_CONTEXT ** filter_pred)
345 {
346  LF_TRAN_ENTRY *t_entry = thread_get_tran_entry (thread_p, THREAD_TS_FPCACHE);
347  FPCACHE_ENTRY *fpcache_entry = NULL;
348  int error_code = NO_ERROR;
349 
350  assert (filter_pred != NULL && *filter_pred == NULL);
351 
352  if (fpcache_Enabled)
353  {
354  /* Try to find available filter predicate expression in cache. */
355  ATOMIC_INC_64 (&fpcache_Stat_lookup, 1);
356 
357  error_code = lf_hash_find (t_entry, &fpcache_Ht, btid, (void **) &fpcache_entry);
358  if (error_code != NO_ERROR)
359  {
360  ASSERT_ERROR ();
361  return error_code;
362  }
363  if (fpcache_entry == NULL)
364  {
365  /* Entry not found. */
366  ATOMIC_INC_64 (&fpcache_Stat_miss, 1);
367  ATOMIC_INC_64 (&fpcache_Stat_clone_miss, 1);
368  }
369  else
370  {
371  /* Hash-table entry found. Try to claim a filter predicate expression, if there is any available. */
372  ATOMIC_INC_64 (&fpcache_Stat_hit, 1);
373  if (fpcache_entry->clone_stack_head >= 0)
374  {
375  /* Available filter predicate expression. */
376  assert (fpcache_entry->clone_stack_head < fpcache_Clone_stack_size);
377  *filter_pred = fpcache_entry->clone_stack[fpcache_entry->clone_stack_head--];
378  ATOMIC_INC_64 (&fpcache_Stat_clone_hit, 1);
379  ATOMIC_INC_32 (&fpcache_Clone_counter, -1);
380  }
381  else
382  {
383  /* No filter predicate expression is available. */
384  ATOMIC_INC_64 (&fpcache_Stat_clone_miss, 1);
385  }
386  /* Unlock hash-table entry. */
387  pthread_mutex_unlock (&fpcache_entry->mutex);
388  }
389  }
390 
391  if (*filter_pred == NULL)
392  {
393  /* Allocate new filter predicate expression. */
394  /* Use global heap as other threads may also use this filter predicate expression. */
395  HL_HEAPID old_private_heap = db_change_private_heap (thread_p, 0);
396  error_code =
397  stx_map_stream_to_filter_pred (thread_p, filter_pred, or_pred->pred_stream, or_pred->pred_stream_size);
398  if (error_code != NO_ERROR)
399  {
400  ASSERT_ERROR ();
401  }
402  (void) db_change_private_heap (thread_p, old_private_heap);
403  }
404  return NO_ERROR;
405 }

407 /*
408  * fpcache_retire () - Retire filter predicate expression; if the filter predicate hash entry is already at maximum
409  * capacity, the predicate expression must be freed.
410  *
411  * return : Error code.
412  * thread_p (in) : Thread entry.
413  * class_oid (in) : Class OID (of index).
414  * btid (in) : B-tree ID.
415  * filter_pred (in) : Filter predicate expression.
416  */
417 int
418 fpcache_retire (THREAD_ENTRY * thread_p, OID * class_oid, BTID * btid, PRED_EXPR_WITH_CONTEXT * filter_pred)
419 {
420  LF_TRAN_ENTRY *t_entry = thread_get_tran_entry (thread_p, THREAD_TS_FPCACHE);
421  FPCACHE_ENTRY *fpcache_entry = NULL;
422  int error_code = NO_ERROR;
423  int inserted = 0;
424 
425  if (fpcache_Enabled)
426  {
427  /* Try to retire in cache entry. */
428  ATOMIC_INC_64 (&fpcache_Stat_add, 1);
429  error_code = lf_hash_find_or_insert (t_entry, &fpcache_Ht, btid, (void **) &fpcache_entry, &inserted);
430  if (error_code != NO_ERROR)
431  {
432  ASSERT_ERROR ();
433  return error_code;
434  }
435  if (fpcache_entry != NULL)
436  {
437  if (inserted)
438  {
439  /* Newly inserted. We must set class_oid. */
440  COPY_OID (&fpcache_entry->class_oid, class_oid);
441 
442  ATOMIC_INC_32 (&fpcache_Entry_counter, 1);
443  ATOMIC_INC_64 (&fpcache_Stat_add, 1);
444 
445  if (fpcache_Entry_counter >= fpcache_Soft_capacity)
446  {
447  /* Try cleanup. */
448  fpcache_cleanup (thread_p);
449  }
450  }
451  else
452  {
453  /* Entry is older. Safe-guard: class OID must match. */
454  assert (OID_EQ (&fpcache_entry->class_oid, class_oid));
455  }
456  /* save filter_pred for later usage. */
457  if (fpcache_entry->clone_stack_head < fpcache_Clone_stack_size - 1)
458  {
459  /* Can save filter predicate expression. */
460  fpcache_entry->clone_stack[++fpcache_entry->clone_stack_head] = filter_pred;
461  filter_pred = NULL;
462  ATOMIC_INC_64 (&fpcache_Stat_clone_add, 1);
463  ATOMIC_INC_32 (&fpcache_Clone_counter, 1);
464  }
465  else
466  {
467  /* No room for another filter predicate expression. */
468  ATOMIC_INC_64 (&fpcache_Stat_clone_discard, 1);
469  }
470  gettimeofday (&fpcache_entry->time_last_used, NULL);
471  pthread_mutex_unlock (&fpcache_entry->mutex);
472  }
473  else
474  {
475  /* Unexpected. */
476  assert (false);
477  error_code = ER_FAILED;
478  }
479  }
480 
481  if (filter_pred != NULL)
482  {
483  /* Filter predicate expression could not be cached. Free it. */
484  HL_HEAPID old_private_heap = db_change_private_heap (thread_p, 0);
485  stx_free_additional_buff (thread_p, filter_pred->unpack_info);
486  stx_free_xasl_unpack_info (filter_pred->unpack_info);
487  db_private_free_and_init (thread_p, filter_pred->unpack_info);
488  (void) db_change_private_heap (thread_p, old_private_heap);
489  }
490  return error_code;
491 }

493 /*
494  * fpcache_remove_by_class () - Remove all filter predicate cache entries belonging to the given class.
495  *
496  * return : Void.
497  * thread_p (in) : Thread entry.
498  * class_oid (in) : Class OID.
499  */
500 void
501 fpcache_remove_by_class (THREAD_ENTRY * thread_p, OID * class_oid)
502 {
503 #define FPCACHE_DELETE_BTIDS_SIZE 1024
504 
505  LF_HASH_TABLE_ITERATOR iter;
506  LF_TRAN_ENTRY *t_entry = thread_get_tran_entry (thread_p, THREAD_TS_FPCACHE);
507  FPCACHE_ENTRY *fpcache_entry;
508  int success = 0;
509  BTID delete_btids[FPCACHE_DELETE_BTIDS_SIZE];
510  int n_delete_btids = 0;
511  int btid_index = 0;
512  bool finished = false;
513 
514  if (!fpcache_Enabled)
515  {
516  return;
517  }
518 
519  while (!finished)
520  {
521  lf_hash_create_iterator (&iter, t_entry, &fpcache_Ht);
522 
523  while (true)
524  {
525  /* Start by iterating to next hash entry. */
526  fpcache_entry = (FPCACHE_ENTRY *) lf_hash_iterate (&iter);
527 
528  if (fpcache_entry == NULL)
529  {
530  /* Finished hash. */
531  finished = true;
532  break;
533  }
534 
535  if (OID_EQ (&fpcache_entry->class_oid, class_oid))
536  {
537  /* Save entry to be deleted after the iteration.
538  * We cannot delete from hash while iterating. The lock-free transaction used by iterator cannot be used
539  * for delete too (and we have just one transaction for each thread).
540  */
541  delete_btids[n_delete_btids++] = fpcache_entry->btid;
542 
543  if (n_delete_btids == FPCACHE_DELETE_BTIDS_SIZE)
544  {
545  /* Free mutex. */
546  pthread_mutex_unlock (&fpcache_entry->mutex);
547  /* Full buffer. Interrupt iteration, delete entries collected so far and then start over. */
548  lf_tran_end_with_mb (t_entry);
549  break;
550  }
551  }
552  }
553 
554  /* Delete collected btids. */
555  for (btid_index = 0; btid_index < n_delete_btids; btid_index++)
556  {
557  if (lf_hash_delete (t_entry, &fpcache_Ht, &delete_btids[btid_index], &success) != NO_ERROR)
558  {
559  /* Unexpected. */
560  assert (false);
561  }
562  else if (success)
563  {
564  /* Successfully removed. */
565  ATOMIC_INC_32 (&fpcache_Entry_counter, -1);
566  ATOMIC_INC_64 (&fpcache_Stat_discard, 1);
567  }
568  else
569  {
570  /* Unexpected. */
571  assert (false);
572  }
573  }
574  n_delete_btids = 0;
575  }
576 
577 #undef FPCACHE_DELETE_BTIDS_SIZE
578 }

634 /*
635 * fpcache_cleanup () - Cleanup filter predicate cache when soft capacity is exceeded.
636 *
637 * return : Void.
638 * thread_p (in) : Thread entry.
639  */
640 static void
641 fpcache_cleanup (THREAD_ENTRY * thread_p)
642 {
643  LF_HASH_TABLE_ITERATOR iter;
644  LF_TRAN_ENTRY *t_entry = thread_get_tran_entry (thread_p, THREAD_TS_FPCACHE);
645  FPCACHE_ENTRY *fpcache_entry = NULL;
646  FPCACHE_CLEANUP_CANDIDATE candidate;
647  int candidate_index;
648  int success;
649 
650  /* We can allow only one cleanup process at a time. There is no point in duplicating this work. Therefore, anyone
651  * trying to do the cleanup should first try to set fpcache_Cleanup_flag. */
652  if (!ATOMIC_CAS_32 (&fpcache_Cleanup_flag, 0, 1))
653  {
654  /* Somebody else does the cleanup. */
655  return;
656  }
657  if (fpcache_Entry_counter <= fpcache_Soft_capacity)
658  {
659  /* Already cleaned up. */
660  if (!ATOMIC_CAS_32 (&fpcache_Cleanup_flag, 1, 0))
661  {
662  assert_release (false);
663  fpcache_Cleanup_flag = 0;
664  }
665  return;
666  }
667 
668  /* Start cleanup. */
669 
670  /* The cleanup is a two-step process:
671  * 1. Iterate through hash and select candidates for cleanup. The least recently used entries are sorted into a binary
672  * heap.
673  * NOTE: the binary heap does not story references to hash entries; it stores copies from the candidate keys and
674  * last used timer of course to sort the candidates.
675  * 2. Remove collected candidates from hash. Entries must be unfix and no flags must be set.
676  */
677 
678  assert (fpcache_Cleanup_bh->element_count == 0);
679  fpcache_Cleanup_bh->element_count = 0;
680 
681  /* Collect candidates for cleanup. */
682  lf_hash_create_iterator (&iter, t_entry, &fpcache_Ht);
683 
684  while ((fpcache_entry = (FPCACHE_ENTRY *) lf_hash_iterate (&iter)) != NULL)
685  {
686  candidate.btid = fpcache_entry->btid;
687  candidate.time_last_used = fpcache_entry->time_last_used;
688 
689  (void) bh_try_insert (fpcache_Cleanup_bh, &candidate, NULL);
690  }
691 
692  /* Remove candidates from filter predicate cache. */
693  for (candidate_index = 0; candidate_index < fpcache_Cleanup_bh->element_count; candidate_index++)
694  {
695  /* Get candidate at candidate_index. */
696  bh_element_at (fpcache_Cleanup_bh, candidate_index, &candidate);
697 
698  /* Try delete. */
699  if (lf_hash_delete (t_entry, &fpcache_Ht, &candidate.btid, &success) != NO_ERROR)
700  {
701  ASSERT_ERROR ();
702  continue;
703  }
704  if (success)
705  {
706  ATOMIC_INC_64 (&fpcache_Stat_cleanup_entry, 1);
707  ATOMIC_INC_64 (&fpcache_Stat_discard, 1);
708  ATOMIC_INC_32 (&fpcache_Entry_counter, -1);
709  }
710  }
711 
712  /* Reset binary heap. */
713  fpcache_Cleanup_bh->element_count = 0;
714 
715  ATOMIC_INC_64 (&fpcache_Stat_cleanup, 1);
716  if (!ATOMIC_CAS_32 (&fpcache_Cleanup_flag, 1, 0))
717  {
718  assert_release (false);
719  fpcache_Cleanup_flag = 0;
720  }
721 }
722 
723 /*
724  * fpcache_compare_cleanup_candidates () - Compare cleanup candidates by their time_last_used. Oldest candidates are
725  * considered "greater".
726  *
727  * return : BH_CMP_RESULT:
728  * BH_GT if left is older.
729  * BH_LT if right is older.
730  * BH_EQ if left and right are equal.
731  * left (in) : Left FPCACHE cleanup candidate.
732  * right (in) : Right FPCACHE cleanup candidate.
733  * ignore_arg (in) : Ignored.
734  */
735 static BH_CMP_RESULT
736 fpcache_compare_cleanup_candidates (const void *left, const void *right, BH_CMP_ARG ingore_arg)
737 {
738  struct timeval left_timeval = ((FPCACHE_CLEANUP_CANDIDATE *) left)->time_last_used;
739  struct timeval right_timeval = ((FPCACHE_CLEANUP_CANDIDATE *) right)->time_last_used;
740 
741  /* Lesser means placed in binary heap. So return BH_LT for older timeval. */
742  if (left_timeval.tv_sec < right_timeval.tv_sec)
743  {
744  return BH_LT;
745  }
746  else if (left_timeval.tv_sec == right_timeval.tv_sec)
747  {
748  return BH_EQ;
749  }
750  else
751  {
752  return BH_GT;
753  }
754 }

758  *
759  * return : Void.

762 void
763 fpcache_drop_all (THREAD_ENTRY * thread_p)

776

778 }
