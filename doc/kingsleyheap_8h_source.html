65  inline size_t size2Class (const size_t sz)
66  {
67 #if defined(__sparc) && defined(__GNUC__)
68  // Adapted from _Hacker's Delight_, by Henry Warren (p.80)
69  size_t x = sz;
70  x = x | (x >> 1);
71  x = x | (x >> 2);
72  x = x | (x >> 4);
73  x = x | (x >> 8);
74  x = x | (x >> 16);
75  return popc (x) - 3;
76 #else
77  if (sz < 128)
78  {
79  assert (class2Size (cl[sz >> 3]) >= sz);
80  return cl[(sz - 1) >> 3];
81  }
82  else
83  {
84  //
85  // We know that the object is more than 128 bytes long,
86  // so we can avoid iterating 5 times.
87  //
88  int c = 5;
89  size_t sz1 = ((sz - 1) >> 5);
90  while (sz1 > 7)
91  {
92  sz1 >>= 1;
93  c++;
94  }
95  assert (class2Size (c) >= sz);
96  return c;
97  }
