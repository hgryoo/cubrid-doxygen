363 static char *
364 ldr_get_token (char *str, char **out_str, char start, char end)
365 {
366  char *p = NULL;
367 
368  if (out_str)
369  {
370  *out_str = NULL;
371  }
372 
373  if (str == NULL)
374  {
375  return NULL;
376  }
377 
378  if (*str == start)
379  {
380  str++;
381  p = str;
382  }
383  while (*str != end && *str != '\0')
384  {
385  str++;
386  }
387 
388  if (*str != end)
389  {
390  return NULL;
391  }
392 
393  *str = '\0';
394  *out_str = p;
395  return str + 1;
396 }

398 /*
399  * ldr_compare_attribute_with_meta -
400  * return: NO_ERROR if successful, ER_FAILED otherwise
401  */
402 static int
403 ldr_compare_attribute_with_meta (char *table_name, char *meta, DB_ATTRIBUTE * attribute)
404 {
405  int error = NO_ERROR;
406  char *p, *name_str, *type_str, *order_str, *storage_order_str, *shared_str;
407  int type, order, storage_order, shared;
408 
409  if (meta == NULL || attribute == NULL)
410  {
411  return ER_FAILED;
412  }
413 
414  p = meta;
415  p = ldr_get_token (p, &name_str, '[', ']');
416  p = ldr_get_token (p, &type_str, '(', ')');
417  p = ldr_get_token (p, &order_str, '(', ')');
418  p = ldr_get_token (p, &storage_order_str, '(', ')');
419  p = ldr_get_token (p, &shared_str, '(', ')');
420  if (name_str == NULL || type_str == NULL || order_str == NULL || storage_order_str == NULL)
421  {
422  print_log_msg (1, "\nCan not build meta: %s", table_name);
423  error = ER_FAILED;
424  goto compare_end;
425  }
426  if (parse_int (&type, type_str, 10) != NO_ERROR || parse_int (&order, order_str, 10) != NO_ERROR
427  || parse_int (&storage_order, storage_order_str, 10) != NO_ERROR || (shared_str[0] != 'S'
428  && shared_str[0] != 'I'))
429  {
430  print_log_msg (1, "\nCan not build meta: %s", table_name);
431  error = ER_FAILED;
432  goto compare_end;
433  }
434  shared = shared_str[0] == 'S';
435 
436  if (type != db_attribute_type (attribute))
437  {
438  print_log_msg (1, "\nThe type of attribute (%s.%s)" " does not match with the meta information.", table_name,
439  db_attribute_name (attribute));
440  error = ER_FAILED;
441  goto compare_end;
442  }
443  if (order != db_attribute_order (attribute))
444  {
445  print_log_msg (1, "\nThe order of attribute (%s.%s)" " does not match with the meta information.", table_name,
446  db_attribute_name (attribute));
447  error = ER_FAILED;
448  goto compare_end;
449  }
450  if (storage_order != attribute->storage_order)
451  {
452  print_log_msg (1, "\nThe storage order of attribute (%s.%s)" " does not match with the meta information.",
453  table_name, db_attribute_name (attribute));
454  error = ER_FAILED;
455  goto compare_end;
456  }
457  if (shared != db_attribute_is_shared (attribute))
458  {
459  print_log_msg (1, "\nThe shared flag of attribute (%s.%s)" " does not match with the meta information.",
460  table_name, db_attribute_name (attribute));
461  error = ER_FAILED;
462  goto compare_end;
463  }
464  if (strcmp (name_str, db_attribute_name (attribute)) != 0)
465  {
466  print_log_msg (1, "\nThe name of attribute (%s.%s)" " does not match with the meta information.", table_name,
467  db_attribute_name (attribute));
468  error = ER_FAILED;
469  goto compare_end;
470  }
471 
472 compare_end:
473  return error;
474 }

476 /*
477  * ldr_compare_storage_order -
478  * return: NO_ERROR if successful, ER_FAILED otherwise
479  * compare_tables(in): two tables in a string (ex. foo:bar,apple:banana)
480  */
481 static int
482 ldr_compare_storage_order (FILE * schema_file)
483 {
484  char *table_name;
485  char *p, *colon, *comma;
486  DB_ATTRIBUTE *attribute;
487  DB_OBJECT *table_object;
488  char line[LINE_MAX * 100];
489  size_t line_size = LINE_MAX * 100;
490  int error = NO_ERROR;
491 
492  if (schema_file == NULL)
493  {
494  return error;
495  }
496 
497  if (fseek (schema_file, 0, SEEK_SET) == -1)
498  {
499  print_log_msg (1, "\nCan not build meta...");
500  return ER_FAILED;
501  }
502 
503  while (true)
504  {
505  p = fgets (line, (int) line_size - 1, schema_file);
506  if (p == NULL)
507  {
508  break;
509  }
510 
511  if (strncmp (p, "-- !META! ", 10) != 0)
512  {
513  continue;
514  }
515  p += 10;
516  colon = strchr (p, ':');
517  if (colon == NULL)
518  {
519  print_log_msg (1, "\nInvalid meta: %s", line);
520  error = ER_FAILED;
521  goto compare_end;
522  }
523 
524  *colon = '\0';
525  p = ldr_get_token (p, &table_name, '[', ']');
526  table_object = db_find_class (table_name);
527  if (table_object == NULL)
528  {
529  print_log_msg (1, "\nCan not find table: %s", table_name);
530  error = ER_FAILED;
531  goto compare_end;
532  }
533 
534  p = colon + 1;
535  comma = p;
536  attribute = db_get_attributes (table_object);
537  for (; attribute && comma; attribute = db_attribute_next (attribute))
538  {
539  if (db_attribute_class (attribute) != table_object)
540  {
541  continue;
542  }
543 
544  comma = strchr (p, ',');
545  if (comma)
546  {
547  *comma = '\0';
548  }
549  if (ldr_compare_attribute_with_meta (table_name, p, attribute) != NO_ERROR)
550  {
551  print_log_msg (1,
552  "\nThe table %s is not suitable to be"
553  " replicated since it is different from the original.", table_name);
554  error = ER_FAILED;
555  goto compare_end;
556  }
557  p = comma + 1;
558  }
559 
560  if (attribute || comma)
561  {
562  print_log_msg (1, "\nThe number of columns of %s is different" " from meta information.", table_name);
563  error = ER_FAILED;
564  goto compare_end;
565  }
566  }
567 
568 compare_end:
569  return error;
570 }

1145 /*
1146  * ldr_exec_query_from_file - execute queries from file
1147  * return: 0 if successful, non-zero otherwise
1148  * file_name(in): file path
1149  * file(in): FILE *
1150  * start_line(in): start line
1151  * commit_period(in): commit period
1152  */
1153 static int
1154 ldr_exec_query_from_file (const char *file_name, FILE * input_stream, int *start_line, int commit_period)
1155 {
1156  DB_SESSION *session = NULL;
1157  DB_QUERY_RESULT *res = NULL;
1158  int error = NO_ERROR;
1159  int stmt_cnt, stmt_id = 0, stmt_type;
1160  int executed_cnt = 0;
1161  int parser_start_line_no;
1162  int parser_end_line_no = 1;
1163  int check_line_no = true;
1164 
1165  if ((*start_line) > 1)
1166  {
1167  int line_count = *start_line - 1;
1168 
1169  do
1170  {
1171  int c = fgetc (input_stream);
1172  if (c == EOF)
1173  {
1174  print_log_msg (1,
1175  msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_LOADDB, LOADDB_MSG_UNREACHABLE_LINE),
1176  file_name, *start_line);
1177  error = ER_GENERIC_ERROR;
1178  goto end;
1179  }
1180  else if (c == '\n')
1181  {
1182  line_count--;
1183  }
1184  }
1185  while (line_count > 0);
1186  }
1187 
1188  check_line_no = false;
1189  session = db_make_session_for_one_statement_execution (input_stream);
1190  if (session == NULL)
1191  {
1192  print_log_msg (1, "ERROR: %s\n", db_error_string (3));
1193  assert (er_errid () != NO_ERROR);
1194  error = er_errid ();
1195  goto end;
1196  }
1197 
1198  util_arm_signal_handlers (&ldr_exec_query_interrupt_handler, &ldr_exec_query_interrupt_handler);
1199 
1200  while (1)
1201  {
1202  if (interrupt_query)
1203  {
1204  if (er_errid () != ER_INTERRUPTED)
1205  {
1206  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_INTERRUPTED, 0);
1207  }
1208  error = er_errid ();
1209  db_close_session (session);
1210  goto end;
1211  }
1212  parser_start_line_no = parser_end_line_no;
1213 
1214  stmt_cnt = db_parse_one_statement (session);
1215  if (stmt_cnt > 0)
1216  {
1217  db_get_parser_line_col (session, &parser_end_line_no, NULL);
1218  stmt_id = db_compile_statement (session);
1219  }
1220 
1221  if (stmt_cnt <= 0 || stmt_id <= 0)
1222  {
1223  DB_SESSION_ERROR *session_error;
1224  int line, col;
1225  if ((session_error = db_get_errors (session)) != NULL)
1226  {
1227  do
1228  {
1229  session_error = db_get_next_error (session_error, &line, &col);
1230  if (line >= 0)
1231  {
1232  print_log_msg (1, "In %s line %d,\n", file_name, line + (*start_line));
1233  print_log_msg (1, "ERROR: %s \n", db_error_string (3));
1234  assert (er_errid () != NO_ERROR);
1235  error = er_errid ();
1236  }
1237  }
1238  while (session_error);
1239  }
1240  db_close_session (session);
1241  break;
1242  }
1243 
1244  stmt_type = db_get_statement_type (session, stmt_id);
1245 
1246  res = (DB_QUERY_RESULT *) NULL;
1247  error = db_execute_statement (session, stmt_id, &res);
1248 
1249  if (error < 0)
1250  {
1251  print_log_msg (1, "ERROR: %s\n", db_error_string (3));
1252  db_close_session (session);
1253  break;
1254  }
1255  executed_cnt++;
1256  error = db_query_end (res);
1257  if (error < 0)
1258  {
1259  print_log_msg (1, "ERROR: %s\n", db_error_string (3));
1260  db_close_session (session);
1261  break;
1262  }
1263 
1264  if (stmt_type == CUBRID_STMT_COMMIT_WORK || (commit_period && (executed_cnt % commit_period == 0)))
1265  {
1266  db_commit_transaction ();
1267  print_log_msg (Verbose_commit, "%8d statements executed. Commit transaction at line %d\n", executed_cnt,
1268  parser_end_line_no);
1269  *start_line = parser_end_line_no + 1;
1270  }
1271  print_log_msg ((int) Verbose, "Total %8d statements executed.\r", executed_cnt);
1272  fflush (stdout);
1273  }
1274 
1275 end:
1276  if (error < 0)
1277  {
1278  db_abort_transaction ();
1279  }
1280  else
1281  {
1282  *start_line = parser_end_line_no + 1;
1283  print_log_msg (1, "Total %8d statements executed.\n", executed_cnt);
1284  fflush (stdout);
1285  db_commit_transaction ();
1286  }
1287  return error;
1288 }
