61 /*
62  * ERROR REPORTING
63  *
64  * NOTE: There are two levels of error handling here. When a call
65  * is made to a ldr_ function, those functions will return a int
66  * and use er_set to set the global error state. For those errors,
67  * we simply use db_error_string to get the text and display it.
68  *
69  * Other errors are detected by the action routines in this file.
70  * For these, we don't use er_set since it isn't really necessary. Instead
71  * we just have message catalog entries for the error text we wish

116 /*
117  * parse_error - Called when we have some sort of parsing problem
118  * return: void
119  * token_type(in): incoming token type
120  * token(in): token string
121  */
122 static void
123 parse_error (DB_TYPE token_type, const char *token)
124 {
125  display_error_line (0);
126 
127  /*
128  * Note, this is always called after a successful call
129  * to ldr_add_attribute() via get_value(). At this point
130  * the current value counter has been incremented to the
131  * next value so we have to use ldr_prev_att_name() to
132  * get the name of the attribute we're having problems with.
133  */
134  fprintf (stderr, msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_LOADDB, LOADDB_MSG_PARSE_ERROR), token,
135  db_get_type_name (token_type), ldr_prev_att_name (), ldr_class_name ());
136 
137  ldr_invalid_object ();
138 }

141 /*
142  * domain_error - called when we have trouble converting an incoming
143  * token to a destination value.
144  * return: void
145  * token_type(in): incoming token type
146  * Note:
147  * Usually, this kind of error will be caught by the ldr_ module in the
148  * call to ldr_add_value. If not, we display error messages here.
149  */
150 static void
151 domain_error (DB_TYPE token_type)
152 {
153  SM_DOMAIN *domain;
154 
155  display_error_line (0);
156  domain = ldr_att_domain ();
157  if (domain == NULL)
158  fprintf (stderr, msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_LOADDB, LOADDB_MSG_MISSING_DOMAIN),
159  ldr_att_name (), ldr_class_name ());
160  else
161  {
162  if (pr_is_set_type (TP_DOMAIN_TYPE (domain)))
163  {
164  fprintf (stderr, msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_LOADDB, LOADDB_MSG_SET_DOMAIN_ERROR),
165  ldr_att_name (), ldr_class_name (), db_get_type_name (token_type));
166  }
167  else
168  {
169  if (token_type == DB_TYPE_SET)
170  fprintf (stderr, msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_LOADDB, LOADDB_MSG_UNEXPECTED_SET),
171  ldr_att_name (), ldr_class_name (), ldr_att_domain_name ());
172  else
173  fprintf (stderr, msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_LOADDB, LOADDB_MSG_UNEXPECTED_TYPE),
174  ldr_att_name (), ldr_class_name (), ldr_att_domain_name (), db_get_type_name (token_type));
175  }
176  }
177  ldr_invalid_object ();
178 }

565  * act_int - assign an integer value
566  * return: void
567  * token(in): int token value
568  */
569 void
570 act_int (char *token)
571 {
572  DB_VALUE temp, *value;
573  int error;
574  TP_DOMAIN domain;
575  TP_DOMAIN *domain_ptr;
576 
577  value = get_value (DB_TYPE_INTEGER);
578  if (value != NULL)
579  {

583  domain_error (DB_TYPE_INTEGER);
584  return;

586 
587  db_make_string (&temp, token);
588  error = tp_value_cast (&temp, value, domain_ptr, false);
589  if (error)
590  {
591  parse_error (TP_DOMAIN_TYPE (domain_ptr), token);
592  }
593  }
594 }

629 /*
630  * act_monetary - Assign a monetary value.
631  * return: void
632  * token(in): token string
633  */
634 void
635 act_monetary (char *token)
636 {
637  DB_VALUE *value;
638  int args;
639  const unsigned char *p = (const unsigned char *) token;
640  double amt;
641 
642  value = get_value (DB_TYPE_MONETARY);
643  if (value != NULL)
644  {
645  if (DB_VALUE_DOMAIN_TYPE (value) != DB_TYPE_MONETARY)
646  {
647  domain_error (DB_TYPE_MONETARY);
648  }
649  else
650  {
651  switch (lang_id ())
652  {
653  case INTL_LANG_KOREAN:
654  if (p[0] == '\\')
655  {
656  token += 1;
657  }
658  else if (p[0] == 0xa3 && p[1] == 0xdc)
659  {
660  token += 2;
661  }
662  break;
663  default:
664  if (p[0] == '$')
665  {
666  token += 1;
667  }
668  }
669  args = sscanf (token, "%lf", &amt);
670 
671  if (args != 1)
672  parse_error (DB_TYPE_MONETARY, token);
673  else
674  {
675  switch (lang_id ())
676  {
677  case INTL_LANG_KOREAN:
678  db_make_monetary (value, DB_CURRENCY_WON, amt);
679  break;
680  default:
681  db_make_monetary (value, DB_CURRENCY_DOLLAR, amt);
682  break;
683  }
684  }
685  }
686  }
687 }

724 /*
725  * act_time - assign a time value
726  * return: void
727  * token(in): token string
728  * ttype(in): time format type
729  */
730 void
731 act_time (char *token, int ttype)
732 {
733  DB_VALUE *value;
734  int args, hour, minute, second;
735  char half[8];
736 
737  value = get_value (DB_TYPE_TIME);
738  if (value != NULL)
739  {
740  if (DB_VALUE_DOMAIN_TYPE (value) != DB_TYPE_TIME)
741  {
742  domain_error (DB_TYPE_TIME);
743  }
744  else
745  {
746  switch (ttype)
747  {
748  case 1:
749  args = sscanf (token, "%d:%d", &hour, &minute);
750  if (args != 2)
751  {
752  parse_error (DB_TYPE_TIME, token);
753  }
754  else
755  {
756  db_make_time (value, hour, minute, 0);
757  }
758  break;
759  case 2:
760  args = sscanf (token, "%d:%d:%d", &hour, &minute, &second);
761  if (args != 3)
762  {
763  parse_error (DB_TYPE_TIME, token);
764  }
765  else
766  {
767  db_make_time (value, hour, minute, second);
768  }
769  break;
770  case 3:
771  args = sscanf (token, "%d:%d %7s", &hour, &minute, half);
772  if (args != 3)
773  parse_error (DB_TYPE_TIME, token);
774  else
775  {
776  if (strcasecmp (half, "am") == 0)
777  {
778  db_make_time (value, hour, minute, 0);
779  }
780  else
781  {
782  db_make_time (value, hour + 12, minute, 0);
783  }
784  }
785  break;
786  case 4:
787  args = sscanf (token, "%d:%d:%d %7s", &hour, &minute, &second, half);
788  if (args != 4)
789  parse_error (DB_TYPE_TIME, token);
790  else
791  {
792  if (strcasecmp (half, "am") == 0)
793  {
794  db_make_time (value, hour, minute, second);
795  }
796  else
797  {
798  db_make_time (value, hour + 12, minute, second);
799  }
800  }
801  break;
802  default:
803  DB_MAKE_NULL (value);
804  break;
805  }
806  }
807  }
808 }
