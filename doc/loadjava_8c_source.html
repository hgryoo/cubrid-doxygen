86 /*
87  * filecopy() - copy a file
88  * return: 0 if success, otherwise errno
89  * fn_src(in): source filename
90  * fn_dst(in): destination filename
91  */
92 static int
93 filecopy (const char *fn_src, const char *fn_dst)
94 {
95  size_t bytesr;
96  int retval = 0;
97  FILE *fh_src = NULL;
98  FILE *fh_dst = NULL;
99  char buff[COPY_BUFFER_SIZE];
100  char c;
101 
102  fh_src = fopen (fn_src, "rb");
103  if (fh_src == NULL)
104  {
105  fprintf (stderr, "'%s' cannot open.\n", fn_src);
106  return errno;
107  }
108 
109  if (!Force_overwrite)
110  {
111  fh_dst = fopen (fn_dst, "r");
112  if (fh_dst != NULL)
113  {
114  fclose (fh_dst);
115  fprintf (stdout, "'%s' is exist. overwrite? (y/n): ", fn_dst);
116  c = getchar ();
117  if (c != 'Y' && c != 'y')
118  {
119  fclose (fh_src);
120  fprintf (stdout, "loadjava is canceled\n");
121  return 0;
122  }
123  }
124  }
125  fh_dst = fopen (fn_dst, "w+b");
126 
127  if (fh_dst == NULL)
128  {
129  fprintf (stderr, "'%s' cannot open.\n", fn_dst);
130  retval = errno;
131  }
132  else
133  {
134  while (retval == 0)
135  {
136  bytesr = fread (buff, 1, COPY_BUFFER_SIZE, fh_src);
137  if (bytesr < 1 || bytesr > COPY_BUFFER_SIZE)
138  {
139  if (feof (fh_src))
140  {
141  break;
142  }
143  else
144  {
145  retval = ferror (fh_src);
146  }
147  }
148  else
149  {
150  if (fwrite (buff, 1, bytesr, fh_dst) != bytesr)
151  {
152  retval = ferror (fh_dst);
153  break;
154  }
155  }
156  }
157  }
158 
159  if (fh_dst)
160  {
161  fclose (fh_dst);
162  }
163  if (fh_src)
164  {
165  fclose (fh_src);
166  }
167  return retval;
168 }
