355 sl_write_update_sql (DB_OTMPL * inst_tp, DB_VALUE * key)
356 {
357  int result;
358 
359  if (strcmp (sm_ch_name ((MOBJ) (inst_tp->class_)), "db_serial") != 0)
360  {
361  /* ordinary tables */
362  string_buffer update_strbuf;
363 
364  update_strbuf ("UPDATE [%s] SET ", sm_ch_name ((MOBJ) (inst_tp->class_)));
365  sl_print_update_att_set (update_strbuf, inst_tp->assignments, inst_tp->nassigns);
366  update_strbuf (" WHERE ");
367  if (sl_print_pk (update_strbuf, inst_tp->class_, key) != NO_ERROR)
368  {
369  return ER_FAILED;
370  }
371  update_strbuf (";");
372 
373  string_buffer select_strbuf;
374 
375  if (sl_print_select (select_strbuf, inst_tp->class_, key) != NO_ERROR)
376  {
377  return ER_FAILED;
378  }
379 
380  return sl_write_sql (update_strbuf, &select_strbuf);
381  }
382  else
383  {
384  /* db_serial */
385  DB_VALUE *cur_value = sl_find_att_value ("current_val", inst_tp->assignments, inst_tp->nassigns);
386  DB_VALUE *incr_value = sl_find_att_value ("increment_val", inst_tp->assignments, inst_tp->nassigns);
387 
388  if (cur_value == NULL || incr_value == NULL)
389  {
390  return ER_FAILED;
391  }
392 
393  DB_VALUE next_value;
394 
395  result = numeric_db_value_add (cur_value, incr_value, &next_value);
396  if (result != NO_ERROR)
397  {
398  return ER_FAILED;
399  }
400 
401  string_buffer serial_name_strbuf;
402 
403  sl_print_att_value (serial_name_strbuf, "name", inst_tp->assignments, inst_tp->nassigns);
404  char *serial_name = trim_single_quote ((char *) serial_name_strbuf.get_buffer (), serial_name_strbuf.len ());
405 
406  string_buffer alter_strbuf;
407  char str_next_value[NUMERIC_MAX_STRING_SIZE];
408 
409  alter_strbuf ("ALTER SERIAL [%s] START WITH %s;", serial_name,
410  numeric_db_value_print (&next_value, str_next_value));
411 
412  return sl_write_sql (alter_strbuf, NULL);
413  }

500 static int
501 sl_write_sql (string_buffer & query, string_buffer * select)
502 {
503  time_t curr_time;
504  char time_buf[20];
505 
506  assert (query.get_buffer () != NULL);
507 
508  if (log_fp == NULL)
509  {
510  if ((log_fp = sl_log_open ()) == NULL)
511  {
512  return ER_FAILED;
513  }
514  }
515 
516  curr_time = time (NULL);
517  strftime (time_buf, sizeof (time_buf), "%Y-%m-%d %H:%M:%S", localtime (&curr_time));
518 
519  /* -- datetime | sql_id | is_ddl | select length | query length */
520  fprintf (log_fp, "-- %s | %u | %zu | %zu\n", time_buf, ++sl_Info.last_inserted_sql_id,
521  (select == NULL) ? 0 : select->len (), query.len ());
522 
523  /* print select for verifying data consistency */
524  if (select != NULL)
525  {
526  /* -- select_length select * from tbl_name */
527  fprintf (log_fp, "-- ");
528  fwrite (select->get_buffer (), sizeof (char), select->len (), log_fp);
529  fputc ('\n', log_fp);
530  }
531 
532  /* print SQL query */
533  fwrite (query.get_buffer (), sizeof (char), query.len (), log_fp);
534  fputc ('\n', log_fp);
535 
536  fflush (log_fp);
537 
538  sl_write_catalog ();
539 
540  fseek (log_fp, 0, SEEK_END);
541  if (ftell (log_fp) >= SL_LOG_FILE_MAX_SIZE)
542  {
543  log_fp = sl_open_next_file (log_fp);
544  }
545 
546  return NO_ERROR;
547 }
