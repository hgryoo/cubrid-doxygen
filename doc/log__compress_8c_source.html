35 /*
36  * log_zip - compress(zip) log data into LOG_ZIP
37  * return: true on success, false on failure
38  * log_zip(in/out): LOG_ZIP structure allocated by log_zip_alloc
39  * length(in): length of given data
40  * data(in): log data to be compressed
41  */
42 bool
43 log_zip (LOG_ZIP * log_zip, LOG_ZIP_SIZE_T length, const void *data)
44 {
45  lzo_uint zip_len = 0;
46  LOG_ZIP_SIZE_T buf_size;
47  int rc;
48 
49  assert (length > 0 && data != NULL);
50  assert (log_zip != NULL);
51 
52  log_zip->data_length = 0;
53 
54  buf_size = LOG_ZIP_BUF_SIZE (length);
55  if (buf_size > log_zip->buf_size)
56  {
57  if (log_zip->log_data)
58  {
59  free_and_init (log_zip->log_data);
60  }
61 
62  log_zip->log_data = (unsigned char *) malloc (buf_size);
63  if (log_zip->log_data == NULL)
64  {
65  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, (size_t) buf_size);
66  }
67  log_zip->buf_size = buf_size;
68  }
69 
70  if (log_zip->log_data == NULL)
71  {
72  log_zip->data_length = 0;
73  log_zip->buf_size = 0;
74  return false;
75  }
76 
77  /* save original data length */
78  memcpy (log_zip->log_data, &length, sizeof (LOG_ZIP_SIZE_T));
79 
80  rc =
81  lzo1x_1_compress ((lzo_bytep) data, (lzo_uint) length, log_zip->log_data + sizeof (LOG_ZIP_SIZE_T), &zip_len,
82  log_zip->wrkmem);
83  if (rc == LZO_E_OK)
84  {
85  log_zip->data_length = (LOG_ZIP_SIZE_T) zip_len + sizeof (LOG_ZIP_SIZE_T);
86  /* if the compressed data length >= orginal length, then it means that compression failed */
87  if (log_zip->data_length < length)
88  {
89  return true;
90  }
91  }
92  return false;
93 }

95 /*
96  * log_unzip - decompress(unzip) log data into LOG_ZIP
97  * return: true on success, false on failure
98  * log_unzip(out): LOG_ZIP structure allocated by log_zip_alloc
99  * length(in): length of given data
100  * data(out): compressed log data
101  */
102 bool
103 log_unzip (LOG_ZIP * log_unzip, LOG_ZIP_SIZE_T length, void *data)
104 {
105  lzo_uint unzip_len;
106  LOG_ZIP_SIZE_T org_len;
107  LOG_ZIP_SIZE_T buf_size;
108  int rc;
109 
110  assert (length > 0 && data != NULL);
111  assert (log_unzip != NULL);
112 
113  /* get original legnth from the compressed data */
114  memcpy (&org_len, data, sizeof (LOG_ZIP_SIZE_T));
115 
116  if (org_len <= 0)
117  return false;
118  unzip_len = (lzo_uint) org_len;
119  buf_size = LOG_ZIP_BUF_SIZE (org_len);
120  length -= sizeof (LOG_ZIP_SIZE_T);
121 
122  if (buf_size > log_unzip->buf_size)
123  {
124  if (log_unzip->log_data)
125  {
126  free_and_init (log_unzip->log_data);
127  }
128 
129  log_unzip->log_data = (unsigned char *) malloc (buf_size);
130  if (log_unzip->log_data == NULL)
131  {
132  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, (size_t) buf_size);
133  }
134  log_unzip->buf_size = buf_size;
135  }
136 
137  if (log_unzip->log_data == NULL)
138  {
139  log_unzip->data_length = 0;
140  log_unzip->buf_size = 0;
141  return false;
142  }
143 
144  rc =
145  lzo1x_decompress_safe ((lzo_bytep) data + sizeof (LOG_ZIP_SIZE_T), (lzo_uint) length, log_unzip->log_data,
146  &unzip_len, NULL);
147 
148  if (rc == LZO_E_OK)
149  {
150  log_unzip->data_length = (LOG_ZIP_SIZE_T) unzip_len;
151  /* if the uncompressed data length != original length, then it means that uncompression failed */
152  if (unzip_len == (lzo_uint) org_len)
153  {
154  return true;
155  }
156  }
157  return false;
158 }

160 /*
161  * log_diff - make log diff - redo data XORed with undo data
162  * return: true
163  * undo_length(in): length of undo data
164  * undo_data(in): undo log data
165  * redo_length(in): length of redo data
166  * redo_data(in/out) redo log data; set as side effect
167  */
168 bool
169 log_diff (LOG_ZIP_SIZE_T undo_length, const void *undo_data, LOG_ZIP_SIZE_T redo_length, void *redo_data)
170 {
171  LOG_ZIP_SIZE_T i, size;
172  unsigned char *p, *q;
173 
174  assert (undo_length > 0 && undo_data != NULL);
175  assert (redo_length > 0 && redo_data != NULL);
176 
177  size = MIN (undo_length, redo_length);
178 
179  /* redo = redo xor undo */
180  p = (unsigned char *) redo_data;
181  q = (unsigned char *) undo_data;
182  for (i = 0; i < size; i++)
183  {
184  *(p++) ^= *(q++);
185  }
186 
187  return true;
188 }

209 
210  return NULL;
211  }
212  log_zip->data_length = 0;
213 
214  log_zip->log_data = (lzo_bytep) malloc ((size_t) buf_size);
215  if (log_zip->log_data == NULL)
216  {
217  free_and_init (log_zip);
218  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, (size_t) buf_size);
219  return NULL;
220  }
221  log_zip->buf_size = buf_size;
222 
223  if (is_zip)
224  {
225  /* lzo method is best speed : LZO1X_1_MEM_COMPRESS */
226  log_zip->wrkmem = (lzo_bytep) malloc (LZO1X_1_MEM_COMPRESS);
227  if (log_zip->wrkmem == NULL)
228  {
229  free_and_init (log_zip->log_data);
230  free_and_init (log_zip);
231  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, (size_t) LZO1X_1_MEM_COMPRESS);
232  return NULL;
233  }
234  }
235  else
236  {
237  log_zip->wrkmem = NULL;
238  }
239 
240  return log_zip;
