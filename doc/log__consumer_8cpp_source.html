134  void execute (cubthread::entry &thread_ref) override
135  {
136  stream_entry *se = NULL;
137  using tasks_map = std::unordered_map <MVCCID, applier_worker_task *>;
138  tasks_map repl_tasks;
139  tasks_map nonexecutable_repl_tasks;
140 
141  while (true)
142  {
143  bool should_stop = false;
144  m_lc.pop_entry (se, should_stop);
145 
146  if (should_stop)
147  {
148  break;
149  }
150 
151  if (se->is_group_commit ())
152  {
153  assert (se->get_data_packed_size () == 0);
154 
155  /* wait for all started tasks to finish */
156  m_lc.wait_for_tasks ();
157 
158  for (tasks_map::iterator it = repl_tasks.begin ();
159  it != repl_tasks.end ();
160  it++)
161  {
162  /* check last stream entry of task */
163  applier_worker_task *my_repl_applier_worker_task = it->second;
164  if (my_repl_applier_worker_task->has_commit ())
165  {
166  m_lc.execute_task (it->second);
167  }
168  else if (my_repl_applier_worker_task->has_abort ())
169  {
170  /* just drop task */
171  }
172  else
173  {
174  assert (it->second->get_entries_cnt () > 0);
175  nonexecutable_repl_tasks.insert (std::make_pair (it->first, it->second));
176  }
177  }
178  repl_tasks.clear ();
179  if (nonexecutable_repl_tasks.size () > 0)
180  {
181  repl_tasks = nonexecutable_repl_tasks;
182  nonexecutable_repl_tasks.clear ();
183  }
184 
185  /* deleted the group commit stream entry */
186  delete se;
187  }
188  else
189  {
190  MVCCID mvccid = se->get_mvccid ();
191  auto it = repl_tasks.find (mvccid);
192 
193  if (it != repl_tasks.end ())
194  {
195  /* already a task with same MVCCID, add it to existing task */
196  applier_worker_task *my_repl_applier_worker_task = it->second;
197  my_repl_applier_worker_task->add_repl_stream_entry (se);
198 
199  assert (my_repl_applier_worker_task->get_entries_cnt () > 0);
200  }
201  else
202  {
203  applier_worker_task *my_repl_applier_worker_task = new applier_worker_task (se, m_lc);
204  repl_tasks.insert (std::make_pair (mvccid, my_repl_applier_worker_task));
205 
206  assert (my_repl_applier_worker_task->get_entries_cnt () > 0);
207  }
208 
209  /* stream entry is deleted by applier task thread */
210  }
211  }
212 
213  }
214 
215  private:
216  log_consumer &m_lc;
217  };
