129 /*
130  * logwr_to_physical_pageid -
131  *
132  * return:
133  * logical_pageid(in):
134  * Note:
135  */
136 LOG_PHY_PAGEID
137 logwr_to_physical_pageid (LOG_PAGEID logical_pageid)
138 {
139  LOG_PHY_PAGEID phy_pageid;
140 
141  if (logical_pageid == LOGPB_HEADER_PAGE_ID)
142  {
143  phy_pageid = 0;
144  }
145  else
146  {
147  LOG_PAGEID tmp_pageid;
148 
149  tmp_pageid = logical_pageid - logwr_Gl.hdr.fpageid;
150 
151  if (tmp_pageid >= logwr_Gl.hdr.npages)
152  {
153  tmp_pageid %= logwr_Gl.hdr.npages;
154  }
155  else if (tmp_pageid < 0)
156  {
157  tmp_pageid = (logwr_Gl.hdr.npages - ((-tmp_pageid) % logwr_Gl.hdr.npages));
158  }
159  tmp_pageid++;
160  if (tmp_pageid > logwr_Gl.hdr.npages)
161  {
162  tmp_pageid %= logwr_Gl.hdr.npages;
163  }
164 
165  assert (tmp_pageid <= PAGEID_MAX);
166  phy_pageid = (LOG_PHY_PAGEID) tmp_pageid;
167  }
168 
169  return phy_pageid;
170 }

172 /*
173  * logwr_fetch_header_page -
174  *
175  * return:
176  * log_pgptr(out):
177  * vol_fd(in):
178  * Note:
179  */
180 static int
181 logwr_fetch_header_page (LOG_PAGE * log_pgptr, int vol_fd)
182 {
183  LOG_PAGEID pageid;
184  LOG_PHY_PAGEID phy_pageid;
185 
186  assert (log_pgptr != NULL);
187 
188  /*
189  * Page is contained in the active log.
190  * Find the corresponding physical page and read the page form disk.
191  */
192  pageid = LOGPB_HEADER_PAGE_ID;
193  phy_pageid = logwr_to_physical_pageid (pageid);
194 
195  if (fileio_read (NULL, vol_fd, log_pgptr, phy_pageid, LOG_PAGESIZE) == NULL)
196  {
197  er_set (ER_FATAL_ERROR_SEVERITY, ARG_FILE_LINE, ER_LOG_READ, 3, pageid, phy_pageid, logwr_Gl.active_name);
198  return ER_LOG_READ;
199  }
200  else
201  {
202  if (log_pgptr->hdr.logical_pageid != pageid)
203  {
204  er_set (ER_FATAL_ERROR_SEVERITY, ARG_FILE_LINE, ER_LOG_PAGE_CORRUPTED, 1, pageid);
205  return ER_LOG_PAGE_CORRUPTED;
206  }
207  }
208 
209  return NO_ERROR;
210 }

228 
229  if (!fileio_is_volume_exist (logwr_Gl.active_name))

235  else
236  {
237  /* Mount the active log and read the log header */
238  logwr_Gl.append_vdes =

240  if (logwr_Gl.append_vdes == NULL_VOLDES)
241  {

244  }
245  else
246  {
247  error = logwr_fetch_header_page (log_pgptr, logwr_Gl.append_vdes);
248  if (error != NO_ERROR)
249  {
250  return error;
251  }
252 
253  logwr_Gl.hdr = *((LOG_HEADER *) log_pgptr->area);
254 
255  assert (log_pgptr->hdr.logical_pageid == LOGPB_HEADER_PAGE_ID);
256  assert (log_pgptr->hdr.offset == NULL_OFFSET);
257  }
258  }
259 
260  return NO_ERROR;

333 /*
334  * logwr_initialize - Initialize logwr_Gl structure
335  *
336  * return:
337  *
338  * db_name(in):
339  * log_path(in):
340  * mode(in):
341  *
342  * Note:
343  */
344 static int
345 logwr_initialize (const char *db_name, const char *log_path, int mode, LOG_PAGEID start_pageid)
346 {
347  int log_nbuffers;
348  int error;
349  char *at_char = NULL;
350 
351  /* signal processing */
352 #if defined(WINDOWS)
353  (void) os_set_signal_handler (SIGABRT, logwr_shutdown_by_signal);
354  (void) os_set_signal_handler (SIGINT, logwr_shutdown_by_signal);
355  (void) os_set_signal_handler (SIGTERM, logwr_shutdown_by_signal);
356 #else /* ! WINDOWS */
357  (void) os_set_signal_handler (SIGSTOP, logwr_shutdown_by_signal);
358  (void) os_set_signal_handler (SIGTERM, logwr_shutdown_by_signal);
359  (void) os_set_signal_handler (SIGPIPE, SIG_IGN);
360 #endif /* ! WINDOWS */
361 
362  /* set the db name and log path */
363  strncpy (logwr_Gl.db_name, db_name, PATH_MAX - 1);
364  if ((at_char = strchr (logwr_Gl.db_name, '@')) != NULL)
365  {
366  *at_char = '\0';
367  logwr_Gl.hostname = at_char + 1;
368  }
369  strncpy (logwr_Gl.log_path, log_path, PATH_MAX - 1);
370  /* set the mode */
371  logwr_Gl.mode = (LOGWR_MODE) mode;
372 
373  /* set the active log file path */
374  fileio_make_log_active_name (logwr_Gl.active_name, log_path, logwr_Gl.db_name);
375  /* set the log info file path */
376  fileio_make_log_info_name (logwr_Gl.loginf_path, log_path, logwr_Gl.db_name);
377  /* background archive file path */
378  fileio_make_log_archive_temp_name (logwr_Gl.bg_archive_name, log_path, logwr_Gl.db_name);
379  log_nbuffers = LOGWR_COPY_LOG_BUFFER_NPAGES + 1;
380 
381  if (logwr_Gl.logpg_area == NULL)
382  {
383  logwr_Gl.logpg_area_size = log_nbuffers * LOG_PAGESIZE;
384  logwr_Gl.logpg_area = (char *) malloc (logwr_Gl.logpg_area_size);
385  if (logwr_Gl.logpg_area == NULL)
386  {
387  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, (size_t) logwr_Gl.logpg_area_size);
388  logwr_Gl.logpg_area_size = 0;
389  return ER_OUT_OF_VIRTUAL_MEMORY;
390  }
391  }
392 
393  if (logwr_Gl.toflush == NULL)
394  {
395  int i;
396 
397  logwr_Gl.max_toflush = log_nbuffers - 1;
398  logwr_Gl.toflush = (LOG_PAGE **) calloc (logwr_Gl.max_toflush, sizeof (logwr_Gl.toflush));
399  if (logwr_Gl.toflush == NULL)
400  {
401  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1,
402  logwr_Gl.max_toflush * sizeof (logwr_Gl.toflush));
403  logwr_Gl.max_toflush = 0;
404  return ER_OUT_OF_VIRTUAL_MEMORY;
405  }
406  for (i = 0; i < logwr_Gl.max_toflush; i++)
407  {
408  logwr_Gl.toflush[i] = NULL;
409  }
410  }
411 
412  error = logwr_read_log_header ();
413  if (error != NO_ERROR)
414  {
415  return error;
416  }
417 
418  logwr_Gl.start_pageid = start_pageid;
419  if (logwr_Gl.start_pageid >= NULL_PAGEID && logwr_Gl.hdr.nxarv_pageid != NULL_PAGEID)
420  {
421  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_HA_GENERIC_ERROR, 1, "Replication transaction log already exist");
422  return ER_HA_GENERIC_ERROR;
423  }
424 
425  logwr_Gl.action = LOGWR_ACTION_NONE;
426 
427  logwr_Gl.last_arv_fpageid = logwr_Gl.hdr.nxarv_pageid;
428  logwr_Gl.last_arv_num = logwr_Gl.hdr.nxarv_num;
429 
430  logwr_Gl.force_flush = false;
431  logwr_Gl.last_flush_time.tv_sec = 0;
432  logwr_Gl.last_flush_time.tv_usec = 0;
433 
434  logwr_Gl.ori_nxarv_pageid = NULL_PAGEID;
435 
436  if (prm_get_bool_value (PRM_ID_LOG_BACKGROUND_ARCHIVING))
437  {
438  BACKGROUND_ARCHIVING_INFO *bg_arv_info;
439  bg_arv_info = &logwr_Gl.bg_archive_info;
440 
441  if (fileio_is_volume_exist (logwr_Gl.bg_archive_name) == true)
442  {
443  bg_arv_info->vdes =
444  fileio_mount (NULL, logwr_Gl.bg_archive_name, logwr_Gl.bg_archive_name, LOG_DBLOG_ARCHIVE_VOLID, true,
445  false);
446  if (bg_arv_info->vdes == NULL_VOLDES)
447  {
448  return ER_IO_MOUNT_FAIL;
449  }
450 
451  error = logwr_read_bgarv_log_header ();
452  if (error != NO_ERROR)
453  {
454  if (error == ER_LOG_INCOMPATIBLE_DATABASE)
455  {
456  /* Considering the case of upgrading, if it read magic from bg archive file which is not include
457  * magic, make new bg archive file. */
458  fileio_dismount (NULL, bg_arv_info->vdes);
459  fileio_unformat (NULL, logwr_Gl.bg_archive_name);
460 
461  bg_arv_info->vdes = NULL_VOLDES;
462  }
463  else
464  {
465  return error;
466  }
467  }
468  }
469 
470  /* if bg archive file does not exist or magic is diff, */
471  /* create new bg archive file */
472  if (bg_arv_info->vdes == NULL_VOLDES)
473  {
474  bg_arv_info->vdes =
475  fileio_format (NULL, logwr_Gl.db_name, logwr_Gl.bg_archive_name, LOG_DBLOG_BG_ARCHIVE_VOLID,
476  logwr_Gl.hdr.npages + 1, false, false, false, LOG_PAGESIZE, 0, false);
477  if (bg_arv_info->vdes == NULL_VOLDES)
478  {
479  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_HA_GENERIC_ERROR, 1,
480  "Unable to create temporary archive log");
481  return ER_HA_GENERIC_ERROR;
482  }
483 
484  bg_arv_info->start_page_id = logwr_Gl.hdr.nxarv_pageid;
485  bg_arv_info->current_page_id = NULL_PAGEID;
486  bg_arv_info->last_sync_pageid = logwr_Gl.hdr.nxarv_pageid;
487 
488  error = logwr_flush_bgarv_header_page ();
489  if (error != NO_ERROR)
490  {
491  return error;
492  }
493  }
494  }
495 
496  return NO_ERROR;
497 }

549 /*
550  * logwr_set_hdr_and_flush_info -
551  *
552  * return:
553  * Note:
554  */
555 int
556 logwr_set_hdr_and_flush_info (void)
557 {
558  LOG_PAGE *log_pgptr = NULL, *last_pgptr;
559  char *p;
560  int num_toflush = 0;
561 
562  /* Set the flush information */
563  p = logwr_Gl.logpg_area + LOG_PAGESIZE;
564  while (p < (logwr_Gl.logpg_area + logwr_Gl.logpg_fill_size))
565  {
566  log_pgptr = (LOG_PAGE *) p;
567  logwr_Gl.toflush[num_toflush++] = log_pgptr;
568  p += LOG_PAGESIZE;
569  }
570 
571  last_pgptr = log_pgptr;
572  logwr_Gl.num_toflush = num_toflush;
573 
574  /* get original next archive pageid */
575  logwr_Gl.ori_nxarv_pageid = logwr_Gl.hdr.nxarv_pageid;
576 
577  /* Set the header and action information */
578  if (num_toflush > 0)
579  {
580  log_pgptr = (LOG_PAGE *) logwr_Gl.logpg_area;
581  logwr_Gl.hdr = *((LOG_HEADER *) log_pgptr->area);
582  logwr_Gl.loghdr_pgptr = log_pgptr;
583 
584  /* Initialize archive info if it is not set */
585  if (logwr_Gl.last_arv_fpageid == NULL_PAGEID || logwr_Gl.last_arv_num < 0)
586  {
587  logwr_Gl.last_arv_fpageid = logwr_Gl.hdr.nxarv_pageid;
588  logwr_Gl.last_arv_num = logwr_Gl.hdr.nxarv_num;
589  if (prm_get_bool_value (PRM_ID_LOG_BACKGROUND_ARCHIVING))
590  {
591  logwr_Gl.bg_archive_info.start_page_id = logwr_Gl.last_arv_fpageid;
592  }
593  }
594 
595  /* Check if it need archiving */
596  if (((logwr_Gl.last_arv_num + 1 < logwr_Gl.hdr.nxarv_num)
597  && (logwr_Gl.hdr.ha_file_status == LOG_HA_FILESTAT_ARCHIVED))
598  && (logwr_Gl.last_arv_fpageid <= logwr_Gl.last_recv_pageid))
599  {
600  /* Do delayed archiving */
601  logwr_Gl.action = (LOGWR_ACTION) (logwr_Gl.action | LOGWR_ACTION_ARCHIVING);
602  logwr_Gl.last_arv_lpageid = logwr_Gl.last_recv_pageid;
603  }
604  else if ((logwr_Gl.last_arv_num + 1 == logwr_Gl.hdr.nxarv_num)
605  && (last_pgptr->hdr.logical_pageid >= logwr_Gl.hdr.nxarv_pageid))
606  {
607  logwr_Gl.action = (LOGWR_ACTION) (logwr_Gl.action | LOGWR_ACTION_ARCHIVING);
608  logwr_Gl.last_arv_lpageid = logwr_Gl.hdr.nxarv_pageid - 1;
609  }
610 
611  if (last_pgptr != NULL && last_pgptr->hdr.logical_pageid < logwr_Gl.hdr.eof_lsa.pageid)
612  {
613  /* There are left several pages to get from the server */
614  logwr_Gl.last_recv_pageid = last_pgptr->hdr.logical_pageid;
615  logwr_Gl.action = (LOGWR_ACTION) (logwr_Gl.action | LOGWR_ACTION_DELAYED_WRITE);
616  }
617  else
618  {
619  logwr_Gl.last_recv_pageid = logwr_Gl.hdr.eof_lsa.pageid;
620 
621  if (logwr_Gl.hdr.perm_status == LOG_PSTAT_HDRFLUSH_INPPROCESS || logwr_Gl.action & LOGWR_ACTION_DELAYED_WRITE)
622  {
623  /* In case that it finishes delay write */
624  logwr_Gl.action = (LOGWR_ACTION) (logwr_Gl.action & ~LOGWR_ACTION_DELAYED_WRITE);
625  }
626  }
627  }
628  else
629  {
630  /* If it gets only the header page, compares both of the headers. There is no update for the header information */
631  LOG_HEADER hdr;
632  log_pgptr = (LOG_PAGE *) logwr_Gl.logpg_area;
633  hdr = *((LOG_HEADER *) log_pgptr->area);
634

838 /*
839  * logwr_flush_all_append_pages -
840  *
841  * return:
842  * Note:
843  */
844 static int
845 logwr_flush_all_append_pages (void)
846 {
847  LOG_PAGE *pgptr, *prv_pgptr;
848  LOG_PAGEID pageid, prv_pageid;
849  int idxflush;
850  bool need_sync;
851  int flush_page_count;
852  int i;
853 
854  idxflush = -1;
855  prv_pgptr = NULL;
856  need_sync = false;
857  flush_page_count = 0;
858 
859  for (i = 0; i < logwr_Gl.num_toflush; i++)
860  {
861  pgptr = logwr_Gl.toflush[i];
862 
863  if (idxflush != -1 && prv_pgptr != NULL)
864  {
865  /*
866  * This append log page should be dirty and contiguous to previous
867  * append page. If it is not, we need to flush the accumulated pages
868  * up to this point, and then start accumulating pages again.
869  */
870  pageid = pgptr->hdr.logical_pageid;
871  prv_pageid = prv_pgptr->hdr.logical_pageid;
872 
873  if ((pageid != prv_pageid + 1)
874  || (logwr_to_physical_pageid (pageid) != logwr_to_physical_pageid (prv_pageid) + 1))
875  {
876  /*
877  * This page is not contiguous.
878  *
879  * Flush the accumulated contiguous pages
880  */
881  if (logwr_writev_append_pages (&logwr_Gl.toflush[idxflush], i - idxflush) == NULL)
882  {
883  assert (er_errid () != NO_ERROR);
884  return er_errid ();
885  }
886  else
887  {
888  need_sync = true;
889 
890  /*
891  * Start over the accumulation of pages
892  */
893 
894  flush_page_count += i - idxflush;
895  idxflush = -1;
896  }
897  }
898  }
899 
900  if (idxflush == -1)
901  {
902  /*
903  * This page should be included in the flush
904  */
905  idxflush = i;
906  }
907 
908  /* prv_pgptr was not pgptr's previous buffer. prv_pgptr was the first buffer to flush, so only 2 continous pages
909  * always were flushed together. */
910  prv_pgptr = pgptr;
911  }
912 
913  /*
914  * If there are any accumulated pages, flush them at this point
915  */
916 
917  if (idxflush != -1)
918  {
919  int page_toflush = logwr_Gl.num_toflush - idxflush;
920 
921  /* last countious pages */
922  if (logwr_writev_append_pages (&logwr_Gl.toflush[idxflush], page_toflush) == NULL)
923  {
924  assert (er_errid () != NO_ERROR);
925  return er_errid ();
926  }
927  else
928  {
929  need_sync = true;
930  flush_page_count += page_toflush;
931  pgptr = logwr_Gl.toflush[idxflush + page_toflush - 1];
932  }
933  }
934 
935  /*
936  * Make sure that all of the above log writes are synchronized with any
937  * future log writes. That is, the pages should be stored on physical disk.
938  */
939  if (need_sync == true
940  && fileio_synchronize (NULL, logwr_Gl.append_vdes, logwr_Gl.active_name, FILEIO_SYNC_ONLY) == NULL_VOLDES)
941  {
942  assert (er_errid () != NO_ERROR);
943  return er_errid ();
944  }
945 
946  /* It's for dual write. */
947  if (need_sync == true && prm_get_bool_value (PRM_ID_LOG_BACKGROUND_ARCHIVING))
948  {
949  if (fileio_synchronize (NULL, logwr_Gl.bg_archive_info.vdes, logwr_Gl.bg_archive_name,
950  FILEIO_SYNC_ONLY) == NULL_VOLDES)
951  {
952  assert (er_errid () != NO_ERROR);
953  return er_errid ();
954  }
955  }
956 
957  /* Initialize flush info */
958  for (i = 0; i < logwr_Gl.num_toflush; i++)
959  {
960  logwr_Gl.toflush[i] = NULL;
961  }
962  logwr_Gl.num_toflush = 0;
963 
964  er_log_debug (ARG_FILE_LINE, "logwr_write_log_pages, flush_page_count(%d)\n", flush_page_count);
965 
966  return NO_ERROR;
967 }
