107 /* Process the remaining bytes in the internal buffer and the usual
108  prolog according to the standard and write the result to RESBUF. */
109 void *
110 md5_finish_ctx (struct md5_ctx *ctx, void *resbuf)
111 {
112  /* Take yet unprocessed bytes into account. */
113  UINT32 bytes = (UINT32) ctx->buflen;
114  size_t size = (bytes < 56) ? 64 / 4 : 64 * 2 / 4;
115 
116  /* Now count remaining bytes. */
117  ctx->total[0] += bytes;
118  if (ctx->total[0] < bytes)
119  ++ctx->total[1];
120 
121  /* Put the 64-bit file length in *bits* at the end of the buffer. */
122  ctx->buffer[size - 2] = SWAP (ctx->total[0] << 3);
123  ctx->buffer[size - 1] = SWAP ((ctx->total[1] << 3) | (ctx->total[0] >> 29));
124 
125  memcpy (&((char *) ctx->buffer)[bytes], fillbuf, (size - 2) * 4 - bytes);
126 
127  /* Process last bytes. */
128  md5_process_block (ctx->buffer, size * 4, ctx);
129 
130  return md5_read_ctx (ctx, resbuf);
131 }

223 void
224 md5_process_bytes (const void *buffer, size_t len, struct md5_ctx *ctx)
225 {
226  /* When we already have some bits in our internal buffer concatenate both inputs first. */
227  if (ctx->buflen != 0)
228  {
229  size_t left_over = ctx->buflen;
230  size_t add = 128 - left_over > len ? len : 128 - left_over;
231 
232  memcpy (&((char *) ctx->buffer)[left_over], buffer, add);
233  ctx->buflen += add;
234 
235  if (ctx->buflen > 64)
236  {
237  md5_process_block (ctx->buffer, ctx->buflen & ~63, ctx);
238 
239  ctx->buflen &= 63;
240  /* The regions in the following copy operation cannot overlap. */
241  memcpy (ctx->buffer, &((char *) ctx->buffer)[(left_over + add) & ~63], ctx->buflen);
242  }
243 
244  buffer = (const char *) buffer + add;
245  len -= add;
246  }
247 
248  /* Process available complete blocks. */
249  if (len >= 64)
250  {
251 #if !_STRING_ARCH_unaligned
252 # define alignof(type) offsetof (struct { char c; type x; }, x)
253 # define UNALIGNED_P(p) (((size_t) p) % alignof (UINT32) != 0)
254  if (UNALIGNED_P (buffer))
255  while (len > 64)
256  {
257  md5_process_block (memcpy (ctx->buffer, buffer, 64), 64, ctx);
258  buffer = (const char *) buffer + 64;
259  len -= 64;
260  }
261  else
262 #endif
263  {
264  md5_process_block (buffer, len & ~63, ctx);
265  buffer = (const char *) buffer + (len & ~63);
266  len &= 63;
267  }
268  }
269 
270  /* Move remaining bytes in internal buffer. */
271  if (len > 0)
272  {
273  size_t left_over = ctx->buflen;
274 
275  memcpy (&((char *) ctx->buffer)[left_over], buffer, len);
276  left_over += len;
277  if (left_over >= 64)
278  {
279  md5_process_block (ctx->buffer, 64, ctx);
280  left_over -= 64;
281  memcpy (ctx->buffer, &ctx->buffer[16], left_over);
282  }
283  ctx->buflen = left_over;
284  }
285 }

301 md5_process_block (const void *buffer, size_t len, struct md5_ctx *ctx)
302 {
303  UINT32 correct_words[16];
304  const UINT32 *words = buffer;
305  size_t nwords = len / sizeof (UINT32);
306  const UINT32 *endp = words + nwords;
307  UINT32 A = ctx->A;
308  UINT32 B = ctx->B;
309  UINT32 C = ctx->C;
310  UINT32 D = ctx->D;
311 
312  /* First increment the byte count. RFC 1321 specifies the possible length of the file up to 2^64 bits. Here we only
313  * compute the number of bytes. Do a double word increment. */
314  ctx->total[0] += (UINT32) len;
315  if (ctx->total[0] < len)
316  ++ctx->total[1];
317 
318  /* Process all bytes in the buffer with 64 bytes in each round of the loop. */
319  while (words < endp)
320  {
321  UINT32 *cwp = correct_words;
322  UINT32 A_save = A;
323  UINT32 B_save = B;
324  UINT32 C_save = C;
325  UINT32 D_save = D;
326 
327  /* First round: using the given function, the context and a constant the next context is computed. Because the
328  * algorithms processing unit is a 32-bit word and it is determined to work on words in little endian byte order
329  * we perhaps have to change the byte order before the computation. To reduce the work for the next steps we
330  * store the swapped words in the array CORRECT_WORDS. */
331 
332 #define OP(a, b, c, d, s, T) \
333  do \
334  { \
335  a += FF (b, c, d) + (*cwp++ = SWAP (*words)) + T; \
336  ++words; \
337  CYCLIC (a, s); \
338  a += b; \
339  } \
340  while (0)

344 #define CYCLIC(w, s) (w = (w << s) | (w >> (32 - s)))
