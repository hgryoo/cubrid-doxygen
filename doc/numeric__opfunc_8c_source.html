322 /*
323  * numeric_add_dec_str () -
324  * arg1(in) : ptr to a DEC_STRING
325  * arg2(in) : ptr to a DEC_STRING
326  * answer(out): ptr to a DEC_STRING
327  *
328  * Note: This routine adds two DEC_STRINGs and returns the result. It assumes
329  * that arg1 and arg2 have the same scaling.
330  */
331 
332 static void
333 numeric_add_dec_str (DEC_STRING * arg1, DEC_STRING * arg2, DEC_STRING * answer)
334 {
335  unsigned int answer_bit = 0;
336  int digit;
337  char arg1_dec, arg2_dec;
338 
339  /* Loop through the characters setting answer */
340  for (digit = TWICE_NUM_MAX_PREC - 1; digit >= 0; digit--)
341  {
342  arg1_dec = arg1->digits[digit];
343  arg2_dec = arg2->digits[digit];
344 
345  if (arg1_dec == -1)
346  {
347  arg1_dec = 0;
348 
349  if (answer_bit < 10)
350  {
351  break; /* pass through the leftmost digits */
352  }
353  }
354 
355  if (arg2_dec == -1)
356  {
357  /* is not first element */
358  assert (digit < TWICE_NUM_MAX_PREC - 1);
359 
360  arg2_dec = 0;
361  }
362 
363  assert (arg1_dec >= 0);
364  assert (arg2_dec >= 0);
365 
366  answer_bit = (arg1_dec + arg2_dec) + (answer_bit >= 10);
367  answer->digits[digit] = fast_mod[answer_bit];
368  }
369 }

503 /*
504  * numeric_double_shift_bit () -
505  * return:
506  * arg1(in) : DB_C_NUMERIC
507  * arg2(in) : DB_C_NUMERIC
508  * numbits(in): integer number of bits to shift
509  * lsb(out) : DB_C_NUMERIC
510  * msb(out) : DB_C_NUMERIC
511  * is_long_num(in) : is long NUMERIC.
512  *
513  * Note: This routine returns lsb, msb shifted by numbits from arg1, arg2.
514  * Bits that are shifted out of arg1 are placed into LSB of arg2.
515  * only arg1 and lsb may be long NUMERIC.
516  */
517 static void
518 numeric_double_shift_bit (DB_C_NUMERIC arg1, DB_C_NUMERIC arg2, int numbits, DB_C_NUMERIC lsb, DB_C_NUMERIC msb,
519  bool is_long_num)
520 {
521  /* the largest buf size of DB_C_NUMERIC */
522  unsigned char local_arg1[DB_NUMERIC_BUF_SIZE * DB_LONG_NUMERIC_MULTIPLIER];
523  unsigned char local_arg2[DB_NUMERIC_BUF_SIZE]; /* copy of a DB_C_NUMERIC */
524  unsigned int digit;
525  unsigned int buf_size;
526 
527  if (is_long_num)
528  {
529  buf_size = DB_NUMERIC_BUF_SIZE * DB_LONG_NUMERIC_MULTIPLIER;
530  }
531  else
532  {
533  buf_size = DB_NUMERIC_BUF_SIZE;
534  }
535 
536  /* Copy args into local variables */
537  numeric_copy_long (local_arg1, arg1, is_long_num);
538  numeric_copy (local_arg2, arg2);
539 
540  /* Loop through all but last word of msb shifting bits */
541  for (digit = 0; digit < DB_NUMERIC_BUF_SIZE - 1; digit++)
542  {
543  msb[digit] = (local_arg2[digit] << numbits) | (local_arg2[digit + 1] >> (8 - numbits));
544  }
545 
546  /* Do last word of msb separately using upper word of lsb */
547  msb[DB_NUMERIC_BUF_SIZE - 1] = (local_arg2[DB_NUMERIC_BUF_SIZE - 1] << numbits) | (local_arg1[0] >> (8 - numbits));
548 
549  /* Loop through all but last word of lsb shifting bits */
550  for (digit = 0; digit < buf_size - 1; digit++)
551  {
552  lsb[digit] = (local_arg1[digit] << numbits) | (local_arg1[digit + 1] >> (8 - numbits));
553  }
554 
555  /* Do last word of lsb separately. */
556  lsb[buf_size - 1] = local_arg1[buf_size - 1] << numbits;
557 }
