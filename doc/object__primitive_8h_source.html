205 /* PRIMITIVE TYPE MACROS
206  *
207  * These provide a shell around the calling of primitive type operators
208  * so they look nicer. We used to have "VAL" oriented functions here too
209  * but with the new DB_VALUE system, these aren't really necessary any more.
210  * It probably wouldn't hurt to phase these out either.
211  */
212 
213 /* PRIM_INITMEM
214  *
215  * Initialize an attribute value in instance memory
216  * For fixed witdh attributes, it tries to zero out the memory. This isn't a
217  * requirement but it makes the object block look cleaner in the debugger.
218  * For variable width attributes, it makes sure the pointers are NULL.
219  */
220 #define PRIM_INITMEM(type, mem, domain) (*((type)->initmem))(mem, domain)
221 
222 /* PRIM_SETMEM
223  * Assign a value into instance memory, copy the value.
224  */
225 #define PRIM_SETMEM(type, domain, mem, value) (*((type)->setmem))(mem, domain, value)
226 
227 /* PRIM_GETMEM
228  * Get a value from instance memory, copy the value.
229  */
230 #define PRIM_GETMEM(type, domain, mem, value) \
231  (*((type)->getmem))(mem, domain, value, true)
232 
233 /* PRIM_GETMEM_NOCOPY
234  * The "nocopy" option for "getmem" is intended only for internal processes
235  * that need to access the values but don't want the overhead of copying,
236  * particularly for strings. The only user of this right now is the UNIQUE
237  * register logic.
238  */
239 #define PRIM_GETMEM_NOCOPY(type, domain, mem, value) \
240  (*((type)->getmem))(mem, domain, value, false)
241 
242 /* PRIM_FREEMEM
243  * Free any external storage in instance memory that may have been allocated on
244  * behalf of a type.
245  */
246 #define PRIM_FREEMEM(type, mem) \
247  if (type->freemem != NULL) (*((type)->freemem))(mem)
248 
249 /* PRIM_READ
250  * Read a value from a disk buffer into instance memory.
251  */
252 #define PRIM_READ(type, domain, tr, mem, len) \
253  (*((type)->data_readmem))(tr, mem, domain, len)
254 
255 /* PRIM_WRITE
256  * Write a value from instance memory into a disk buffer.
257  */
258 #define PRIM_WRITE(type, domain, tr, mem) (*((type)->data_writemem))(tr, mem, domain)
259 
260 #define PRIM_WRITEVAL(type, buf, val) (*((type)->writeval))(buf, val)
261 #define PRIM_LENGTHVAL(type, val, size) (*((type)->lengthval))(val, size)
262 
263 /* PRIM_SET_NULL
264  * set a db_value to NULL
265  */
266 #define PRIM_SET_NULL(db_value_ptr) \
267  ((db_value_ptr)->domain.general_info.is_null = 1, \
268  (db_value_ptr)->need_clear = false)
