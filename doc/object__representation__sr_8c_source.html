140  * record(in): disk record
141  */

143 orc_diskrep_from_record (THREAD_ENTRY * thread_p, RECDES * record)
144 {
145  DISK_ATTR *att, *att_fixed, *att_variable;
146  OR_ATTRIBUTE *or_att;
147  int i, j, k, n_attributes, n_btstats;
148  BTREE_STATS *bt_statsp;

151  OR_CLASSREP *or_rep = NULL;
152  OR_INDEX *or_idx = NULL;
153 
154  VPID root_vpid;
155  PAGE_PTR root;
156  BTREE_ROOT_HEADER *root_header = NULL;
157  BTID_INT btid_int;

159  or_rep = or_get_classrep (record, NULL_REPRID);
160  if (or_rep == NULL)
161  {
162  goto error;
163  }
164 
165  rep = (DISK_REPR *) malloc (sizeof (DISK_REPR));
166  if (rep == NULL)
167  {
168  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (DISK_REPR));
169  goto error;
170  }
171 
172  rep->id = or_rep->id;
173  rep->n_fixed = 0;
174  rep->n_variable = 0;
175  rep->fixed_length = or_rep->fixed_length;
176 #if 0 /* reserved for future use */
177  rep->repr_reserved_1 = 0;
178 #endif
179  rep->fixed = NULL;
180  rep->variable = NULL;
181 
182  /* Calculate the number of fixed and variable length attributes */
183  n_attributes = or_rep->n_attributes;
184  or_att = or_rep->attributes;
185  for (i = 0; i < n_attributes; i++, or_att++)
186  {
187  if (or_att->is_fixed)
188  {
189  (rep->n_fixed)++;
190  }
191  else
192  {
193  (rep->n_variable)++;
194  }
195  }
196 
197  if (rep->n_fixed)
198  {
199  rep->fixed = (DISK_ATTR *) malloc (sizeof (DISK_ATTR) * rep->n_fixed);
200  if (rep->fixed == NULL)
201  {
202  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (DISK_ATTR) * rep->n_fixed);
203  goto error;
204  }
205  memset (rep->fixed, 0x0, sizeof (DISK_ATTR) * rep->n_fixed);
206  }
207 
208  if (rep->n_variable)
209  {
210  rep->variable = (DISK_ATTR *) malloc (sizeof (DISK_ATTR) * rep->n_variable);
211  if (rep->variable == NULL)
212  {
213  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (DISK_ATTR) * rep->n_variable);
214  goto error;
215  }
216  memset (rep->variable, 0x0, sizeof (DISK_ATTR) * rep->n_variable);
217  }
218 
219  /* Copy the attribute information */
220  att_fixed = rep->fixed;
221  att_variable = rep->variable;
222  or_att = or_rep->attributes;
223 
224  for (i = 0; i < n_attributes; i++, or_att++)
225  {
226  if (or_att->is_fixed)
227  {
228  att = att_fixed;
229  att_fixed++;
230  }
231  else
232  {
233  att = att_variable;
234  att_variable++;
235  }
236 
237  if (att == NULL)
238  {
239  goto error;
240  }
241 
242  att->type = or_att->type;
243  att->id = or_att->id;
244  att->location = or_att->location;
245  att->position = or_att->position;
246  att->val_length = or_att->default_value.val_length;
247  att->value = or_att->default_value.value;
248  or_att->default_value.value = NULL;
249  att->classoid = or_att->classoid;
250 
251  /* initialize B+tree statistics information */
252 
253  n_btstats = att->n_btstats = or_att->n_btids;
254  if (n_btstats > 0)
255  {
256  att->bt_stats = (BTREE_STATS *) malloc (sizeof (BTREE_STATS) * n_btstats);
257  if (att->bt_stats == NULL)
258  {
259  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (BTREE_STATS) * n_btstats);
260  goto error;
261  }
262  memset (att->bt_stats, 0, sizeof (BTREE_STATS) * n_btstats);
263 
264  for (j = 0, bt_statsp = att->bt_stats; j < n_btstats; j++, bt_statsp++)
265  {
266  bt_statsp->btid = or_att->btids[j];
267 
268  bt_statsp->leafs = 0;
269  bt_statsp->pages = 0;
270  bt_statsp->height = 0;
271  bt_statsp->keys = 0;
272  bt_statsp->has_function = 0;
273  for (k = 0; k < or_rep->n_indexes; k++)
274  {
275  or_idx = &or_rep->indexes[k];
276  if (or_idx && BTID_IS_EQUAL (&or_idx->btid, &bt_statsp->btid) && or_idx->func_index_info
277  && or_idx->func_index_info->col_id == 0)
278  {
279  bt_statsp->has_function = 1;
280  break;
281  }
282  }
283 
284  bt_statsp->key_type = NULL;
285  bt_statsp->pkeys_size = 0;
286  bt_statsp->pkeys = NULL;
287 
288 #if 0 /* reserved for future use */
289  for (k = 0; k < BTREE_STATS_RESERVED_NUM; k++)
290  {
291  bt_statsp->reserved[k] = 0;
292  }
293 #endif
294 
295  /* read B+tree Root page header info */
296  root_vpid.pageid = bt_statsp->btid.root_pageid;
297  root_vpid.volid = bt_statsp->btid.vfid.volid;
298 
299  if (VPID_ISNULL (&root_vpid))
300  {
301  /* after create the catalog record of the class, and before create the catalog record of the
302  * constraints for the class currently, does not know BTID */
303  continue;
304  }
305 
306  root = pgbuf_fix (thread_p, &root_vpid, OLD_PAGE, PGBUF_LATCH_READ, PGBUF_UNCONDITIONAL_LATCH);
307  if (root == NULL)
308  {
309  goto error;
310  }
311 
312  (void) pgbuf_check_page_ptype (thread_p, root, PAGE_BTREE);
313 
314  root_header = btree_get_root_header (thread_p, root);
315  if (root_header == NULL)
316  {
317  pgbuf_unfix_and_init (thread_p, root);
318  goto error;
319  }
320 
321  /* construct BTID_INT structure */
322  btid_int.sys_btid = &bt_statsp->btid;
323  if (btree_glean_root_header_info (thread_p, root_header, &btid_int) != NO_ERROR)
324  {
325  pgbuf_unfix_and_init (thread_p, root);
326  goto error;
327  }
328 
329  pgbuf_unfix_and_init (thread_p, root);
330 
331  bt_statsp->key_type = btid_int.key_type;
332  if (TP_DOMAIN_TYPE (bt_statsp->key_type) == DB_TYPE_MIDXKEY)
333  {
334  bt_statsp->pkeys_size = tp_domain_size (bt_statsp->key_type->setdomain);
335  }
336  else
337  {
338  bt_statsp->pkeys_size = 1;
339  }
340 
341  /* cut-off to stats */
342  if (bt_statsp->pkeys_size > BTREE_STATS_PKEYS_NUM)
343  {
344  bt_statsp->pkeys_size = BTREE_STATS_PKEYS_NUM;
345  }
346 
347  bt_statsp->pkeys = (int *) malloc (bt_statsp->pkeys_size * sizeof (int));
348  if (bt_statsp->pkeys == NULL)
349  {
350  bt_statsp->pkeys_size = 0;
351  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1,
352  bt_statsp->pkeys_size * sizeof (int));
353  goto error;
354  }
355 
356  assert (bt_statsp->pkeys_size <= BTREE_STATS_PKEYS_NUM);
357  for (k = 0; k < bt_statsp->pkeys_size; k++)
358  {
359  bt_statsp->pkeys[k] = 0;
360  }
361  } /* for (j = 0, ...) */
362  }
363  else
364  {
365  att->bt_stats = NULL;
366  }
367  }
368 
369  or_free_classrep (or_rep);
370  return (rep);
371 
372 error:
373  if (rep != NULL)
374  {
375  orc_free_diskrep (rep);
376  }
377 
378  if (or_rep != NULL)
379  {
380  or_free_classrep (or_rep);
381  }
382 
383  return (NULL);
384 }

498 /*
499  * orc_subclasses_from_record () - Extracts the OID's of the immediate
500  * subclasses
501  * return: error code
502  * record(in): record containing a class
503  * array_size(out): pointer to int containing max size of array
504  * array_ptr(out): pointer to OID array
505  *
506  * Note: The array is maintained as an array of OID's, the last element in the
507  * array will satisfy the OID_ISNULL() test. The array_size has
508  * the number of actual elements allocated in the array which may be more
509  * than the number of slots that have non-NULL OIDs.
510  * The function adds the subclass oids to the existing array. If the
511  * array is not large enough, it is reallocated using realloc.
512  */
513 int
514 orc_subclasses_from_record (RECDES * record, int *array_size, OID ** array_ptr)
515 {
516  int error = NO_ERROR;
517  OID *array;
518  char *ptr;
519  int max, insert, i, newsize, nsubs;
520  char *subset = NULL;
521 
522  nsubs = 0;
523 
524  if (!OR_VAR_IS_NULL (record->data, ORC_SUBCLASSES_INDEX))
525  {
526  subset = (char *) (record->data) + OR_VAR_OFFSET (record->data, ORC_SUBCLASSES_INDEX);
527  nsubs = OR_SET_ELEMENT_COUNT (subset);
528  }
529 
530  if (nsubs)
531  {
532  max = *array_size;
533  array = *array_ptr;
534  if (array == NULL)
535  {
536  max = 0;
537  }
538 
539  /* find the last element in the array */
540  for (i = 0; i < max && !OID_ISNULL (&array[i]); i++)
541  {
542  ;
543  }
544  insert = i;
545 
546  /*
547  * check for array extension.
548  * Add one in the comparison since a NULL_OID is set at the end of the
549  * array
550  */
551  if (array == NULL || (insert + nsubs + 1) > max)
552  {
553  newsize = insert + nsubs + 10;
554  if (array == NULL)
555  {
556  array = (OID *) malloc (newsize * sizeof (OID));
557  }
558  else
559  {
560  array = (OID *) realloc (array, newsize * sizeof (OID));
561  }
562 
563  if (array == NULL)
564  {
565  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, newsize * sizeof (OID));
566  return ER_OUT_OF_VIRTUAL_MEMORY;
567  }
568 
569  for (i = max; i < newsize; i++)
570  {
571  OID_SET_NULL (&array[i]);
572  }
573 
574  max = newsize;
575  }
576 
577  /* Advance past the set header, the domain size, and the "object" domain. Note that this assumes we are not using
578  * a bound bit array even though this is a fixed width homogeneous set. Probably not a good assumption. */
579  ptr = subset + OR_SET_HEADER_SIZE + OR_INT_SIZE + OR_INT_SIZE;
580 
581  assert (array != NULL);
582 
583  /* add the new OIDs */
584  for (i = 0; i < nsubs; i++)
585  {
586  OR_GET_OID (ptr, &array[insert + i]);
587  ptr += OR_OID_SIZE;
588  }
589 
590  OID_SET_NULL (&array[insert + nsubs]);
591 
592  /* return these in case there were changes */
593  *array_size = max;
594  *array_ptr = array;
595  }
596 
597  return error;
598 }

604  * record(in): record containing a class
605  * array_size(out): pointer to int containing max size of array
606  * array_ptr(out): pointer to OID array
607  *

610  * the number of actual elements allocated in the array which may be

628  nsupers = 0;
629  if (OR_VAR_IS_NULL (record->data, ORC_SUPERCLASSES_INDEX))

842 /*

851  * max_classes(out):

854  * Note: This routine gets checks the class to see if it has an attribute named
855  * attr_name, that has a source class equal to source_class. This would
856  * mean that the given class participates in the hierachy. We add its
857  * heap and attribute id to the arrays and recurse for any subclasses
858  * that it might have.
859  */
860 static int
861 or_get_hierarchy_helper (THREAD_ENTRY * thread_p, OID * source_class, OID * class_, BTID * btid, OID ** class_oids,
862  HFID ** hfids, int *num_classes, int *max_classes, int *partition_local_index)
863 {

869  OR_CLASSREP *or_rep = NULL;

871  HFID hfid;
872  HEAP_SCANCACHE scan_cache;
873 
874  (void) heap_scancache_quick_start_root_hfid (thread_p, &scan_cache);
875 
876  if (heap_get_class_record (thread_p, class_, &record, &scan_cache, COPY) != S_SUCCESS)
877  {
878  goto error;
879  }

881  or_rep = or_get_classrep (&record, NULL_REPRID);
882  if (or_rep == NULL)
883  {
884  goto error;

890  {
891  if (BTID_IS_EQUAL (&(or_index->btid), btid))
892  {
893  found = 1;
894  }
895  }

900  * partition has it's own btree */

902  {
903  *partition_local_index = 1;
904  }
905  else
906  {
907  goto success;
908  }
909  }
910 
911  /*
912  * For each subclass, recurse ...
913  * Unfortunately, this information is not available in the OR_CLASSREP
914  * structure, so we'll digress into the RECDES structure for it. It
915  * might be a good idea to add subclass information to the OR_CLASSREP
916  * structure.
917  */
918  nsubs = 0;
919  if (!OR_VAR_IS_NULL (record.data, ORC_SUBCLASSES_INDEX))
920  {
921  subset = (char *) (record.data) + OR_VAR_OFFSET (record.data, ORC_SUBCLASSES_INDEX);
922  nsubs = OR_SET_ELEMENT_COUNT (subset);
923  }
924

928  * a bound bit array even though this is a fixed width homogeneous set. Probably not a good assumption. */

939  }
940

942  }

946  or_class_hfid (&record, &hfid);

948  {
949  goto success;
950  }
951

953  for (i = 0; i < *num_classes; i++)
954  {
955  if (*class_oids != NULL && OID_EQ (class_, &((*class_oids)[i])))
956  {
957  goto success;
958  }
959  }
960 
961  /* do we need to extend the arrays? */
962  if ((*num_classes + 1) > *max_classes)
963  {
964  newsize = *max_classes + OR_ARRAY_EXTENT;
965 
966  if (*class_oids == NULL)
967  {
968  *class_oids = (OID *) malloc (newsize * sizeof (OID));
969  }
970  else
971  {
972  *class_oids = (OID *) realloc (*class_oids, newsize * sizeof (OID));
973  }
974 
975  if (*class_oids == NULL)
976  {
977  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, newsize * sizeof (OID));
978  goto error;
979  }
980 
981  if (*hfids == NULL)
982  {
983  *hfids = (HFID *) malloc (newsize * sizeof (HFID));
984  }
985  else
986  {
987  *hfids = (HFID *) realloc (*hfids, newsize * sizeof (HFID));
988  }
989 
990  if (*hfids == NULL)
991  {
992  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, newsize * sizeof (HFID));
993  goto error;
994  }
995 
996  *max_classes = newsize;
997  }
998 
999  if (*class_oids == NULL || *hfids == NULL)
1000  {
1001  goto error;
1002  }
1003 
1004  COPY_OID (&((*class_oids)[*num_classes]), class_);
1005  (*hfids)[*num_classes] = hfid;
1006  *num_classes += 1;
1007 
1008 success:
1009  or_free_classrep (or_rep);
1010  (void) heap_scancache_end (thread_p, &scan_cache);
1011  return NO_ERROR;
1012 
1013 error:
1014  if (or_rep != NULL)
1015  {
1016  or_free_classrep (or_rep);
1017  }
1018 
1019  (void) heap_scancache_end (thread_p, &scan_cache);
1020 
1021  assert (er_errid () != NO_ERROR);
1022  return er_errid ();
1023 }
