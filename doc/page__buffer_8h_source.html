85 #define PGBUF_WATCHER_SET_GROUP(w,hfid) \
86  do { \
87  if ((hfid) == NULL || (hfid)->vfid.volid == NULL_VOLID \
88  || (hfid)->hpgid == NULL_PAGEID || HFID_IS_NULL (hfid)) \
89  { \
90  VPID_SET_NULL (&((w)->group_id)); \
91  } \
92  else \
93  { \
94  (w)->group_id.volid = (hfid)->vfid.volid; \
95  (w)->group_id.pageid = (hfid)->hpgid; \
96  } \
97  } while (0)
98 
99 #define PGBUF_WATCHER_COPY_GROUP(w_dst,w_src) \
100  do { \
101  assert ((w_src) != NULL); \
102  assert ((w_dst) != NULL); \
103  assert (!VPID_ISNULL (&((w_src)->group_id))); \
104  VPID_COPY (&((w_dst)->group_id), &((w_src)->group_id)); \
105  } while (0)
106 
107 #define PGBUF_WATCHER_RESET_RANK(w,rank) \
108  do { \
109  (w)->initial_rank = (rank); \
110  } while (0)

158 typedef enum
159 {
160  OLD_PAGE = 0, /* Fetch page that should be allocated and already existing either in page buffer or on
161  * disk. Must pass validation test and must be fixed from disk if it doesn't exist in
162  * buffer. */
163  NEW_PAGE, /* Fetch newly allocated page. Must pass validation test but it can be created directly
164  * in buffer without fixing from disk. */
165  OLD_PAGE_IF_IN_BUFFER, /* Fetch existing page only if is valid and if it exists in page buffer. Page may be
166  * deallocated or flushed and invalidated from buffer, in which case fixing page is not
167  * necessary. */
168  OLD_PAGE_PREVENT_DEALLOC, /* Fetch existing page and mark its memory buffer, to prevent deallocation. */
169  OLD_PAGE_DEALLOCATED, /* Fetch page that has been deallocated. */
170  OLD_PAGE_MAYBE_DEALLOCATED, /* Fetch page that maybe was deallocated. */
171 } PAGE_FETCH_MODE;
