252 /*
253  * pruningset_union () - perform intersection on two PRUNING_BITSET objects
254  * return : void
255  * left (in/out) :
256  * right (in) :
257  */
258 static void
259 pruningset_union (PRUNING_BITSET * left, const PRUNING_BITSET * right)
260 {
261  unsigned int max, i;
262 
263  max = BITSET_LENGTH (left);
264  if (BITSET_LENGTH (left) > BITSET_LENGTH (right))
265  {
266  max = BITSET_LENGTH (right);
267  }
268 
269  for (i = 0; i < max; i++)
270  {
271  left->set[i] |= right->set[i];
272  }
273 }

287 /*
288  * pruningset_popcount () - return the number of elements which are set in a
289  * PRUNING_BITSET object
290  * return : number of elements set
291  * s (in) : PRUNING_BITSET object
292  */
293 static int
294 pruningset_popcount (const PRUNING_BITSET * s)
295 {
296  /* we expect to have only a few bits set, so the Brian Kernighan algorithm should be the fastest. */
297  int count = 0;
298  unsigned i, v;
299 
300  for (i = 0; i < BITSET_LENGTH (s); i++)
301  {
302  v = s->set[i];
303  for (; v != 0; count++)
304  {
305  v &= v - 1;
306  }
307  }
308 
309  if (count > s->count)
310  {
311  return s->count;
312  }
313 
314  return count;
315 }

330 /*
331  * pruningset_iterator_next () - advance the iterator
332  * return : next element
333  * it (in/out) : iterator
334  */
335 static int
336 pruningset_iterator_next (PRUNING_BITSET_ITERATOR * it)
337 {
338  unsigned int i, j;
339  unsigned int word;
340  int pos = 0;
341 
342  if (it->next >= it->set->count)
343  {
344  it->next = -1;
345  return -1;
346  }
347 
348  for (i = it->next / BITS_IN_WORD; i < BITSET_LENGTH (it->set); i++)
349  {
350  if (it->set->set[i] == 0)
351  {
352  continue;
353  }
354 
355  word = it->set->set[i];
356  pos = it->next % BITS_IN_WORD;
357 
358  for (j = pos; j < BITS_IN_WORD; j++)
359  {
360  if ((word & (1 << j)) != 0)
361  {
362  it->next = i * BITS_IN_WORD + j + 1;
363  return (i * BITS_IN_WORD + j);
364  }
365  }
366 
367  it->next = (i + 1) * BITS_IN_WORD;
368  }
369 
370  it->next = -1;
371 
372  return -1;
373 }

394  *

402 {

406  old_heap = db_change_private_heap (NULL, 0);
407

411  {
412  int i;
413 
414  for (i = 0; i < entry->count; i++)
415  {
416  db_seq_free (entry->partitions[i].values);
417  }
418  free_and_init (entry->partitions);
419  }
420 
421  free_and_init (entry);
422  }
423 
424  db_change_private_heap (NULL, old_heap);
425 
426  return NO_ERROR;
427 }

466 /*
467  * partition_pruning_context_to_cache_entry () - create a cache entry from
468  * a pruning context
469  * return : cache entry object
470  * pinfo (in) : pruning context
471  */
472 static PARTITION_CACHE_ENTRY *
473 partition_pruning_context_to_cache_entry (PRUNING_CONTEXT * pinfo)
474 {
475  PARTITION_CACHE_ENTRY *entry_p = NULL;
476  int i = 0;
477  HL_HEAPID old_heap_id = 0;
478 
479  assert (pinfo != NULL);
480 
481  entry_p = (PARTITION_CACHE_ENTRY *) malloc (sizeof (PARTITION_CACHE_ENTRY));
482  if (entry_p == NULL)
483  {
484  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (PARTITION_CACHE_ENTRY));
485  pinfo->error_code = ER_FAILED;
486  goto error_return;
487  }
488  entry_p->partitions = NULL;
489  entry_p->count = 0;
490 
491  COPY_OID (&entry_p->class_oid, &pinfo->root_oid);
492  entry_p->attr_id = pinfo->attr_id;
493 
494  entry_p->count = pinfo->count;
495 
496  entry_p->partitions = (OR_PARTITION *) malloc (entry_p->count * sizeof (OR_PARTITION));
497  if (entry_p->partitions == NULL)
498  {
499  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1,
500  entry_p->count * sizeof (PARTITION_CACHE_ENTRY));
501  pinfo->error_code = ER_FAILED;
502  goto error_return;
503  }
504 
505  /* Change private heap to 0 to use malloc/free for entry_p allocated values. These values outlast the current thread
506  * heap */
507  old_heap_id = db_change_private_heap (pinfo->thread_p, 0);
508  for (i = 0; i < entry_p->count; i++)
509  {
510  COPY_OID (&entry_p->partitions[i].class_oid, &pinfo->partitions[i].class_oid);
511 
512  HFID_COPY (&entry_p->partitions[i].class_hfid, &pinfo->partitions[i].class_hfid);
513 
514  entry_p->partitions[i].partition_type = pinfo->partitions[i].partition_type;
515 
516  entry_p->partitions[i].rep_id = pinfo->partitions[i].rep_id;
517 
518  if (pinfo->partitions[i].values != NULL)
519  {
520  entry_p->partitions[i].values = db_seq_copy (pinfo->partitions[i].values);
521  if (entry_p->partitions[i].values == NULL)
522  {
523  pinfo->error_code = ER_FAILED;
524  goto error_return;
525  }
526  }
527  }
528 
529  /* restore heap id */
530  db_change_private_heap (pinfo->thread_p, old_heap_id);
531 
532  return entry_p;
533 
534 error_return:
535  if (entry_p != NULL)
536  {
537  if (entry_p->partitions != NULL)
538  {
539  int j;
540 
541  for (j = 0; j < i; j++)
542  {
543  if (entry_p->partitions[j].values != NULL)
544  {
545  db_seq_free (entry_p->partitions[j].values);
546  }
547  }
548 
549  free_and_init (entry_p->partitions);
550  }
551 
552  free_and_init (entry_p);
553  }
554 
555  if (old_heap_id != 0)
556  {
557  db_change_private_heap (pinfo->thread_p, old_heap_id);
558  }
559 
560  return NULL;
561 }

563 /*
564  * partition_cache_pruning_context () - cache a pruning context
565  * return : error code or NO_ERROR
566  * pinfo (in) : pruning context
567  * exists (out) : return true if put into cache success,
568  * return false if the key already exists in hash map.
569  */
570 static int
571 partition_cache_pruning_context (PRUNING_CONTEXT * pinfo, bool * already_exists)
572 {
573  PARTITION_CACHE_ENTRY *entry_p = NULL;
574  OID *oid_key = NULL;
575  const void *val;
576 
577  if (!PARTITION_IS_CACHE_INITIALIZED ())
578  {
579  return NO_ERROR;
580  }
581 
582  assert (pinfo != NULL);
583 
584  /* Check if this class is being modified by this transaction. If this is the case, we can't actually use this cache
585  * entry because the next request might have already modified the partitioning schema and we won't know it */
586  if (log_is_class_being_modified (pinfo->thread_p, &pinfo->root_oid))
587  {
588  return NO_ERROR;
589  }
590 
591  entry_p = partition_pruning_context_to_cache_entry (pinfo);
592  if (entry_p == NULL)
593  {
594  return ER_FAILED;
595  }
596 
597  oid_key = &entry_p->class_oid;
598  if (csect_enter (pinfo->thread_p, CSECT_PARTITION_CACHE, INF_WAIT) != NO_ERROR)
599  {
600  return ER_FAILED;
601  }
602 
603  val = mht_put_if_not_exists (db_Partition_Ht, oid_key, entry_p);
604  if (val == NULL)
605  {
606  csect_exit (pinfo->thread_p, CSECT_PARTITION_CACHE);
607  return ER_FAILED;
608  }
609 
610  if (val != entry_p)
611  {
612  partition_free_cache_entry (entry_p);
613  *already_exists = true;
614  }
615  else
616  {
617  *already_exists = false;
618  }
619 
620  csect_exit (pinfo->thread_p, CSECT_PARTITION_CACHE);
621 
622  return NO_ERROR;
623 }

625 /*
626  * partition_load_context_from_cache () - load pruning info from cache
627  * return : true if the pruning context was found in cache, false otherwise
628  * pinfo (in/out) : pruning context
629  * is_modfied (in/out) : true if the class schema is being modified
630  *
631  * Note: If the class schema is being modified by the current transaction, the
632  * cache info is not used since it might not reflect the latest changes.
633  * There is no way that the class schema is being modified by another
634  * transaction than the one which is requesting the current pruning context.
635  * Any schema changes place a X_LOCK on the class and the transaction
636  * performing the operation has exclusive access to it (and implicitly to the
637  * entry in the cache). Because of this, we only have to look for the class
638  * in the modified list of the current transaction. If this changes (classes
639  * are no longer exclusively locked for alter), we have to review this
640  * function.
641  */
642 static bool
643 partition_load_context_from_cache (PRUNING_CONTEXT * pinfo, bool * is_modfied)
644 {
645  PARTITION_CACHE_ENTRY *entry_p = NULL;
646

852 /*
853  * pruningset_to_spec_list () - convert a pruningset to an array of
854  * PARTITION_SPEC_TYPE elements
855  * return : error code or NO_ERROR
856  * pinfo (in) : pruning context
857  * pruned (in) : pruned partitions
858  */
859 static int
860 pruningset_to_spec_list (PRUNING_CONTEXT * pinfo, const PRUNING_BITSET * pruned)
861 {
862  int cnt = 0, i = 0, pos = 0, error = NO_ERROR;
863  PARTITION_SPEC_TYPE *spec = NULL;
864  bool is_index = false;
865  char *btree_name = NULL;
866  OID *master_oid = NULL;
867  BTID *master_btid = NULL;
868  PRUNING_BITSET_ITERATOR it;
869 
870  cnt = pruningset_popcount (pruned);
871  if (cnt == 0)
872  {
873  /* pruning did not find any partition, just return */
874  error = NO_ERROR;
875  goto cleanup;
876  }
877 
878  if (pinfo->spec->access == ACCESS_METHOD_INDEX || pinfo->spec->access == ACCESS_METHOD_INDEX_KEY_INFO)
879  {
880  /* we have to load information about the index used so we can duplicate it for each partition */
881  is_index = true;
882  master_oid = &ACCESS_SPEC_CLS_OID (pinfo->spec);
883  master_btid = &pinfo->spec->indexptr->btid;
884  error =
885  heap_get_indexinfo_of_btid (pinfo->thread_p, master_oid, master_btid, NULL, NULL, NULL, NULL, &btree_name,
886  NULL);
887  if (error != NO_ERROR)
888  {
889  goto cleanup;
890  }
891  }
892 
893  spec = (PARTITION_SPEC_TYPE *) db_private_alloc (pinfo->thread_p, cnt * sizeof (PARTITION_SPEC_TYPE));
894  if (spec == NULL)
895  {
896  error = ER_FAILED;
897  goto cleanup;
898  }
899 
900  pruningset_iterator_init (pruned, &it);
901 
902  pos = 0;
903  for (i = 0, pos = pruningset_iterator_next (&it); i < cnt && pos >= 0; i++, pos = pruningset_iterator_next (&it))
904  {
905  COPY_OID (&spec[i].oid, &pinfo->partitions[pos + 1].class_oid);
906  HFID_COPY (&spec[i].hfid, &pinfo->partitions[pos + 1].class_hfid);
907 
908  if (i == cnt - 1)
909  {
910  spec[i].next = NULL;
911  }
912  else
913  {
914  spec[i].next = &spec[i + 1];
915  }
916 
917  if (is_index)
918  {
919  error = heap_get_index_with_name (pinfo->thread_p, &spec[i].oid, btree_name, &spec[i].btid);
920 
921  if (error != NO_ERROR)
922  {
923  goto cleanup;
924  }
925  }
926  }
927 
928 cleanup:
929  if (btree_name != NULL)
930  {
931  /* heap_get_indexinfo_of_btid calls strdup on the index name so we have to free it with free_and_init */
932  free_and_init (btree_name);
933  }
934 
935  pinfo->spec->curent = NULL;
936 
937  if (error != NO_ERROR)
938  {
939  if (spec != NULL)
940  {
941  db_private_free (pinfo->thread_p, spec);
942  }
943  pinfo->spec->parts = NULL;
944  }
945  else
946  {
947  pinfo->spec->parts = spec;
948  }
949 
950  return error;
951 }

953 /*
954  * partition_do_regu_variables_match () - check if two regu variables match
955  * return :
956  * pinfo (in) :
957  * left (in) :
958  * right (in) :
959  */
960 static bool
961 partition_do_regu_variables_match (PRUNING_CONTEXT * pinfo, const REGU_VARIABLE * left, const REGU_VARIABLE * right)
962 {
963  if (left == NULL || right == NULL)
964  {
965  return left == right;
966  }
967 
968  if (left->type != right->type)
969  {
970  return false;
971  }
972 
973  switch (left->type)
974  {
975  case TYPE_DBVAL:
976  /* use dbval */
977  if (tp_value_compare (&left->value.dbval, &right->value.dbval, 1, 0) != DB_EQ)
978  {
979  return false;
980  }
981  else
982  {
983  return true;
984  }
985 
986  case TYPE_CONSTANT:
987  /* use varptr */
988  if (tp_value_compare (left->value.dbvalptr, right->value.dbvalptr, 1, 1) != DB_EQ)
989  {
990  return false;
991  }
992  else
993  {
994  return true;
995  }
996 
997  case TYPE_POS_VALUE:
998  {
999  /* use val_pos for host variable references */
1000  DB_VALUE *val_left, *val_right;
1001 
1002  val_left = (DB_VALUE *) pinfo->vd->dbval_ptr + left->value.val_pos;
1003  val_right = (DB_VALUE *) pinfo->vd->dbval_ptr + right->value.val_pos;
1004 
1005  if (tp_value_compare (val_left, val_right, 1, 1) != DB_EQ)
1006  {
1007  return false;
1008  }
1009  else
1010  {
1011  return true;
1012  }
1013  }
1014 
1015  case TYPE_INARITH:
1016  case TYPE_OUTARITH:
1017  /* use arithptr */
1018  if (left->value.arithptr->opcode != right->value.arithptr->opcode)
1019  {
1020  /* different operations */
1021  return false;
1022  }
1023 
1024  /* check misc_operand for EXTRACT, etc */
1025  if (left->value.arithptr->misc_operand != right->value.arithptr->misc_operand)
1026  {
1027  return false;
1028  }
1029 
1030  /* check left operand */
1031  if (!partition_do_regu_variables_match (pinfo, left->value.arithptr->leftptr, right->value.arithptr->leftptr))
1032  {
1033  return false;
1034  }
1035 
1036  /* check right operand */
1037  if (!partition_do_regu_variables_match (pinfo, left->value.arithptr->rightptr, right->value.arithptr->rightptr))
1038  {
1039  return false;
1040  }
1041 
1042  /* check third operand */
1043  if (!partition_do_regu_variables_match (pinfo, left->value.arithptr->thirdptr, right->value.arithptr->thirdptr))
1044  {
1045  return false;
1046  }
1047 
1048  return true;
1049 
1050  case TYPE_ATTR_ID:
1051  /* use attr_descr */
1052  return left->value.attr_descr.id == right->value.attr_descr.id;
1053 
1054  default:
1055  return false;
1056  }
1057 
1058  return false;
1059 }

1061 /*
1062  * partition_prune_list () - Perform pruning for LIST type partitions
1063  * return : match status
1064  * pinfo (in) : pruning context
1065  * val (in) : the value to which the partition expression is compared
1066  * op (in) : operator to apply
1067  * pruned (in/out): pruned partitions
1068  */
1069 static MATCH_STATUS
1070 partition_prune_list (PRUNING_CONTEXT * pinfo, const DB_VALUE * val, const PRUNING_OP op, PRUNING_BITSET * pruned)
1071 {
1072  OR_PARTITION *part;
1073  int size = 0, i = 0, j;
1074  DB_SEQ *part_collection = NULL;
1075  DB_COLLECTION *val_collection = NULL;
1076  MATCH_STATUS status = MATCH_NOT_FOUND;
1077 
1078  for (i = 0; i < PARTITIONS_COUNT (pinfo); i++)
1079  {
1080  part = &pinfo->partitions[i + 1];
1081  part_collection = part->values;
1082 
1083  size = db_set_size (part_collection);
1084  if (size < 0)
1085  {
1086  return MATCH_NOT_FOUND;
1087  }
1088 
1089  switch (op)
1090  {
1091  case PO_IN:
1092  {
1093  DB_VALUE col;
1094  bool found = false;
1095 
1096  /* Perform intersection on part_values and val. If the result is not empty then this partition should be
1097  * added to the pruned list */
1098  if (!db_value_type_is_collection (val))
1099  {
1100  return MATCH_NOT_FOUND;
1101  }
1102 
1103  val_collection = db_get_set (val);
1104  for (j = 0; j < size; j++)
1105  {
1106  db_set_get (part_collection, j, &col);
1107  if (db_set_ismember (val_collection, &col))
1108  {
1109  pr_clear_value (&col);
1110  found = true;
1111  break;
1112  }
1113  pr_clear_value (&col);
1114  }
1115 
1116  if (found)
1117  {
1118  /* add this partition */
1119  pruningset_add (pruned, i);
1120  }
1121 
1122  status = MATCH_OK;
1123  break;
1124  }
1125 
1126  case PO_NOT_IN:
1127  {
1128  DB_VALUE col;
1129  bool found = false;
1130 
1131  /* Perform intersection on part_values and val. If the result is empty then this partition should be added
1132  * to the pruned list */
1133  if (!db_value_type_is_collection (val))
1134  {
1135  status = MATCH_NOT_FOUND;
1136  }
1137 
1138  val_collection = db_get_set (val);
1139  for (j = 0; j < size; j++)
1140  {
1141  db_set_get (part_collection, j, &col);
1142  if (db_set_ismember (val_collection, &col))
1143  {
1144  pr_clear_value (&col);
1145  found = true;
1146  break;
1147  }
1148  pr_clear_value (&col);
1149  }
1150 
1151  if (!found)
1152  {
1153  /* add this partition */
1154  pruningset_add (pruned, i);
1155  }
1156 
1157  status = MATCH_OK;
1158  break;
1159  }
1160 
1161  case PO_IS_NULL:
1162  /* Add this partition if part_values contains val. */
1163  if (db_set_has_null (part_collection))
1164  {
1165  /* add this partition */
1166  pruningset_add (pruned, i);
1167  /* Since partition values are disjoint sets this is the only possible result */
1168  return MATCH_OK;
1169  }
1170  break;
1171 
1172  case PO_EQ:
1173  /* Add this partition if part_values contains val. */
1174  if (db_set_ismember (part_collection, (DB_VALUE *) val))
1175  {
1176  /* add this partition */
1177  pruningset_add (pruned, i);
1178  /* Since partition values are disjoint sets this is the only possible result */
1179  return MATCH_OK;
1180  }
1181  break;
1182 
1183  case PO_NE:
1184  case PO_LT:
1185  case PO_LE:
1186  case PO_GT:
1187  case PO_GE:
1188  default:
1189  return status = MATCH_NOT_FOUND;
1190  }
1191  }
1192 
1193  return status;
1194 }

1196 /*
1197  * partition_prune_hash () - Perform pruning for HASH type partitions
1198  * return : match status
1199  * pinfo (in) : pruning context
1200  * val (in) : the value to which the partition expression is compared
1201  * op (in) : operator to apply
1202  * pruned (in/out): pruned partitions
1203  */
1204 static MATCH_STATUS
1205 partition_prune_hash (PRUNING_CONTEXT * pinfo, const DB_VALUE * val_p, const PRUNING_OP op, PRUNING_BITSET * pruned)
1206 {
1207  int idx = 0;
1208  int hash_size = PARTITIONS_COUNT (pinfo);
1209  TP_DOMAIN *col_domain = NULL;
1210  DB_VALUE val;
1211  MATCH_STATUS status = MATCH_NOT_FOUND;
1212 
1213  db_make_null (&val);
1214 
1215  col_domain = pinfo->partition_pred->func_regu->domain;
1216  switch (op)
1217  {
1218  case PO_EQ:
1219  if (TP_DOMAIN_TYPE (col_domain) != DB_VALUE_TYPE (val_p))
1220  {
1221  /* We have a problem here because type checking might not have coerced val to the type of the column. If this
1222  * is the case, we have to do it here */
1223  if (tp_value_cast (val_p, &val, col_domain, false) == DOMAIN_INCOMPATIBLE)
1224  {
1225  /* We cannot set an error here because this is not considered to be an error when scanning regular
1226  * tables. We can consider this predicate to be always false so status to MATCH_OK to simulate the case
1227  * when no appropriate partition was found. */
1228  er_clear ();
1229  status = MATCH_OK;
1230  }
1231  }
1232  else
1233  {
1234  pr_clone_value (val_p, &val);
1235  }
1236 
1237  idx = mht_get_hash_number (hash_size, &val);
1238 
1239  /* Start from 1 because we're using position 0 for the master class */
1240  pruningset_add (pruned, idx);
1241 
1242  status = MATCH_OK;
1243  break;
1244 
1245  case PO_IN:
1246  {
1247  DB_COLLECTION *values = NULL;
1248  int size = 0, i, idx;
1249 
1250  if (!db_value_type_is_collection (val_p))
1251  {
1252  /* This is an error and it should be handled outside of the pruning environment */
1253  status = MATCH_NOT_FOUND;
1254  goto cleanup;
1255  }
1256 
1257  values = db_get_set (val_p);
1258  size = db_set_size (values);
1259  if (size < 0)
1260  {
1261  pinfo->error_code = ER_FAILED;
1262  status = MATCH_NOT_FOUND;
1263  goto cleanup;
1264  }
1265 
1266  for (i = 0; i < size; i++)
1267  {
1268  DB_VALUE col;
1269 
1270  if (db_set_get (values, i, &col) != NO_ERROR)
1271  {
1272  pinfo->error_code = ER_FAILED;
1273  status = MATCH_NOT_FOUND;
1274  goto cleanup;
1275  }
1276 
1277  if (TP_DOMAIN_TYPE (col_domain) != DB_VALUE_TYPE (&col))
1278  {
1279  /* A failed coercion is not an error in this case, we should just skip over it */
1280  if (tp_value_cast (val_p, &val, col_domain, false) == DOMAIN_INCOMPATIBLE)
1281  {
1282  pr_clear_value (&col);
1283  er_clear ();
1284  continue;
1285  }
1286  }
1287 
1288  idx = mht_get_hash_number (hash_size, &col);
1289  pruningset_add (pruned, idx);
1290 
1291  pr_clear_value (&col);
1292  }
1293 
1294  status = MATCH_OK;
1295  break;
1296  }
1297 
1298  case PO_IS_NULL:
1299  /* first partition */
1300  pruningset_add (pruned, 0);
1301  status = MATCH_OK;
1302  break;
1303 
1304  default:
1305  status = MATCH_NOT_FOUND;
1306  break;
1307  }
1308 
1309 cleanup:
1310  pr_clear_value (&val);
1311 
1312  return status;
1313 }

1318  * pinfo (in) : pruning context
1319  * val(in) : the value to which the partition expression is compared
1320  * op (in) : operator to apply
1321  * pruned (in/out) : pruned partitions
1322  */
1323 static MATCH_STATUS
1324 partition_prune_range (PRUNING_CONTEXT * pinfo, const DB_VALUE * val, const PRUNING_OP op, PRUNING_BITSET * pruned)
1325 {
1326  int i = 0, error = NO_ERROR;
1327  int added = 0;
1328  OR_PARTITION *part;
1329  DB_VALUE min, max;
1330  int rmin = DB_UNK, rmax = DB_UNK;
1331  MATCH_STATUS status;
1332 
1333  db_make_null (&min);
1334  db_make_null (&max);
1335 
1336  for (i = 0; i < PARTITIONS_COUNT (pinfo); i++)
1337  {
1338  part = &pinfo->partitions[i + 1];
1339 
1340  pr_clear_value (&min);
1341  pr_clear_value (&max);
1342 
1343  error = db_set_get (part->values, 0, &min);
1344  if (error != NO_ERROR)
1345  {
1346  pinfo->error_code = error;
1347  status = MATCH_NOT_FOUND;
1348  goto cleanup;
1349  }
1350 
1351  error = db_set_get (part->values, 1, &max);
1352  if (error != NO_ERROR)
1353  {
1354  pinfo->error_code = error;
1355  status = MATCH_NOT_FOUND;
1356  goto cleanup;
1357  }
1358 
1359  if (DB_IS_NULL (&min))
1360  {
1361  /* MINVALUE */
1362  rmin = DB_LT;
1363  }
1364  else
1365  {
1366  rmin = tp_value_compare (&min, val, 1, 1);
1367  }
1368 
1369  if (DB_IS_NULL (&max))
1370  {
1371  /* MAXVALUE */
1372  rmax = DB_LT;
1373  }
1374  else
1375  {
1376  if (op == PO_GT)
1377  {
1378  /* Partitioning interval is stored as [min, max). Try to convert it to [min, max--] in order to handle
1379  * some limit cases like val > max-- which should not match any partition */
1380  (void) partition_decrement_value (&max);
1381  }
1382  rmax = tp_value_compare (val, &max, 1, 1);
1383  }
1384 
1385  status = MATCH_OK;
1386  switch (op)
1387  {
1388  case PO_EQ:
1389  /* Filter is part_expr = value. Find the *only* partition for which min <= value < max */
1390  if ((rmin == DB_EQ || rmin == DB_LT) && rmax == DB_LT)
1391  {
1392  pruningset_add (pruned, i);
1393  ++added;
1394  /* no need to look any further */
1395  goto cleanup;
1396  }
1397  break;
1398 
1399  case PO_LT:
1400  /* Filter is part_expr < value. All partitions for which min < value qualify */
1401  if (rmin == DB_LT)
1402  {
1403  pruningset_add (pruned, i);
1404  ++added;
1405  }
1406  break;
1407 
1408  case PO_LE:
1409  /* Filter is part_expr <= value. All partitions for which min <= value qualify */
1410  if (rmin == DB_EQ)
1411  {
1412  /* this is the only partition than can qualify */
1413  pruningset_add (pruned, i);
1414  ++added;
1415  goto cleanup;
1416  }
1417  else if (rmin == DB_LT)
1418  {
1419  pruningset_add (pruned, i);
1420  ++added;
1421  }
1422  break;
1423 
1424  case PO_GT:
1425  /* Filter is part_expr > value. All partitions for which value < max qualify */
1426  if (rmax == DB_LT)
1427  {
1428  pruningset_add (pruned, i);
1429  ++added;
1430  }
1431  break;
1432 
1433  case PO_GE:
1434  /* Filter is part_expr > value. All partitions for which value < max qualify */
1435  if (rmax == DB_LT)
1436  {
1437  pruningset_add (pruned, i);
1438  ++added;
1439  }
1440  break;
1441 
1442  case PO_IS_NULL:
1443  if (DB_IS_NULL (&min))
1444  {
1445  pruningset_add (pruned, i);
1446  ++added;
1447  /* no need to look any further */
1448  goto cleanup;
1449  }
1450  break;
1451 
1452  default:
1453  status = MATCH_NOT_FOUND;
1454  goto cleanup;
1455  }
1456  }
1457 
1458 cleanup:
1459  pr_clear_value (&min);
1460  pr_clear_value (&max);
1461 
1462  if (status == MATCH_OK && added == 0)
1463  {
1464  status = MATCH_NOT_FOUND;
1465  }
1466 
1467  return status;
1468 }

1813 /*
1814  * partition_match_pred_expr () - get partitions matching a predicate
1815  * expression
1816  * return : match status
1817  * pinfo (in) : pruning context
1818  * pr (in) : predicate expression
1819  * pruned (in/out): pruned partitions
1820  */
1821 static MATCH_STATUS
1822 partition_match_pred_expr (PRUNING_CONTEXT * pinfo, const PRED_EXPR * pr, PRUNING_BITSET * pruned)
1823 {
1824  MATCH_STATUS status = MATCH_NOT_FOUND;
1825  REGU_VARIABLE *part_expr = pinfo->partition_pred->func_regu;
1826 
1827  if (pr == NULL)
1828  {
1829  assert (false);
1830  return MATCH_NOT_FOUND;
1831  }
1832 
1833  switch (pr->type)
1834  {
1835  case T_PRED:
1836  {
1837  /* T_PRED contains conjunctions and disjunctions of PRED_EXPR. Get partitions matching the left PRED_EXPR and
1838  * partitions matching the right PRED_EXPR and merge or intersect the lists */
1839  PRUNING_BITSET left_set, right_set;
1840  MATCH_STATUS lstatus, rstatus;
1841  BOOL_OP op = pr->pe.pred.bool_op;
1842 
1843  if (op != B_AND && op != B_OR)
1844  {
1845  /* only know how to handle AND/OR predicates */
1846  status = MATCH_NOT_FOUND;
1847  break;
1848  }
1849 
1850  pruningset_init (&left_set, PARTITIONS_COUNT (pinfo));
1851  pruningset_init (&right_set, PARTITIONS_COUNT (pinfo));
1852 
1853  lstatus = partition_match_pred_expr (pinfo, pr->pe.pred.lhs, &left_set);
1854  rstatus = partition_match_pred_expr (pinfo, pr->pe.pred.rhs, &right_set);
1855 
1856  if (op == B_AND)
1857  {
1858  /* do intersection between left and right */
1859  if (lstatus == MATCH_NOT_FOUND)
1860  {
1861  /* pr->pe.pred.lhs does not refer part_expr so return right */
1862  pruningset_copy (pruned, &right_set);
1863  status = rstatus;
1864  }
1865  else if (rstatus == MATCH_NOT_FOUND)
1866  {
1867  /* pr->pe.pred.rhs does not refer part_expr so return right */
1868  pruningset_copy (pruned, &left_set);
1869  status = lstatus;
1870  }
1871  else
1872  {
1873  status = MATCH_OK;
1874  pruningset_intersect (&left_set, &right_set);
1875  pruningset_copy (pruned, &left_set);
1876  }
1877  }
1878  else
1879  {
1880  /* this is the OR operator so union the two sets */
1881  if (lstatus == MATCH_NOT_FOUND || rstatus == MATCH_NOT_FOUND)
1882  {
1883  status = MATCH_NOT_FOUND;
1884  }
1885  else
1886  {
1887  pruningset_union (&left_set, &right_set);
1888  pruningset_copy (pruned, &left_set);
1889 
1890  status = MATCH_OK;
1891  }
1892  }
1893  break;
1894  }
1895 
1896  case T_EVAL_TERM:
1897  switch (pr->pe.eval_term.et_type)
1898  {
1899  case T_COMP_EVAL_TERM:
1900  {
1901  /* see if part_expr matches right or left */
1902  REGU_VARIABLE *left, *right;
1903  PRUNING_OP op;
1904 
1905  left = pr->pe.eval_term.et.et_comp.lhs;
1906  right = pr->pe.eval_term.et.et_comp.rhs;
1907  op = partition_rel_op_to_pruning_op (pr->pe.eval_term.et.et_comp.rel_op);
1908 
1909  status = MATCH_NOT_FOUND;
1910  if (partition_do_regu_variables_match (pinfo, left, part_expr))
1911  {
1912  status = partition_prune (pinfo, right, op, pruned);
1913  }
1914  else if (partition_do_regu_variables_match (pinfo, right, part_expr))
1915  {
1916  status = partition_prune (pinfo, left, op, pruned);
1917  }
1918  break;
1919  }
1920 
1921  case T_ALSM_EVAL_TERM:
1922  {
1923  REGU_VARIABLE *regu, *list;
1924  PRUNING_OP op;
1925 
1926  regu = pr->pe.eval_term.et.et_alsm.elem;
1927  list = pr->pe.eval_term.et.et_alsm.elemset;
1928  op = partition_rel_op_to_pruning_op (pr->pe.eval_term.et.et_alsm.rel_op);
1929  /* adjust rel_op based on the QL_FLAG of the alsm eval node */
1930  if (pr->pe.eval_term.et.et_alsm.eq_flag == F_SOME)
1931  {
1932  if (op == PO_EQ)
1933  {
1934  op = PO_IN;
1935  }
1936  else if (op == PO_NE)
1937  {
1938  op = PO_NOT_IN;
1939  }
1940  }
1941  if (partition_do_regu_variables_match (pinfo, regu, part_expr))
1942  {
1943  status = partition_prune (pinfo, list, op, pruned);
1944  }
1945  }
1946  break;
1947 
1948  case T_LIKE_EVAL_TERM:
1949  case T_RLIKE_EVAL_TERM:
1950  /* Don't know how to work with LIKE/RLIKE expressions yet. There are some cases in which we can deduce a
1951  * range from the like pattern. */
1952  status = MATCH_NOT_FOUND;
1953  break;
1954  }
1955  break;
1956 
1957  case T_NOT_TERM:
1958  status = MATCH_NOT_FOUND;
1959  break;
1960  }
1961 
1962  return status;
1963 }

1965 /*
1966  * partition_match_key_range () - perform pruning using a key_range
1967  * return : pruned list
1968  * pinfo (in) : pruning context
1969  * partitions (in) : partitions to prune
1970  * key_range (in) : key range
1971  * status (in/out) : pruning status
1972  */
1973 static MATCH_STATUS
1974 partition_match_key_range (PRUNING_CONTEXT * pinfo, const KEY_RANGE * key_range, PRUNING_BITSET * pruned)
1975 {
1976  PRUNING_OP lop = PO_INVALID, rop = PO_INVALID;
1977  MATCH_STATUS lstatus, rstatus;
1978  PRUNING_BITSET left, right;
1979 
1980  switch (key_range->range)
1981  {
1982  case NA_NA:
1983  /* v1 and v2 are N/A, so that no range is defined */
1984  return MATCH_NOT_FOUND;
1985 
1986  case GE_LE:
1987  /* v1 <= key <= v2 */
1988  lop = PO_GE;
1989  rop = PO_LE;
1990  break;
1991 
1992  case GE_LT:
1993  /* v1 <= key < v2 */
1994  lop = PO_GE;
1995  rop = PO_LT;
1996  break;
1997 
1998  case GT_LE:
1999  /* v1 < key <= v2 */
2000  lop = PO_GT;
2001  rop = PO_LE;
2002  break;
2003 
2004  case GT_LT:
2005  /* v1 < key < v2 */
2006  lop = PO_GT;
2007  rop = PO_LT;
2008  break;
2009 
2010  case GE_INF:
2011  /* v1 <= key (<= the end) */
2012  lop = PO_GE;
2013  rop = PO_INVALID;
2014  break;
2015 
2016  case GT_INF:
2017  /* v1 < key (<= the end) */
2018  lop = PO_GT;
2019  rop = PO_INVALID;
2020  break;
2021 
2022  case INF_LE:
2023  /* (the beginning <=) key <= v2 */
2024  lop = PO_INVALID;
2025  rop = PO_LE;
2026  break;
2027 
2028  case INF_LT:
2029  /* (the beginning <=) key < v2 */
2030  lop = PO_INVALID;
2031  rop = PO_LT;
2032  break;
2033 
2034  case INF_INF:
2035  /* the beginning <= key <= the end */
2036  lop = PO_INVALID;
2037  rop = PO_INVALID;
2038  break;
2039 
2040  case EQ_NA:
2041  /* key = v1, v2 is N/A */
2042  lop = PO_EQ;
2043  rop = PO_INVALID;
2044  break;
2045 
2046  case LE_GE:
2047  case LE_GT:
2048  case LT_GE:
2049  case LT_GT:
2050  case NEQ_NA:
2051  lop = PO_INVALID;
2052  rop = PO_INVALID;
2053  break;
2054  }
2055 
2056  pruningset_init (&left, PARTITIONS_COUNT (pinfo));
2057  pruningset_init (&right, PARTITIONS_COUNT (pinfo));
2058 
2059  /* prune left */
2060  if (lop == PO_INVALID)
2061  {
2062  lstatus = MATCH_NOT_FOUND;
2063  }
2064  else
2065  {
2066  lstatus = partition_prune (pinfo, key_range->key1, lop, &left);
2067  if (pinfo->error_code != NO_ERROR)
2068  {
2069  return MATCH_NOT_FOUND;
2070  }
2071  }
2072 
2073  /* prune right */
2074  if (rop == PO_INVALID)
2075  {
2076  rstatus = MATCH_NOT_FOUND;
2077  }
2078  else
2079  {
2080  rstatus = partition_prune (pinfo, key_range->key2, rop, &right);
2081  if (pinfo->error_code != NO_ERROR)
2082  {
2083  return MATCH_NOT_FOUND;
2084  }
