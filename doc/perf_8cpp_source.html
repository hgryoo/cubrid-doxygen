71  statset_definition::statset_definition (std::initializer_list<stat_definition> defs)
72  : m_stat_count (defs.size ())
73  , m_value_count (0)
74  , m_stat_defs (NULL)
75  , m_value_names (NULL)
76  {
77  // copy definitions
78  m_stat_defs = new stat_definition[defs.size ()];
79  std::size_t stat_index = 0;
80  for (auto def_it : defs)
81  {
82  if (def_it.m_id != stat_index)
83  {
84  // statset_definition is bad; crash program
85  throw std::runtime_error ("statset_definition is bad");
86  }
87  m_stat_defs[stat_index] = def_it; // copy definitions
88 
89  // set offset and increment value count
90  m_stat_defs[stat_index].m_offset = m_value_count;
91  m_value_count += def_it.get_value_count ();
92 
93  // increment index
94  stat_index++;
95  }
96 
97  // names for all values
98  m_value_names = new std::string[m_value_count];
99  std::size_t value_index = 0;
100  for (stat_index = 0; stat_index < m_stat_count; stat_index++)
101  {
102  assert (value_index == m_stat_defs[stat_index].m_offset);
103  for (std::size_t def_name_index = 0; def_name_index < m_stat_defs[stat_index].get_value_count ();
104  def_name_index++)
105  {
106  m_value_names[value_index++] = m_stat_defs[stat_index].m_names[def_name_index];
107  }
108  }
109  }
