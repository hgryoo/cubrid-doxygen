49 connection_object_init(Connection_Object *self, PyObject *args, PyObject *kwargs)
50 {
51  static char *kwList[] = {"host", "port", "db", "user", "password", NULL};
52  char *host = "localhost";
53  int port = 30000;
54  char *db = "demodb";
55  char *user = "public";
56  char *passwd = "";
57  int res, res2;
58  char db_ver[16];
59  T_CCI_ERROR error;
60 
61  if (!PyArg_ParseTupleAndKeywords(
62  args, kwargs, "|sisss", kwList, &host, &port, &db, &user, &passwd)) {
63  return -1;
64  }
65 
66  self->handle = 0;
67  self->host[0] = '\0';
68  self->port = 0;
69  self->db[0] = '\0';
70  self->user[0] = '\0';
71 
72  res = cci_connect(host, port, db, user, passwd);
73 
74  if (res < 0) {
75  handle_error(res, NULL);
76  return -1;
77  }
78 
79  res2 = cci_get_db_version(res, db_ver, sizeof(db_ver));
80  if (res2 < 0) {
81  cci_disconnect(res, &error);
82  handle_error(res2, NULL);
83  return -1;
84  }
85 
86  res2 = cci_end_tran(res, CCI_TRAN_COMMIT, &error);
87  if (res2 < 0) {
88  cci_disconnect(res, &error);
89  handle_error(res2, &error);
90  return -1;
91  }
92 
93  self->handle = res;
94  strcpy(self->host, host);
95  self->port = port;
96  strcpy(self->db, db);
97  strcpy(self->user, user);
