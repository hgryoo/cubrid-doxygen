117  if (cubrid_err_msgs[i].err == err_code) {

120  }
121  }

125 PyObject *
126 handle_error(int e, T_CCI_ERROR *error)
127 {
128  PyObject *t;
129  int err_code;
130  char msg[1024];
131  char *err_msg = NULL, *facility_msg;
132 
133  if (e == CCI_ER_DBMS) {
134  facility_msg = "DBMS";
135  if (error) {
136  err_code = error->err_code;
137  err_msg = error->err_msg;
138  } else {
139  err_code = 0;
140  err_msg = "Unknown DBMS Error";
141  }
142  } else {
143  if (get_error_msg(e, &err_msg) < 0) {
144  strcpy(err_msg, "Unknown Error");
145  }
146  err_code = e;
147 
148  if (e > -1000) {
149  facility_msg = "CCI";
150  } else if (e > -2000) {
151  facility_msg = "CAS";
152  } else if (e > -2000) {
153  facility_msg = "CLIENT";
154  } else {
155  facility_msg = "UNKNOWN";
156  }
157  }
158 
159  sprintf(msg, "ERROR: %s, %d, %s", facility_msg, err_code, err_msg);
160 
161  if (!(t = PyTuple_New(2))) return NULL;
162 
163  PyTuple_SetItem(t, 0, PyInt_FromLong((long)e));
164  PyTuple_SetItem(t, 1, PyString_FromString(msg));
165 
166  PyErr_SetObject(cubrid_error, t);
167  Py_DECREF(t);
168 
169  return NULL;
170 }
