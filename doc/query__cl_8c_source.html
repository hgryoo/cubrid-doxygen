38 /*
39  * prepare_query () - Prepares a query for later (and repetitive)
40  * execution
41  * return : Error code
42  * context (in) : query string; used for hash key of the XASL cache
43  * stream (in/out) : XASL stream, size, xasl_id & xasl_header;
44  * set to NULL if you want to look up the XASL cache
45  *
46  * NOTE: If stream->xasl_header is not NULL, also XASL node header will be
47  * requested from server.
48  */
49 int
50 prepare_query (COMPILE_CONTEXT * context, XASL_STREAM * stream)
51 {
52  int ret = NO_ERROR;
53 
54  assert (context->sql_hash_text);
55 
56  /* if QO_PARAM_LEVEL indicate no execution, just return */
57  if (qo_need_skip_execution ())
58  {
59  return NO_ERROR;
60  }
61 
62  /* allocate XASL_ID, the caller is responsible to free this */
63  stream->xasl_id = (XASL_ID *) malloc (sizeof (XASL_ID));
64  if (stream->xasl_id == NULL)
65  {
66  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (XASL_ID));
67  return ER_OUT_OF_VIRTUAL_MEMORY;
68  }
69 
70  /* send XASL stream to the server and get XASL_ID */
71  ret = qmgr_prepare_query (context, stream);
72  if (ret != NO_ERROR)
73  {
74  free_and_init (stream->xasl_id);
75  ASSERT_ERROR ();
76  return ret;
77  }
78 
79  /* if the query is not found in the cache */
80  if (stream->buffer == NULL && stream->xasl_id && XASL_ID_IS_NULL (stream->xasl_id))
81  {
82  free_and_init (stream->xasl_id);
83  }
84 
85  assert (ret == NO_ERROR);
86 
87  return ret;
88 }

90 /*
91  * execute_query () - Execute a prepared query
92  * return: Error code
93  * xasl_id(in) : XASL file id that was a result of prepare_query()
94  * query_idp(out) : query id to be used for getting results
95  * var_cnt(in) : number of host variables
96  * varptr(in) : array of host variables (query input parameters)
97  * list_idp(out) : query result file id (QFILE_LIST_ID)
98  * flag(in) : flag
99  * clt_cache_time(in) :
100  * srv_cache_time(in) :
101  */
102 int
103 execute_query (const XASL_ID * xasl_id, QUERY_ID * query_idp, int var_cnt, const DB_VALUE * varptr,
104  QFILE_LIST_ID ** list_idp, QUERY_FLAG flag, CACHE_TIME * clt_cache_time, CACHE_TIME * srv_cache_time)
105 {
106  int query_timeout;
107  int ret = NO_ERROR;
108 
109  *list_idp = NULL;
110 
111  /* if QO_PARAM_LEVEL indicate no execution, just return */
112  if (qo_need_skip_execution ())
113  {
114  return NO_ERROR;
115  }
116 
117  query_timeout = tran_get_query_timeout ();
118  /* send XASL file id and host variables to the server and get QFILE_LIST_ID */
119  *list_idp =
120  qmgr_execute_query (xasl_id, query_idp, var_cnt, varptr, flag, clt_cache_time, srv_cache_time, query_timeout);
121 
122  if (*list_idp == NULL)
123  {
124  return ((ret = er_errid ()) == NO_ERROR) ? ER_FAILED : ret;
125  }
126 
127  assert (ret == NO_ERROR);
128 
129  return ret;
130 }
