219 struct qo_info
220 {
221  struct qo_info *next;
222 
223  /*
224  * The environment relative to which all of the following sets, etc.
225  * make sense.
226  */
227  QO_ENV *env;
228 
229  /*
230  * The Planner instance to which this Info node belongs. I wish
231  * we didn't have to do this, but there are just enough occassions
232  * where we need the back pointer that it is easier just to include
233  * it and be done with it. This pointer is NULL if this info node
234  * has been "detached" from the planner; this happens when the
235  * winning plan is "finalized".
236  */
237  QO_PLANNER *planner;
238 
239  /*
240  * The lowest-cost plan without regard to ordering.
241  */
242  QO_PLANVEC best_no_order;
243 
244  /*
245  * The set of nodes joined by the plans at this node.
246  */
247  BITSET nodes;
248 
249  /*
250  * All of the terms accounted for by the plans in this node and their
251  * descendents, i.e., the complement of the terms remaining to be
252  * dealt with.
253  */
254  BITSET terms;
255 
256  /*
257  * The equivalence classes represented by all of the attributes
258  * (segments) joined together in this node.
259  */
260  BITSET eqclasses;
261 
262  /*
263  * 'projected_segs' is the set of segments (attributes) that need to
264  * be projected from this plan produced by this node in order to
265  * satisfy the needs of upper level plans. 'projected_size' is the
266  * number of bytes per record required by 'projected_segs'.
267  * 'cardinality' is the estimated cardinality of the results produced
268  * by plans at this node.
269  */
270  BITSET projected_segs;
271  double cardinality;
272 
273  /*
274  * One plan for each equivalence class, in each case the best we have
275  * seen so far. This vector is NULL after a node is detached.
276  */
277  QO_PLANVEC *planvec;
278 
279  int projected_size;
280 
281  /*
282  * The last join level.
283  */
284  int join_unit;
285 
286  /*
287  * `detached' is true iff the node has been detached; we can no
288  * longer just use the value of `plans' as the indicator because
289  * dependent derived tables can give rise to join graphs that couple
290  * nodes together without creating an equivalence class, and thus
291  * without the need to populate the `plans' vector.
292  */
293  int detached;
294 };

296 struct qo_planner
297 {
298  /*
299  * The struct that encapsulates the information involved in searching
300  * for an optimal query plan.
301  */
302 
303  /*
304  * The environment that supplies the various nodes, edges, segments, etc.
305  */
306  QO_ENV *env;
307 
308  /*
309  * The relations being considered in this join; there are N of them.
310  */
311  QO_NODE *node;
312 
313  /*
314  * The join terms (e.g., employee.dno = dept.dno); there are T of
315  * them, E of which are actual edges in the join graph. node_mask is
316  * a bit mask used to mask out non-node terms from node bitsets.
317  * M is simply 2^N, and is the size of the join_info vector that will
318  * be allocated.
319  */
320  QO_TERM *term;
321  unsigned int N;
322  unsigned int E, M, T;
323  unsigned long node_mask;
324 
325  /*
326  * The path segments involved in the various join terms, and the
327  * equivalence classes implied by those joins (e.g., if we have join
328  * terms c1 = c2 and c2 = c3, (c1,c2,c3) is an equivalence class for
329  * the purposes of determining sort orderings).
330  */
331  QO_SEGMENT *segment;
332 
333  QO_EQCLASS *eqclass;
334 
335  /*
336  * The partitions (strong components) of the join graph.
337  */
338  QO_PARTITION *partition;
339 
340  /*
341  * The last join level.
342  */
343  int join_unit;
344  unsigned int S;
345  unsigned int EQ;
346  unsigned int P;
347 
348  /*
349  * The (level-1 correlated) subqueries used in this query.
350  */
351  QO_SUBQUERY *subqueries;
352  BITSET all_subqueries;
353  unsigned int Q;
354 
355  /*
356  * The final set of segments to be projected out of the top-level
357  * plan produced by this planner.
358  */
359  BITSET final_segs;
360 
361 
362  QO_INFO **node_info;
363  QO_INFO **join_info;
364  QO_INFO **cp_info;
365  QO_INFO *best_info;
366 
367  QO_PLAN *worst_plan;
368  QO_INFO *worst_info;
369 
370  /* alloced info list */
371  QO_INFO *info_list;
372 
373  /*
374  * true iff qo_planner_cleanup() needs to be called before freeing
375  * this planner. This is needed to help clean up after aborts, when
376  * control flow takes an unexpected longjmp.
377  */
378  int cleanup_needed;
379 };
