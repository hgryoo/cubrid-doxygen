140 /*
141  * repl_log_info_alloc - log info area allocation
142  *
143  * return: Error Code
144  *
145  * tdes(in): transaction descriptor
146  * arr_size(in): array size to be allocated
147  * need_realloc(in): 0-> initial allocation (malloc), otherwise "realloc"
148  *
149  * NOTE:This function allocates the memory for the log info area of the
150  * target transaction. It is called when the transaction tries to to
151  * insert/update/delete operation. If the transaction do read operation
152  * only, no memory is allocated.
153  *
154  * The allocation size is defined by a constant - REPL_LOG_INFO_ALLOC_SIZE
155  * We need to set the size for the user request ?
156  */
157 static int
158 repl_log_info_alloc (LOG_TDES * tdes, int arr_size, bool need_realloc)
159 {
160  int i = 0, k;
161  int error = NO_ERROR;
162 
163  if (need_realloc == false)
164  {
165  i = arr_size * DB_SIZEOF (LOG_REPL_RECORD);
166  tdes->repl_records = (LOG_REPL_RECORD *) malloc (i);
167  if (tdes->repl_records == NULL)
168  {
169  error = ER_REPL_ERROR;
170  er_set (ER_WARNING_SEVERITY, ARG_FILE_LINE, ER_REPL_ERROR, 1, "can't allocate memory");
171  return error;
172  }
173  tdes->num_repl_records = arr_size;
174  k = 0;
175  }
176  else
177  {
178  i = tdes->num_repl_records + arr_size;
179  tdes->repl_records = (LOG_REPL_RECORD *) realloc (tdes->repl_records, i * DB_SIZEOF (LOG_REPL_RECORD));
180  if (tdes->repl_records == NULL)
181  {
182  error = ER_REPL_ERROR;
183  er_set (ER_WARNING_SEVERITY, ARG_FILE_LINE, ER_REPL_ERROR, 1, "can't allocate memory");
184  return error;
185  }
186  k = tdes->num_repl_records;
187  tdes->num_repl_records = i;
188  }
189 
190  for (i = k; i < tdes->num_repl_records; i++)
191  {
192  tdes->repl_records[i].repl_data = NULL;
193  }
194 
195  return error;
196 }

198 /*
199  * repl_add_update_lsa - update the LSA of the target transaction lo
200  *
201  * return: NO_ERROR or error code
202  *
203  * inst_oid(in): OID of the instance
204  *
205  * NOTE:For update operation, the server does "Heap stuff" after "Index Stuff".
206  * In order to reduce the cost of replication log, we generates a
207  * replication log at the point of indexing processing step.
208  * (During index processing, the primary key value is fetched ..)
209  * After Heap operation, we have to set the target LSA into the
210  * replication log.
211  *
212  * For update case, this function is called by locator_update_force().
213  * In the case of insert/delete cases, when the replication log info. is
214  * generated, the server already has the target transaction log(HEAP_INSERT
215  * or HEAP_DELETE).
216  * But, for the udpate case, the server doesn't has the target log when
217  * it generates the replication log. So, the server has to find out the
218  * location of replication record and match with the target transaction
219  * log after heap_update(). This is done by locator_update_force().
220  */
221 int
222 repl_add_update_lsa (THREAD_ENTRY * thread_p, const OID * inst_oid)
223 {
224  int tran_index;
225  LOG_TDES *tdes;
226  LOG_REPL_RECORD *repl_rec;
227  int i;
228  bool find = false;
229  int error = NO_ERROR;
230 
231  tran_index = LOG_FIND_THREAD_TRAN_INDEX (thread_p);
232 
233  tdes = LOG_FIND_TDES (tran_index);
234  if (tdes == NULL)
235  {
236  return ER_FAILED;
237  }
238 
239  /* If suppress_replication flag is set, do not write replication log. */
240  if (tdes->suppress_replication != 0)
241  {
242  return NO_ERROR;
243  }
244 
245  for (i = tdes->cur_repl_record - 1; i >= 0; i--)
246  {
247  repl_rec = (LOG_REPL_RECORD *) (&tdes->repl_records[i]);
248  if (OID_EQ (&repl_rec->inst_oid, inst_oid) && !LSA_ISNULL (&tdes->repl_update_lsa))
249  {
250  assert (repl_rec->rcvindex == RVREPL_DATA_UPDATE || repl_rec->rcvindex == RVREPL_DATA_UPDATE_START
251  || repl_rec->rcvindex == RVREPL_DATA_UPDATE_END);
252  if (repl_rec->rcvindex == RVREPL_DATA_UPDATE || repl_rec->rcvindex == RVREPL_DATA_UPDATE_START
253  || repl_rec->rcvindex == RVREPL_DATA_UPDATE_END)
254  {
255  LSA_COPY (&repl_rec->lsa, &tdes->repl_update_lsa);
256  LSA_SET_NULL (&tdes->repl_update_lsa);
257  LSA_SET_NULL (&tdes->repl_insert_lsa);
258  find = true;
259  break;
260  }
261  }
262  }
263 
264  if (find == false)
265  {
266  er_log_debug (ARG_FILE_LINE, "can't find out the UPDATE LSA");
267  }
268 
269  return error;
270 }

272 /*
273  * repl_log_insert - insert a replication info to the transaction descriptor
274  *
275  * return: NO_ERROR or error code
276  *
277  * class_oid(in): OID of the class
278  * inst_oid(in): OID of the instance
279  * log_type(in): log type (DATA or SCHEMA)
280  * rcvindex(in): recovery index (INSERT or DELETE or UPDATE)
281  * key_dbvalue(in): Primary Key value
282  *
283  * NOTE:insert a replication log info to the transaction descriptor (tdes)
284  */
285 int
286 repl_log_insert (THREAD_ENTRY * thread_p, const OID * class_oid, const OID * inst_oid, LOG_RECTYPE log_type,
287  LOG_RCVINDEX rcvindex, DB_VALUE * key_dbvalue, REPL_INFO_TYPE repl_info)
288 {
289  int tran_index;
290  LOG_TDES *tdes;
291  LOG_REPL_RECORD *repl_rec;
292  char *class_name;
293  char *ptr;
294  int error = NO_ERROR, strlen;
295 
296  tran_index = LOG_FIND_THREAD_TRAN_INDEX (thread_p);
297  tdes = LOG_FIND_TDES (tran_index);
298  if (tdes == NULL)
299  {
300  return ER_FAILED;
301  }
302 
303  /* If suppress_replication flag is set, do not write replication log. */
304  if (tdes->suppress_replication != 0)
305  {
306  /* clear repl lsa in tdes since no replication log will be written */
307  LSA_SET_NULL (&tdes->repl_insert_lsa);
308  LSA_SET_NULL (&tdes->repl_update_lsa);
309 
310  return NO_ERROR;
311  }
312 
313  /* check the replication log array status, if we need to alloc? */
314  if (REPL_LOG_IS_NOT_EXISTS (tran_index)
315  && ((error = repl_log_info_alloc (tdes, REPL_LOG_INFO_ALLOC_SIZE, false)) != NO_ERROR))
316  {
317  return error;
318  }
319  /* the replication log array is full? re-alloc? */
320  else if (REPL_LOG_IS_FULL (tran_index)
321  && (error = repl_log_info_alloc (tdes, REPL_LOG_INFO_ALLOC_SIZE, true)) != NO_ERROR)
322  {
323  return error;
324  }
325 
326  repl_rec = (LOG_REPL_RECORD *) (&tdes->repl_records[tdes->cur_repl_record]);
327  repl_rec->repl_type = log_type;
328 
329  repl_rec->rcvindex = rcvindex;
330  if (rcvindex == RVREPL_DATA_UPDATE)
331  {
332  switch (repl_info)
333  {
334  case REPL_INFO_TYPE_RBR_START:
335  repl_rec->rcvindex = RVREPL_DATA_UPDATE_START;
336  break;
337  case REPL_INFO_TYPE_RBR_END:
338  repl_rec->rcvindex = RVREPL_DATA_UPDATE_END;
339  break;
340  case REPL_INFO_TYPE_RBR_NORMAL:
341  default:
342  break;
343  }
344 
345  }
346  COPY_OID (&repl_rec->inst_oid, inst_oid);
347 
348  /* make the common info for the data replication */
349  if (log_type == LOG_REPLICATION_DATA)
350  {
351  char *ptr_to_packed_key_value_size = NULL;
352  int packed_key_len = 0;
353 
354  if (heap_get_class_name (thread_p, class_oid, &class_name) != NO_ERROR || class_name == NULL)
355  {
356  ASSERT_ERROR_AND_SET (error);
357  if (error == NO_ERROR)
358  {
359  error = ER_REPL_ERROR;
360  er_set (ER_WARNING_SEVERITY, ARG_FILE_LINE, ER_REPL_ERROR, 1, "can't get class_name");
361  }
362  return error;
363  }
364 
365  repl_rec->length = OR_INT_SIZE; /* packed_key_value_size */
366  repl_rec->length += or_packed_string_length (class_name, &strlen);
367  repl_rec->length += OR_VALUE_ALIGNED_SIZE (key_dbvalue);
368 
369  ptr = (char *) malloc (repl_rec->length);
370  if (ptr == NULL)
371  {
372  error = ER_REPL_ERROR;
373  er_set (ER_WARNING_SEVERITY, ARG_FILE_LINE, ER_REPL_ERROR, 1, "can't allocate memory");
374  free_and_init (class_name);
375  return error;
376  }
377 
378 #if !defined (NDEBUG)
379  /* Suppress valgrind complaint. */
380  memset (ptr, 0, repl_rec->length);
381 #endif // DEBUG
382 
383  repl_rec->repl_data = ptr;
384 
385  /* first 4 bytes are for packed_key_len which will be filled after packing the value. */
386  ptr_to_packed_key_value_size = ptr;
387  ptr += OR_INT_SIZE;
388 
389  ptr = or_pack_string_with_length (ptr, class_name, strlen);
390  ptr = or_pack_mem_value (ptr, key_dbvalue, &packed_key_len);
391 
392  /* fill the length of disk image of pk */
393  or_pack_int (ptr_to_packed_key_value_size, packed_key_len);
394 
395  free_and_init (class_name);
396  }
397  else
398  {
399  repl_rec->repl_data = NULL;
400  repl_rec->length = 0;
401  }
402  repl_rec->must_flush = LOG_REPL_COMMIT_NEED_FLUSH;
403 
404  switch (rcvindex)
405  {
406  case RVREPL_DATA_INSERT:
407  if (!LSA_ISNULL (&tdes->repl_insert_lsa))
408  {
409  LSA_COPY (&repl_rec->lsa, &tdes->repl_insert_lsa);
410  LSA_SET_NULL (&tdes->repl_insert_lsa);
411  LSA_SET_NULL (&tdes->repl_update_lsa);
412  }
413  break;
414  case RVREPL_DATA_UPDATE:
415  /*
416  * for the update case, this function is called before the heap
417  * file update, so we don't need to LSA for update log here.
418  */
419  LSA_SET_NULL (&repl_rec->lsa);
420  break;
421  case RVREPL_DATA_DELETE:
422  /*
423  * for the delete case, we don't need to find out the target
424  * LSA. Delete is operation is possible without "After Image"
425  */
426  if (LSA_ISNULL (&tdes->tail_lsa))
427  {
428  LSA_COPY (&repl_rec->lsa, &log_Gl.prior_info.prior_lsa);
429  }
430  else
431  {
432  LSA_COPY (&repl_rec->lsa, &tdes->tail_lsa);
433  }
434  break;
435  default:
436  break;
437  }
438  tdes->cur_repl_record++;
439 
440  /* if flush marking is started, mark "must_flush" at current log except the log conflicts with previous logs due to
441  * same instance update */
442  if (tdes->fl_mark_repl_recidx != -1)
443  {
444  LOG_REPL_RECORD *recsp = tdes->repl_records;
445  int i;
446 
447  for (i = 0; i < tdes->fl_mark_repl_recidx; i++)
448  {
449  if (recsp[i].must_flush == LOG_REPL_COMMIT_NEED_FLUSH && OID_EQ (&recsp[i].inst_oid, &repl_rec->inst_oid))
450  {
451  break;
452  }
453  }
454 
455  if (i >= tdes->fl_mark_repl_recidx)
456  {
457  repl_rec->must_flush = LOG_REPL_NEED_FLUSH;
458  }
459  }
460 
461  return error;
462 }

464 /*
465  * repl_log_insert_schema - insert a replication info(schema) to the
466  * transaction descriptor
467  *
468  * return: NO_ERROR or error code
469  *
470  * repl_schema(in):
471  *

473  * descriptor (tdes)
474  */
475 int
476 repl_log_insert_statement (THREAD_ENTRY * thread_p, REPL_INFO_SBR * repl_info)
477 {
478  int tran_index;
479  LOG_TDES *tdes;
480  LOG_REPL_RECORD *repl_rec;
481  char *ptr;
482  int error = NO_ERROR, strlen1, strlen2, strlen3, strlen4;
483 
484  tran_index = LOG_FIND_THREAD_TRAN_INDEX (thread_p);
485  tdes = LOG_FIND_TDES (tran_index);
486  if (tdes == NULL)
487  {
488  return ER_FAILED;
489  }
490 
491  /* If suppress_replication flag is set, do not write replication log. */
492  if (tdes->suppress_replication != 0)
493  {
494  return NO_ERROR;
495  }
496 
497  /* check the replication log array status, if we need to alloc? */
498  if (REPL_LOG_IS_NOT_EXISTS (tran_index)
499  && ((error = repl_log_info_alloc (tdes, REPL_LOG_INFO_ALLOC_SIZE, false)) != NO_ERROR))
500  {
501  return error;
502  }
503  /* the replication log array is full? re-alloc? */
504  else if (REPL_LOG_IS_FULL (tran_index)
505  && (error = repl_log_info_alloc (tdes, REPL_LOG_INFO_ALLOC_SIZE, true)) != NO_ERROR)
506  {
507  return error;
508  }
509 
510  repl_rec = (LOG_REPL_RECORD *) (&tdes->repl_records[tdes->cur_repl_record]);
511  repl_rec->repl_type = LOG_REPLICATION_STATEMENT;
512  repl_rec->rcvindex = RVREPL_STATEMENT;
513  repl_rec->must_flush = LOG_REPL_COMMIT_NEED_FLUSH;
514  OID_SET_NULL (&repl_rec->inst_oid);
515 
516  /* make the common info for the schema replication */
517  repl_rec->length = (OR_INT_SIZE /* REPL_INFO_SCHEMA.statement_type */
518  + or_packed_string_length (repl_info->name, &strlen1)
519  + or_packed_string_length (repl_info->stmt_text, &strlen2)
520  + or_packed_string_length (repl_info->db_user, &strlen3)
521  + or_packed_string_length (repl_info->sys_prm_context, &strlen4));
522 
523  repl_rec->repl_data = (char *) malloc (repl_rec->length);
524  if (repl_rec->repl_data == NULL)
525  {
526  error = ER_REPL_ERROR;
527  er_set (ER_WARNING_SEVERITY, ARG_FILE_LINE, ER_REPL_ERROR, 1, "can't allocate memory");
528  return error;
529  }
530  ptr = repl_rec->repl_data;
531  ptr = or_pack_int (ptr, repl_info->statement_type);
532  ptr = or_pack_string_with_length (ptr, repl_info->name, strlen1);
533  ptr = or_pack_string_with_length (ptr, repl_info->stmt_text, strlen2);
534  ptr = or_pack_string_with_length (ptr, repl_info->db_user, strlen3);
535  ptr = or_pack_string_with_length (ptr, repl_info->sys_prm_context, strlen4);
536 
537  er_log_debug (ARG_FILE_LINE,
538  "repl_log_insert_statement: repl_info_sbr { type %d, name %s, stmt_txt %s, user %s, "
539  "sys_prm_context %s }\n", repl_info->statement_type, repl_info->name, repl_info->stmt_text,
540  repl_info->db_user, repl_info->sys_prm_context);
541  LSA_COPY (&repl_rec->lsa, &tdes->tail_lsa);
542 
543  if (tdes->fl_mark_repl_recidx != -1 && tdes->cur_repl_record >= tdes->fl_mark_repl_recidx)
544  {
545  /*
546  * statement replication does not check log conflicts, so
547  * use repl_start_flush_mark with caution.
548  */
549  repl_rec->must_flush = LOG_REPL_NEED_FLUSH;
550  }
551 
552  tdes->cur_repl_record++;
553 
554  return error;
555 }
