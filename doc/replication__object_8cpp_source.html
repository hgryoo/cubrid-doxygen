63  {
64  size_t i;
65  const single_row_repl_entry *other_t = dynamic_cast<const single_row_repl_entry *> (other);
66 
67  if (other_t == NULL
68  || m_type != other_t->m_type
69  || strcmp (m_class_name, other_t->m_class_name) != 0
70  || db_value_compare (&m_key_value, &other_t->m_key_value) != DB_EQ
71  || changed_attributes.size () != other_t->changed_attributes.size ()
72  || m_new_values.size () != other_t->m_new_values.size ())
73  {
74  return false;
75  }
76 
77  for (i = 0; i < changed_attributes.size (); i++)
78  {
79  if (changed_attributes[i] != other_t->changed_attributes[i])
80  {
81  return false;
82  }
83  }
84 
85  for (i = 0; i < m_new_values.size (); i++)
86  {
87  if (db_value_compare (&m_new_values[i], &other_t->m_new_values[i]) != DB_EQ)
88  {
89  return false;
90  }
91  }
92 
93  return true;

123  size_t single_row_repl_entry::get_packed_size (cubpacking::packer *serializator)
124  {
125  size_t i;
126  size_t entry_size = 0;
127 
128  /* we assume that offset start has already MAX_ALIGNMENT */
129 
130  /* type of packed object + type of RBR entry */
131  entry_size += 2 * serializator->get_packed_int_size (entry_size);
132 
133  entry_size += serializator->get_packed_int_vector_size (entry_size, (int) changed_attributes.size ());
134 
135  entry_size += serializator->get_packed_small_string_size (m_class_name, entry_size);
136 
137  entry_size += serializator->get_packed_db_value_size (m_key_value, entry_size);
138 
139  /* count of new_values */
140  entry_size += serializator->get_packed_int_size (entry_size);
141 
142  for (i = 0; i < m_new_values.size (); i++)
143  {
144  entry_size += serializator->get_packed_db_value_size (m_new_values[i], entry_size);
145  }
146
