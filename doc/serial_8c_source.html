263 /*
264  * xserial_get_next_value () -
265  * return: NO_ERROR, or ER_status
266  * result_num(out) :
267  * oid_p(in) :
268  * cached_num(in) :
269  * num_alloc(in) :
270  * is_auto_increment(in) :
271  * force_set_last_insert_id(in):
272  */
273 int
274 xserial_get_next_value (THREAD_ENTRY * thread_p, DB_VALUE * result_num, const OID * oid_p, int cached_num,
275  int num_alloc, int is_auto_increment, bool force_set_last_insert_id)
276 {
277  int ret = NO_ERROR, granted;
278  SERIAL_CACHE_ENTRY *entry;
279  bool is_cache_mutex_locked = false;
280  bool is_oid_locked = false;
281 #if defined (SERVER_MODE)
282  int rc;
283 #endif /* SERVER_MODE */
284 
285  assert (oid_p != NULL);
286  assert (result_num != NULL);
287 
288  CHECK_MODIFICATION_NO_RETURN (thread_p, ret);
289  if (ret != NO_ERROR)
290  {
291  return ret;
292  }
293 
294  if (num_alloc < 1)
295  {
296  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_QPROC_INVALID_PARAMETER, 0);
297  return ER_FAILED;
298  }
299 
300  if (cached_num <= 1)
301  {
302  /* not used serial cache */
303  ret = xserial_get_next_value_internal (thread_p, result_num, oid_p, num_alloc);
304  }
305  else
306  {
307  /* used serial cache */
308  granted = LK_NOTGRANTED;
309 
310  try_again:
311  rc = pthread_mutex_lock (&serial_Cache_pool.cache_pool_mutex);
312  is_cache_mutex_locked = true;
313 
314  entry = (SERIAL_CACHE_ENTRY *) mht_get (serial_Cache_pool.ht, oid_p);
315  if (entry != NULL)
316  {
317  ret = serial_get_next_cached_value (thread_p, entry, num_alloc);
318  if (ret != NO_ERROR)
319  {
320  goto exit;
321  }
322  pr_clone_value (&entry->cur_val, result_num);
323  }
324  else
325  {
326  if (OID_ISNULL (&serial_Cache_pool.db_serial_class_oid))
327  {
328  ret = serial_load_attribute_info_of_db_serial (thread_p);
329  }
330 
331  if (ret == NO_ERROR)
332  {
333  if (is_oid_locked == false)
334  {
335  granted = lock_object (thread_p, oid_p, &serial_Cache_pool.db_serial_class_oid, X_LOCK, LK_COND_LOCK);
336 
337  if (granted != LK_GRANTED)
338  {
339  /* release mutex, get OID lock, and restart */
340  pthread_mutex_unlock (&serial_Cache_pool.cache_pool_mutex);
341  is_cache_mutex_locked = false;
342  granted =
343  lock_object (thread_p, oid_p, &serial_Cache_pool.db_serial_class_oid, X_LOCK, LK_UNCOND_LOCK);
344  if (granted == LK_GRANTED)
345  {
346  is_oid_locked = true;
347  goto try_again;
348  }
349  }
350  else
351  {
352  is_oid_locked = true;
353  }
354  }
355 
356  if (granted != LK_GRANTED)
357  {
358  assert (er_errid () != NO_ERROR);
359  ret = er_errid ();
360  }
361  else
362  {
363  ret = xserial_get_next_value_internal (thread_p, result_num, oid_p, num_alloc);
364  assert (is_oid_locked == true);
365  (void) lock_unlock_object (thread_p, oid_p, &serial_Cache_pool.db_serial_class_oid, X_LOCK, true);
366  is_oid_locked = false;
367  }
368  }
369  }
370 
371  if (is_cache_mutex_locked == true)
372  {
373  pthread_mutex_unlock (&serial_Cache_pool.cache_pool_mutex);
374  is_cache_mutex_locked = false;
375  }
376  }
377 
378  if (ret == NO_ERROR && is_auto_increment == GENERATE_AUTO_INCREMENT)
379  {
380  /* we update current insert id for this session here */
381  /* Note that we ignore an error during updating current insert id. */
382  (void) xsession_set_cur_insert_id (thread_p, result_num, force_set_last_insert_id);
383  }
384 
385 exit:
386  if (is_cache_mutex_locked)
387  {
388  pthread_mutex_unlock (&serial_Cache_pool.cache_pool_mutex);
389  is_cache_mutex_locked = false;
390  }
391 
392  if (is_oid_locked)
393  {
394  (void) lock_unlock_object (thread_p, oid_p, &serial_Cache_pool.db_serial_class_oid, X_LOCK, true);
395  is_oid_locked = false;
396  }
397 
398  return ret;
399 }

401 /*
402  * serial_get_next_cached_value () -
403  * return: NO_ERROR, or ER_status
404  * entry(in/out) :
405  * num_alloc(in) :
406  */
407 static int
408 serial_get_next_cached_value (THREAD_ENTRY * thread_p, SERIAL_CACHE_ENTRY * entry, int num_alloc)
409 {
410  DB_VALUE cmp_result;
411  DB_VALUE next_val;
412  int error, nturns;
413  bool exhausted = false;
414 
415  assert (1 <= num_alloc);
416 
417  /* check if cached numbers were already exhausted */
418  if (num_alloc == 1)
419  {
420  error = numeric_db_value_compare (&entry->cur_val, &entry->last_cached_val, &cmp_result);
421  if (error != NO_ERROR)
422  {
423  return error;
424  }
425  if (db_get_int (&cmp_result) == 0)
426  {
427  /* entry is cached to number of cached_num */
428  exhausted = true;
429  }
430  }
431  else
432  {
433  error =
434  serial_get_nth_value (&entry->inc_val, &entry->cur_val, &entry->min_val, &entry->max_val, &entry->cyclic,
435  num_alloc, &next_val);
436 
437  error = numeric_db_value_compare (&next_val, &entry->last_cached_val, &cmp_result);
438  if (error != NO_ERROR)
439  {
440  return error;
441  }
442 
443  if (db_get_int (&cmp_result) >= 0)
444  {
445  exhausted = true;
446  }
447 
448  }
449 
450  /* consumed all cached value */
451  if (exhausted == true)
452  {
453  nturns = CEIL_PTVDIV (num_alloc, entry->cached_num);
454 
455  error =
456  serial_get_nth_value (&entry->inc_val, &entry->last_cached_val, &entry->min_val, &entry->max_val,
457  &entry->cyclic, (nturns * entry->cached_num), &entry->last_cached_val);
458 
459  if (error != NO_ERROR)
460  {
461  return error;
462  }
463 
464  /* cur_val of db_serial is updated to last_cached_val of entry */
465  error = serial_update_cur_val_of_serial (thread_p, entry);
466  if (error != NO_ERROR)
467  {
468  return error;
469  }
470  }
471 
472  /* get next value */
473  if (num_alloc == 1)
474  {
475  error =
476  serial_get_nth_value (&entry->inc_val, &entry->cur_val, &entry->min_val, &entry->max_val, &entry->cyclic,
477  num_alloc, &next_val);
478  if (error != NO_ERROR)
479  {
480  return error;
481  }
482  }
483 
484  pr_clone_value (&next_val, &entry->cur_val);
485 
486  return NO_ERROR;
487 }

926 /*
927  * serial_get_nth_value () - get Nth next_value
928  * return: NO_ERROR, or ER_status
929  * inc_val(in) :
930  * cur_val(in) :
931  * min_val(in) :
932  * max_val(in) :
933  * cyclic(in) :
934  * nth(in) :
935  * result_val(out) :
936  */
937 static int
938 serial_get_nth_value (DB_VALUE * inc_val, DB_VALUE * cur_val, DB_VALUE * min_val, DB_VALUE * max_val, DB_VALUE * cyclic,
939  int nth, DB_VALUE * result_val)
940 {
941  DB_VALUE tmp_val, cmp_result, add_val;
942  unsigned char num[DB_NUMERIC_BUF_SIZE];
943  int inc_val_flag;
944  int ret;
945 
946  inc_val_flag = numeric_db_value_is_positive (inc_val);
947  if (inc_val_flag < 0)
948  {
949  return ER_FAILED;
950  }
951 
952  /* Now calculate next value */
953  if (nth > 1)
954  {
955  numeric_coerce_int_to_num (nth, num);
956  db_make_numeric (&tmp_val, num, DB_MAX_NUMERIC_PRECISION, 0);
957  numeric_db_value_mul (inc_val, &tmp_val, &add_val);
958  }
959  else
960  {
961  PR_SHARE_VALUE (inc_val, &add_val);
962  }
963 
964  /* inc_val_flag (1 or 0) */
965  if (inc_val_flag > 0)
966  {
967  ret = numeric_db_value_sub (max_val, &add_val, &tmp_val);
968  if (ret != NO_ERROR)
969  {
970  return ret;
971  }
972  ret = numeric_db_value_compare (cur_val, &tmp_val, &cmp_result);
973  if (ret != NO_ERROR)
974  {
975  return ret;
976  }
977 
978  /* cur_val + inc_val * cached_num > max_val */
979  if (db_get_int (&cmp_result) > 0)
980  {
981  if (db_get_int (cyclic))
982  {
983  PR_SHARE_VALUE (min_val, result_val);
984  }
985  else
986  {
987  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_QPROC_SERIAL_RANGE_OVERFLOW, 0);
988  return ER_QPROC_SERIAL_RANGE_OVERFLOW;
989  }
990  }
991  else
992  {
993  (void) numeric_db_value_add (cur_val, &add_val, result_val);
994  }
995  }
996  else
997  {
998  ret = numeric_db_value_sub (min_val, &add_val, &tmp_val);
999  if (ret != NO_ERROR)
1000  {
1001  return ret;
1002  }
1003  ret = numeric_db_value_compare (cur_val, &tmp_val, &cmp_result);
1004  if (ret != NO_ERROR)
1005  {
1006  return ret;
1007  }
1008 
1009  /* cur_val + inc_val * cached_num < min_val */
1010  if (db_get_int (&cmp_result) < 0)
1011  {
1012  if (db_get_int (cyclic))
1013  {
1014  PR_SHARE_VALUE (max_val, result_val);
1015  }
1016  else
1017  {
1018  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_QPROC_SERIAL_RANGE_OVERFLOW, 0);
1019  return ER_QPROC_SERIAL_RANGE_OVERFLOW;
1020  }
1021  }
1022  else
1023  {
1024  (void) numeric_db_value_add (cur_val, &add_val, result_val);
1025  }
1026  }
1027 
1028  return NO_ERROR;
1029 }
