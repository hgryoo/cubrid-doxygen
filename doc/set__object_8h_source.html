84 struct setobj

87  DB_TYPE coltype;
88  int size; /* valid indexes from 0 to size -1 aka the number of represented values in the
89  * collection */
90  int lastinsert; /* the last value insertion point 0 to size. */
91  int topblock; /* maximum index of an allocated block. This is the maximum non-NULL db_value pointer
92  * index of array. array[topblock] should be non-NULL. array[topblock+1] will be a NULL
93  * pointer for future expansion. */
94  int arraytop; /* maximum indexable pointer in array the valid indexes for array are 0 to arraytop
95  * inclusive Generally this may be greater than topblock */
96  int topblockcount; /* This is the max index of the top block Since it may be shorter than a standard sized
97  * block for space efficiency. */
98  DB_VALUE **array;
99 
100  /* not stored on disk, attached at run time by the schema */
101  struct tp_domain *domain;
102 
103  /* external reference list */
104  DB_COLLECTION *references;
105 
106  /* clear if we can't guarantee sort order, always on for sequences */
107  unsigned sorted:1;
108 
109  /* set if we can't guarantee that there are no temporary OID's in here */
110  unsigned may_have_temporary_oids:1;
111 };
