79 fn_get_shard_key_default (const char *shard_key, T_SHARD_U_TYPE type, const void *value, int value_len)
80 {
81  int modular_key;
82 
83  if (value == NULL)
84  {
85  PROXY_LOG (PROXY_LOG_MODE_ERROR, "Invalid shard key value. " "Shard key value couldn't be NUll.");
86  return ERROR_ON_ARGUMENT;
87  }
88 
89  modular_key = shm_proxy_p->shard_key_modular;
90  if (modular_key < 0)
91  {
92  PROXY_LOG (PROXY_LOG_MODE_ERROR,
93  "Invalid modular key. " "Shard modular key value couldn't be negative integer. " "(modular_key:%d).",
94  modular_key);
95  return ERROR_ON_MAKE_SHARD_KEY;
96  }
97 
98  if (type == SHARD_U_TYPE_INT || type == SHARD_U_TYPE_UINT)
99  {
100  unsigned int ival;
101  ival = (unsigned int) (*(unsigned int *) value);
102  return ival % modular_key;
103  }
104  else if (type == SHARD_U_TYPE_SHORT || type == SHARD_U_TYPE_USHORT)
105  {
106  unsigned short sval;
107  sval = (unsigned short) (*(unsigned short *) value);
108  return sval % modular_key;
109  }
110  else if (type == SHARD_U_TYPE_BIGINT || type == SHARD_U_TYPE_UBIGINT)
111  {
112  UINT64 lval;
113  lval = (UINT64) (*(UINT64 *) value);
114  return lval % modular_key;
115  }
116  else
117  {
118  PROXY_LOG (PROXY_LOG_MODE_ERROR, "Unexpected shard key type. " "(type:%d).", type);
119  return ERROR_ON_ARGUMENT;
120  }
121 }
