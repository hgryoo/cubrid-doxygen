72 static int
73 shard_stmt_lru_insert (T_SHARD_STMT * stmt_p)
74 {
75  if ((stmt_p->num_pinned != 0) || (stmt_p->lru_prev != NULL) || (stmt_p->lru_next != NULL))
76  {
77  PROXY_DEBUG_LOG ("Invalid statement status. statement(%s).", shard_str_stmt (stmt_p));
78 
79  assert (false);
80  return -1;
81  }
82 
83  stmt_p->lru_next = NULL;
84  stmt_p->lru_prev = shard_Stmt.mru;
85 
86  if (shard_Stmt.mru)
87  {
88  shard_Stmt.mru->lru_next = stmt_p;
89  }
90  else
91  {
92  shard_Stmt.lru = stmt_p;
93  }
94 
95  shard_Stmt.mru = stmt_p;
96 
97  return 0;
98 }

100 static int
101 shard_stmt_lru_delete (T_SHARD_STMT * stmt_p)
102 {
103  ENTER_FUNC ();
104 
105  if ((stmt_p->lru_prev == NULL && shard_Stmt.lru != stmt_p) || (stmt_p->lru_next == NULL && shard_Stmt.mru != stmt_p))
106  {
107  PROXY_DEBUG_LOG ("Invalid statement lru list. " "(stmt_p:%p, lru:%p, mru:%p). " "statement(%s).", stmt_p,
108  shard_Stmt.lru, shard_Stmt.mru, shard_str_stmt (stmt_p));
109 
110  assert (false);
111 
112  EXIT_FUNC ();
113  return -1;
114  }
115 
116  if (stmt_p->lru_next)
117  {
118  stmt_p->lru_next->lru_prev = stmt_p->lru_prev;
119  }
120  else
121  {
122  shard_Stmt.mru = stmt_p->lru_prev;
123  }
124 
125  if (stmt_p->lru_prev)
126  {
127  stmt_p->lru_prev->lru_next = stmt_p->lru_next;
128  }
129  else
130  {
131  shard_Stmt.lru = stmt_p->lru_next;
132  }
133 
134  stmt_p->lru_prev = NULL;
135  stmt_p->lru_next = NULL;
136 
137  EXIT_FUNC ();
138  return 0;

198 static T_BROKER_VERSION
199 shard_stmt_make_protocol_version (T_BROKER_VERSION client_version)
200 {
201  /* protocol version used in only shard_Stmt->stmt_entry's client_version */
202  if (client_version < CAS_MAKE_VER (8, 3, 0))
203  {
204  /* old protocol */
205  return CAS_MAKE_VER (8, 2, 0);
206  }
207  else if (client_version < CAS_MAKE_VER (8, 4, 0))
208  {
209  /* error indicator protocol added */
210  return CAS_MAKE_VER (8, 3, 0);
211  }
212  else if (client_version <= CAS_PROTO_MAKE_VER (PROTOCOL_V1))
213  {
214  /* prepare result info added */
215  return CAS_PROTO_MAKE_VER (PROTOCOL_V1);
216  }
217  /* send columns meta-data with the result for executing */
218  /* PROTOCOL_V2 is current protocol version */
219  return CAS_PROTO_CURRENT_VER;
220 }

245  {

412 static T_SHARD_STMT *
413 shard_stmt_new_internal (int stmt_type, char *sql_stmt, int ctx_cid, unsigned int ctx_uid,
414  T_BROKER_VERSION client_version)
415 {
416  int error;
417  int i, num_cas;
418  T_SHARD_STMT *stmt_p = NULL;
419  T_PROXY_CONTEXT *ctx_p = NULL;
420 
421  assert ((stmt_type != SHARD_STMT_TYPE_SCHEMA_INFO && sql_stmt != NULL)
422  || (stmt_type == SHARD_STMT_TYPE_SCHEMA_INFO && sql_stmt == NULL && client_version == 0));
423 
424  ctx_p = proxy_context_find (ctx_cid, ctx_uid);
425  assert (ctx_p != NULL);
426 
427  assert (stmt_type != SHARD_STMT_TYPE_PREPARED
428  || shard_stmt_find_by_sql (sql_stmt, ctx_p->database_user, client_version) == NULL);
429 
430  num_cas = shard_Stmt.max_num_shard * shard_Stmt.num_cas_per_shard;
431 
432  stmt_p = shard_stmt_find_unused ();
433  if (stmt_p)
434  {
435  assert (stmt_p->num_pinned == 0);
436  assert (stmt_p->lru_next == NULL && stmt_p->lru_prev == NULL);
437 
438  if (!(stmt_p->num_pinned == 0) || !(stmt_p->lru_next == NULL && stmt_p->lru_prev == NULL))
439  {
440  PROXY_LOG (PROXY_LOG_MODE_ERROR, "Invalid statement LRU. statement(%s).", shard_str_stmt (stmt_p));
441  return NULL;
442  }
443  }
444  else
445  {
446  stmt_p = shard_stmt_get_lru ();
447  if (stmt_p)
448  {
449  shard_stmt_free (stmt_p);
450  }
451  else
452  {
453  PROXY_LOG (PROXY_LOG_MODE_ERROR, "Exceeds max prepared statement. ");
454  return NULL;
455  }
456  }
457 
458  if (stmt_p)
459  {
460  if ((int) stmt_p->num_alloc >= shard_Stmt_max_num_alloc)
461  {
462  stmt_p->num_alloc = 0;
463  }
464  stmt_p->stmt_h_id = (stmt_p->num_alloc * shard_Stmt.max_num_stmt) + stmt_p->index;
465 
466  stmt_p->status = SHARD_STMT_STATUS_IN_PROGRESS;
467  stmt_p->stmt_type = stmt_type;
468  if (stmt_p->stmt_type != SHARD_STMT_TYPE_SCHEMA_INFO)
469  {
470  stmt_p->client_version = shard_stmt_make_protocol_version (client_version);
471 
472  stmt_p->parser = sp_create_parser (sql_stmt);
473  if (stmt_p->parser == NULL)
474  {
475  shard_stmt_free (stmt_p);
476  return NULL;
477  }
478  }
479  else
480  {
481  stmt_p->client_version = 0;
482  assert (stmt_p->parser == NULL);
483  stmt_p->parser = NULL;
484  }
485 
486  stmt_p->ctx_cid = ctx_cid;
487  stmt_p->ctx_uid = ctx_uid;
488  strncpy (stmt_p->database_user, ctx_p->database_user, SRV_CON_DBUSER_SIZE - 1);
489 
490  stmt_p->num_pinned = 0;
491  stmt_p->lru_prev = NULL;
492  stmt_p->lru_next = NULL;
493 
494  /* __FOR_DEBUG */
495  assert (stmt_p->request_buffer_length == 0);
496  assert (stmt_p->request_buffer == NULL);
497  assert (stmt_p->reply_buffer_length == 0);
498  assert (stmt_p->reply_buffer == NULL);
499 
500  stmt_p->request_buffer_length = 0;
501  stmt_p->request_buffer = NULL;
502  stmt_p->reply_buffer_length = 0;
503  stmt_p->reply_buffer = NULL;
504 
505  for (i = 0; i < num_cas; i++)
506  {
507  stmt_p->srv_h_id_ent[i] = SHARD_STMT_INVALID_HANDLE_ID;
508  }
509 
510  error = shard_queue_initialize (&stmt_p->waitq);
511  if (error)
512  {
513  assert (false);
514 
515  shard_stmt_free (stmt_p);
516  return NULL;
517  }
518 
519  stmt_p->num_alloc += 1;
520 
521  if (stmt_p->stmt_type == SHARD_STMT_TYPE_PREPARED)
522  {
523  shard_stmt_put_statement_to_map (sql_stmt, stmt_p);
524  }
525  }
526 
527  return stmt_p;
528 }

627 
628  num_cas = shard_Stmt.max_num_shard * shard_Stmt.num_cas_per_shard;
629

631  {
632  shard_stmt_del_statement_from_map (stmt_p);

634

636  stmt_p->status = SHARD_STMT_STATUS_UNUSED;
637  stmt_p->stmt_type = -1;

642  stmt_p->ctx_uid = 0;
643  stmt_p->database_user[0] = '\0';
644 
645  stmt_p->num_pinned = 0;
