2  * sp_tree.c
3  *
4  * Implementation for splay binary search tree.
5  * Copyright (C) 2001-2004 Farooq Mela.
6  *
7  * $Id: sp_tree.c,v 1.12 2001/12/12 06:44:51 farooq Exp farooq $
8  *
9  * cf. [Sleator and Tarjan, 1985], [Tarjan 1985], [Tarjan 1983]
10  *
11  * A single operation on a splay tree has a worst-case time complexity of O(N),
12  * but a series of M operations have a time complexity of O(M lg N), and thus
13  * the amortized time complexity of an operation is O(lg N). More specifically,
14  * a series of M operations on a tree with N nodes will runs in O((N+M)lg(N+M))
15  * time. Splay trees work by "splaying" a node up the tree using a series of
16  * rotations until it is the root each time it is accessed. They are much
17  * simpler to code than most balanced trees, because there is no strict
18  * requirement about maintaining a balance scheme among nodes. When inserting
19  * and searching, we simply splay the node in question up until it becomes the
20  * root of the tree.
21  *
22  * This implementation is a bottom-up, move-to-root splay tree.
23  */

123 void
124 sp_tree_empty(sp_tree *tree, int del)
125 {
126  sp_node *node, *parent;
127 
128  ASSERT(tree != NULL);
129 
130  node = tree->root;
131  while (node) {
132  parent = node->parent;
133  if (node->llink || node->rlink) {
134  node = node->llink ? node->llink : node->rlink;
135  continue;
136  }
137 
138  if (del) {
139  if (tree->key_del)
140  tree->key_del(node->key);
141  if (tree->dat_del)
142  tree->dat_del(node->dat);
143  }
144  FREE(node);
145 
146  if (parent) {
147  if (parent->llink == node)
148  parent->llink = NULL;
149  else
150  parent->rlink = NULL;
151  }
152  node = parent;
153  }
154 
155  tree->root = NULL;
156  tree->count = 0;
157 }

159 /*
160  * XXX Each zig/zig and zig/zag operation can be optimized, but for now we just
161  * use two rotations.
162  */
163 /*
164  * zig_zig_right(T, A):
165  *
166  * C A
167  * / B \
168  * B ==> / \ ==> B
169  * / A C \
170  * A C
171  *
172  * zig_zig_left(T, C):
173  *
174  * A C
175  * \ B /
176  * B ==> / \ ==> B
177  * \ A C /
178  * C A
179  *
180  * zig_zag_right(T, B)
181  *
182  * A A
183  * \ \ B
184  * C ==> B ==> / \
185  * / \ A C
186  * B C
187  *
188  * zig_zag_left(T, B)
189  *
190  * C C
191  * / / B
192  * A ==> B ==> / \
193  * \ / A C
194  * B A
195  */
196 #define SPLAY(t,n) \
197 { \
198  sp_node *p; \
199  \
200  p = (n)->parent; \
201  if (p == (t)->root) { \
202  if (p->llink == (n)) /* zig right */ \
203  rot_right((t), p); \
204  else /* zig left */ \
205  rot_left((t), p); \
206  } else { \
207  if (p->llink == (n)) { \
208  if (p->parent->llink == p) { /* zig zig right */ \
209  rot_right((t), p->parent); \
210  rot_right((t), (n)->parent); \
211  } else { /* zig zag right */ \
212  rot_right((t), p); \
213  rot_left((t), (n)->parent); \
214  } \
215  } else { \
216  if (p->parent->rlink == p) { /* zig zig left */ \
217  rot_left((t), p->parent); \
218  rot_left((t), (n)->parent); \
219  } else { /* zig zag left */ \
220  rot_left((t), p); \
221  rot_right((t), (n)->parent); \
222  } \
223  } \
224  } \
225 }

227 int
228 sp_tree_insert(sp_tree *tree, void *key, void *dat, int overwrite)
229 {
230  int rv = 0; /* shut up GCC */
231  sp_node *node, *parent = NULL;
232 
233  ASSERT(tree != NULL);
234 
235  node = tree->root;
236  while (node) {
237  rv = tree->key_cmp(key, node->key);
238  if (rv < 0)
239  parent = node, node = node->llink;
240  else if (rv > 0)
241  parent = node, node = node->rlink;
242  else {
243  if (overwrite == 0)
244  return 1;
245  if (tree->key_del)
246  tree->key_del(node->key);
247  if (tree->dat_del)
248  tree->dat_del(node->dat);
249  node->key = key;
250  node->dat = dat;
251  return 0;
252  }
253  }
254 
255  if ((node = node_new(key, dat)) == NULL)
256  return -1;
257 
258  if ((node->parent = parent) == NULL) {
259  ASSERT(tree->count == 0);
260  tree->root = node;
261  tree->count = 1;
262  return 0;
263  }
264  if (rv < 0)
265  parent->llink = node;
266  else
267  parent->rlink = node;
268  tree->count++;
269 
270  while (node->parent)
271  SPLAY(tree, node);
272 
273  return 0;
274 }

359 int
360 sp_tree_remove(sp_tree *tree, const void *key, int del)
361 {
362  int rv;
363  sp_node *node, *temp, *out, *parent;
364  void *tmp;
365 
366  ASSERT(tree != NULL);
367 
368  node = tree->root;
369  while (node) {
370  rv = tree->key_cmp(key, node->key);
371  if (rv < 0)
372  node = node->llink;
373  else if (rv > 0)
374  node = node->rlink;
375  else
376  break;
377  }
378 
379  if (node == NULL)
380  return -1;
381 
382  if (node->llink == NULL || node->rlink == NULL) {
383  out = node;
384  } else {
385  /*
386  * This is sure to screw up iterators that were positioned at the node
387  * "out".
388  */
389  for (out = node->rlink; out->llink; out = out->llink)
390  /* void */;
391  SWAP(node->key, out->key, tmp);
392  SWAP(node->dat, out->dat, tmp);
393  }
394 
395  temp = out->llink ? out->llink : out->rlink;
396  parent = out->parent;
397  if (temp)
398  temp->parent = parent;
399  if (parent) {
400  if (parent->llink == out)
401  parent->llink = temp;
402  else
403  parent->rlink = temp;
404  } else {
405  tree->root = temp;
406  }
407  if (del) {
408  if (tree->key_del)
409  tree->key_del(out->key);
410  if (tree->dat_del)
411  tree->dat_del(out->dat);
412  }
413 
414  /* Splay an adjacent node to the root, if possible. */
415  temp =
416  node->parent ? node->parent :
417  node->rlink ? node->rlink :
418  node->llink;
419  if (temp)
420  while (temp->parent)
421  SPLAY(tree, temp);
422 
423  FREE(out);
424  tree->count--;
425 
426  return 0;
427 }

507 static void
508 rot_left(sp_tree *tree, sp_node *node)
509 {
510  sp_node *rlink, *parent;
511 
512  ASSERT(tree != NULL);
513  ASSERT(node != NULL);
514  ASSERT(node->rlink != NULL);
515 
516  rlink = node->rlink;
517  node->rlink = rlink->llink;
518  if (rlink->llink)
519  rlink->llink->parent = node;
520  parent = node->parent;
521  rlink->parent = parent;
522  if (parent) {
523  if (parent->llink == node)
524  parent->llink = rlink;
525  else
526  parent->rlink = rlink;
527  } else {
528  tree->root = rlink;
529  }
530  rlink->llink = node;
531  node->parent = rlink;
532 }
