705  private void completeStatementForCommit() throws SQLException {
706  for (int i = 0; i < statements.size(); i++) {
707  CUBRIDStatement stmt = (CUBRIDStatement) statements.get(i);
708 
709  if (stmt.getHoldability() == ResultSet.HOLD_CURSORS_OVER_COMMIT) {
710  stmt.setCurrentTransaction(false);
711  continue;
712  }
713  else if (stmt instanceof CUBRIDPreparedStatement) {
714  statements.remove(i);
715  if (u_con.brokerInfoStatementPooling() == true)
716  stmt.complete();
717  else
718  stmt.close();
719  } else
720  stmt.complete();
721  }
722  }

724  private void completeAllStatements() throws SQLException {
725  for (int i = 0; i < statements.size(); i++) {
726  CUBRIDStatement stmt = (CUBRIDStatement) statements.get(i);
727 
728  if (stmt.getHoldability() == ResultSet.HOLD_CURSORS_OVER_COMMIT
729  && !stmt.isFromCurrentTransaction()) {
730  continue;
731  }
732 
733  if (stmt instanceof CUBRIDPreparedStatement) {
734 
735  statements.remove(i);
736  if (u_con.brokerInfoStatementPooling() == true)
737  stmt.complete();
738  else
739  stmt.close();
740  } else
741  stmt.complete();
742  }
743  }
