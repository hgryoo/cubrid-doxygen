73  throws SQLException {
74  String dataSourceName = getDataSourceName();

76 
77  if (dataSourceName == null || dataSourceName.length() == 0) {
78  if (getUrl() != null) {
79  CUBRIDDriver driver = new CUBRIDDriver();
80  Properties props = new Properties();
81 
82  if (username != null) {
83  props.setProperty("user", username);
84  }
85  if (passwd != null) {
86  props.setProperty("password", passwd);
87  }
88  con = driver.connect(getUrl(), props);
89  } else {
90  if (username == null) {
91  username = getUser();
92  }
93  if (passwd == null) {
94  passwd = getPassword();
95  }
96  UConnection u_con = UJCIManager.connect(
97  getServerName(),
98  getPortNumber(),
99  getDatabaseName(), username,
100  passwd,
101  getDataSourceID(username));
102  con = new CUBRIDConnection(u_con, null,
103  username);
104  }
105  writeLog("getConnection(" + username + ")");
106  } else {
107  CUBRIDConnectionPoolDataSource cpds;
108 
109  cpds = CUBRIDConnectionPoolManager
110  .getConnectionPoolDataSource(dataSourceName);
111  if (username == null)
112  username = cpds.getUser();
113  if (passwd == null)
114  passwd = cpds.getPassword();
115  con = CUBRIDConnectionPoolManager.getConnection(cpds,
116  username, passwd);
117  }
