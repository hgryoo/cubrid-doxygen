260  CUBRIDXidTable.putXidInfo(xacon_key, xidInfo);
261  } else if (flags == TMJOIN || flags == TMRESUME) {
262  if (xidInfo == null)
263  throw new XAException(XAException.XAER_NOTA);
264 
265  synchronized (xidInfo) {
266  if (xidInfo.status != CUBRIDXidInfo.STATUS_NOFLAG
267  && xidInfo.status != CUBRIDXidInfo.STATUS_SUSPENDED) {
268  throw new XAException(XAException.XAER_PROTO);
269  }

272  throw new XAException(XAException.XAER_RMERR);
273
