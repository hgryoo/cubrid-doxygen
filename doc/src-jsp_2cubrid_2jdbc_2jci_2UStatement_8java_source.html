531  synchronized public void close(boolean close_srv_handle) {
532  if (isClosed == true) {
533  errorHandler.setErrorCode(UErrorCode.ER_IS_CLOSED);
534  return;
535  }
536  relatedConnection.pooled_ustmts.remove(this);
537  currentFirstCursor = cursorPosition = totalTupleNumber = fetchedTupleNumber = 0;
538  isClosed = true;
539  if (stmt_cache != null) {
540  stmt_cache.decr_ref_count();
541  }
542 
543  try {
544  if (!isReturnable
545  && close_srv_handle
546  && (relatedConnection.getAutoCommit() == false
547  || relatedConnection.brokerInfoStatementPooling() == true
548  || ((prepare_flag & UConnection.PREPARE_HOLDABLE) != 0))) {
549  if (getSqlType()) {
550  synchronized (relatedConnection) {
551  outBuffer.newRequest(UFunctionCode.CLOSE_USTATEMENT);
552  outBuffer.addInt(serverHandler);
553  outBuffer.addByte(relatedConnection.getAutoCommit()
554  ? (byte) 1 : (byte) 0);
555  relatedConnection.send_recv_msg();
556  }
557  } else {
558  relatedConnection.deferred_close_handle.add(new Integer(
559  serverHandler));
560  }
561  }
562  } catch (UJciException e) {
563  if (relatedConnection.isActive()) {
564  e.setStackTrace(e.getStackTrace());
565  e.toUError(errorHandler);
566  }
567  } catch (IOException e) {
568  if (relatedConnection.isActive()) {
569  e.setStackTrace(e.getStackTrace());

571  }
572  } finally {
573  currentFirstCursor = cursorPosition = totalTupleNumber = fetchedTupleNumber = 0;
574  isClosed = true;

576  stmt_cache.decr_ref_count();
577  }

697  private void writeExecuteRequest(int maxField, boolean isScrollable,
698  int queryTimeout)
699  throws IOException, UJciException {
700  byte is_auto_commit = (byte) 0, is_forward_only = (byte) 0;
701  long remainingTime = 0;
702 
703  outBuffer.newRequest(UFunctionCode.EXECUTE);
704  outBuffer.addInt(serverHandler);
705  outBuffer.addByte(executeFlag);
706  outBuffer.addInt(maxField < 0 ? 0 : maxField);
707  outBuffer.addInt(0);
708 
709  if (firstStmtType == CUBRIDCommandType.CUBRID_STMT_CALL_SP
710  && bindParameter != null) {
711  outBuffer.addBytes(bindParameter.paramMode);
712  } else {
713  outBuffer.addNull();
714  }
715 
716  /* fetch flag */
717  if (firstStmtType == CUBRIDCommandType.CUBRID_STMT_SELECT) {
718  outBuffer.addByte((byte) 1);
719  } else {
720  outBuffer.addByte((byte) 0);
721  }
722 
723  if (relatedConnection.getAutoCommit() && !isGeneratedKeys) {
724  is_auto_commit = (byte) 1;
725  }
726  outBuffer.addByte(is_auto_commit);
