413  private Value readArgument(DataInputStream dis, int paramSize,
414  int paramType, int mode, int dbType) throws IOException,
415  TypeMismatchException, SQLException {
416  Value arg = null;
417  switch (paramType) {
418  case DB_SHORT:
419  // assert paramSize == 4
420  arg = new ShortValue((short) dis.readInt(), mode, dbType);
421  break;
422  case DB_INT:
423  // assert paramSize == 4
424  arg = new IntValue(dis.readInt(), mode, dbType);
425  break;
426  case DB_BIGINT:
427  // assert paramSize == 8
428  arg = new LongValue(dis.readLong(), mode, dbType);
429  break;
430  case DB_FLOAT:
431  // assert paramSize == 4
432  arg = new FloatValue(dis.readFloat(), mode, dbType);
433  break;
434  case DB_DOUBLE:
435  case DB_MONETARY:
436  // assert paramSize == 8
437  arg = new DoubleValue(dis.readDouble(), mode, dbType);
438  break;
439  case DB_NUMERIC: {
440  byte[] paramValue = new byte[paramSize];
441  dis.readFully(paramValue);
442 
443  int i;
444  for (i = 0; i < paramValue.length; i++) {
445  if (paramValue[i] == 0)
446  break;
447  }
448 
449  byte[] strValue = new byte[i];
450  System.arraycopy(paramValue, 0, strValue, 0, i);
451 
452  arg = new NumericValue(new String(strValue), mode, dbType);
453  }
454  break;
455  case DB_CHAR:
456  case DB_STRING:
457  // assert paramSize == n
458  {
459  byte[] paramValue = new byte[paramSize];
460  dis.readFully(paramValue);
461 
462  int i;
463  for (i = 0; i < paramValue.length; i++) {
464  if (paramValue[i] == 0)
465  break;
466  }
467 
468  byte[] strValue = new byte[i];
469  System.arraycopy(paramValue, 0, strValue, 0, i);
470  arg = new StringValue(new String(strValue), mode, dbType);
471  }
472  break;
473  case DB_DATE:
474  // assert paramSize == 3
475  {
476  int year = dis.readInt();
477  int month = dis.readInt();
478  int day = dis.readInt();
479 
480  arg = new DateValue(year, month, day, mode, dbType);
481  }
482  break;
483  case DB_TIME:
484  // assert paramSize == 3
485  {
486  int hour = dis.readInt();
487  int min = dis.readInt();
488  int sec = dis.readInt();
489  Calendar cal = Calendar.getInstance();
490  cal.set(0, 0, 0, hour, min, sec);
491 
492  arg = new TimeValue(hour, min, sec, mode, dbType);
493  }
494  break;
495  case DB_TIMESTAMP:
496  // assert paramSize == 6
497  {
498  int year = dis.readInt();
499  int month = dis.readInt();
500  int day = dis.readInt();
501  int hour = dis.readInt();

503  int sec = dis.readInt();

505  cal.set(year, month, day, hour, min, sec);
506 
507  arg = new TimestampValue(year, month, day, hour, min, sec, mode,
508  dbType);
509  }
510  break;
511  case DB_DATETIME:

514  int year = dis.readInt();
515  int month = dis.readInt();
516  int day = dis.readInt();
517  int hour = dis.readInt();

527  break;
528  case DB_SET:
529  case DB_MULTISET:
530  case DB_SEQUENCE: {
531  int nCol = dis.readInt();
532  // System.out.println(nCol);
533  arg = new SetValue(readArgumentsForSet(dis, nCol), mode, dbType);
534  }
535  break;
536  case DB_OBJECT: {
537  int page = dis.readInt();
538  short slot = (short) dis.readInt();
539  short vol = (short) dis.readInt();
540 
541  byte[] bOID = new byte[UConnection.OID_BYTE_SIZE];
542  bOID[0] = ((byte) ((page >>> 24) & 0xFF));
543  bOID[1] = ((byte) ((page >>> 16) & 0xFF));
544  bOID[2] = ((byte) ((page >>> 8) & 0xFF));
545  bOID[3] = ((byte) ((page >>> 0) & 0xFF));
546  bOID[4] = ((byte) ((slot >>> 8) & 0xFF));
547  bOID[5] = ((byte) ((slot >>> 0) & 0xFF));
548  bOID[6] = ((byte) ((vol >>> 8) & 0xFF));
549  bOID[7] = ((byte) ((vol >>> 0) & 0xFF));
550 
551  if (jdbcConnection == null) {
552  jdbcConnection = DriverManager
553  .getConnection("jdbc:default:connection:");
554  }
555  arg = new OidValue(new CUBRIDOID((CUBRIDConnection) jdbcConnection,

557  }
558  break;
559  case DB_NULL:
560  arg = new NullValue(mode, dbType);
561  break;
562  default:
563  // unknown type
564  break;
565  }
566  return arg;
567  }

569  private void sendValue(Object result, DataOutputStream dos, int ret_type)
570  throws IOException {
571  if (result == null) {
572  dos.writeInt(DB_NULL);
573  } else if (result instanceof Short) {
574  dos.writeInt(DB_SHORT);
575  dos.writeInt(((Short) result).intValue());
576  } else if (result instanceof Integer) {
577  dos.writeInt(DB_INT);
578  dos.writeInt(((Integer) result).intValue());
579  } else if (result instanceof Long) {
580  dos.writeInt(DB_BIGINT);
581  dos.writeLong(((Long) result).longValue());
582  } else if (result instanceof Float) {
583  dos.writeInt(DB_FLOAT);
584  dos.writeFloat(((Float) result).floatValue());
585  } else if (result instanceof Double) {
586  dos.writeInt(ret_type);
587  dos.writeDouble(((Double) result).doubleValue());
588  } else if (result instanceof BigDecimal) {
589  dos.writeInt(DB_NUMERIC);
590  packAndSendString(((BigDecimal) result).toString(), dos);
591  } else if (result instanceof String) {
592  dos.writeInt(DB_STRING);
593  packAndSendString((String) result, dos);
594  } else if (result instanceof java.sql.Date) {
595  dos.writeInt(DB_DATE);
596  packAndSendString(result.toString(), dos);
597  } else if (result instanceof java.sql.Time) {
598  dos.writeInt(DB_TIME);
599  packAndSendString(result.toString(), dos);
600  } else if (result instanceof java.sql.Timestamp) {
601  dos.writeInt(ret_type);
602 
603  if (ret_type == DB_DATETIME) {
604  packAndSendString(result.toString(), dos);
605  } else {
606  SimpleDateFormat formatter = new SimpleDateFormat(
607  "yyyy-MM-dd HH:mm:ss");
608  packAndSendString(formatter.format(result), dos);
609  }
610  } else if (result instanceof CUBRIDOID) {
611  dos.writeInt(DB_OBJECT);
612  byte[] oid = ((CUBRIDOID) result).getOID();
613  dos.writeInt(UJCIUtil.bytes2int(oid, 0));
614  dos.writeInt(UJCIUtil.bytes2short(oid, 4));
615  dos.writeInt(UJCIUtil.bytes2short(oid, 6));
616  } else if (result instanceof ResultSet) {
617  dos.writeInt(DB_RESULTSET);
618  dos.writeInt(((CUBRIDResultSet) result).getServerHandle());
619  } else if (result instanceof int[]) {
620  int length = ((int[]) result).length;
621  Integer[] array = new Integer[length];
622  for (int i = 0; i < array.length; i++) {
623  array[i] = new Integer(((int[]) result)[i]);
624  }
625  sendValue(array, dos, ret_type);
626  } else if (result instanceof short[]) {
627  int length = ((short[]) result).length;
628  Short[] array = new Short[length];
629  for (int i = 0; i < array.length; i++) {
630  array[i] = new Short(((short[]) result)[i]);
631  }
632  sendValue(array, dos, ret_type);
633  } else if (result instanceof float[]) {
634  int length = ((float[]) result).length;
635  Float[] array = new Float[length];
636  for (int i = 0; i < array.length; i++) {
637  array[i] = new Float(((float[]) result)[i]);
638  }
639  sendValue(array, dos, ret_type);
640  } else if (result instanceof double[]) {
641  int length = ((double[]) result).length;
642  Double[] array = new Double[length];
643  for (int i = 0; i < array.length; i++) {
644  array[i] = new Double(((double[]) result)[i]);
645  }
646  sendValue(array, dos, ret_type);
647  } else if (result instanceof Object[]) {
648  dos.writeInt(ret_type);
649  Object[] arr = (Object[]) result;
650 
651  dos.writeInt(arr.length);
652  for (int i = 0; i < arr.length; i++) {
653  sendValue(arr[i], dos, ret_type);
654  }
655  } else
656  ;
657  }

660  throws IOException {

662 
663  int len = b.length + 1;

669
