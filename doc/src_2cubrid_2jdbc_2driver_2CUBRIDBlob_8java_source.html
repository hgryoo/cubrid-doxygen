103  public byte[] getBytes(long pos, int length) throws SQLException {
104  if (lobHandle == null) {
105  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.invalid_value, null);
106  }
107  if (pos < 1 || length < 0) {
108  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.invalid_value, null);
109  }
110  if (length == 0) {
111  return new byte[0];
112  }
113 
114  pos--; // pos is now offset from 0
115  int real_read_len, read_len, total_read_len = 0;
116 
117  if (pos + length > length()) {
118  length = (int) (length() - pos);
119  }
120 
121  if (length <= 0) {
122  return new byte[0];
123  }
124 
125  byte[] buf = new byte[length];
126 
127  while (length > 0) {
128  read_len = Math.min(length, BLOB_MAX_IO_LENGTH);
129  real_read_len = conn.lobRead(lobHandle.getPackedLobHandle(), pos,
130  buf, total_read_len, read_len);
131 
132  pos += real_read_len;
133  length -= real_read_len;
134  total_read_len += real_read_len;
135 
136  if (real_read_len == 0) {
137  break;
138  }
139  }
140 
141  if (total_read_len < buf.length) {
142  // In common case, this code cannot be executed
143  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.unknown, null);
144  // byte[]new_buf = new byte[total_read_len];
145  // System.arraycopy (buf, 0, new_buf, 0, total_read_len);
146  // return new_buf;
147  } else {
148  return buf;
149  }
150  }

182  public int setBytes(long pos, byte[] bytes, int offset, int len)
183  throws SQLException {
184  if (lobHandle == null) {
185  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.invalid_value, null);
186  }
187  if (pos < 1 || offset < 0 || len < 0) {
188  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.invalid_value, null);
189  }
190  if (offset + len > bytes.length) {
191  throw new IndexOutOfBoundsException();
192  }
193 
194  if (isWritable) {
195  if (length() + 1 != pos) {
196  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.lob_pos_invalid, null);
197  }
198 
199  pos--; // pos is now offset from 0
200 
201  int real_write_len, write_len, total_write_len = 0;
202 
203  while (len > 0) {
204  write_len = Math.min(len, BLOB_MAX_IO_LENGTH);
205  real_write_len = conn.lobWrite(lobHandle.getPackedLobHandle(),
206  pos, bytes, offset, write_len);
207 
208  pos += real_write_len;
209  len -= real_write_len;
210  offset += real_write_len;
211  total_write_len += real_write_len;
212  }
213 
214  if (pos > length()) {
215  lobHandle.setLobSize(pos);
216  }
217 
218  return total_write_len;
219  } else {
220  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.lob_is_not_writable, null);
221  }
222  }
