190  public synchronized int setString(long pos, String str) throws SQLException {
191  if (lobHandle == null) {
192  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.invalid_value, null);
193  }
194  if (pos < 1) {
195  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.invalid_value, null);
196  }
197  if (!isWritable) {
198  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.lob_is_not_writable, null);
199  }
200  if (str == null || str.length() <= 0) {
201  return 0;
202  }
203 
204  if (readClobPartially(pos, 1) != 0) {
205  // only append is allowed.
206  throw conn.createCUBRIDException(CUBRIDJDBCErrorCode.lob_pos_invalid, null);
207  }
208 
209  byte[] bytes = string2bytes(str);
210  int bytes_len = bytes.length;
211  int bytes_offset = 0;
212 
213  while (bytes_len > 0) {
214  int bytesWritten = conn.lobWrite(lobHandle.getPackedLobHandle(),
215  clobBytePos + bytes_offset, bytes, bytes_offset, Math.min(
216  bytes_len, CLOB_MAX_IO_LENGTH));
217 
218  bytes_len -= bytesWritten;
219  bytes_offset += bytesWritten;
220  }
221 
222  lobHandle.setLobSize(clobBytePos + bytes_offset);
223  clobCharLength = length() + str.length();
224 
225  return str.length();
226  }

309  private int readClobPartially(long pos, int length) throws SQLException {
310  if (clobCharLength != -1 && pos > clobCharLength) {
311  clobBytePos = clobNextReadBytePos = lobHandle.getLobSize();
312  clobCharPos = clobCharLength;
313  clobCharBuffer.setLength(0);
314  if (pos == clobCharLength + 1)
315  return 0;
316  else
317  return -1;
318  }
319 
320  pos--; // pos is now offset from 0
321 
322  if (pos < clobCharPos) {
323  clobBytePos = clobNextReadBytePos = 0;
324  clobCharPos = 0;
325  clobCharBuffer.setLength(0);
326  readClob();
327  }
328 
329  while (pos >= clobCharPos + clobCharBuffer.length()) {
330  clobBytePos = clobNextReadBytePos;
331  clobCharPos += clobCharBuffer.length();
332  clobCharBuffer.setLength(0);
333  if (clobNextReadBytePos >= lobHandle.getLobSize()) {
334  return 0;
335  }
336  readClob();
337  }
338 
339  int delete_len = (int) (pos - clobCharPos);
340  if (delete_len > 0) {
341  clobCharPos = pos;
342  clobBytePos += string2bytes(clobCharBuffer.substring(0, delete_len)).length;
343  clobCharBuffer.delete(0, delete_len);
344  }
345 
346  while (length > clobCharBuffer.length()) {
347  if (clobNextReadBytePos >= lobHandle.getLobSize()) {
348  return clobCharBuffer.length();
349  }
350  readClob();
351  }
352 
353  return length;
354  }

356  private void readClob() throws SQLException {
357  if (conn == null || lobHandle == null) {
358  throw new NullPointerException();
359  }
360 
361  int read_len = conn.lobRead(lobHandle.getPackedLobHandle(),
362  clobNextReadBytePos, clobByteBuffer, 0, CLOB_MAX_IO_LENGTH);
363 
364  StringBuffer sb = new StringBuffer(bytes2string(clobByteBuffer, 0,
365  read_len));
366 
367  if (clobNextReadBytePos + read_len >= lobHandle.getLobSize()) // End of CLOB
368  {
369  clobNextReadBytePos += read_len;
370  clobCharLength = clobCharPos + clobCharBuffer.length()
371  + sb.length();
372  } else {
373  clobNextReadBytePos += string2bytes (sb.substring (0, sb.length () - 1)).length;
374  sb.setLength(sb.length() - 1);
375  }
376 
377  clobCharBuffer.append(sb);
378  }
