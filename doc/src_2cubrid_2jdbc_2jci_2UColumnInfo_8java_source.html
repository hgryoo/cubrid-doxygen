161  static byte[] confirmType(byte originalType, byte collectionFlags) throws UJciException {
162  byte typeInfo[];
163 
164  typeInfo = new byte[2];
165  switch (collectionFlags) {
166  case 0:
167  typeInfo[0] = originalType;
168  typeInfo[1] = -1;
169  return typeInfo;
170  case SET_FLAG:
171  typeInfo[0] = UUType.U_TYPE_SET;
172  typeInfo[1] = originalType;
173  if (typeInfo[1] < UUType.U_TYPE_MIN || typeInfo[1] > UUType.U_TYPE_MAX)
174  throw new UJciException(UErrorCode.ER_TYPE_CONVERSION);
175  return typeInfo;
176  case MULTISET_FLAG:
177  typeInfo[0] = UUType.U_TYPE_MULTISET;
178  typeInfo[1] = originalType;
179  if (typeInfo[1] < UUType.U_TYPE_MIN || typeInfo[1] > UUType.U_TYPE_MAX)
180  throw new UJciException(UErrorCode.ER_TYPE_CONVERSION);
181  return typeInfo;
182  case SEQUENCE_FLAG:
183  typeInfo[0] = UUType.U_TYPE_SEQUENCE;
184  typeInfo[1] = originalType;
185  if (typeInfo[1] < UUType.U_TYPE_MIN || typeInfo[1] > UUType.U_TYPE_MAX)
186  throw new UJciException(UErrorCode.ER_TYPE_CONVERSION);
187  return typeInfo;
188  default:
189  typeInfo[0] = UUType.U_TYPE_NULL;
190  typeInfo[1] = -1;
191  }
192  return typeInfo;
193  }
