41 /*
42  * stats_get_statistics () - Get class statistics
43  * return:
44  * classoid(in): OID of the class
45  * timestamp(in):
46  *
47  * Note: This function provides an easier interface for the client for
48  * obtaining statistics to the client side by taking care of the
49  * communication details . (Note that the upper levels shouldn't have to
50  * worry about the communication buffer.)
51  */
52 CLASS_STATS *
53 stats_get_statistics (OID * class_oid_p, unsigned int time_stamp)
54 {
55  CLASS_STATS *stats_p = NULL;
56  char *buffer_p = NULL;
57  int length = -1;
58 
59  buffer_p = stats_get_statistics_from_server (class_oid_p, time_stamp, &length);
60  if (buffer_p)
61  {
62  assert (length > 0);
63 
64  stats_p = stats_client_unpack_statistics (buffer_p);
65  free_and_init (buffer_p);
66  }
67 
68  return stats_p;
69 }

71 /*
72  * qst_client_unpack_statistics () - Unpack the buffer containing statistics
73  * return: CLASS_STATS or NULL in case of error
74  * bufp(in): buffer containing the class statistics
75  *
76  * Note: This function unpacks the statistics on the buffer received from the
77  * server side, and builds a CLASS_STATS structure on the work space
78  * area. This sturucture is returned to the caller.
79  */
80 static CLASS_STATS *
81 stats_client_unpack_statistics (char *buf_p)
82 {
83  CLASS_STATS *class_stats_p;
84  ATTR_STATS *attr_stats_p;
85  BTREE_STATS *btree_stats_p;
86  int max_unique_keys;
87  int i, j, k;
88 
89  if (buf_p == NULL)
90  {
91  return NULL;
92  }
93 
94  class_stats_p = (CLASS_STATS *) db_ws_alloc (sizeof (CLASS_STATS));
95  if (class_stats_p == NULL)
96  {
97  return NULL;
98  }
99 
100  class_stats_p->time_stamp = (unsigned int) OR_GET_INT (buf_p);
101  buf_p += OR_INT_SIZE;
102 
103  class_stats_p->heap_num_objects = OR_GET_INT (buf_p);
104  buf_p += OR_INT_SIZE;
105  if (class_stats_p->heap_num_objects < 0)
106  {
107  assert (false);
108  class_stats_p->heap_num_objects = 0;
109  }
110 
111  class_stats_p->heap_num_pages = OR_GET_INT (buf_p);
112  buf_p += OR_INT_SIZE;
113  if (class_stats_p->heap_num_pages < 0)
114  {
115  assert (false);
116  class_stats_p->heap_num_pages = 0;
117  }
118 
119  /* to get the doubtful statistics to be updated, need to clear timestamp */
120  if (class_stats_p->heap_num_objects == 0 || class_stats_p->heap_num_pages == 0)
121  {
122  class_stats_p->time_stamp = 0;
123  }
124 
125  class_stats_p->n_attrs = OR_GET_INT (buf_p);
126  buf_p += OR_INT_SIZE;
127  if (class_stats_p->n_attrs == 0)
128  {
129  db_ws_free (class_stats_p);
130  return NULL;
131  }
132 
133  class_stats_p->attr_stats = (ATTR_STATS *) db_ws_alloc (class_stats_p->n_attrs * sizeof (ATTR_STATS));
134  if (class_stats_p->attr_stats == NULL)
135  {
136  db_ws_free (class_stats_p);
137  return NULL;
138  }
139 
140  for (i = 0, attr_stats_p = class_stats_p->attr_stats; i < class_stats_p->n_attrs; i++, attr_stats_p++)
141  {
142  attr_stats_p->id = OR_GET_INT (buf_p);
143  buf_p += OR_INT_SIZE;
144 
145  attr_stats_p->type = (DB_TYPE) OR_GET_INT (buf_p);
146  buf_p += OR_INT_SIZE;
147 
148  attr_stats_p->n_btstats = OR_GET_INT (buf_p);
149  buf_p += OR_INT_SIZE;
150 
151  if (attr_stats_p->n_btstats <= 0)
152  {
153  attr_stats_p->bt_stats = NULL;
154  continue;
155  }
156 
157  attr_stats_p->bt_stats = (BTREE_STATS *) db_ws_alloc (attr_stats_p->n_btstats * sizeof (BTREE_STATS));
158  if (attr_stats_p->bt_stats == NULL)
159  {
160  stats_free_statistics (class_stats_p);
161  return NULL;
162  }
163  memset (attr_stats_p->bt_stats, 0, attr_stats_p->n_btstats * sizeof (BTREE_STATS));
164 
165  for (j = 0, btree_stats_p = attr_stats_p->bt_stats; j < attr_stats_p->n_btstats; j++, btree_stats_p++)
166  {
167  OR_GET_BTID (buf_p, &btree_stats_p->btid);
168  buf_p += OR_BTID_ALIGNED_SIZE;
169 
170  btree_stats_p->leafs = OR_GET_INT (buf_p);
171  buf_p += OR_INT_SIZE;
172 
173  btree_stats_p->pages = OR_GET_INT (buf_p);
174  buf_p += OR_INT_SIZE;
175 
176  btree_stats_p->height = OR_GET_INT (buf_p);
177  buf_p += OR_INT_SIZE;
178 
179  btree_stats_p->has_function = OR_GET_INT (buf_p);
180  buf_p += OR_INT_SIZE;
181 
182  btree_stats_p->keys = OR_GET_INT (buf_p);
183  buf_p += OR_INT_SIZE;
184 
185  buf_p = or_unpack_domain (buf_p, &btree_stats_p->key_type, 0);
186 
187  if (TP_DOMAIN_TYPE (btree_stats_p->key_type) == DB_TYPE_MIDXKEY)
188  {
189  btree_stats_p->pkeys_size = tp_domain_size (btree_stats_p->key_type->setdomain);
190  }
191  else
192  {
193  btree_stats_p->pkeys_size = 1;
194  }
195 
196  /* cut-off to stats */
197  if (btree_stats_p->pkeys_size > BTREE_STATS_PKEYS_NUM)
198  {
199  btree_stats_p->pkeys_size = BTREE_STATS_PKEYS_NUM;
200  }
201 
202  btree_stats_p->pkeys = (int *) db_ws_alloc (btree_stats_p->pkeys_size * sizeof (int));
203  if (btree_stats_p->pkeys == NULL)
204  {
205  stats_free_statistics (class_stats_p);
206  return NULL;
207  }
208 
209  assert (btree_stats_p->pkeys_size <= BTREE_STATS_PKEYS_NUM);
210  for (k = 0; k < btree_stats_p->pkeys_size; k++)
211  {
212  btree_stats_p->pkeys[k] = OR_GET_INT (buf_p);
213  buf_p += OR_INT_SIZE;
214  }
215  }
216  }
217 
218  /* correct estimated num_objects with unique keys */
219  max_unique_keys = OR_GET_INT (buf_p);
220  buf_p += OR_INT_SIZE;
221  if (max_unique_keys > 0)
222  {
223  class_stats_p->heap_num_objects = max_unique_keys;
224  }
225 
226  /* validate key stats info */
227  assert (class_stats_p->heap_num_objects >= 0);
228  for (i = 0, attr_stats_p = class_stats_p->attr_stats; i < class_stats_p->n_attrs; i++, attr_stats_p++)
229  {
230  for (j = 0, btree_stats_p = attr_stats_p->bt_stats; j < attr_stats_p->n_btstats; j++, btree_stats_p++)
231  {
232  assert (btree_stats_p->keys >= 0);
233  btree_stats_p->keys = MIN (btree_stats_p->keys, class_stats_p->heap_num_objects);
234  for (k = 0; k < btree_stats_p->pkeys_size; k++)
235  {
236  assert (btree_stats_p->pkeys[k] >= 0);
237  btree_stats_p->pkeys[k] = MIN (btree_stats_p->pkeys[k], btree_stats_p->keys);
238  }
239  }
240  }
241 
242  return class_stats_p;
243 }

245 /*
246  * stats_free_statistics () - Frees the given CLASS_STAT structure
247  * return: void
248  * class_statsp(in): class statistics to be freed
249  */
250 void
251 stats_free_statistics (CLASS_STATS * class_statsp)
252 {
253  ATTR_STATS *attr_statsp;
254  int i, j;
255 
256  if (class_statsp)
257  {
258  if (class_statsp->attr_stats)
259  {
260  for (i = 0, attr_statsp = class_statsp->attr_stats; i < class_statsp->n_attrs; i++, attr_statsp++)
261  {
262  if (attr_statsp->bt_stats)
263  {
264  for (j = 0; j < attr_statsp->n_btstats; j++)
265  {
266  if (attr_statsp->bt_stats[j].pkeys)
267  {
268  db_ws_free (attr_statsp->bt_stats[j].pkeys);
269  attr_statsp->bt_stats[j].pkeys = NULL;
270  }
271  }
272 
273  db_ws_free (attr_statsp->bt_stats);
274  attr_statsp->bt_stats = NULL;
275  }
276  }
277  db_ws_free (class_statsp->attr_stats);
278  class_statsp->attr_stats = NULL;
279  }
280 
281  db_ws_free (class_statsp);
282  }
