348 /* Replace existing data in record at offset_to_data and size old_data_size
349  * with new_data of size new_data_size.
350  */
351 #define RECORD_REPLACE_DATA(record, offset_to_data, old_data_size, \
352  new_data_size, new_data) \
353  do \
354  { \
355  assert ((record) != NULL); \
356  assert ((record)->data != NULL); \
357  assert ((offset_to_data) >= 0 && (offset_to_data) <= (record)->length); \
358  assert ((old_data_size) >= 0 && (new_data_size) >= 0); \
359  assert ((offset_to_data) + (old_data_size) <= (record)->length); \
360  if ((old_data_size) != (new_data_size)) \
361  { \
362  /* We may need to move data inside record. */ \
363  if ((offset_to_data) + (old_data_size) < (record)->length) \
364  { \
365  /* Move data inside record. */ \
366  memmove ((record)->data + (offset_to_data) + (new_data_size), \
367  (record)->data + (offset_to_data) + (old_data_size), \
368  (record)->length - (offset_to_data) - (old_data_size)); \
369  } \
370  /* Update record length. */ \
371  (record)->length += (new_data_size) - (old_data_size); \
372  } \
373  /* Copy new data (if any). */ \
374  if ((new_data_size) > 0) \
375  { \
376  memcpy ((record)->data + (offset_to_data), new_data, new_data_size); \
377  } \
378  } \
379  while (false)
