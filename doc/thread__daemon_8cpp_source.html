77  void
78  daemon::stop_execution (void)
79  {
80  // note: this must not be called concurrently
81 
82  if (m_looper.stop ())
83  {
84  // already stopped
85  return;
86  }
87 
88  if (m_func_on_stop)
89  {
90  // to interrupt execution context
91  m_func_on_stop ();
92  }
93 
94  // make sure thread will wakeup
95  wakeup ();
96 
97  // then wait for thread to finish
98  m_thread.join ();
99  }

202  void
203  daemon::loop_without_context (daemon *daemon_arg, task_without_context *exec_arg, const char *name)
204  {
205  (void) name; // suppress unused parameter warning
206  // its purpose is to help visualize daemon thread stacks
207 
208  daemon_arg->register_stat_start ();
209 
210  while (!daemon_arg->m_looper.is_stopped ())
211  {
212  // execute task
213  exec_arg->execute ();
214  daemon_arg->register_stat_execute ();
215 
216  // take a break
217  daemon_arg->pause ();
218  daemon_arg->register_stat_pause ();
219  }
220 
221  // retire task
222  exec_arg->retire ();
223  }
