34  // cubthread::task
35  //
36  // description:
37  // abstract class for thread tasks
38  //
39  // templates:
40  // Context - thread execution context, a helper/cache structure that can be passed to multiple tasks.
41  //
42  // how to use:
43  // 1. specialize task with Context
44  // e.g. in CUBRID we have entry_task which uses entry as context; see thread_entry_task.hpp
45  //
46  // 2. extend specialized task<custom_context> and override virtual functions
47  // override execute (Context &) to define task execution
48  // [optional] override retire function; by default it deletes task
49  //
50  // 3. execute multiple tasks using same context:
51  // custom_context context; // this is a dummy example
52  // custom_task* task_p = NULL; // using custom_task = task<custom_context>
53  //
54  // for (task_p = get_task (); task_p != NULL; task_p = get_task ())
55  // {
56  // task_p->execute (context);
57  // task_p->retire (); // this will delete task_p
58  // }
59  //
60  template <typename Context>
61  class task
62  {
63  public:
64  using context_type = Context;
65 
66  task (void) = default;
67 
68  // abstract class requires virtual destructor
69  virtual ~task (void) = default;
70 
71  // virtual functions to be implemented by inheritors
72  virtual void execute (context_type &) = 0;
73 
74  // implementation of task's retire function.
75  virtual void retire (void)
76  {
77  delete this;
78  }
79  };

97  // context_manager
98  //
99  // description:
100  // complex tasks may require bulky context that can take a long time to construct/destruct.
101  // context_manager abstract class is designed to pool context and hand them quickly on demand.
102  //
103  // templates:
104  // Context - thread execution context, a helper/cache structure that can be passed to multiple tasks
105  //
106  // how to use:
107  // 1. specialize context_manager with custom context
108  // e.g. see CUBRID implementation for entry_manager.
109  //
110  // 2. override create_context, retire_context and recycle_context functions
111  //
112  // 3. execute multiple tasks using same context:
113  // custom_context_manager context_mgr; // using custom_context_manager = context_manager<custom_context>
114  // custom_context& context_ref = context_mgr.create_context ();
115  // custom_task* task_p = NULL; // using custom_task = task<custom_context>
116  //
117  // for (task_p = get_task (); task_p != NULL; task_p = get_task ())
118  // {
119  // task_p->execute (context_ref);
120  // task_p->retire (); // this will delete task_p
121  // // context_mgr.recycle_context (); // optional operation before reusing context
122  // }
123  //
124  // context_mgr.retire_context (context_ref);
125  //
126  // [optional]
127  // 4. if task execution can take a very long time and you need to force stop it, you can use stop_execution:
128  // 4.1. implement context_manager::stop_execution; should notify context to stop.
129  // 4.2. you have to check stop notifications in custom_task::execute
130  // 4.3. call context_mgr.stop_execution (context_ref).
131  //
132  template <typename Context>
133  class context_manager
134  {
135  public:
136  using context_type = Context;
137 
138  // abstract class requires virtual destructor
139  virtual ~context_manager () = default;
140 
141  virtual context_type &create_context (void) = 0; // create a new thread context; cannot fail
142  virtual void retire_context (context_type &) = 0; // retire the thread context
143  virtual void recycle_context (context_type &) // recycle context before reuse
144  {
145  // usage and implementation is optional
146  }
147  virtual void stop_execution (context_type &) // notify context to stop execution
148  {
149  // usage and implementation is optional
150  }
