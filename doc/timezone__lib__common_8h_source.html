36 /*
37  * The defines below are types of the data representation for the "ON" column
38  * from the daylight saving rules. The "ON" column can be either a number
39  * representing a day of the month, an identifier like "last<Weekday>" of the
40  * specified month, such as "lastSun" or "lastFri", or a rule representing a
41  * particular weekday after a specific day of the month, such as "Fri>=24".
42  */
43 typedef enum
44 {
45  TZ_DS_TYPE_FIXED = 0,
46  TZ_DS_TYPE_VAR_SMALLER,
47  TZ_DS_TYPE_VAR_GREATER
48 } TZ_DS_TYPE;

50 /*
51  * The values below are needed for identifying and properly computing time
52  * when a daylight saving event occurs, namely to track the suffixes
53  * 's' (standard local time), 'g', 'u' and 'z' (GMT/UTC/Zulu), 'w' or
54  * none (local wall clock time), from the AT column in daylight saving rules.
55  */
56 typedef enum
57 {
58  TZ_TIME_TYPE_LOCAL_STD = 0,
59  TZ_TIME_TYPE_LOCAL_WALL,
60  TZ_TIME_TYPE_UTC
61 } TZ_TIME_TYPE;

63 /*
64  * The defines below are used for identifying the type of timezone library and
65  * files which the gen_tz utility should use as input, and what to generate
66  * as output. A C file to be compiled into a shared library is always
67  * generated; the question is what to include in it, and weather or not it is
68  * necessary to also generate either a new or an updated C file containing
69  * timezone names and IDs (which should be included in the new CUBRID release)
70  * TZ_GEN_TYPE_NEW - generate the TZ/ID arrays from scratch;
71  * - include all the data from the TZ database into the
72  * C file to be compiled into the TZ shared library
73  * TZ_GEN_TYPE_UPDATE - don't generate any TZ/ID arrays
74  * - use the existing TZ/ID arrays to identify which
75  * timezones need to be updated;
76  * - no new time zones are added;
77  * TZ_GEN_TYPE_EXTEND - this flag is used when generating a new timezone library
78  * using the old library and the new timezone data
79  *
80  *
81  *
82  */
83 typedef enum
84 {
85  TZ_GEN_TYPE_NEW = 0,
86  TZ_GEN_TYPE_UPDATE,
87  TZ_GEN_TYPE_EXTEND
88 } TZ_GEN_TYPE;
