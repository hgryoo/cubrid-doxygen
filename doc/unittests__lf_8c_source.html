779 {
780 #define MAX_THREADS 64
781

783  static LF_TRAN_SYSTEM ts;
784  pthread_t threads[MAX_THREADS];
785  char msg[256];
786  int i;

788  sprintf (msg, "freelist (transaction=%s, %d threads)", test_local_tran ? "n" : "y", nthreads);
789  begin (msg);
790 
791  /* initialization */
792  if (nthreads > MAX_THREADS)
793  {
794  return fail ("too many threads");
795  }
796 
797  if (lf_tran_system_init (&ts, nthreads) != NO_ERROR)
798  {
799  return fail ("transaction system init");
800  }
801 
802  if (lf_freelist_init (&freelist, 100, 100, edesc, &ts) != NO_ERROR)
803  {
804  return fail ("freelist init");
805  }
806 
807  /* multithreaded test */
808  for (i = 0; i < nthreads; i++)
809  {
810  if (pthread_create (&threads[i], NULL, (test_local_tran ? test_freelist_proc_local_tran : test_freelist_proc),
811  (void *) &freelist) != NO_ERROR)
812  {
813  return fail ("thread create");
814  }
815  }
816 
817  for (i = 0; i < nthreads; i++)
818  {
