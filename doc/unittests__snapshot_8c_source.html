322 static int
323 logtb_initialize_mvcc_testing (int num_threads, THREAD_ENTRY ** thread_array)
324 {
325  LOG_ADDR_TDESAREA *area = NULL; /* Contiguous area for new transaction indices */
326  size_t size, area_size;
327  int i;
328  THREAD_ENTRY *thread_p;
329  int error_code = NO_ERROR;
330  LOG_TDES *tdes;
331 
332  if (num_threads == 0 || thread_array == NULL)
333  {
334  return ER_FAILED;
335  }
336 
337  log_Gl.trantable.area = NULL;
338  log_Gl.trantable.all_tdes = NULL;
339 
340  size = num_threads * sizeof (THREAD_ENTRY);
341  *thread_array = (THREAD_ENTRY *) malloc (size);
342  if (*thread_array == NULL)
343  {
344  error_code = ER_OUT_OF_VIRTUAL_MEMORY;
345  goto error;
346  }
347  memset (*thread_array, 0, size);
348  for (i = 0; i < num_threads; i++)
349  {
350  thread_p = *thread_array + i;
351  thread_p->type = TT_WORKER; /* init */
352  thread_p->index = i;
353  thread_p->tran_index = i + 1; /* quick fix to avoid issue in logtb_get_mvcc_snapshot - LOG_SYSTEM_TRAN_INDEX */
354  }
355 
356  size = num_threads * sizeof (*log_Gl.trantable.all_tdes);
357  log_Gl.trantable.all_tdes = (LOG_TDES **) malloc (size);
358  if (log_Gl.trantable.all_tdes == NULL)
359  {
360  error_code = ER_OUT_OF_VIRTUAL_MEMORY;
361  goto error;
362  }
363 
364  area_size = num_threads * sizeof (LOG_TDES) + sizeof (LOG_ADDR_TDESAREA);
365  area = (LOG_ADDR_TDESAREA *) malloc (area_size);
366  if (area == NULL)
367  {
368  error_code = ER_OUT_OF_VIRTUAL_MEMORY;
369  goto error;
370  }
371 
372  area->tdesarea = ((LOG_TDES *) ((char *) area + sizeof (LOG_ADDR_TDESAREA)));
373  area->next = NULL;
374 
375  /*
376  * Initialize every newly created transaction descriptor index
377  */
378  for (i = 0; i < num_threads; i++)
379  {
380  tdes = log_Gl.trantable.all_tdes[i] = &area->tdesarea[i];
381  logtb_initialize_tdes_for_mvcc_testing (tdes, i);
382  }
383 
384  log_Gl.trantable.area = area;
385  log_Gl.trantable.num_total_indices = num_threads;
386 
387  error_code = logtb_initialize_mvcctable ();
388  if (error_code != NO_ERROR)
389  {
390  goto error;
391  }
392  log_Gl.hdr.mvcc_next_id = MVCCID_FIRST;
393 
394  return NO_ERROR;
395 
396 error:
397  if (*thread_array)
398  {
399  free_and_init (*thread_array);
400  }
401 
402  if (log_Gl.trantable.all_tdes)
403  {
404  free_and_init (log_Gl.trantable.all_tdes);
405  }
406 
407  if (log_Gl.trantable.area)
408  {
409  free_and_init (log_Gl.trantable.area);
410  }
411 
412  return error_code;
413 }

468 THREAD_RET_T THREAD_CALLING_CONVENTION
469 test_mvcc_get_snapshot (void *param)
470 {
471  int i;
472  THREAD_ENTRY *thread_p = (THREAD_ENTRY *) param;
473  LOG_TDES *tdes = LOG_FIND_TDES (LOG_FIND_THREAD_TRAN_INDEX (thread_p));
474  unsigned int local_count_snapshots = 0;
475  MVCC_INFO *curr_mvcc_info = &tdes->mvccinfo;
476  volatile MVCCID *p_transaction_lowest_active_mvccid =
477  LOG_FIND_TRAN_LOWEST_ACTIVE_MVCCID (LOG_FIND_THREAD_TRAN_INDEX (thread_p));
478 
479  for (i = 0; i < NOPS_SNAPSHOT; i++)
480  {
481  if (logtb_get_mvcc_snapshot (thread_p) != NULL)
482  {
483  local_count_snapshots++;
484  }
485 
486  /* Invalidate snapshot */
487  ATOMIC_TAS_64 (p_transaction_lowest_active_mvccid, MVCCID_NULL);
488  curr_mvcc_info->recent_snapshot_lowest_active_mvccid = MVCCID_NULL;
489  MVCC_CLEAR_MVCC_INFO (curr_mvcc_info);
490  }
491 
492  ATOMIC_INC_64 (&count_snapshots, local_count_snapshots);
493  fprintf (stdout, "snapshot worker thread (%p) is leaving\n", thread_p);
494  fflush (stdout);
495 
496  return (THREAD_RET_T) 0;
497 }

499 THREAD_RET_T THREAD_CALLING_CONVENTION
500 test_new_mvcc_complete (void *param)
501 {
502  int i;
503  THREAD_ENTRY *thread_p = (THREAD_ENTRY *) param;
504  LOG_TDES *tdes = LOG_FIND_TDES (LOG_FIND_THREAD_TRAN_INDEX (thread_p));
505  unsigned int local_count_complete = 0;
506  bool committed = true;
507  volatile MVCCID *p_transaction_lowest_active_mvccid =
508  LOG_FIND_TRAN_LOWEST_ACTIVE_MVCCID (LOG_FIND_THREAD_TRAN_INDEX (thread_p));
509  MVCCID mvccid;
510 
511  for (i = 0; i < NOPS_COMPLPETE; i++)
512  {
513  mvccid = logtb_get_current_mvccid (thread_p);
514  if (mvccid == MVCCID_NULL)
515  {
516  abort ();
517  }
518 
519  logtb_complete_mvcc (thread_p, tdes, committed);
520  committed = !committed;
521 
522  /* here we may test whether bit was set */
523  local_count_complete++;
524 
525  if (*p_transaction_lowest_active_mvccid != MVCCID_NULL)
526  {
527  /* set transaction lowest active MVCCID to null to allow VACUUM advancing */
528  ATOMIC_TAS_64 (p_transaction_lowest_active_mvccid, MVCCID_NULL);
529  }
530  }
531 
532  ATOMIC_INC_64 (&count_complete, local_count_complete);
533  fprintf (stdout, "complete worker thread (%p) is leaving\n", thread_p);
534  fflush (stdout);
535 
536  return (THREAD_RET_T) 0;
537 }

568 static int
569 test_mvcc_operations (int num_snapshot_threads, int num_complete_threads, int num_oldest_mvccid_threads,
570  THREAD_ENTRY * thread_array)
571 {
572  int i;
573  int numthreads;
574 #define MAX_THREADS 100
575  pthread_t threads[MAX_THREADS];
576  int idx_thread_entry;
577  char msg[256];
578 
579  numthreads = num_snapshot_threads + num_complete_threads + num_oldest_mvccid_threads;
580  sprintf (msg, "test_mvcc_operations (%d snapshot threads, %d complete threads, %d oldest threads)",
581  num_snapshot_threads, num_complete_threads, num_oldest_mvccid_threads);
582  begin (msg);
583 
584  if (num_snapshot_threads < 0 || num_complete_threads < 0 || num_oldest_mvccid_threads < 0)
585  {
586  printf (" %s: %s\n", "FAILED", "negative number of threads not allowed");
587  return ER_FAILED;
588  }
589 
590  if (numthreads > MAX_THREADS)
591  {
592  printf (" %s: %s\n", "FAILED", "too many threads");
593  return ER_FAILED;
594  }
595 
596  count_snapshots = count_complete = count_oldest = 0;
597  idx_thread_entry = 0;
598  for (i = 0; i < num_snapshot_threads; i++, idx_thread_entry++)
599  {
600  if (pthread_create (&threads[idx_thread_entry], NULL, test_mvcc_get_snapshot,
601  (void *) (thread_array + idx_thread_entry)) != NO_ERROR)
602  {
603  printf (" %s: %s\n", "FAILED", "thread create error");
604  return ER_FAILED;
605  }
606  }
607 
608  for (i = 0; i < num_complete_threads; i++, idx_thread_entry++)
609  {
610  if (pthread_create (&threads[idx_thread_entry], NULL, test_new_mvcc_complete,
611  (void *) (thread_array + idx_thread_entry)) != NO_ERROR)
612  {
613  printf (" %s: %s\n", "FAILED", "thread create error");
614  return ER_FAILED;
615  }
616  }
617
