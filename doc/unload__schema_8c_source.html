188 /*
189  * filter_system_classes - Goes through a class list removing system classes.
190  * return: void
191  * class_list(in): class list to filter
192  */
193 static void
194 filter_system_classes (DB_OBJLIST ** class_list)
195 {
196  DB_OBJLIST *cl, *prev, *next;
197 
198  for (cl = *class_list, prev = NULL, next = NULL; cl != NULL; cl = next)
199  {
200  next = cl->next;
201  if (!db_is_system_class (cl->op))
202  {
203  prev = cl;
204  }
205  else
206  {
207  if (prev == NULL)
208  {
209  *class_list = next;
210  }
211  else
212  {
213  prev->next = next;
214  }
215  /*
216  * class_list links were allocated via ml_ext_alloc_link, so we must
217  * free them via ml_ext_free_link. Otherwise, we can crash.
218  */
219  ml_ext_free_link (cl);
220  }
221  }
222 }

257 /*
258  * is_dependent_class - determines if a particular class results in a
259  * dependency.
260  * return: non-zero if a dependency exists
261  * class(in): class to ponder
262  * unordered(in): remaining unordered class list
263  * ordered(in): ordered class list
264  * Note:
265  * This is determined by seeint if the class is NOT already on the
266  * ordered list AND the class IS on the unordered list.
267  * If the class is not on the unordered list, then we have to
268  * assume that the user is responsible for defining it at the
269  * appropriate time. This will also handle the case
270  * where we have dependencies on system defined classes since these
271  * won't be in the original class list.
272  */
273 static int
274 is_dependent_class (DB_OBJECT * class_, DB_OBJLIST * unordered, DB_OBJLIST * ordered)
275 {
276  return (!ml_find (ordered, class_) && ml_find (unordered, class_));
277 }

280 /*
281  * check_domain_dependencies - checks for dependencies on the classes that are
282  * found in a domain specification.
283  * return: non-zero if there are dependencies
284  * domain(in): domain to ponder
285  * this_class(in): class to ckeck
286  * unordered(in): remaining unordered class list
287  * ordered(in): ordered class list
288  */
289 static int
290 check_domain_dependencies (DB_DOMAIN * domain, DB_OBJECT * this_class, DB_OBJLIST * unordered, DB_OBJLIST * ordered)
291 {
292  DB_DOMAIN *d, *setdomain;
293  DB_OBJECT *class_;
294  int dependencies;
295 
296  dependencies = 0;
297  for (d = domain; d != NULL && !dependencies; d = db_domain_next (d))
298  {
299  setdomain = db_domain_set (d);
300  if (setdomain != NULL)
301  {
302  dependencies = check_domain_dependencies (setdomain, this_class, unordered, ordered);
303  }
304  else
305  {
306  class_ = db_domain_class (d);
307  if (class_ != NULL && class_ != this_class)
308  {
309  dependencies = is_dependent_class (class_, unordered, ordered);
310  }
311  }
312  }
313  return (dependencies);
314 }

316 /*
317  * has_dependencies - checks to see if a class has any remaining dependencnes
318  * return: non zero if there are dependencies remaining
319  * mop(in): class to ponder
320  * unordered(in): remaining unordered list
321  * ordered(in): ordered list being built
322  * conservative(in): if set detect a depencency on any class that is used
323  * as the domain of an attribute in this class.
324  */
325 static int
326 has_dependencies (DB_OBJECT * mop, DB_OBJLIST * unordered, DB_OBJLIST * ordered, int conservative)
327 {
328  DB_OBJLIST *supers, *su;
329  DB_ATTRIBUTE *att;
330  DB_DOMAIN *domain;
331  int dependencies;
332 
333  dependencies = 0;
334 
335  supers = db_get_superclasses (mop);
336  for (su = supers; su != NULL && !dependencies; su = su->next)
337  {
338  dependencies = is_dependent_class (su->op, unordered, ordered);
339  }
340 
341  /*
342  * if we're doing a conservative dependency check, look at the domains
343  * of each attribute.
344  */
345  if (!dependencies && conservative)
346  {
347  for (att = db_get_attributes (mop); att != NULL && !dependencies; att = db_attribute_next (att))
348  {
349  domain = db_attribute_domain (att);
350  dependencies = check_domain_dependencies (domain, mop, unordered, ordered);
351  }
352  }
353 
354  return (dependencies);
355 }

358 /*
359  * order_classes - transfers class list to ordered list if they have no
360  * dependencies
361  * return: number of classes transfered
362  * class_list(in): remaining unordered list
363  * order_list(out): order list we're building
364  * conservative(in): if set detect a depencency on any class that is used
365  * as the domain of an attribute in this class.
366  * Note:
367  * This makes a pass on the unordered class list transfering
368  * classes to the ordered list if they have no dependencies.
369  * We may make many passes over this list but there will be at least
370  * one class moved on each pass. If a pass is made and no classes
371  * can be moved, in indicates a circular depenency that cannot be
372  * handled by this algorithm. For normal classes, this would
373  * be an error condition since it indicates a cycle in the class
374  * hierarcy which isn't allowed.
375  * For proxy classes, it indicates circular references between the
376  * attributes that make up the OBJECT_ID which also isn't allowed.
377  * It can also indicate a bug in the dependency algorithm.
378  */
379 static int
380 order_classes (DB_OBJLIST ** class_list, DB_OBJLIST ** order_list, int conservative)
381 {
382  DB_OBJLIST *cl, *o, *next, *prev, *last;
383  int add_count;
384 
385  add_count = 0;
386 
387  for (cl = *class_list, prev = NULL, next = NULL; cl != NULL; cl = next)
388  {
389  next = cl->next;
390 
391  if (has_dependencies (cl->op, *class_list, *order_list, conservative))
392  {
393  prev = cl;
394  }
395  else
396  {
397  /* no dependencies, move it to the other list */
398  if (prev == NULL)
399  {
400  *class_list = next;
401  }
402  else
403  {
404  prev->next = next;
405  }
406 
407  /* append it on the order list */
408  cl->next = NULL;
409  for (o = *order_list, last = NULL; o != NULL; o = o->next)
410  {
411  last = o;
412  }
413 
414  if (last == NULL)
415  {
416  *order_list = cl;
417  }
418  else
419  {
420  last->next = cl;
421  }
422 
423  add_count++;
424  }
425  }
426 
427  return (add_count);
428 }

454 /*
455  * force_one_class - pick the top class off the 'class_list' and append to
456  * the 'order_list'
457  * return: void
458  * class_list(out): remaining unordered classes
459  * order_list(out): ordered class list
460  * Note:
461  * This is called when a depencency cycle is detected that we can't
462  * figure out. So we at least get the full schema dumped to the
463  * output file, pick the top class off the list and hopefully this
464  * will break the cycle.
465  * The user will then have to go in and hand edit the output file
466  * so that it can be loaded.
467  */
468 static void
469 force_one_class (DB_OBJLIST ** class_list, DB_OBJLIST ** order_list)
470 {
471  DB_OBJLIST *cl, *o, *last;
472 
473  emit_cycle_warning ();
474 
475  cl = *class_list;
476  *class_list = cl->next;
477 
478  cl->next = NULL;
479  for (o = *order_list, last = NULL; o != NULL; o = o->next)
480  {
481  last = o;
482  }
483 
484  if (last == NULL)
485  {
486  *order_list = cl;
487  }
488  else
489  {
490  last->next = cl;
491  }
492 }

495 /*
496  * get_ordered_classes - takes a list of classes to dump and returns a list
497  * of classes ordered according to their definition dependencies.
498  * return: ordered class list
499  * class_table(in): classes to dump
500  */
501 static DB_OBJLIST *
502 get_ordered_classes (MOP * class_table)
503 {
504  DB_OBJLIST *classes, *ordered;
505  int count, i;
506 
507  ordered = NULL;
508 
509  /*
510  * if class_table is passed, use it to initialize the list, otherwise
511  * get it from the API.
512  */
513  if (class_table == NULL)
514  {
515  classes = sm_fetch_all_classes (1, DB_FETCH_READ);
516  if (classes == NULL)
517  {
518  return NULL;
519  }
520 
521  filter_system_classes (&classes);
522  if (classes == NULL)
523  { /* no user class */
524  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_NODATA_TOBE_UNLOADED, 0);
525  return NULL;
526  }
527 
528  if (input_filename && required_class_only)
529  {
530  filter_unrequired_classes (&classes);
531  }
532  }
533  else
534  {
535  classes = NULL;
536  for (i = 0; class_table[i] != NULL; i++)
537  {
538  if (ml_ext_add (&classes, class_table[i], NULL))
539  {
540  /* memory error */
541  ml_ext_free (classes);
542  return NULL;
543  }
544  }
545  }
546 
547  while (classes != NULL)
548  {
549 
550  count = order_classes (&classes, &ordered, 1);
551  if (count == 0)
552  {
553  /*
554  * didn't find any using the conservative ordering, try the
555  * more relaxed one.
556  */
557  count = order_classes (&classes, &ordered, 0);
558  if (count == 0)
559  {
560  force_one_class (&classes, &ordered);
561  }
562  }
563  }
564  return (ordered);
565 }

602  * export_serial - export db_serial
603  * return: NO_ERROR if successful, error code otherwise
604  * outfp(in/out): out FILE pointer
605  */
606 static int
607 export_serial (FILE * outfp)
608 {
609  int error = NO_ERROR;
610  int i;
611  DB_QUERY_RESULT *query_result;
612  DB_QUERY_ERROR query_error;
613  DB_VALUE values[SERIAL_VALUE_INDEX_MAX], diff_value, answer_value;
614  DB_DOMAIN *domain;
615  char str_buf[NUMERIC_MAX_STRING_SIZE];
616 
617  /*
618  * You must check SERIAL_VALUE_INDEX enum defined on the top of this file
619  * when changing the following query. Notice the order of the result.
620  */
621  const char *query =
622  "select [name], [owner].[name], " "[current_val], " "[increment_val], " "[max_val], " "[min_val], " "[cyclic], "
623  "[started], " "[cached_num], " "[comment] " "from [db_serial] where [class_name] is null and [att_name] is null";
624 
625  db_make_null (&diff_value);
626  db_make_null (&answer_value);
627 
628  error = db_compile_and_execute_local (query, &query_result, &query_error);
629  if (error < 0)
630  {
631  goto err;
632  }
633 
634  error = db_query_first_tuple (query_result);
635  if (error != DB_CURSOR_SUCCESS)
636  {
637  goto err;
638  }
639 
640  do
641  {
642  for (i = 0; i < SERIAL_VALUE_INDEX_MAX; i++)
643  {
644  error = db_query_get_tuple_value (query_result, i, &values[i]);
645  if (error != NO_ERROR)
646  {
647  goto err;
648  }
649 
650  /* Validation of the result value */
651  switch (i)
652  {
653  case SERIAL_OWNER_NAME:
654  {
655  if (DB_IS_NULL (&values[i]) || DB_VALUE_TYPE (&values[i]) != DB_TYPE_STRING)
656  {
657  db_make_string (&values[i], "PUBLIC");
658  }
659  }
660  break;
661 
662  case SERIAL_NAME:
663  {
664  if (DB_IS_NULL (&values[i]) || DB_VALUE_TYPE (&values[i]) != DB_TYPE_STRING)
665  {
666  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_INVALID_SERIAL_VALUE, 0);
667  error = ER_INVALID_SERIAL_VALUE;
668  goto err;
669  }
670  }
671  break;
672 
673  case SERIAL_CURRENT_VAL:
674  case SERIAL_INCREMENT_VAL:
675  case SERIAL_MAX_VAL:
676  case SERIAL_MIN_VAL:
677  {
678  if (DB_IS_NULL (&values[i]) || DB_VALUE_TYPE (&values[i]) != DB_TYPE_NUMERIC)
679  {
680  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_INVALID_SERIAL_VALUE, 0);
681  error = ER_INVALID_SERIAL_VALUE;
682  goto err;
683  }
684  }
685  break;
686 
687  case SERIAL_CYCLIC:
688  case SERIAL_STARTED:
689  case SERIAL_CACHED_NUM:
690  {
691  if (DB_IS_NULL (&values[i]) || DB_VALUE_TYPE (&values[i]) != DB_TYPE_INTEGER)
692  {
693  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_INVALID_SERIAL_VALUE, 0);
694  error = ER_INVALID_SERIAL_VALUE;
695  goto err;
696  }
697  }
698  break;
699 
700  case SERIAL_COMMENT:
701  {
702  if (DB_IS_NULL (&values[i]) == false && DB_VALUE_TYPE (&values[i]) != DB_TYPE_STRING)
703  {
704  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_INVALID_SERIAL_VALUE, 0);
705  error = ER_INVALID_SERIAL_VALUE;
706  goto err;
707  }
708  }
709  break;
710 
711  default:
712  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_INVALID_SERIAL_VALUE, 0);
713  error = ER_INVALID_SERIAL_VALUE;
714  goto err;
715  }
716  }
717 
718  if (db_get_int (&values[SERIAL_STARTED]) == 1)
