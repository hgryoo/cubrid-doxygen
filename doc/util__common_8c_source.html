115 /*
116  * check_database_name_local() - check validation of the name of a database
117  * return: error code
118  * name(in): the name of a database
119  * existing_or_new_db(in): whether db is existing or new one
120  */
121 static int
122 check_database_name_local (const char *name, int existing_or_new_db)
123 {
124  int status = NO_ERROR;
125  int i = 0;
126 
127  if (name[0] == '#')
128  {
129  status = ER_GENERIC_ERROR;
130  }
131  else
132  {
133  for (i = 0; name[i] != 0; i++)
134  {
135  if (isspace (name[i]) || name[i] == '/' || name[i] == '\\' || !isprint (name[i])
136  || (existing_or_new_db == NEW_DATABASE && name[i] == '@'))
137  {
138  status = ER_GENERIC_ERROR;
139  break;
140  }
141  }
142  }
143 
144  if (status == ER_GENERIC_ERROR)
145  {
146  const char *message = utility_get_generic_message (MSGCAT_UTIL_GENERIC_BAD_DATABASE_NAME);
147  if (message != NULL)
148  {
149  PRINT_AND_LOG_ERR_MSG (message, name[i], name);
150  }
151  }
152  return status;
153 }

348 /*
349  * utility_keyword_search
350  */
351 int
352 utility_keyword_search (UTIL_KEYWORD * keywords, int *keyval_p, char **keystr_p)
353 {
354  UTIL_KEYWORD *keyp;
355 
356  if (*keyval_p >= 0 && *keystr_p == NULL)
357  {
358  /* get keyword string from keyword value */
359  for (keyp = keywords; keyp->keyval >= 0; keyp++)
360  {
361  if (*keyval_p == keyp->keyval)
362  {
363  *keystr_p = const_cast < char *>(keyp->keystr);
364  return NO_ERROR;
365  }
366  }
367  }
368  else if (*keyval_p < 0 && *keystr_p != NULL)
369  {
370  /* get keyword value from keyword string */
371  for (keyp = keywords; keyp->keystr != NULL; keyp++)
372  {
373  if (!strcasecmp (*keystr_p, keyp->keystr))
374  {
375  *keyval_p = keyp->keyval;
376  return NO_ERROR;
377  }
378  }
379  }
380  return ER_FAILED;
381 }

383 /*
384  * utility_localtime - transform date and time to broken-down time
385  * return: 0 if success, otherwise -1
386  * ts(in): pointer of time_t data value
387  * result(out): pointer of struct tm which will store the broken-down time
388  */
389 int
390 utility_localtime (const time_t * ts, struct tm *result)
391 {
392  struct tm *tm_p, tm_val;
393 
394  if (result == NULL)
395  {
396  return -1;
397  }
398 
399  tm_p = localtime_r (ts, &tm_val);
400  if (tm_p == NULL)
401  {
402  memset (result, 0, sizeof (struct tm));
403  return -1;
404  }
405 
406  memcpy (result, tm_p, sizeof (struct tm));
407  return 0;
408 }

647  * util_make_ha_conf -
648  *
649  * return:
650  *
651  * NOTE:
652  */
653 int
654 util_make_ha_conf (HA_CONF * ha_conf)
655 {
656  int error = NO_ERROR;
657  int i, num_ha_nodes;
658  char *ha_db_list_p = NULL;
659  char *ha_node_list_p = NULL, **ha_node_list_pp = NULL;
660  char *ha_sync_mode_p = NULL, **ha_sync_mode_pp = NULL;
661  const char *ha_copy_log_base_p;
662  int ha_max_mem_size;
663  bool is_replica_node;
664 
665  error =
666  util_get_ha_parameters (&ha_node_list_p, &ha_db_list_p, &ha_sync_mode_p, &ha_copy_log_base_p, &ha_max_mem_size);
667  if (error != NO_ERROR)
668  {
669  return error;
670  }
671 
672  ha_conf->db_names = util_split_ha_db (ha_db_list_p);
673  if (ha_conf->db_names == NULL)
674  {
675  const char *message = utility_get_generic_message (MSGCAT_UTIL_GENERIC_NO_MEM);
676  fprintf (stderr, message);
677 
678  error = ER_GENERIC_ERROR;
679  goto ret;
680  }
681 
682  ha_node_list_pp = util_split_ha_node (ha_node_list_p);
683  if (ha_node_list_pp == NULL)
684  {
685  const char *message = utility_get_generic_message (MSGCAT_UTIL_GENERIC_NO_MEM);
686  fprintf (stderr, message);
687 
688  error = ER_GENERIC_ERROR;
689  goto ret;
690  }
691 
692  for (i = 0; ha_node_list_pp[i] != NULL;)
693  {
694  i++;
695  }
696  num_ha_nodes = i;
697 
698  ha_conf->node_conf = (HA_NODE_CONF *) malloc (sizeof (HA_NODE_CONF) * num_ha_nodes);
699  if (ha_conf->node_conf == NULL)
700  {
701  const char *message = utility_get_generic_message (MSGCAT_UTIL_GENERIC_NO_MEM);
702  fprintf (stderr, message);
703 
704  error = ER_GENERIC_ERROR;
705  goto ret;
706  }
707  memset ((void *) ha_conf->node_conf, 0, sizeof (HA_NODE_CONF) * num_ha_nodes);
708  ha_conf->num_node_conf = num_ha_nodes;
709 
710  /* set ha_sync_mode */
711  is_replica_node = util_is_replica_node ();
712  if (is_replica_node == true)
713  {
