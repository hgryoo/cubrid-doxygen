113 /*
114  * backupdb() - backupdb main routine
115  * return: EXIT_SUCCESS/EXIT_FAILURE
116  */
117 int
118 backupdb (UTIL_FUNCTION_ARG * arg)
119 {
120  UTIL_ARG_MAP *arg_map = arg->arg_map;
121  char er_msg_file[PATH_MAX];
122  const char *database_name;
123  const char *backup_path = NULL;
124  bool remove_log_archives = false;
125  int backup_level = FILEIO_BACKUP_FULL_LEVEL;
126  const char *backup_verbose_file = NULL;
127  bool no_check = false;
128  bool check = true;
129  int backup_num_threads;
130  bool compress_flag;
131  bool sa_mode;
132  FILEIO_ZIP_METHOD backup_zip_method = FILEIO_ZIP_NONE_METHOD;
133  FILEIO_ZIP_LEVEL backup_zip_level = FILEIO_ZIP_NONE_LEVEL;
134  bool skip_activelog = false;
135  int sleep_msecs;
136  struct stat st_buf;
137  char real_pathbuf[PATH_MAX];
138  char verbose_file_realpath[PATH_MAX];
139 
140  if (utility_get_option_string_table_size (arg_map) != 1)
141  {
142  goto print_backup_usage;
143  }
144 
145  database_name = utility_get_option_string_value (arg_map, OPTION_STRING_TABLE, 0);
146  if (database_name == NULL)
147  {
148  goto print_backup_usage;
149  }
150 
151  backup_path = utility_get_option_string_value (arg_map, BACKUP_DESTINATION_PATH_S, 0);
152  remove_log_archives = utility_get_option_bool_value (arg_map, BACKUP_REMOVE_ARCHIVE_S);
153  backup_level = utility_get_option_int_value (arg_map, BACKUP_LEVEL_S);
154  backup_verbose_file = utility_get_option_string_value (arg_map, BACKUP_OUTPUT_FILE_S, 0);
155  no_check = utility_get_option_bool_value (arg_map, BACKUP_NO_CHECK_S);
156  check = !no_check;
157  backup_num_threads = utility_get_option_int_value (arg_map, BACKUP_THREAD_COUNT_S);
158  compress_flag = utility_get_option_bool_value (arg_map, BACKUP_COMPRESS_S);
159  skip_activelog = utility_get_option_bool_value (arg_map, BACKUP_EXCEPT_ACTIVE_LOG_S);
160  sleep_msecs = utility_get_option_int_value (arg_map, BACKUP_SLEEP_MSECS_S);
161  sa_mode = utility_get_option_bool_value (arg_map, BACKUP_SA_MODE_S);
162 
163  /* Range checking of input */
164  if (backup_level < 0 || backup_level >= FILEIO_BACKUP_UNDEFINED_LEVEL)
165  {
166  goto print_backup_usage;
167  }
168 
169  if (sa_mode && backup_num_threads > 1)
170  {
171  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_BACKUPDB,
172  BACKUPDB_INVALID_THREAD_NUM_OPT));
173  }
174 
175  if (backup_num_threads < FILEIO_BACKUP_NUM_THREADS_AUTO)
176  {
177  goto print_backup_usage;
178  }
179 
180  if (sleep_msecs < FILEIO_BACKUP_SLEEP_MSECS_AUTO)
181  {
182  goto print_backup_usage;
183  }
184 
185  if (compress_flag)
186  {
187  backup_zip_method = FILEIO_ZIP_LZO1X_METHOD;
188  backup_zip_level = FILEIO_ZIP_LZO1X_DEFAULT_LEVEL;
189  }
190 
191  /* extra validation */
192  if (check_database_name (database_name))
193  {
194  goto error_exit;
195  }
196 
197  if (backup_path != NULL)
198  {
199  memset (real_pathbuf, 0, sizeof (real_pathbuf));
200  if (realpath (backup_path, real_pathbuf) != NULL)
201  {
202  backup_path = real_pathbuf;
203  }
204 
205  // accept directory and FIFO (named pipe) file as backup destination.
206  if (stat (backup_path, &st_buf) != 0)
207  {
208  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_BACKUPDB,
209  BACKUPDB_INVALID_PATH));
210  goto error_exit;
211  }
212  else if (!S_ISDIR (st_buf.st_mode))
213  {
214 #if !defined (WINDOWS)
215  // Unfortunately, Windows does not support FIFO file.
216  if (!S_ISFIFO (st_buf.st_mode))
217 #endif /* !WINDOWS */
218  {
219  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_BACKUPDB,
220  BACKUPDB_INVALID_PATH));
221  goto error_exit;
222  }
223  }
224  }
225 
226  /* error message log file */
227  snprintf (er_msg_file, sizeof (er_msg_file) - 1, "%s_%s.err", database_name, arg->command_name);
228  er_init (er_msg_file, ER_NEVER_EXIT);
229 
230  sysprm_set_force (prm_get_name (PRM_ID_JAVA_STORED_PROCEDURE), "no");
231 
232  AU_DISABLE_PASSWORDS (); /* disable authorization for this operation */
233  db_set_client_type (DB_CLIENT_TYPE_ADMIN_UTILITY);
234  db_login ("DBA", NULL);
235 
236  if (db_restart (arg->command_name, TRUE, database_name) != NO_ERROR)
237  {
238  PRINT_AND_LOG_ERR_MSG ("%s\n", db_error_string (3));
239  goto error_exit;
240  }
241 
242  if (check)
243  {
244  int check_flag = 0;
245 
246  check_flag |= CHECKDB_FILE_TRACKER_CHECK;
247  check_flag |= CHECKDB_HEAP_CHECK_ALLHEAPS;
248  check_flag |= CHECKDB_CT_CHECK_CAT_CONSISTENCY;
249  check_flag |= CHECKDB_BTREE_CHECK_ALL_BTREES;
250  check_flag |= CHECKDB_LC_CHECK_CLASSNAMES;
251 
252  if (db_set_isolation (TRAN_READ_COMMITTED) != NO_ERROR
253  || boot_check_db_consistency (check_flag, 0, 0, NULL) != NO_ERROR)
254  {
255  const char *tmpname;
256 
257  tmpname = er_get_msglog_filename ();
258  if (tmpname == NULL)
259  {
260  tmpname = "/dev/null";
261  }
262  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_CHECKDB,
263  CHECKDB_MSG_INCONSISTENT), tmpname);
264  db_shutdown ();
265  goto error_exit;
266  }
267  }
268 
269  /* some other utilities may need interrupt handler too */
270  if (os_set_signal_handler (SIGINT, backupdb_sig_interrupt_handler) == SIG_ERR)
271  {
272  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_GENERIC_ERROR, 0);
273  PRINT_AND_LOG_ERR_MSG ("%s\n", db_error_string (3));
274  db_shutdown ();
275  goto error_exit;
276  }
277 
278  if (backup_verbose_file && *backup_verbose_file && *backup_verbose_file != '/')
279  {
280  char dirname[PATH_MAX];
281 
282  /* resolve relative path */
283  if (getcwd (dirname, PATH_MAX) != NULL)
284  {
285  snprintf (verbose_file_realpath, PATH_MAX - 1, "%s/%s", dirname, backup_verbose_file);
286  backup_verbose_file = verbose_file_realpath;
287  }
288  }
289 
290  if (boot_backup (backup_path, (FILEIO_BACKUP_LEVEL) backup_level, remove_log_archives, backup_verbose_file,
291  backup_num_threads, backup_zip_method, backup_zip_level, skip_activelog, sleep_msecs) != NO_ERROR)
292  {
293  PRINT_AND_LOG_ERR_MSG ("%s\n", db_error_string (3));
294  db_shutdown ();
295  goto error_exit;
296  }
297 
298  if (db_commit_transaction () != NO_ERROR)
299  {
300  PRINT_AND_LOG_ERR_MSG ("%s\n", db_error_string (3));
301  }
302 
303  db_shutdown ();
304 
305  return EXIT_SUCCESS;
306 
307 print_backup_usage:
308  fprintf (stderr, msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_BACKUPDB, BACKUPDB_MSG_USAGE),
309  basename (arg->argv0));
310  util_log_write_errid (MSGCAT_UTIL_GENERIC_INVALID_ARGUMENT);
311 
312 error_exit:
313  return EXIT_FAILURE;
314 }

409  ext_info.purpose = DB_TEMPORARY_DATA_PURPOSE;
410  }

413  ext_info.purpose = DB_PERMANENT_DATA_PURPOSE;

417  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_ADDVOLDB, ADDVOLDB_MSG_BAD_PURPOSE),

422

434  goto error_exit;
435  }
436

439  er_init (er_msg_file, ER_NEVER_EXIT);
440 
441  /* tuning system parameters */

496 /*
497  * util_get_class_oids_and_index_btid() -
498  * return: OID array/NULL
499  * index_name(in)
500  * index_btid(out)
501  */
502 static OID *
503 util_get_class_oids_and_index_btid (dynamic_array * darray, const char *index_name, BTID * index_btid)
504 {
505  MOP cls_mop;
506  OID *oids;
507  OID *cls_oid;
508  SM_CLASS *cls_sm;
509  SM_CLASS_CONSTRAINT *constraint;
510  char table[SM_MAX_IDENTIFIER_LENGTH];
511  char name[SM_MAX_IDENTIFIER_LENGTH];
512  int i;
513  int num_tables = da_size (darray);
514  MOBJ *obj;
515 
516  oids = (OID *) malloc (sizeof (OID) * num_tables);
517  if (oids == NULL)
518  {
519  perror ("malloc");
520  util_log_write_errid (MSGCAT_UTIL_GENERIC_NO_MEM);
521  return NULL;
522  }
523 
524  for (i = 0; i < num_tables; i++)
525  {
526  if (da_get (darray, i, table) != NO_ERROR)
527  {
528  free (oids);
529  return NULL;
530  }
531 
532  OID_SET_NULL (&oids[i]);
533  if (table == NULL || table[0] == '\0')
534  {
535  continue;
536  }
537 
538  sm_downcase_name (table, name, SM_MAX_IDENTIFIER_LENGTH);
539  cls_mop = locator_find_class (name);
540 
541  obj = (MOBJ *) & cls_sm;
542  ws_find (cls_mop, obj);
543  if (cls_sm == NULL || cls_sm->class_type != SM_CLASS_CT)
544  {
545  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_CHECKDB,
546  CHECKDB_MSG_NO_SUCH_CLASS), table);
547  continue;
548  }
549 
550  cls_oid = ws_oid (cls_mop);
551  if (cls_oid)
552  {
553  oids[i] = *cls_oid;
554  }
555  else
556  {
557  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_CHECKDB,
558  CHECKDB_MSG_NO_SUCH_CLASS), table);
559  continue;
560  }
561 
562  if (index_name != NULL)
563  {
564  constraint = classobj_find_class_index (cls_sm, index_name);
565  if (constraint == NULL)
566  {
567  PRINT_AND_LOG_ERR_MSG (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_CHECKDB,
568  CHECKDB_MSG_NO_SUCH_INDEX), index_name, name);
569  free_and_init (oids);
570  return NULL;
571  }
572 
573  db_constraint_index (constraint, index_btid);
574 
575  assert (i == 0);
576  }
577  }
578 
579  return oids;
580 }
