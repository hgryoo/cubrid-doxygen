43 /*
44  * utility_make_getopt_optstring - makes optstring for getopt_long()
45  * return: optstring
46  * opt_array(in): array of option structure
47  * buf(out): buffer for optstring
48  */
49 char *
50 utility_make_getopt_optstring (const GETOPT_LONG * opt_array, char *buf)
51 {
52  int i;
53  char *p = buf;
54  for (i = 0; opt_array[i].name; i++)
55  {
56  if (opt_array[i].val < 255)
57  {
58  *p++ = (char) opt_array[i].val;
59  if (opt_array[i].has_arg)
60  {
61  *p++ = ':';
62  }
63  }
64  }
65  *p = '\0';
66  return buf;
67 }

78 utility_load_library (DSO_HANDLE * handle, const char *path)
79 {
80  UTILITY_INIT_FUNC init_fn;
81 
82 #if defined(WINDOWS)
83  (*handle) = LoadLibrary (path);
84 #elif defined(_AIX)
85  (*handle) = dlopen (path, RTLD_NOW | RTLD_MEMBER);
86 #else
87  (*handle) = dlopen (path, RTLD_NOW | RTLD_GLOBAL);
88 #endif
89  if ((*handle) == 0)
90  {
91  // todo: generate verbose error
92  return ER_GENERIC_ERROR;
93  }
94 
95  /* initialize library */
96  if (utility_load_symbol (*handle, (DSO_HANDLE *) (&init_fn), UTILITY_INIT_FUNC_NAME) == NO_ERROR
97  && (*init_fn) () == NO_ERROR)
98  {
99  return NO_ERROR;
100  }
101  else
102  {
103  *handle = NULL;
104  return ER_GENERIC_ERROR;
105  }
106 }

110  *
111  * return:
112  *

114  */
115 int
116 utility_load_symbol (DSO_HANDLE library_handle, DSO_HANDLE * symbol_handle, const char *symbol_name)
117 {
118 #if defined(WINDOWS)
119  (*symbol_handle) = GetProcAddress ((HMODULE) library_handle, symbol_name);
120  return (*symbol_handle) == NULL ? ER_GENERIC_ERROR : NO_ERROR;
121 #else
122  (*symbol_handle) = dlsym (library_handle, symbol_name);
123  return (*symbol_handle) == 0 ? ER_GENERIC_ERROR : NO_ERROR;

179 /*
180  * util_parse_argument - parse arguments of an utility
181  *
182  * return:
183  *
184  * NOTE:
185  */
186 int
187 util_parse_argument (UTIL_MAP * util_map, int argc, char **argv)
188 {
189  int status;
190  int option_value;
191  int option_index;
192  char option_string[64];
193  GETOPT_LONG *option = util_map->getopt_long;
194  utility_make_getopt_optstring (option, option_string);
195  status = NO_ERROR;
196  while (status == NO_ERROR)
197  {
198  option_index = 0;
199  option_value = getopt_long (argc, argv, option_string, option, &option_index);
200  if (option_value == -1)
201  {
202  break;
203  }
204  else if (option_value == '?' || option_value == ':')
205  {
206  status = ER_FAILED;
207  return status;
208  }
209  status = util_put_option_value (util_map, option_value, optarg);
210  if (status != NO_ERROR)
211  {
212  fprintf (stderr, "invalid '--%s' option value: %s\n", util_get_option_name (option, option_value), optarg);
213  return ER_FAILED;
214  }
215 
216  if (util_is_password_argument (util_map->utility_index, option_value))
217  {
218  util_hide_password (optarg);
219  }
220  }
221 
222  status = util_parse_string_table (util_map, optind, argc, argv);
223  return status;
224 }

246 /*
247  * util_put_option_value - put parsed an argument's value to the utility map
248  *
249  * return:
250  *
251  * NOTE: An allocated memory may be leaked by the strdup.
252  */
253 static int
254 util_put_option_value (UTIL_MAP * util_map, int arg_ch, const char *option_arg)
255 {
256  int i;
257  UTIL_ARG_MAP *arg_map = util_map->arg_map;
258 
259  for (i = 0; arg_map[i].arg_ch; i++)
260  {
261  if (arg_map[i].arg_ch == arg_ch)
262  {
263  switch (arg_map[i].value_info.value_type)
264  {
265  case ARG_BOOLEAN:
266  arg_map[i].arg_value.i = 1;
267  return NO_ERROR;
268  case ARG_INTEGER:
269  {
270  int value = 0, result;
271 
272  result = parse_int (&value, option_arg, 10);
273 
274  if (result != 0)
275  {
276  return ER_FAILED;
277  }
278 
279  arg_map[i].arg_value.i = value;
280  return NO_ERROR;
281  }
282  case ARG_BIGINT:
283  {
284  int result = 0;
285  INT64 value;
286 
287  result = parse_bigint (&value, option_arg, 10);
288 
289  if (result != 0)
290  {
291  return ER_FAILED;
292  }
293 
294  arg_map[i].arg_value.l = value;
295  return NO_ERROR;
296  }
297  case ARG_STRING:
298  if (option_arg[0] == '-')
299  {
300  return ER_FAILED;
301  }
302 
303  arg_map[i].arg_value.p = strdup (option_arg);
304  return NO_ERROR;
305  default:
306  return ER_FAILED;
307  }
308  return NO_ERROR;
309  }
310  }
311  return ER_FAILED;
312 }

315  * util_parse_string_table - parse non-option arguments
316  *
317  * return:
318  *
319  * NOTE: An allocated memory may be leaked by the malloc and the strdup.
320  */
321 static int
322 util_parse_string_table (UTIL_MAP * util_map, int index, int count, char **argv)
323 {
324  int i;
325  int need_args_num;
326  char **string_table;
327  UTIL_ARG_MAP *string_table_arg = NULL;
328  int num_string_args;
329  for (i = 0; i < util_map->arg_map[i].arg_ch; i++)
330  {
331  if (util_map->arg_map[i].arg_ch == OPTION_STRING_TABLE)
332  {
333  string_table_arg = &util_map->arg_map[i];
334  need_args_num = util_map->need_args_num;
335  break;
336  }
337  }
338  if (string_table_arg == NULL)
339  {
340  return ER_FAILED;
341  }
342  num_string_args = count - index;
343  string_table = (char **) malloc (sizeof (char *) * num_string_args);
344  if (string_table == NULL)
345  {
346  return ER_FAILED;
347  }
348  memset (string_table, 0, sizeof (char *) * num_string_args);
349  for (i = 0; index < count; index++, i++)
350  {
351  string_table[i] = argv[index];
352  /* fprintf (stdout, "%s\n", (*string_table)[i]); */
353  }
354  string_table_arg->arg_value.p = string_table;
355  string_table_arg->value_info.num_strings = num_string_args;
356  if (need_args_num < num_string_args && (util_map->utility_index != COMPACTDB && util_map->utility_index != CHECKDB))
357  {
358  fprintf (stderr, "'%s' argument is not needed.\n",
359  string_table[need_args_num] == NULL ? "" : string_table[need_args_num]);
360  return ER_FAILED;
361  }
362  return NO_ERROR;
363 }
