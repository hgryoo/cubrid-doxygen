177 /*
178  * css_tcp_client_open_with_retry() -
179  * return:
180  * hostname(in):
181  * port(in):
182  * willretry(in):
183  */
184 SOCKET
185 css_tcp_client_open_with_retry (const char *host_name, int port, bool will_retry)
186 {
187  int bool_value;
188  SOCKET s;
189  int err;
190  struct hostent *dest_host;
191  unsigned int remote_ip;
192  struct sockaddr_in addr;
193  int success, numtries;
194 
195  /* Winsock must have been opened by now */
196 
197  /* first try the internet address format */
198  remote_ip = inet_addr (host_name);
199  if (remote_ip == INADDR_NONE)
200  {
201  /* then try a host name */
202  dest_host = gethostbyname (host_name);
203  if (dest_host == NULL)
204  {
205  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_CSS_WINSOCK_HOSTNAME, 1, WSAGetLastError ());
206  return INVALID_SOCKET;
207  }
208  remote_ip = *((unsigned int *) (dest_host->h_addr));
209  }
210 
211  success = 0;
212  if (will_retry)
213  {
214  numtries = 0;
215  }
216  else
217  {
218  numtries = css_Tcp_max_connect_tries - 1;
219  }
220 
221  while (!success && numtries < css_Tcp_max_connect_tries)
222  {
223  s = socket (PF_INET, SOCK_STREAM, IPPROTO_TCP);
224  if (IS_INVALID_SOCKET (s))
225  {
226  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_WINTCP_CANNOT_CREATE_STREAM, 1, WSAGetLastError ());
227  return INVALID_SOCKET;
228  }
229 
230  addr.sin_family = AF_INET;
231  addr.sin_port = htons (port);
232  addr.sin_addr.s_addr = remote_ip;
233 
234  /*
235  * The master deliberately tries to connect to itself once to
236  * prevent multiple masters from running. In the "good" case,
237  * the connect will fail. Printing the message when that happens
238  * makes starting the master on NT disturbing because the users sees
239  * what they think is a bad message so don't print anything here.
240  */
241  if (connect (s, (struct sockaddr *) &addr, sizeof (addr)) != SOCKET_ERROR)
242  {
243  success = 1;
244  }
245  else
246  {
247  err = WSAGetLastError ();
248  (void) closesocket (s);
249 
250  if (err != WSAECONNREFUSED && err != WSAETIMEDOUT)
251  {
252  /* this isn't an error we handle */
253  return INVALID_SOCKET;
254  }
255  else
256  {
257  /*
258  * See tcp.c for Unix platforms for more information.
259  * retry the connection a few times in case the server is
260  * overloaded at the moment.
261  * Should be sleeping here but I can't find a Windows SDK function
262  * to do that.
263  */
264  numtries++;
265  }
266  }
267  }
268 
269  if (!success)
270  {
271  return INVALID_SOCKET;
272  }
273 
274  bool_value = 1;
275  /* ask for the "keep alive" option, ignore errors */
276  if (prm_get_bool_value (PRM_ID_TCP_KEEPALIVE))
277  {
278  (void) setsockopt (s, SOL_SOCKET, SO_KEEPALIVE, (const char *) &bool_value, sizeof (int));
279  }
280 
281  /* ask for NODELAY, this one is rather important */
282  (void) setsockopt (s, IPPROTO_TCP, TCP_NODELAY, (const char *) &bool_value, sizeof (int));
283 
284  return s;
285 }

516 /*
517  * css_tcp_master_open() - initialize for the master server internet
518  * communication
519  * return:
520  * port(in):
521  * sockfd(in):
522  */
523 int
524 css_tcp_master_open (int port, SOCKET * sockfd)
525 {
526  struct sockaddr_in tcp_srv_addr; /* server's internet socket addr */
527  SOCKET sock;
528  int retry_count = 0;
529  int bool_value;
530 
531  /*
532  * We have to create a socket ourselves and bind our well-known
533  * address to it.
534  */
535  if (css_windows_startup () < 0)
536  {
537  return ER_CSS_WINSOCK_STARTUP;
538  }
539 
540  memset ((void *) &tcp_srv_addr, 0, sizeof (tcp_srv_addr));
541  tcp_srv_addr.sin_family = AF_INET;
542  tcp_srv_addr.sin_addr.s_addr = htonl (INADDR_ANY);
543 
544  if (port > 0)
545  {
546  tcp_srv_addr.sin_port = htons (port);
547  }
548  else
549  {
550  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_WINTCP_PORT_ERROR, 0);
551  return ERR_CSS_WINTCP_PORT_ERROR;
552  }
553 
554  /*
555  * Create the socket and Bind our local address so that any
556  * client may send to us.
557  */
558 
559 retry:
560  sock = socket (AF_INET, SOCK_STREAM, 0);
561  if (IS_INVALID_SOCKET (sock))
562  {
563  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_WINTCP_CANNOT_CREATE_STREAM, 1, WSAGetLastError ());
564  *sockfd = sock;
565  return ERR_CSS_WINTCP_CANNOT_CREATE_STREAM;
566  }
567 
568  /*
569  * Allow the new master to rebind the CUBRID port even if there are
570  * clients with open connections from previous masters.
571  */
572  bool_value = 0;
573  setsockopt (sock, SOL_SOCKET, SO_REUSEADDR, (const char *) &bool_value, sizeof (int));
574 
575  bool_value = 1;
576  setsockopt (sock, IPPROTO_TCP, TCP_NODELAY, (const char *) &bool_value, sizeof (int));
577 
578  if (bind (sock, (struct sockaddr *) &tcp_srv_addr, sizeof (tcp_srv_addr)) == SOCKET_ERROR)
579  {
580  if (WSAGetLastError () == WSAEADDRINUSE && retry_count <= 5)
581  {
582  retry_count++;
583  sleep (1);
584  css_shutdown_socket (sock);
585  goto retry;
586  }
587 
588  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_WINTCP_BIND_ABORT, 1, WSAGetLastError ());
589  css_shutdown_socket (sock);
590  *sockfd = sock;
591  return ERR_CSS_WINTCP_BIND_ABORT;
592  }
593 
594  /*
595  * And set the listen parameter, telling the system that we're
596  * ready to accept incoming connection requests.
597  */
598  listen (sock, css_Maximum_server_count);
599 
600  *sockfd = sock;
601  return NO_ERROR;
602 }

604 /*
605  * css_accept() - accept of a request from a client
606  * return:
607  * sockfd(in):
608  */
609 static SOCKET
610 css_accept (SOCKET sockfd)
611 {
612  struct sockaddr_in tcp_cli_addr;
613  SOCKET newsockfd;
614  int clilen, error;
615 
616  while (true)
617  {
618  clilen = sizeof (tcp_cli_addr);
619  newsockfd = accept (sockfd, (struct sockaddr *) &tcp_cli_addr, &clilen);
620 
621  if (IS_INVALID_SOCKET (newsockfd))
622  {
623  error = WSAGetLastError ();
624  if (error == WSAEINTR)
625  {
626  continue;
627  }
628 
629  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_WINTCP_ACCEPT_ERROR, 1, error);
630  return INVALID_SOCKET;
631  }
632 
633  break;
634  }
635 
636  return newsockfd;
637 }

651 /*
652  * css_open_server_connection_socket() - open the socket used by the server
653  * for incoming client connection
654  * requests
655  * return: port id
656  */
657 int
658 css_open_server_connection_socket (void)
659 {
660  struct sockaddr_in tcp_srv_addr; /* server's internet socket addr */
661  SOCKET fd;
662  int get_length;
663  int bool_value;
664 
665 #if !defined(SERVER_MODE)
666  if (css_windows_startup () < 0)
667  {
668  return -1;
669  }
670 #endif /* not SERVER_MODE */
671 
672  /* Create the socket */
673  fd = socket (AF_INET, SOCK_STREAM, 0);
674  if (IS_INVALID_SOCKET (fd))
675  {
676  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_WINTCP_CANNOT_CREATE_STREAM, 1, WSAGetLastError ());
677  return -1;
678  }
679 
680  bool_value = 1;
681  setsockopt (fd, IPPROTO_TCP, TCP_NODELAY, (const char *) &bool_value, sizeof (int));
682 
683  if (prm_get_bool_value (PRM_ID_TCP_KEEPALIVE))
684  {
685  setsockopt (fd, SOL_SOCKET, SO_KEEPALIVE, (const char *) &bool_value, sizeof (int));
686  }
687 
688  /*
689  * Set up an address asking for "any" (the local ?) IP addres
690  * and set the port to zero to it will be automatically assigned.
691  */
692  memset ((void *) &tcp_srv_addr, 0, sizeof (tcp_srv_addr));
693  tcp_srv_addr.sin_family = AF_INET;
694  tcp_srv_addr.sin_addr.s_addr = htonl (INADDR_ANY);
695  tcp_srv_addr.sin_port = 0;
696 
697 
698  /* Bind the socket */
699  if (bind (fd, (struct sockaddr *) &tcp_srv_addr, sizeof (tcp_srv_addr)) == SOCKET_ERROR)
700  {
701  goto error;
702  }
703 
704  /* Determine which port_id the system has assigned. */
705  get_length = sizeof (tcp_srv_addr);
706  if (getsockname (fd, (struct sockaddr *) &tcp_srv_addr, &get_length) == SOCKET_ERROR)
707  {
708  goto error;
709  }
710 
711  /*
712  * Set it up to listen for incoming connections. Note that under Winsock
713  * (NetManage version at least), the backlog parameter is silently limited
714  * to 5, regardless of what is requested.
715  */
716  if (listen (fd, css_Maximum_server_count) == SOCKET_ERROR)
717  {
718  goto error;
719  }
720 
721  css_Server_connection_socket = fd;
722 
723  return (int) ntohs (tcp_srv_addr.sin_port);
724 
725 error:
726  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_WINTCP_BIND_ABORT, 1, WSAGetLastError ());
727  css_shutdown_socket (fd);
728  return -1;
729 }
