45 typedef struct table_info TABLE_INFO;
46 struct table_info
47 {
48  struct table_info *next; /* usual list link */
49  PT_NODE *class_spec; /* SHARED pointer to parse tree entity spec */
50  const char *exposed; /* SHARED pointer to entity spec exposed name */
51  UINTPTR spec_id;
52  PT_NODE *attribute_list; /* is a list of names which appear anywhere in a select statement with the exposed name
53  */
54  VAL_LIST *value_list; /* is a list of DB_VALUES which correspond by position in list to the attributes named
55  * in attribute_list */
56  int is_fetch;
57 };

59 /*
60  * This structure represents the global information needed to be stored
61  * to translate a parse tree into XASL. It is only used in the process of
62  * translation. A new one is "pushed" on the stack every time a query
63  * statement is to be translated, and "popped" when its translation is
64  * complete. Nested select statements will then yield a stack of these
65  * which can be used to resolve attribute names referenced.
66  */
67 typedef enum
68 { UNBOX_AS_VALUE, UNBOX_AS_TABLE } UNBOX;
