140 virtual HRESULT IsValidValue (ULONG iCurSet, DBPROP * pDBProp)
141  {
142 
143 ATLASSERT (pDBProp);
144 
145 if (pDBProp->dwPropertyID == DBPROP_INIT_ASYNCH)
146 
147  {
148 
149 LONG val = V_I4 (&pDBProp->vValue);
150 
151 if (val == 0 || val == DBPROPVAL_ASYNCH_INITIALIZE)
152 
153 return S_OK;
154 
155 return S_FALSE;
156 
157 }
158 
159 if (pDBProp->dwPropertyID == DBPROP_INIT_PROMPT)
160 
161  {
162 
163 short val = V_I2 (&pDBProp->vValue);
164 
165 if (val == DBPROMPT_NOPROMPT || val == DBPROMPT_COMPLETE ||
166 val == DBPROMPT_COMPLETEREQUIRED
167  || val == DBPROMPT_PROMPT)
168 
169 return S_OK;
170 
171 return S_FALSE;
172 
173 }
174 
175 return IDBPropertiesImpl < CCUBRIDDataSource >::IsValidValue (iCurSet, pDBProp);
176 
177 }
