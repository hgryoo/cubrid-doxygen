73 static HRESULT FetchData(int hReq, CCOLUMNSRow &crData)
74 {
75  char *cvalue;
76  int ivalue, ind, res;
77  T_CCI_ERROR err_buf;
78  int nCCIUType, nPrecision, nScale;
79  bool bNullable;
80 
81  res = cci_fetch(hReq, &err_buf);
82  if(res<0) return RaiseError(E_FAIL, 1, __uuidof(IDBSchemaRowset), err_buf.err_msg);
83 
84  res = cci_get_data(hReq, 11, CCI_A_TYPE_STR, &cvalue, &ind);
85  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
86 
87  wcsncpy(crData.m_szTableName, CA2W(cvalue), 128);
88  crData.m_szTableName[128] = 0;
89  _wcsupr(crData.m_szTableName);
90 
91  res = cci_get_data(hReq, 1, CCI_A_TYPE_STR, &cvalue, &ind);
92  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
93 
94  wcsncpy(crData.m_szColumnName, CA2W(cvalue), 128);
95  crData.m_szColumnName[128] = 0;
96  _wcsupr(crData.m_szColumnName);
97 
98  res = cci_get_data(hReq, 2, CCI_A_TYPE_INT, &ivalue, &ind);
99  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
100 
101  nCCIUType = ivalue;
102 
103  if(nCCIUType==-1)
104  crData.m_nDataType = Type::GetStaticTypeInfo(CCI_U_TYPE_STRING).nOLEDBType;
105  else
106  crData.m_nDataType = Type::GetStaticTypeInfo(ivalue).nOLEDBType;
107 
108  res = cci_get_data(hReq, 3, CCI_A_TYPE_INT, &ivalue, &ind);
109  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
110  nScale = ivalue;
111 
112  res = cci_get_data(hReq, 4, CCI_A_TYPE_INT, &ivalue, &ind);
113  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
114  nPrecision = ivalue;
115 
116  res = cci_get_data(hReq, 6, CCI_A_TYPE_INT, &ivalue, &ind);
117  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
118  crData.m_bIsNullable = ( ivalue==0 ? ATL_VARIANT_TRUE : ATL_VARIANT_FALSE );
119  bNullable = (ivalue==0);
120 
121  res = cci_get_data(hReq, 10, CCI_A_TYPE_INT, &ivalue, &ind);
122  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
123  crData.m_ulOrdinalPosition = ivalue;
124 
125  res = cci_get_data(hReq, 9, CCI_A_TYPE_STR, &cvalue, &ind);
126  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
127  if(ind==-1)
128  {
129  crData.m_bColumnHasDefault = ATL_VARIANT_FALSE;
130  crData.m_szColumnDefault[0] = L'';
131  }
132  else
133  {
134  crData.m_bColumnHasDefault = ATL_VARIANT_TRUE;

136  crData.m_szColumnDefault[128] = 0;
137  }

154 HRESULT CSRColumns::Execute(LONG * /*pcRowsAffected*/,
155  ULONG cRestrictions, const VARIANT *rgRestrictions)
156 {
157  ATLTRACE2(atlTraceDBProvider, 2, "CSRColumns::Execute\n");
158 
159  ClearError();
160 
161  int hConn = -1;
162  HRESULT hr = CCUBRIDSession::GetSessionPtr(this)->GetConnectionHandle(&hConn);
163  if(FAILED(hr)) return hr;
164 
165  ULONG ulMaxLen = CCUBRIDSession::GetSessionPtr(this)->GetDataSourcePtr()->PARAM_MAX_STRING_LENGTH;
166 
167  char table_name[1024]; table_name[0] = 0;
168  char column_name[1024]; column_name[0] = 0;
169  GetRestrictions(cRestrictions, rgRestrictions, table_name, column_name);
170 
171  {
172  T_CCI_ERROR err_buf;
173  int hReq = cci_schema_info(hConn, CCI_SCH_ATTRIBUTE,
174  (table_name[0]?table_name:NULL),
175  (column_name[0]?column_name:NULL),
176  CCI_CLASS_NAME_PATTERN_MATCH | CCI_ATTR_NAME_PATTERN_MATCH,
177  &err_buf);
178  if(hReq<0)
179  {
180  ATLTRACE2("cci_schema_info fail\n");
181  return E_FAIL;
182  }
183 
184  int res = cci_cursor(hReq, 1, CCI_CURSOR_FIRST, &err_buf);
185  if(res==CCI_ER_NO_MORE_DATA) goto done;
186  if(res<0) goto error;
187  while(1)
188  {
189  CCOLUMNSRow crData;
190  hr = FetchData(hReq, crData);
191  if(FAILED(hr)) goto error;
192 
193  //MAX STRING LENGTH 반영
194  if (crData.m_ulCharMaxLength != (ULONG)~0 && crData.m_ulCharMaxLength > ulMaxLen)
195  crData.m_ulCharMaxLength = ulMaxLen;
196  if (crData.m_ulCharMaxLength != (ULONG)~0 && crData.m_ulCharOctetLength > ulMaxLen)
197  crData.m_ulCharOctetLength = ulMaxLen;
198
