44 static void GetRestrictions(ULONG cRestrictions, const VARIANT *rgRestrictions, char *table_name, char* index_name)
45 {
46  // restriction이 없다고 항상 cRestrictions==0은 아니다.
47  // 따라서 vt!=VT_EMPTY인지 등도 검사해줘야 한다.
48 
49  if(cRestrictions>=3 && V_VT(&rgRestrictions[2])==VT_BSTR && V_BSTR(&rgRestrictions[2])!=NULL)
50  { // INDEX_NAME restriction
51  CW2A name(V_BSTR(&rgRestrictions[2]));
52  ATLTRACE2("\tIndex Name = %s\n", (LPSTR)name);
53 
54  strncpy(index_name, name, 255);
55  table_name[255] = 0; // ensure zero-terminated string
56  }
57 
58  if(cRestrictions>=5 && V_VT(&rgRestrictions[4])==VT_BSTR && V_BSTR(&rgRestrictions[4])!=NULL)
59  { // TABLE_NAME restriction
60  CW2A name(V_BSTR(&rgRestrictions[4]));
61  ATLTRACE2("\tTable Name = %s\n", (LPSTR)name);
62 
63  strncpy(table_name, name, 255);
64  table_name[255] = 0; // ensure zero-terminated string
65  }
66 }
