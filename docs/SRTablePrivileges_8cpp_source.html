93 HRESULT CSRTablePrivileges::Execute(LONG * /*pcRowsAffected*/,
94  ULONG cRestrictions, const VARIANT *rgRestrictions)
95 {
96  ATLTRACE2(atlTraceDBProvider, 2, "CSRTablePrivileges::Execute\n");
97 
98  ClearError();
99 
100  int hConn = -1;
101  HRESULT hr = CCUBRIDSession::GetSessionPtr(this)->GetConnectionHandle(&hConn);
102  if(FAILED(hr)) return hr;
103 
104  char table_name[1024]; table_name[0] = 0;
105  GetRestrictions(cRestrictions, rgRestrictions, table_name);
106 
107  CComVariant grantee;
108  hr = GetCurrentUser(this, grantee);
109  if(FAILED(hr)) return hr;
110 
111  {
112  T_CCI_ERROR err_buf;
113  int hReq = cci_schema_info(hConn, CCI_SCH_CLASS_PRIVILEGE, (table_name[0]?table_name:NULL),
114  NULL, CCI_CLASS_NAME_PATTERN_MATCH, &err_buf);
115  if(hReq<0)
116  {
117  ATLTRACE2("cci_schema_info fail\n");
118  return RaiseError(E_FAIL, 1, __uuidof(IDBSchemaRowset), err_buf.err_msg);
119  }
120 
121  int res = cci_cursor(hReq, 1, CCI_CURSOR_FIRST, &err_buf);
122  if(res==CCI_ER_NO_MORE_DATA) goto done;
123  if(res<0) goto error;
124 
125  while(1)
126  {
127  CTablePrivilegesRow tprData;
128  wcscpy(tprData.m_szGrantor, L"DBA");
129  wcscpy(tprData.m_szGrantee, V_BSTR(&grantee));

131  if(FAILED(hr))

133  cci_close_req_handle(hReq);

135  }
136 
137  if(hr==S_OK) // S_FALSE면 추가하지 않음
138  {
139  _ATLTRY

141  // TABLE_NAME, PRIVILEGE_TYPE 순으로 정렬해야 한다.
142  size_t nPos;

144  {
145  int res = wcscmp(m_rgRowData[nPos].m_szTableName, tprData.m_szTableName);
146  if(res>0) break;
147  if(res==0 && wcscmp(m_rgRowData[nPos].m_szPrivilegeType, tprData.m_szPrivilegeType)>0) break;
148  }
149  m_rgRowData.InsertAt(nPos, tprData);
150  }
151  _ATLCATCHALL()
152  {
153  ATLTRACE2("out of memory\n");
154  cci_close_req_handle(hReq);
155  return E_OUTOFMEMORY;
156  }
157  }
158 
159  res = cci_cursor(hReq, 1, CCI_CURSOR_CURRENT, &err_buf);
160  if(res==CCI_ER_NO_MORE_DATA) goto done;
161  if(res<0) goto error;
162  }
163 
164 error:
165  ATLTRACE2("fail to fetch data\n");

167  return RaiseError(E_FAIL, 1, __uuidof(IDBSchemaRowset), err_buf.err_msg);
