45 {
46  // restriction이 없다고 항상 cRestrictions==0은 아니다.
47  // 따라서 vt!=VT_EMPTY인지 등도 검사해줘야 한다.
48 
49  if(cRestrictions>=3 && V_VT(&rgRestrictions[2])==VT_BSTR && V_BSTR(&rgRestrictions[2])!=NULL)
50  { // TABLE_NAME restriction
51  CW2A name(V_BSTR(&rgRestrictions[2]));
52  ATLTRACE2("\tTable Name = %s\n", (LPSTR)name);
53 
54  strncpy(table_name, name, 255);
55  table_name[255] = 0; // ensure zero-terminated string
56  }
57 
58  if(cRestrictions>=4 && V_VT(&rgRestrictions[3])==VT_BSTR && V_BSTR(&rgRestrictions[3])!=NULL)
59  { // TABLE_NAME restriction
60  CW2A name(V_BSTR(&rgRestrictions[3]));
61  ATLTRACE2("\tTable Type = %s\n", (LPSTR)name);
62 
63  *table_type = 3; // wrong restriction
64  if (!strcmp(name, "TABLE"))
65  *table_type = 2;
66  else if (!strcmp(name, "VIEW"))
67  *table_type = 1;
68  else if (!strcmp(name, "SYSTEM TABLE"))
69  *table_type = 0;
70  }
71 }

76 static HRESULT FetchData(int hReq, CTablesInfoRow &tirData, int table_type)
77 {
78  char *value;
79  int int_value;
80  int ind, res;
81  T_CCI_ERROR err_buf;
82 
83  res = cci_fetch(hReq, &err_buf);
84  if(res<0) return RaiseError(E_FAIL, 1, __uuidof(IDBSchemaRowset), err_buf.err_msg);
85 
86  res = cci_get_data(hReq, 1, CCI_A_TYPE_STR, &value, &ind);
87  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
88  wcscpy(tirData.m_szTableName, CA2W(value));
89 
90  res = cci_get_data(hReq, 2, CCI_A_TYPE_INT, &int_value, &ind);
91  if(res<0) return RaiseError(E_FAIL, 0, __uuidof(IDBSchemaRowset));
92 
93  if (table_type >= 0 && table_type != int_value)
94  return S_FALSE;
95 
96  if (int_value == 2)
97  wcscpy(tirData.m_szTableType, L"TABLE");
98  else if (int_value == 1)
99  wcscpy(tirData.m_szTableType, L"VIEW");
100  else if (int_value == 0)
101  wcscpy(tirData.m_szTableType, L"SYSTEM TABLE");
102 
103  tirData.m_bBookmarks = ATL_VARIANT_TRUE;
104  tirData.m_bBookmarkType = DBPROPVAL_BMK_NUMERIC;
105  tirData.m_bBookmarkDatatype = DBTYPE_BYTES;
106  tirData.m_bBookmarkMaximumLength = sizeof(DBROWCOUNT);
107 
108  return S_OK;
109 }

124  if (hReq < 0)

127  res = cci_execute(hReq, CCI_EXEC_QUERY_ALL, 0, &error);

129  return S_OK;

131  res = cci_cursor(hReq, 1, CCI_CURSOR_FIRST, &error);

134  res = cci_fetch(hReq, &error);
135  if(res<0) return S_OK;
136 
137  res = cci_get_data(hReq, 1, CCI_A_TYPE_INT, &cardinality->QuadPart, &ind);
138  if(res<0) return S_OK;
139 
140  cci_close_req_handle(hReq);
141 
142  return S_OK;
143 }

145 HRESULT CSRTablesInfo::Execute(LONG* pcRowsAffected,
146  ULONG cRestrictions, const VARIANT *rgRestrictions)
147 {
148  ATLTRACE2(atlTraceDBProvider, 2, "CSRTablesInfo::Execute\n");
149 
150  ClearError();
151 
152  int hConn = -1;
153  HRESULT hr = CCUBRIDSession::GetSessionPtr(this)->GetConnectionHandle(&hConn);
154  if(FAILED(hr)) return hr;
155 
156  char table_name[256]; table_name[0] = 0;
157  int table_type = -1;
158 
159  GetRestrictions(cRestrictions, rgRestrictions, table_name, &table_type);
160  {
161  T_CCI_ERROR err_buf;
162  int hReq = cci_schema_info(hConn, CCI_SCH_CLASS, (table_name[0]?table_name:NULL),
163  NULL, CCI_CLASS_NAME_PATTERN_MATCH, &err_buf);
164  if(hReq<0)
165  {
166  ATLTRACE2("cci_schema_info fail\n");
167  return E_FAIL;
168  }
169 
170  int res = cci_cursor(hReq, 1, CCI_CURSOR_FIRST, &err_buf);
171  if(res==CCI_ER_NO_MORE_DATA) goto done;
172  if(res<0) goto error;
173 
174  while(1)
175  {
176  CTablesInfoRow tirData;
177  hr = FetchData(hReq, tirData, table_type);
178  if(FAILED(hr))
179  {
180  cci_close_req_handle(hReq);
181  return hr;
182  }
183  HRESULT hrCard = GetCardinality(hConn, tirData.m_szTableName, &tirData.m_ulCardinality);
184  if(FAILED(hrCard))
185  {
186  cci_close_req_handle(hReq);
187  return hrCard;
188  };
189 
190  if(hr==S_OK) // S_FALSE면 추가하지 않음
191  {
