20 /*
21  * area_alloc.c - Area memory manager
22  *
23  * Note:
24  * Allocation areas provide a way to block allocate structures and maintain
25  * a free list. Useful for small structures that are used frequently.
26  * Used for allocation and freeing of many small objects of the same size.
27  *
28  * These areas are NOT allocated within the "workspace" memory so that
29  * they can be used for structures that need to serve as roots to the
30  * garbage collector.
31  */

102 /*
103  * area_final - Shut down the area manager
104  * return: none
105  *
106  * Note: Will be called during system shutdown
107  */
108 void
109 area_final (void)
110 {
111  AREA *area, *next;
112 
113  for (area = area_List, next = NULL; area != NULL; area = next)
114  {
115  next = area->next;
116  area_flush (area);
117  free_and_init (area);
118  }
119  area_List = NULL;
120 
121  Set_Ref_Area = Set_Obj_Area = NULL;
122  pthread_mutex_destroy (&area_List_lock);

125 /*
126  * area_create - Build a new area and add it to the global list
127  * return: created AREA or NULL if fail
128  * name(in):
129  * element_size(in):
130  * alloc_count(in):
131  *
132  * Note:
133  */
134 AREA *
135 area_create (const char *name, size_t element_size, size_t alloc_count)
136 {
137  AREA *area;
138  size_t adjust;
139 #if defined (SERVER_MODE)
140  int rv;
141 #endif /* SERVER_MODE */
142 
143  area = (AREA *) malloc (sizeof (AREA));
144  if (area == NULL)
145  {
146  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (AREA));
147  return NULL;
148  }
149  area->blockset_list = NULL;
150 
151  if (name == NULL)
152  {
153  area->name = NULL;
154  }
155  else
156  {
157  area->name = strdup (name);
158  if (area->name == NULL)
159  {
160  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, (size_t) (strlen (name) + 1));
161  goto error;
162  }
163  }
164 
165 #if !defined (NDEBUG)
166  /* Reserve space for memory checking */
167  element_size += AREA_PREFIX_SIZE;
168 #endif /* !NDEBUG */
169 
170  /* always make sure element size is a double word multiple */
171  adjust = element_size % 8;
172  if (adjust)
173  {
174  element_size += 8 - adjust;
175  }
176  area->element_size = element_size;
177 
178  /* adjust alloc count for lf_bitmap */
179  area->alloc_count = LF_BITMAP_COUNT_ALIGN (alloc_count);
180  area->block_size = area->element_size * area->alloc_count;
181 
182  area->n_allocs = 0;
183  area->n_frees = 0;
184 #if defined (SERVER_MODE)
185  area->failure_function = NULL;
186 #else
187  area->failure_function = ws_abort_transaction;
188 #endif
189 
190  area->blockset_list = area_alloc_blockset (area);
191  if (area->blockset_list == NULL)
192  {
193  goto error;
194  }
195 
196  area->hint_block = area_alloc_block (area);
197  if (area->hint_block == NULL)
198  {
199  goto error;
200  }
201  area->blockset_list->items[0] = area->hint_block;
202  area->blockset_list->used_count++;
203 
204  pthread_mutex_init (&area->area_mutex, NULL);
205 
206  rv = pthread_mutex_lock (&area_List_lock);
207  area->next = area_List;
208  area_List = area;
209  pthread_mutex_unlock (&area_List_lock);
210 
211  return area;
212 
213 error:
214 
215  if (area->name)
216  {
217  free_and_init (area->name);
218  }
219 
220  if (area->blockset_list != NULL)
221  {
222  free_and_init (area->blockset_list);
223  }
224 
225  free_and_init (area);
226 
227  return NULL;
228 }

230 /*
231  * area_destroy - Removes an area
232  * return: none
233  * area(in): AREA tp destroy
234  */
235 void
236 area_destroy (AREA * area)
237 {
238  AREA *a, *prev;
239 #if defined(SERVER_MODE)
240  int rv;
241 #endif /* SERVER_MODE */
242 
243  assert (area != NULL);
244 
245  rv = pthread_mutex_lock (&area_List_lock);
246 
247  for (prev = NULL, a = area_List; a != NULL && a != area; a = a->next)
248  {
249  prev = a;
250  }
251 
252  if (a != NULL)
253  {
254  if (prev == NULL)
255  {
256  area_List = a->next;
257  }
258  else
259  {
260  prev->next = a->next;
261  }
262  }
263 
264  pthread_mutex_unlock (&area_List_lock);
265 
266  area_flush (area);
267 
268  free_and_init (area);
269 }

271 /*
272  * area_alloc_block - Allocate a new block for an area
273  * return: the address of area block, if error, return NULL.
274  * area(in): AREA
275  * thrd_index(in): thread index
276  *
277  * Note: this is called by area_alloc, and lock is also held by area_alloc
278  */
279 static AREA_BLOCK *
280 area_alloc_block (AREA * area)
281 {
282  AREA_BLOCK *new_block;
283  size_t total;
284 
285  assert (area != NULL);
286 
287  total = area->block_size + sizeof (AREA_BLOCK);
288 
289  new_block = (AREA_BLOCK *) malloc (total);
290  if (new_block == NULL)
291  {
292  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, total);
293  if (area->failure_function != NULL)
294  {
295  (*(area->failure_function)) ();
296  }
297 
298  return NULL;
299  }
300 
301  if (lf_bitmap_init (&new_block->bitmap, LF_BITMAP_LIST_OF_CHUNKS, (int) area->alloc_count, LF_AREA_BITMAP_USAGE_RATIO)
302  != NO_ERROR)
303  {
304  goto error;
305  }
306  assert ((int) area->alloc_count == new_block->bitmap.entry_count);
307 
308  new_block->data = ((char *) new_block) + sizeof (AREA_BLOCK);
309 
310  return new_block;
311 
312 error:
313  free_and_init (new_block);
314 
315  return NULL;
316 }

324 static AREA_BLOCKSET_LIST *
325 area_alloc_blockset (AREA * area)
326 {

328

330  if (new_blockset == NULL)
331  {
332  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (AREA_BLOCKSET_LIST));
333  if (area->failure_function != NULL)
334  {
335  (*(area->failure_function)) ();

337

348 /*
349  * area_alloc - Allocate a new element from an area
350  * return: pointer to the element allocated
351  * area(in):
352  *
353  * Note: The element will be taken from the area's free list,
354  * otherwise a new block will be allocated and the element
355  * taken from there
356  */
357 void *
358 area_alloc (AREA * area)
359 {
360  AREA_BLOCKSET_LIST *blockset;
361  AREA_BLOCK *block, *hint_block;
362  int used_count, i, entry_idx;
363  char *entry_ptr;
364 #if defined(SERVER_MODE)
365  int rv;
366 #endif /* SERVER_MODE */
367 #if !defined (NDEBUG)
368  int *prefix;
369 #endif /* !NDEBUG */
370 
371  assert (area != NULL);
372 
373  /* Step 1: find a free entry from the hint block */
374  hint_block = VOLATILE_ACCESS (area->hint_block, AREA_BLOCK *);
375  entry_idx = lf_bitmap_get_entry (&hint_block->bitmap);
376  if (entry_idx != -1)
377  {
378  block = hint_block;
379  goto found;
380  }
381 
382  /* Step 2: if not found, find a free entry from the blockset lists */
383  for (blockset = area->blockset_list; blockset != NULL;
384  blockset = VOLATILE_ACCESS (blockset->next, AREA_BLOCKSET_LIST *))
385  {
386  used_count = VOLATILE_ACCESS (blockset->used_count, int);
387  for (i = 0; i < used_count; i++)
388  {
389  block = VOLATILE_ACCESS (blockset->items[i], AREA_BLOCK *);
390 
391  entry_idx = lf_bitmap_get_entry (&block->bitmap);
392  if (entry_idx != -1)
393  {
394  /* change the hint block */
395  hint_block = VOLATILE_ACCESS (area->hint_block, AREA_BLOCK *);
396  if (LF_BITMAP_IS_FULL (&hint_block->bitmap) && !LF_BITMAP_IS_FULL (&block->bitmap))
397  {
398  ATOMIC_CAS_ADDR (&area->hint_block, hint_block, block);
399  }
400 
401  goto found;
402  }
403  }
404  }
405 
406  /* Step 3: if not found, add a new block. Then find free entry in this new block.
407  * Only one thread is allowed to add a new block at a moment.
408  */
409  rv = pthread_mutex_lock (&area->area_mutex);
410 
411  if (area->hint_block != hint_block)
412  {
413  /* someone may change the hint block */
414  block = area->hint_block;
415  entry_idx = lf_bitmap_get_entry (&block->bitmap);
416  if (entry_idx != -1)
417  {
418  pthread_mutex_unlock (&area->area_mutex);
419  goto found;
420  }
421  }
422 
423  block = area_alloc_block (area);
424  if (block == NULL)
425  {
426  pthread_mutex_unlock (&area->area_mutex);
427  /* error has been set */
428  return NULL;
429  }
430 
431  /* alloc free entry from this new block */
432  entry_idx = lf_bitmap_get_entry (&block->bitmap);
433  assert (entry_idx != -1);
434 
435  if (area_insert_block (area, block) != NO_ERROR)
436  {
437  lf_bitmap_destroy (&block->bitmap);
438  free_and_init (block);
439 
440  pthread_mutex_unlock (&area->area_mutex);
441  /* error has been set */
442  return NULL;
443  }
444 
445  /* always set new block as hint_block */
446  area->hint_block = block;
447 
448  pthread_mutex_unlock (&area->area_mutex);
449 
450 found:
451 
452 #if defined(SERVER_MODE)
453  /* do not count in SERVER_MODE */
454  /* ATOMIC_INC_32 (&area->n_allocs, 1); */
455 #else
456  area->n_allocs++;
457 #endif
458 
459  entry_ptr = block->data + area->element_size * entry_idx;
460 
461 #if !defined (NDEBUG)
462  prefix = (int *) entry_ptr;
463  *prefix = AREA_PREFIX_INITED;
464 
465  entry_ptr += AREA_PREFIX_SIZE;
466 #endif /* !NDEBUG */
467 
468  assert (entry_ptr < (block->data + area->block_size));
469 
470  return ((void *) entry_ptr);
471 }

474  * area_validate - validate that a pointer is within range in an area
475  * return: NO_ERROR if ok (address in range) or ER_AREA_ILLEGAL_POINTER

479  * Note: ER_AREA_ILLEGAL_POINTER will be set if fails.

503 /*
504  * area_free - Free an element in an area
505  * return: error code
506  * area(in): AREA
507  * ptr(in): pointer to the element
508  *
509  * Note: Validation is performed; the element is simply pushed on the free list
510  */
511 int
512 area_free (AREA * area, void *ptr)
513 {
514  AREA_BLOCK *block, *hint_block;
515  char *entry_ptr;
516  int error, entry_idx;
517  int offset = -1;
518 #if !defined (NDEBUG)
519  int *prefix;
520 #endif /* !NDEBUG */
521 
522  assert (area != NULL);
523 
524  if (ptr == NULL)
525  {
526  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_AREA_ILLEGAL_POINTER, 0);
527  assert (ptr != NULL);
528  return ER_AREA_ILLEGAL_POINTER;
529  }
530 
531 #if !defined (NDEBUG)
532  entry_ptr = ((char *) ptr) - AREA_PREFIX_SIZE;
533 #else
534  entry_ptr = (char *) ptr;
535 #endif /* !NDEBUG */
536 
537  block = area_find_block (area, (const void *) entry_ptr);
538  if (block == NULL)
539  {
540  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_AREA_ILLEGAL_POINTER, 0);
541  assert (false);
542  return ER_AREA_ILLEGAL_POINTER;
543  }
544 
545  offset = (int) (entry_ptr - block->data);
546  if (offset < 0 || offset % area->element_size != 0)
547  {
548  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_AREA_ILLEGAL_POINTER, 0);
549  assert (false);
550  return ER_AREA_ILLEGAL_POINTER;
551  }
552 
553 #if !defined (NDEBUG)
554  prefix = (int *) entry_ptr;
555  if ((*prefix) != AREA_PREFIX_INITED)
556  {
557  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_AREA_FREE_TWICE, 0);
558  assert ((*prefix) == AREA_PREFIX_INITED);
559  return ER_AREA_FREE_TWICE;
560  }
561  *prefix = AREA_PREFIX_FREED;
562 #endif /* !NDEBUG */
563 
564  entry_idx = offset / (int) area->element_size;
565 
566  assert (entry_idx >= 0 && entry_idx < (int) area->alloc_count);
567 
568  error = lf_bitmap_free_entry (&block->bitmap, entry_idx);
569  if (error != NO_ERROR)
570  {
571  return error;
572  }
573 
574  /* change hint block if needed */
575  hint_block = VOLATILE_ACCESS (area->hint_block, AREA_BLOCK *);
576  if (LF_BITMAP_IS_FULL (&hint_block->bitmap) && !LF_BITMAP_IS_FULL (&block->bitmap))
577  {
578  ATOMIC_CAS_ADDR (&area->hint_block, hint_block, block);
579  }
580 
581 #if defined(SERVER_MODE)
582  /* do not count in SERVER_MODE */
583  /* ATOMIC_INC_32 (&area->n_frees, 1); */
584 #else
585  area->n_frees++;
586 #endif
587 
588  return NO_ERROR;
589 }

591 /*
592  * area_flush - Free all storage allocated for an area
593  * return: none
594  * area(in): AREA to free
595  *
596  * Note: Normally called as part of final
597  */
598 void
599 area_flush (AREA * area)
600 {
601  AREA_BLOCKSET_LIST *blockset, *next_blockset;
602  AREA_BLOCK *block;
603  int i;
604 
605  assert (area != NULL);
606 
607  for (blockset = area->blockset_list; blockset != NULL; blockset = next_blockset)
608  {
609  next_blockset = blockset->next;
610 
611  for (i = 0; i < blockset->used_count; i++)
612  {
613  block = blockset->items[i];
614 
615  lf_bitmap_destroy (&block->bitmap);
616  free_and_init (block);
617  blockset->items[i] = NULL;
618  }
619 
620  free_and_init (blockset);
621  }
622  area->blockset_list = NULL;
623 
624  pthread_mutex_destroy (&area->area_mutex);
625 
626  if (area->name != NULL)
627  {
628  free_and_init (area->name);
629  }
630 
631 }

633 /*
634  * area_add_block --- insert block into the blockset list
635  * return: none
636  * area(in): area descriptor
637  * new_block(in): the new area_block pointer
638  *
639  * Note: This function is protected by area_mutex, which mean only
640  * 1 thread can insert block in the same time.
641  */
642 static int
643 area_insert_block (AREA * area, AREA_BLOCK * new_block)
644 {
645  AREA_BLOCKSET_LIST **last_blockset_p;
646  AREA_BLOCKSET_LIST *blockset, *new_blockset;
647  int used_count;
648 
649  assert (area != NULL && new_block != NULL);
650 
651  last_blockset_p = &area->blockset_list;
652  /* find an available blockset and insert new_block into it */
653  for (blockset = area->blockset_list; blockset != NULL; blockset = blockset->next)
654  {
655  last_blockset_p = &blockset->next;
656 
657  used_count = blockset->used_count;
658  if (used_count == AREA_BLOCKSET_SIZE)
659  {
660  /* no room */
661  continue;
662  }
663  /* each blockset owns one block at least */
664  assert (used_count >= 1);
665 
666  /* If it fits, insert new_block to the last slot of this blockset. We don't shift/re-sort the blockset to manage
667  * sorted order. Our policy may require more space but greatly reduces the complexity of logic. */
668  if (blockset->items[used_count - 1] < new_block)
669  {
670  blockset->items[used_count] = new_block;
671 
672  /* Use full barrier to ensure that above assignment done before increase used_count */
673  ATOMIC_INC_32 (&blockset->used_count, 1);
674 
675  return NO_ERROR;
676  }
677  }
678 
679  /* If there's no available blockset, we need to allocate a new blockset */
680  new_blockset = area_alloc_blockset (area);
681  if (new_blockset == NULL)
682  {
683  assert (er_errid () != NO_ERROR);
684  return er_errid ();
685  }
686 
687  /* insert new_block to the first slot */
688  new_blockset->items[0] = new_block;
689 
690  /* Use full barrier to ensure that above assignment done before increase used_count */
691  ATOMIC_INC_32 (&new_blockset->used_count, 1);
692 
693  /* append the new blockset to the end of blockset list */
694  assert ((*last_blockset_p) == NULL);
695  *last_blockset_p = new_blockset;
696 
697  return NO_ERROR;
698 }

700 /*
701  * area_find_block -- find related block in blockset list.
702  *
703  * return: the related block pointer, if not found, return NULL
704  * area(in): area descriptor
705  * ptr(in): the entry pointer
706  *
707  * Note: This function does not require area_mutex.
708  */
709 static AREA_BLOCK *
710 area_find_block (AREA * area, const void *ptr)
711 {
712  AREA_BLOCKSET_LIST *blockset;
713  AREA_BLOCK *first_block, *last_block, *block;
714  int middle, left, right, pos;
715  int used_count;
716 
717  /* find the related block in blockset list */
718  for (blockset = area->blockset_list; blockset != NULL;
719  blockset = VOLATILE_ACCESS (blockset->next, AREA_BLOCKSET_LIST *))
720  {
721  used_count = VOLATILE_ACCESS (blockset->used_count, int);
722  /* each blocskset owns one block at least */
723  assert (used_count >= 1);
724 
725  first_block = blockset->items[0];
726  if ((char *) ptr < first_block->data)
727  {
728  continue; /* less than min address */
729  }
730 
731  last_block = VOLATILE_ACCESS (blockset->items[used_count - 1], AREA_BLOCK *);
732  if (last_block->data + area->block_size <= (char *) ptr)
733  {
734  continue; /* large than max address */
735  }
736 
737  assert ((first_block->data <= (char *) ptr) && ((char *) ptr < last_block->data + area->block_size));
738 
739  left = 0;
740  right = used_count - 1;
741 
742  /* binary search in this blockset */
743  while (left <= right)
744  {
745  middle = (left + right) / 2;
746
