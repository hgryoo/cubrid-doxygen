37  /*
38  * Implementation of a enhanced Bipartite circular buffer : returns a contiguous memory for append
39  * and allows reads in past appended (and committed) data.
40  *
41  * The buffer has an active region (m_ptr_start_a -> m_ptr_end_a) which servers the appends
42  * The append pointer (m_ptr_append) is located in this range (m_ptr_start_a <= m_ptr_append <= m_ptr_end_a)
43  * The m_ptr_end_a servers as a "reserve margin" limit : the range (append -> end_a)
44  * is the immediate "future" data (to be reserved). When the reserve margin cannot be guaranteed till end of buffer,
45  * the B region is activated (normally is inactive, and when active it always resides the the beginning of the buffer).
46  *
47  * The append consists for two steps:
48  * 1. reserve (amount) : increments the append pointer with the desired amount
49  * 2. commit (ptr) : increases either the lower limit of A region, or lower limit of previous "era" of region A
50  * (which is the range : m_ptr_prev_gen_committed -> m_ptr_prev_gen_last_reserved), depending on the value of ptr
51  *
52  * Interlaced commit/reserve calls may be performed (concurrent calls are handled at upper layer).
53  * For proper functionality of the buffer, it is assumed that commits are always performed in the same order
54  * as original reserve : R1, R2, C1, R3, C2, C3; (an invalid order would be: R1, R2, C2, C1). This should be
55  * handled by upper layer.
56  *
57  * The reads are performed in three steps:
58  * 1. get_read_ranges : returns valid read ranges
59  * The reads cannot enter either of region A or region B (start_a -> end_a; buffer_start -> end_b) to prevent
60  * interfering with appenders. Also, it may read only data which was actually previously written
61  * (cannot read past 'm_ptr_prev_gen_committed'). The call returns two ranges:
62  * i) trail_b + amount_b (the range after region B), in case region B is de-activated, this range extends from start
63  * buffer to start of region A
64  * ii) trail_a + amount_a (range after region A), this is the oldest data in the buffer and extends from end of A
65  * till m_ptr_prev_gen_committed.
66  * 2. start_read (ptr, amount) : the buffer is split into equal sized pages (template parameter);
67  * each page has a fixed count (m_read_fcnt) and a bit flag in m_read_flags.
68  * start_read increments the fix count of only the first of the affected pages, and sets the corresponding bit
69  * it returns the page id which was latched (to be used by the end_read function)
70  * 3. end_read (page_id) : the reverse of start_read (decrements the fix count,
71  * and clears the bit if count reaches zero)
72  *
73  * The reserve may fail for several reasons:
74  * a) amount is too big - static condition (amount > threshold)
75  * (this is subject to change, we should probably set by configuration/ratio of capacity)
76  * b) amount is too big - dynamic condition : the active region (difference between reserved and committed)
77  * is too large, and no contiguous range may be served
78  * c) the append pointer cannot be extended because readers are pending
79  * in all cases, NULL is returned; it is the responsibility of upper layer to handle this cases
80  * (preferably to prevent them)
81  */

114  const char *reserve (const size_t amount)
115  {
116  const char *reserved_ptr = NULL;
117  bool need_take_margin = true;
118 
119  /* TODO : adaptive reserve margin */
120  /* TODO : a possible optimization would be to increase m_ptr_end_a with page increments,
121  * instead of keeping it consistently ahead of append with a fixed amount */
122  if (amount > m_reserve_margin)
123  {
124  assert (amount < m_capacity / 10);
125  if (amount < m_capacity / 10)
126  {
127  m_reserve_margin = amount;
128  }
129  else
130  {
131  return NULL;
132  }
133  }
134 
135  if (m_ptr_append + amount > m_buffer_end)
136  {
137  /* not enough till end of whole buffer; try switch to region B */
138  if (is_region_b_used () == false)
139  {
140  assert (false);
141 
142  activate_region_b ();
143  }
144  switch_to_region_b ();
145  /* fall through */
146  }
147 
148  if (m_ptr_append + amount > m_ptr_end_a)
149  {
150  /* enough till end of buffer, but need to extend region A */
151  take_margin_in_region_a (amount);
152  need_take_margin = false;
153  /* fall through */
154  }
155 
156  assert (m_ptr_append + amount <= m_ptr_end_a);
157 
158  /* check that no readers are pending inside new region A */
159  if (check_readers (m_ptr_append, m_ptr_append + amount) == false)
160  {
161  /* would block */
162  return NULL;
163  }
164  reserved_ptr = m_ptr_append;
165  m_ptr_append += amount;
166 
167  if (need_take_margin)
168  {
169  take_margin_in_region_a (0);
170  }
171 
172  return reserved_ptr;
173  };

175  int commit (const char *ptr)
176  {
177  assert (ptr >= m_buffer);
178 
179  if (m_ptr_start_a <= ptr && ptr <= m_ptr_end_a)
180  {
181  /* a later reserve is committed before the commit of a earlier reserve */
182  assert (m_ptr_prev_gen_committed <= m_ptr_prev_gen_last_reserved);
183 
184  if (m_ptr_prev_gen_committed < m_ptr_prev_gen_last_reserved)
185  {
186  m_ptr_prev_gen_committed = m_ptr_prev_gen_last_reserved;
187  }
188 
189  m_ptr_start_a = ptr;
190  }
191  else
192  {
193  assert (m_ptr_prev_gen_committed != NULL);
194  assert (m_ptr_prev_gen_last_reserved != NULL);
195 
196  assert (m_ptr_prev_gen_committed <= ptr && ptr <= m_ptr_prev_gen_last_reserved);
197 
198  m_ptr_prev_gen_committed = ptr;
199  }
200 
201  return NO_ERROR;
202  };

204  int start_read (const char *ptr, const size_t amount, mem::buffer_latch_read_id &latched_page_idx)
205  {
206  int start_page_idx;
207 
208 #if !defined(NDEBUG)
209  if (is_region_b_used ())
210  {
211  if (is_range_overlap (ptr, amount, m_buffer, m_ptr_end_b - m_buffer))
212  {
213  return ER_FAILED;
214  }
215  }
216  else
217  {
218  if (m_ptr_prev_gen_committed != NULL
219  && is_range_overlap (ptr, amount, m_ptr_prev_gen_committed, m_buffer_end - m_ptr_prev_gen_committed))
220  {
221  return ER_FAILED;
222  }
223  }
224 
225  if (is_range_overlap (ptr, amount, m_ptr_start_a, m_ptr_end_a - m_ptr_start_a))
226  {
227  return ER_FAILED;
228  }
229 #endif
230 
231  start_page_idx = get_page_from_ptr (ptr);
232  /* since append/reserve pointers always cycle in future of the buffer, it is enough to latch only
233  * the first page (corresponding to start of read range),
234  * the appender will not be able to pass beyond a latched page
235  */
236 
237  /* TODO : avoid to set read latch on a page if append is in that page
238  * Instead we try to latch the page before it (logically in the past): if it is the first page,
239  * we try to latch the last page.
240  * if the previous page cannot be latched the read is denied
241  */
242 
243  if (is_ptr_in_page (m_ptr_append, start_page_idx))
244  {
245  start_page_idx--;
246  if (start_page_idx < 0)
247  {
248  start_page_idx = P - 1;
249  }
250 
251  /* check that this page is not reserved */
252  const char *latch_start_ptr = m_buffer + start_page_idx * m_read_page_size;
253 
254  if (is_region_b_used ())
255  {
256  if (is_range_overlap (latch_start_ptr, m_read_page_size, m_buffer, m_ptr_end_b - m_buffer))
257  {
258  return ER_FAILED;
259  }
260  }
261 
262  if (is_range_overlap (latch_start_ptr, m_read_page_size, m_ptr_start_a, m_ptr_end_a - m_ptr_start_a))
263  {
264  return ER_FAILED;
265  }
266  }
267 
268  assert (is_ptr_in_page (m_ptr_append, start_page_idx) == false);
269 
270  m_read_fcnt[start_page_idx]++;
271  m_read_flags.set (start_page_idx);
272  latched_page_idx = start_page_idx;
273 
274  return NO_ERROR;
275  };

277  int end_read (const mem::buffer_latch_read_id &page_idx)
278  {
279  assert (page_idx >= 0 && page_idx < P);

293  void get_read_ranges (const char *&trail_b, size_t &amount_trail_b, const char *&trail_a, size_t &amount_trail_a)
294  {
295  if (is_region_b_used ())
296  {
297  /* when region B is active there is gap (data may be found but we want to prevent readers in B),
298  * which does not guarantee contiguous data in trail B + trail A, we give up trail A completely
299  * to avoid accounting for the gap.
300  *
301  * | B region | trail B | A region | trail A |
302  * |----------|---------------------|-------------------------|---------|
303  * | Past | Past |Present Future | Far past|
304  * | (avoid R)|
305  *
306  */
307  trail_b = m_ptr_end_b;
308  amount_trail_b = m_ptr_start_a - m_ptr_end_b;
309  amount_trail_a = 0;
310  }
311  else
312  {
313  amount_trail_b = m_ptr_start_a - m_buffer;
314  trail_b = m_buffer;
315 
316  if (m_ptr_prev_gen_committed != NULL)
317  {
318  assert (m_ptr_prev_gen_committed == m_ptr_prev_gen_last_reserved
319  || m_ptr_start_a == m_buffer);
320 
321  trail_a = m_ptr_end_a;
322  amount_trail_a = m_ptr_prev_gen_committed - m_ptr_end_a;
323  }
324  else
325  {
326  amount_trail_a = 0;
327  }
328  }

338  void init (const size_t capacity)
339  {
340  m_capacity = DB_ALIGN (capacity, 4 * 1024);
341  m_capacity = MIN (m_capacity, 100 * 1024 * 1024);
342  m_read_page_size = m_capacity / P;
343  m_capacity = DB_ALIGN (m_capacity, m_read_page_size);
344 
345  m_read_page_size = m_capacity / P;
346 
347  m_reserve_margin = m_capacity / 10;
348  m_reserve_margin = DB_ALIGN (m_reserve_margin, MAX_ALIGNMENT);
349  m_reserve_margin = MAX (m_reserve_margin, 10 * 1024);
350  m_reserve_margin = MIN (m_reserve_margin, 10 * 1024 * 1024);
351 
352  assert (m_reserve_margin < m_capacity);
353 
354  m_buffer = new char[m_capacity];
355  m_buffer_end = m_buffer + m_capacity;
356 
357  m_ptr_start_a = m_buffer;
358  m_ptr_append = m_ptr_start_a;
359  m_ptr_end_a = m_ptr_append + m_reserve_margin;
360 
361  m_ptr_end_b = NULL;
362 
363  m_ptr_prev_gen_committed = NULL;
364  m_ptr_prev_gen_last_reserved = NULL;

393  {
