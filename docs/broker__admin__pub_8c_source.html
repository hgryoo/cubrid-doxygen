185 static int
186 broker_create_dir (const char *new_dir)
187 {
188  char *p, path[BROKER_PATH_MAX];
189 
190  if (new_dir == NULL)
191  return -1;
192 
193  strcpy (path, new_dir);
194  trim (path);
195 
196 #if defined(WINDOWS)
197  unix_style_path (path);
198 #endif /* WINDOWS */
199 
200  p = path;
201 #if defined(WINDOWS)
202  if (path[0] == '/')
203  p = path + 1;
204  else if (strlen (path) > 3 && path[2] == '/')
205  p = path + 3;
206 #else /* WINDOWS */
207  if (path[0] == '/')
208  {
209  p = path + 1;
210  }
211 #endif /* WINDOWS */
212 
213  while (p != NULL)
214  {
215  p = strchr (p, '/');
216  if (p != NULL)
217  *p = '\0';
218  if (access (path, F_OK) < 0)
219  {
220  if (mkdir (path, 0777) < 0)
221  {
222  return -1;
223  }
224  }
225  if (p != NULL)
226  {
227  *p = '/';
228  p++;
229  }
230  }
231  return 0;
232 }

248 int
249 admin_start_cmd (T_BROKER_INFO * br_info, int br_num, int master_shm_id, bool acl_flag, char *acl_file)
250 {
251  int i;
252  int res = 0;
253  char path[BROKER_PATH_MAX];
254  char upper_broker_name[BROKER_NAME_LEN];
255  T_SHM_BROKER *shm_br;
256  T_SHM_APPL_SERVER *shm_as_p = NULL;
257  T_SHM_PROXY *shm_proxy_p = NULL;
258 
259  if (br_num <= 0)
260  {
261  strcpy (admin_err_msg, "Cannot start CUBRID Broker. (number of broker is 0)");
262  return -1;
263  }
264  chdir ("..");
265  broker_create_dir (get_cubrid_file (FID_VAR_DIR, path, BROKER_PATH_MAX));
266  broker_create_dir (get_cubrid_file (FID_CAS_TMP_DIR, path, BROKER_PATH_MAX));
267  broker_create_dir (get_cubrid_file (FID_AS_PID_DIR, path, BROKER_PATH_MAX));
268  broker_create_dir (get_cubrid_file (FID_SQL_LOG_DIR, path, BROKER_PATH_MAX));
269  broker_create_dir (get_cubrid_file (FID_SLOW_LOG_DIR, path, BROKER_PATH_MAX));
270  broker_create_dir (get_cubrid_file (FID_CUBRID_ERR_DIR, path, BROKER_PATH_MAX));
271 #if !defined(WINDOWS)
272  broker_create_dir (get_cubrid_file (FID_SQL_LOG2_DIR, path, BROKER_PATH_MAX));
273  broker_create_dir (get_cubrid_file (FID_SOCK_DIR, path, BROKER_PATH_MAX));
274 #endif /* !WINDOWS */
275 
276 
277  for (i = 0; i < br_num; i++)
278  {
279 #if !defined(WINDOWS)
280  /* prevent the broker from hanging due to an excessively long path socket path length =
281  * sock_path[broker_name].[as_index] */
282  if (strlen (path) + strlen (br_info[i].name) + 1 + NUM_OF_DIGITS (br_info[i].appl_server_max_num) >
283  MEMBER_SIZE (struct sockaddr_un, sun_path) - 1)
284  {
285  snprintf (admin_err_msg, sizeof (admin_err_msg) - 1, "The socket path is too long (>%d): %s",
286  MEMBER_SIZE (struct sockaddr_un, sun_path), path);
287  return -1;
288  }
289 #endif /* !WINDOWS */
290  broker_create_dir (br_info[i].log_dir);
291  broker_create_dir (br_info[i].slow_log_dir);
292  broker_create_dir (br_info[i].err_log_dir);
293  broker_create_dir (br_info[i].access_log_dir);
294 
295  if (br_info[i].shard_flag == ON)
296  {
297  broker_create_dir (br_info[i].proxy_log_dir);
298  }
299  }
300  chdir (envvar_bindir_file (path, BROKER_PATH_MAX, ""));
301 
302  /* create master shared memory */
303  shm_br = broker_shm_initialize_shm_broker (master_shm_id, br_info, br_num, acl_flag, acl_file);
304 
305  if (shm_br == NULL)
306  {
307  strcpy (admin_err_msg, "failed to initialize broker shared memory");
308  return -1;
309  }
310 
311  for (i = 0; i < br_num; i++)
312  {
313  if (br_info[i].shard_flag == OFF)
314  {
315  continue;
316  }
317 
318  if (br_info[i].shard_db_password[0] == '\0')
319  {
320  get_shard_db_password (&shm_br->br_info[i]);
321  }
322 
323  get_upper_str (upper_broker_name, BROKER_NAME_LEN, br_info[i].name);
324 
325  snprintf (shard_db_password_env_str[i], sizeof (shard_db_password_env_str[i]), "%s_SHARD_DB_PASSWORD=",
326  upper_broker_name);
327  putenv (shard_db_password_env_str[i]);
328  }
329 
330  for (i = 0; i < br_num; i++)
331  {
332  shm_as_p = NULL;
333  shm_proxy_p = NULL;
334 
335  if (br_info[i].service_flag == ON)
336  {
337  if (br_info[i].shard_flag == ON)
338 
339  {
340  shm_proxy_p = shard_shm_initialize_shm_proxy (&(shm_br->br_info[i]));
341 
342  if (shm_proxy_p == NULL)
343  {
344  sprintf (admin_err_msg, "%s: failed to initialize proxy shared memory.", br_info->name);
345 
346  res = -1;
347  break;
348  }
349 
350  if (shard_shm_check_max_file_open_limit (&shm_br->br_info[i], shm_proxy_p) < 0)
351  {
352  res = -1;
353  break;
354  }
355  }
356 
357  shm_as_p = broker_shm_initialize_shm_as (&(shm_br->br_info[i]), shm_proxy_p);
358  if (shm_as_p == NULL)
359  {
360  sprintf (admin_err_msg, "%s: failed to initialize appl server shared memory.", br_info->name);
361 
362  res = -1;
363  break;
364  }
365 
366  if (access_control_set_shm (shm_as_p, &shm_br->br_info[i], shm_br, admin_err_msg) < 0)
367  {
368  res = -1;
369  break;
370  }
371 
372  res = br_activate (&(shm_br->br_info[i]), master_shm_id, shm_br);
373 
374  if (shm_br->br_info[i].shard_flag == ON && res == 0)
375  {
376  res = check_shard_conn (shm_as_p, shm_proxy_p);
377  }
378 
379  if (res < 0)
380  {
381  break;
382  }
383 
384  if (shm_as_p)
385  {
386  uw_shm_detach (shm_as_p);
387  }
388  if (shm_proxy_p)
389  {
390  uw_shm_detach (shm_proxy_p);
391  }
392 
393  }
394  }
395 
396  if (res < 0)
397  {
398  char err_msg_backup[ADMIN_ERR_MSG_SIZE];
399  memcpy (err_msg_backup, admin_err_msg, ADMIN_ERR_MSG_SIZE);
400 
401  /* if shm_as_p == NULL then, it is expected that failed creating shared memory */
402  if (shm_as_p == NULL)
403  {
404  if (shm_br->br_info[i].shard_flag == ON && shm_proxy_p)
405  {
406  uw_shm_detach (shm_proxy_p);
407  shm_proxy_p = NULL;
408 
409  uw_shm_destroy (shm_br->br_info[i].proxy_shm_id);
410  }
411 
412  --i;
413  }
414 
415  for (; i >= 0; i--)
416  {
417  br_inactivate (&(shm_br->br_info[i]));
418 
419  uw_shm_destroy ((shm_br->br_info[i].appl_server_shm_id));
420  if (shm_br->br_info[i].shard_flag == ON)
421  {
422  uw_shm_destroy ((shm_br->br_info[i].proxy_shm_id));
423  }
424  }
425 
426  memcpy (admin_err_msg, err_msg_backup, ADMIN_ERR_MSG_SIZE);
427  }
428  if (shm_br)
429  {
430  uw_shm_detach (shm_br);
431  }
432 
433  if (res < 0)
434  {
435  uw_shm_destroy (master_shm_id);
436 
437  if (shm_as_p)
438  {
439  uw_shm_detach (shm_as_p);
440  }
441  if (shm_proxy_p)
442  {
443  uw_shm_detach (shm_proxy_p);
444  }
445  }
446 #if defined(WINDOWS)
447  else
448  {
449  char shm_id_env_str[128];
450  sprintf (shm_id_env_str, "%s=%d", MASTER_SHM_KEY_ENV_STR, master_shm_id);
451  putenv (shm_id_env_str);
452  run_child (NAME_UC_SHM);
453  }
454 #endif /* WINDOWS */
455 
456  return res;
457 }

516 int
517 admin_add_cmd (int master_shm_id, const char *broker)
518 {
519  T_SHM_BROKER *shm_br;
520  T_SHM_APPL_SERVER *shm_appl_server;
521  int i, br_index;
522  int appl_shm_key = 0, as_index;
523  char **env = NULL;
524  int env_num;
525 
526  shm_br = (T_SHM_BROKER *) uw_shm_open (master_shm_id, SHM_BROKER, SHM_MODE_ADMIN);
527  if (shm_br == NULL)
528  {
529  SHM_OPEN_ERR_MSG (admin_err_msg, uw_get_error_code (), uw_get_os_error_code ());
530  return -1;
531  }
532  br_index = -1;
533  for (i = 0; i < shm_br->num_broker; i++)
534  {
535  if (strcasecmp (broker, shm_br->br_info[i].name) == 0)
536  {
537  appl_shm_key = shm_br->br_info[i].appl_server_shm_id;
538  br_index = i;
539  break;
540  }
541  }
542  if (br_index < 0)
543  {
544  sprintf (admin_err_msg, "Cannot find broker [%s]", broker);
545  uw_shm_detach (shm_br);
546  return -1;
547  }
548  if (shm_br->br_info[br_index].shard_flag == ON)
549  {
550  uw_shm_detach (shm_br);
551  return 0;
552  }
553 
554  if (shm_br->br_info[br_index].auto_add_appl_server == ON)
555  {
556  uw_shm_detach (shm_br);
557  return 0;
558  }
559 
560  shm_appl_server = (T_SHM_APPL_SERVER *) uw_shm_open (appl_shm_key, SHM_APPL_SERVER, SHM_MODE_ADMIN);
561  if (shm_appl_server == NULL)
562  {
563  SHM_OPEN_ERR_MSG (admin_err_msg, uw_get_error_code (), uw_get_os_error_code ());
564  uw_shm_detach (shm_br);
565  return -1;
566  }
567 
568  if (shm_br->br_info[br_index].appl_server_num >= shm_br->br_info[br_index].appl_server_max_num)
569  {
570  strcpy (admin_err_msg, "Cannot add appl server\n");
571  uw_shm_detach (shm_br);
572  uw_shm_detach (shm_appl_server);
573  return -1;
574  }
575 
576  as_index = shm_br->br_info[br_index].appl_server_num;
577 
578  (shm_br->br_info[br_index].appl_server_num)++;
579  (shm_appl_server->num_appl_server)++;
580 
581  env = make_env (shm_br->br_info[br_index].source_env, &env_num);
582 
583  as_activate (shm_br, &(shm_br->br_info[br_index]), shm_appl_server, &(shm_appl_server->as_info[as_index]), as_index,
584  env, env_num);
585 
586  shm_appl_server->as_info[as_index].service_flag = SERVICE_ON;
