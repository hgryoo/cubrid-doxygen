503  int num_as;

505  char client_ip_str[16];
506 
507  if (admin_common (br_info, &num_broker, &master_shm_id, admin_log_file, err_msg, 0, NULL, NULL) < 0)
508  {
509  return -1;
510  }
511 
512  shm_br = (T_SHM_BROKER *) uw_shm_open (master_shm_id, SHM_BROKER, SHM_MODE_ADMIN);
513  if (shm_br == NULL)
514  {
515  strcpy (err_msg, "Error: shared memory open error");
516  return -1;
517  }
518  br_index = -1;
519  for (i = 0; i < shm_br->num_broker; i++)
520  {
521  if (strcasecmp (br_name, shm_br->br_info[i].name) == 0)
522  {
523  appl_shm_key = shm_br->br_info[i].appl_server_shm_id;
524  br_index = i;
525  break;
526  }
527  }
528  if (br_index < 0)
529  {
530  sprintf (err_msg, "Error: cannot find broker [%s]", br_name);
531  goto as_info_error;
532  }
533  shm_appl = (T_SHM_APPL_SERVER *) uw_shm_open (appl_shm_key, SHM_APPL_SERVER, SHM_MODE_ADMIN);
534  if (shm_appl == NULL)
535  {
536  strcpy (err_msg, "Error: shared memory open error");
537  goto as_info_error;
538  }
539 
540  num_as = shm_br->br_info[br_index].appl_server_max_num;
541  as_info = (T_AS_INFO *) malloc (sizeof (T_AS_INFO) * num_as);
542  if (as_info == NULL)
543  {
544  strcpy (err_msg, strerror (errno));
545  goto as_info_error;
546  }
547  memset (as_info, 0, sizeof (T_AS_INFO) * num_as);
548 
549  for (i = 0; i < shm_br->br_info[br_index].appl_server_max_num; i++)
550  {
551  T_PSINFO proc_info;
552 
553  as_info[i].service_flag = shm_appl->as_info[i].service_flag;
554  if (shm_appl->as_info[i].service_flag != SERVICE_ON)
555  {
556  continue;
557  }
558 
559  as_info[i].pid = shm_appl->as_info[i].pid;
560  as_info[i].psize = getsize (as_info[i].pid);
561  memset (&proc_info, 0, sizeof (T_PSINFO));
562  get_psinfo (as_info[i].pid, &proc_info);
563  as_info[i].num_thr = proc_info.num_thr;
564  as_info[i].cpu_time = proc_info.cpu_time;
565  as_info[i].pcpu = proc_info.pcpu;
566  as_info[i].num_request = shm_appl->as_info[i].num_request;
567 #if defined(WINDOWS)
568  if (shm_appl->use_pdh_flag == TRUE)
569  {
570  float pct_cpu;
571 
572  pct_cpu = shm_appl->as_info[i].pdh_pct_cpu;
573  if (pct_cpu >= 0)
574  {
575  as_info[i].pcpu = pct_cpu;
576  }
577  as_info[i].psize = shm_appl->as_info[i].pdh_workset;
578  as_info[i].cpu_time = shm_appl->as_info[i].cpu_time;
579  }
580 #endif /* WINDOWS */
581 #if defined(WINDOWS)
582  as_info[i].as_port = shm_appl->as_info[i].as_port;
583 #else /* WINDOWS */
