420 int
421 ut_time_string (char *buf, struct timeval *time_val)
422 {
423  struct tm tm, *tm_p;
424  time_t sec;
425  int millisec;
426 
427  if (buf == NULL)
428  {
429  return 0;
430  }
431 
432  if (time_val == NULL)
433  {
434  struct timeb tb;
435 
436  /* current time */
437  (void) ftime (&tb);
438  sec = tb.time;
439  millisec = tb.millitm;
440  }
441  else
442  {
443  sec = time_val->tv_sec;
444  millisec = time_val->tv_usec / 1000;
445  }
446 
447  tm_p = localtime_r (&sec, &tm);
448  tm.tm_mon++;
449 
450  buf[0] = ((tm.tm_year % 100) / 10) + '0';
451  buf[1] = (tm.tm_year % 10) + '0';
452  buf[2] = '-';
453  buf[3] = (tm.tm_mon / 10) + '0';
454  buf[4] = (tm.tm_mon % 10) + '0';
455  buf[5] = '-';
456  buf[6] = (tm.tm_mday / 10) + '0';
457  buf[7] = (tm.tm_mday % 10) + '0';
458  buf[8] = ' ';
459  buf[9] = (tm.tm_hour / 10) + '0';
460  buf[10] = (tm.tm_hour % 10) + '0';
461  buf[11] = ':';
462  buf[12] = (tm.tm_min / 10) + '0';
463  buf[13] = (tm.tm_min % 10) + '0';
464  buf[14] = ':';
465  buf[15] = (tm.tm_sec / 10) + '0';
466  buf[16] = (tm.tm_sec % 10) + '0';
467  buf[17] = '.';
468  buf[20] = (millisec % 10) + '0';
469  millisec /= 10;
470  buf[19] = (millisec % 10) + '0';
471  millisec /= 10;
472  buf[18] = (millisec % 10) + '0';
473  buf[21] = '\0';
474 
475  return 21;
476 }

578 {

585  assert (false);

587 
588  val = strtod (size_str, &end);
589  if (end == (char *) size_str)
590  {
591  return -1.0;
592  }
593

607  }
608  else if ((strcasecmp (unit, "k") == 0) || (strcasecmp (unit, "kb") == 0))
609  {
610  /* kilo */
611  }
612  else if ((strcasecmp (unit, "m") == 0) || (strcasecmp (unit, "mb") == 0))
613  {
614  /* mega */
615  val = val * ONE_K;
616  }
617  else if ((strcasecmp (unit, "g") == 0) || (strcasecmp (unit, "gb") == 0))
618  {
619  /* giga */
620  val = val * ONE_M;
621  }
622  else
623  {
624  return -1.0;
625  }
626
