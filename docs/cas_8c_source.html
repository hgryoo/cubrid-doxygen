380 cas_send_connect_reply_to_driver (T_CAS_PROTOCOL protocol, SOCKET client_sock_fd, char *cas_info)
381 {
382  char msgbuf[CAS_CONNECTION_REPLY_SIZE + 8];
383  char *p = msgbuf;

385  int v;
386 
387 #if !defined(CAS_FOR_ORACLE) && !defined(CAS_FOR_MYSQL)

391 #endif /* !CAS_FOR_ORACLE && !CAS_FOR_MYSQL */

393  if (DOES_CLIENT_UNDERSTAND_THE_PROTOCOL (protocol, PROTOCOL_V4))
394  {
395  v = htonl (CAS_CONNECTION_REPLY_SIZE);

397  else if (DOES_CLIENT_UNDERSTAND_THE_PROTOCOL (protocol, PROTOCOL_V3))
398  {
399  v = htonl (CAS_CONNECTION_REPLY_SIZE_V3);
400  }
401  else
402  {
403  v = htonl (CAS_CONNECTION_REPLY_SIZE_PRIOR_PROTOCOL_V3);
404  }
405  memcpy (p, &v, sizeof (int));
406  p += sizeof (int);
407  if (cas_info_size > 0)
408  {
409  memcpy (p, cas_info, cas_info_size);
410  p += cas_info_size;
411  }
412  v = htonl (getpid ());
413  memcpy (p, &v, CAS_PID_SIZE);
414  p += CAS_PID_SIZE;
415  memcpy (p, cas_bi_get_broker_info (), BROKER_INFO_SIZE);
416  p += BROKER_INFO_SIZE;
417  if (DOES_CLIENT_UNDERSTAND_THE_PROTOCOL (protocol, PROTOCOL_V4))
418  {
419  v = htonl (shm_as_index + 1);
420  memcpy (p, &v, CAS_PID_SIZE);
421  p += CAS_PID_SIZE;
422  }
423  if (DOES_CLIENT_UNDERSTAND_THE_PROTOCOL (protocol, PROTOCOL_V3))
424  {
425  memcpy (p, sessid, DRIVER_SESSION_SIZE);
426  p += DRIVER_SESSION_SIZE;
427  }
428  else
429  {
430 #if !defined(CAS_FOR_ORACLE) && !defined(CAS_FOR_MYSQL)
431  v = htonl (db_get_session_id ());
432 #else
433  v = 0;
434 #endif /* !CAS_FOR_ORACLE && !CAS_FOR_MYSQL */
435  memcpy (p, &v, SESSION_ID_SIZE);
436  p += SESSION_ID_SIZE;
437  }
438  net_write_stream (client_sock_fd, msgbuf, p - msgbuf);
439 }

539  SLEEP_SEC (1);
540  }
541  while (as_info->uts_status == UTS_STATUS_RESTART || as_info->uts_status == UTS_STATUS_STOP);

552 #endif
553 
554  /* This is a only use in proxy-cas internal message */
555  req_info.client_version = CAS_PROTO_CURRENT_VER;
556 
557  set_cas_info_size ();
558 
559  gettimeofday (&cas_start_time, NULL);
560 
561 #if defined(CAS_FOR_ORACLE) || defined(CAS_FOR_MYSQL)
562  snprintf (cas_db_name, MAX_HA_DBINFO_LENGTH, "%s", shm_appl->shard_conn_info[shm_shard_id].db_name);
