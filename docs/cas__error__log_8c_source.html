142 static void
143 cas_log_write_internal (const char *fmt, ...)
144 {
145  va_list ap;

149  va_start (ap, fmt);
150  p = buf;

152  n = ut_time_string (p, NULL);
153  len -= n;
154  p += n;
155  if (len > 0)
156  {
157  n = vsnprintf (p, len, fmt, ap);
158  len -= n;
159  p += n;
160  }
161  fwrite (buf, (p - buf), 1, error_log_fp);
162 
163  fflush (error_log_fp);
164  va_end (ap);

168 cas_error_log_write (int dbms_errno, const char *dbms_errmsg)
169 {
170  if (error_log_fp == NULL)
171  {
172  return;
173  }
174 
175  if (eid >= INT_MAX)
176  {
177  eid = 0;
178  }
179  cas_log_write_internal (" DBMS ERROR [ERR_CODE : %d, ERR_MSG : %s] EID = %d", dbms_errno,
180  (dbms_errmsg != NULL) ? dbms_errmsg : "-", ++eid);
181  fputc ('\n', error_log_fp);
182 
183  saved_error_log_fpos = ftell (error_log_fp);
184 
185  if ((saved_error_log_fpos / 1000) > shm_appl->sql_log_max_size)
186  {
187  cas_error_log_close (true);
188  cas_error_log_backup ();
189  cas_error_log_open (NULL);
190  }
