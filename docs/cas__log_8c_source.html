114 static char *
115 make_sql_log_filename (T_CUBRID_FILE_ID fid, char *filename_buf, size_t buf_size, const char *br_name)
116 {
117 #ifndef LIBCAS_FOR_JSP
118  char dirname[BROKER_PATH_MAX];
119 
120  assert (filename_buf != NULL);
121 
122  get_cubrid_file (fid, dirname, BROKER_PATH_MAX);
123  switch (fid)
124  {
125  case FID_SQL_LOG_DIR:
126  if (cas_shard_flag == ON)
127  {
128  snprintf (filename_buf, buf_size, "%s%s_%d_%d_%d.sql.log", dirname, br_name, shm_proxy_id + 1, shm_shard_id,
129  shm_shard_cas_id + 1);
130  }
131  else
132  {
133  snprintf (filename_buf, buf_size, "%s%s_%d.sql.log", dirname, br_name, shm_as_index + 1);
134  }
135  break;
136  case FID_SLOW_LOG_DIR:
137  if (cas_shard_flag == ON)
138  {
139  snprintf (filename_buf, buf_size, "%s%s_%d_%d_%d.slow.log", dirname, br_name, shm_proxy_id + 1, shm_shard_id,
140  shm_shard_cas_id + 1);
141  }
142  else
143  {
144  snprintf (filename_buf, buf_size, "%s%s_%d.slow.log", dirname, br_name, shm_as_index + 1);
145  }
146  break;
147  default:
148  assert (0);
149  snprintf (filename_buf, buf_size, "unknown.log");
150  break;
151  }
152  return filename_buf;
153 #endif /* LIBCAS_FOR_JSP */
154  return NULL;

157 void
158 cas_log_open (char *br_name)
159 {
160 #ifndef LIBCAS_FOR_JSP
161  if (log_fp != NULL)
162  {
163  cas_log_close (true);
164  }
165 
166  if (as_info->cur_sql_log_mode != SQL_LOG_MODE_NONE)
167  {
168  if (br_name != NULL)
169  {
170  if (as_info->cas_log_reset == CAS_LOG_RESET_REOPEN)
171  {
172  set_cubrid_file (FID_SQL_LOG_DIR, shm_appl->log_dir);
173  }
174 
175  make_sql_log_filename (FID_SQL_LOG_DIR, log_filepath, BROKER_PATH_MAX, br_name);
176  }
177 
178  /* note: in "a+" mode, output is always appended */
179  log_fp = cas_fopen (log_filepath, "r+");
180  if (log_fp != NULL)
181  {
182  cas_fseek (log_fp, 0, SEEK_END);
183  saved_log_fpos = cas_ftell (log_fp);
184  }
185  else
186  {
187  log_fp = cas_fopen (log_filepath, "w");
188  saved_log_fpos = 0;
189  }
190 
191  if (log_fp)
192  {
193  setvbuf (log_fp, sql_log_buffer, _IOFBF, SQL_LOG_BUFFER_SIZE);
194  }
195  }
196  else
197  {
198  log_fp = NULL;
199  saved_log_fpos = 0;
200  }
201  as_info->cas_log_reset = 0;
202 #endif /* LIBCAS_FOR_JSP */
203 }

309 void
310 cas_log_end (int mode, int run_time_sec, int run_time_msec)
311 {
312 #ifndef LIBCAS_FOR_JSP
313  if (log_fp == NULL && as_info->cur_sql_log_mode != SQL_LOG_MODE_NONE)
314  {
315  cas_log_open (shm_appl->broker_name);
316  }
317 
318  if (log_fp != NULL)
319  {
320  bool abandon = false;
321 
322  /* 'mode' will be either ALL, ERROR, or TIMEOUT */
323  switch (mode)
324  {
325  case SQL_LOG_MODE_ALL:
326  /* if mode == ALL, write log regardless sql_log_mode */
327  break;
328  case SQL_LOG_MODE_ERROR:
329  /* if mode == ERROR, write log if sql_log_mode == ALL || ERROR || NOTICE */
330  if (as_info->cur_sql_log_mode == SQL_LOG_MODE_NONE || as_info->cur_sql_log_mode == SQL_LOG_MODE_TIMEOUT)
331  {
332  abandon = true;
333  }
334  break;
335  case SQL_LOG_MODE_TIMEOUT:
336  /* if mode == TIMEOUT, write log if sql_log_mode == ALL || TIMEOUT || NOTICE */
337  if (as_info->cur_sql_log_mode == SQL_LOG_MODE_NONE || as_info->cur_sql_log_mode == SQL_LOG_MODE_ERROR)
338  {
339  abandon = true;
340  }
341  /* if mode == TIMEOUT and sql_log_mode == TIMEOUT || NOTICE, check if it timed out */
342  else if (as_info->cur_sql_log_mode == SQL_LOG_MODE_TIMEOUT
343  || as_info->cur_sql_log_mode == SQL_LOG_MODE_NOTICE)
344  {
345  /* check timeout */
346  if ((run_time_sec * 1000 + run_time_msec) < shm_appl->long_transaction_time)
347  {
348  abandon = true;
349  }
350  }
351  break;
352  /* if mode == NONE, write log if sql_log_mode == ALL */
353  case SQL_LOG_MODE_NONE:
354  if (as_info->cur_sql_log_mode != SQL_LOG_MODE_ALL)
355  {
356  abandon = true;
357  }
358  break;
359  case SQL_LOG_MODE_NOTICE:
360  default:
361  /* mode NOTICE or others are unexpected values; do not write log */
362  abandon = true;
363  break;
364  }
365 
366  if (abandon)
367  {
368  cas_log_write_and_set_savedpos (log_fp, "%s", "END OF LOG\n\n");
369  }
370  else
371  {
372  if (run_time_sec >= 0 && run_time_msec >= 0)
373  {
374  cas_log_write (0, false, "*** elapsed time %d.%03d\n", run_time_sec, run_time_msec);
375  }
376  saved_log_fpos = cas_ftell (log_fp);
377 
378  if ((saved_log_fpos / 1000) > shm_appl->sql_log_max_size)
379  {
380  cas_log_close (true);
381  cas_log_backup (FID_SQL_LOG_DIR);
382  cas_log_open (NULL);
383  }
384  else
385  {
386  cas_log_write_and_set_savedpos (log_fp, "%s", "END OF LOG\n\n");
387  }
388  }
389  }
390 #endif /* LIBCAS_FOR_JSP */
391 }

393 static void
394 cas_log_write_internal (FILE * fp, struct timeval *log_time, unsigned int seq_num, bool do_flush, const char *fmt,
395  va_list ap)
396 {
397  char *buf, *p;
398  int len, n;
399 
400  p = buf = cas_log_buffer;
401  len = CAS_LOG_BUFFER_SIZE;
402  n = ut_time_string (p, log_time);
403  len -= n;
404  p += n;
405 
406  if (len > 0)
407  {
408  n = snprintf (p, len, " (%u) ", seq_num);
409  len -= n;
410  p += n;
411  if (len > 0)
412  {
413  n = vsnprintf (p, len, fmt, ap);
414  if (n >= len)
415  {
416  /* string is truncated and trailing '\0' is included */
417  n = len - 1;
418  }
419  len -= n;
420  p += n;
421  }
422  }
423 
424  cas_fwrite (buf, (p - buf), 1, fp);
425 
426  if (do_flush == true)
427  {
428  cas_fflush (fp);
429  }
430 }

774 int
775 cas_access_log (struct timeval *start_time, int as_index, int client_ip_addr, char *dbname, char *dbuser,
776  ACCESS_LOG_TYPE log_type)
777 {
778 #ifndef LIBCAS_FOR_JSP
779  FILE *fp;

787  char session_id_buf[16];

790 
791  t1 = start_time->tv_sec;
792  t2 = end_time.tv_sec;

794  {
795  return -1;
796  }
797  ct1.tm_year += 1900;

832  fp = access_log_open (access_log_file);

834  {
835  return -1;
836  }
837  }
838  }
839 
840  ut_get_ipv4_string (clt_ip_str, sizeof (clt_ip_str), (unsigned char *) (&client_ip_addr));

844 #if !defined(CAS_FOR_ORACLE) && !defined(CAS_FOR_MYSQL)

896 static FILE *
897 access_log_open (char *log_file_name)
898 {
899  FILE *fp;
900  int ret;
901  char *tmp_dirname;
902  char *tmp_filename;
903 
904  if (log_file_name == NULL)
905  return NULL;
906 
907 #if defined (WINDOWS)
908  fp = cas_fopen_and_lock (log_file_name, "a");

911  * ent-of-file. So we don't need to lock to the opened file. */

929 #if defined (WINDOWS)

931 #else
932  fp = cas_fopen (log_file_name, "a");
933 #endif

935  {
936  return NULL;
937  }
938  }
939  else
940  {
941  return NULL;
942  }
943  }
944  else

946  return NULL;
947  }
948  }
