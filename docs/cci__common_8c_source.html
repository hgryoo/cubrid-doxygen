114 /*
115  * cci_mht_5str_pseudo_key() - hash string key into pseudo integer key
116  * return: pseudo integer key
117  * key(in): string key to hash
118  * key_size(in): size of key or -1 when unknown
119  *
120  * Note: Based on hash method reported by Diniel J. Bernstein.
121  */
122 static unsigned int
123 cci_mht_5str_pseudo_key (void *key, int key_size)
124 {
125  unsigned int hash = 5381;
126  int i = 0;
127  char *k;
128 
129  assert (key != NULL);
130  k = (char *) key;
131 
132  if (key_size == -1)
133  {
134  int c;
135  while ((c = *(k + i++)) != 0)
136  {
137  hash = ((hash << 5) + hash) + c; /* hash * 33 + c */
138  }
139  }
140  else
141  {
142  for (; i < key_size; i++)
143  {
144  hash = ((hash << 5) + hash) + *(k + i);
145  }
146  }
147 
148  hash += ~(hash << 9);
149  hash ^= ((hash >> 14) | (i << 18)); /* >>> */
150  hash += (hash << 4);
151  hash ^= ((hash >> 10) | (i << 22)); /* >>> */
152 
153  return hash;
154 }

219 /* Ceiling of positive division */
220 #define CEIL_PTVDIV(dividend, divisor) \
221  (((dividend) == 0) ? 0 : (((dividend) - 1) / (divisor)) + 1)
222 
223 static unsigned int
224 cci_mht_calculate_htsize (unsigned int ht_size)
225 {
226  int left, right, middle; /* indices for binary search */
227 
228  if (ht_size > cci_mht_Primes[NPRIMES - 1])
229  {
230  /* get a power of two */
231  if (!((ht_size & (ht_size - 1)) == 0))
232  {
233  /* Turn off some bits but the left most one */
234  while (!(ht_size & (ht_size - 1)))
235  {
236  ht_size &= (ht_size - 1);
237  }
238  ht_size <<= 1;
239  }
240  }
241  else
242  {
243  /* we can assign a primary number; binary search */
244  for (middle = 0, left = 0, right = NPRIMES - 1; left <= right;)
245  {
246  middle = CEIL_PTVDIV ((left + right), 2);
247  if (ht_size == cci_mht_Primes[middle])
248  {
249  break;
250  }
251  else if (ht_size > cci_mht_Primes[middle])
252  {
253  left = middle + 1;
254  }
255  else
256  {
257  right = middle - 1;
258  }
259  }
260  /* If we didn't find the size, get the larger size and not the small one */
261  if (ht_size > cci_mht_Primes[middle] && middle < (NPRIMES - 1))
262  {
263  middle++;
264  }
265  ht_size = cci_mht_Primes[middle];
266  }
267 
268  return ht_size;
269 }

271 /*
272  * cci_mht_create - create a hash table
273  * return: hash table
274  * name(in): name of hash table
275  * est_size(in): estimated number of entries
276  * hash_func(in): hash function
277  * cmp_func(in): key compare function
278  *
279  * Note: Create a new hash table. The estimated number of entries for
280  * the hash table may be adjusted (to a prime number) in order to
281  * obtain certain favorable circumstances when the table is
282  * created, when the table is almost full and there are at least
283  * 5% of collisions. 'hash_func' must return an integer between 0 and
284  * table_size - 1. 'cmp_func' must return TRUE if key1 = key2,
285  * otherwise, FALSE.
286  */
287 CCI_MHT_TABLE *
288 cci_mht_create (char *name, int est_size, HASH_FUNC hash_func, CMP_FUNC cmp_func)
289 {
290  CCI_MHT_TABLE *ht;
291  CCI_HENTRY_PTR *hvector; /* Entries of hash table */
292  unsigned int ht_estsize;
293  unsigned int size;
294 
295  assert (hash_func != NULL && cmp_func != NULL);
296 
297  /* Get a good number of entries for hash table */
298  if (est_size <= 0)
299  {
300  est_size = 2;
301  }
302 
303  ht_estsize = cci_mht_calculate_htsize ((unsigned int) est_size);
304 
305  /* Allocate the header information for hash table */
306  ht = (CCI_MHT_TABLE *) MALLOC (sizeof (CCI_MHT_TABLE));
307  if (ht == NULL)
308  {
309  return NULL;
310  }
311 
312  /* Allocate the hash table entry pointers */
313  size = ht_estsize * sizeof (CCI_HENTRY_PTR);
314  hvector = (CCI_HENTRY_PTR *) MALLOC (size);
315  if (hvector == NULL)
316  {
317  FREE (ht);
318  return NULL;
319  }
320 
321  ht->hash_func = hash_func;
322  ht->cmp_func = cmp_func;
323  ht->name = name;
324  ht->table = hvector;
325  ht->act_head = NULL;
326  ht->act_tail = NULL;
327  ht->prealloc_entries = NULL;
328  ht->size = ht_estsize;
329  ht->rehash_at = (unsigned int) (ht_estsize * CCI_MHT_REHASH_TRESHOLD);
330  ht->nentries = 0;
331  ht->nprealloc_entries = 0;
332  ht->ncollisions = 0;
333 
334  /* Initialize each of the hash entries */
335  for (; ht_estsize > 0; ht_estsize--)
336  {
337  *hvector++ = NULL;
338  }
339 
340  return ht;
341 }

343 /*
344  * cci_mht_rehash - rehash all entires of a hash table
345  * return: error code
346  * ht(in/out): hash table to rehash
347  *
348  * Note: Expand a hash table. All entries are rehashed onto a larger
349  * hash table of entries.
350  */
351 static int
352 cci_mht_rehash (CCI_MHT_TABLE * ht)
353 {
354  CCI_HENTRY_PTR *new_hvector; /* New entries of hash table */
355  CCI_HENTRY_PTR *hvector; /* Entries of hash table */
356  CCI_HENTRY_PTR hentry; /* A hash table entry. linked list */
357  CCI_HENTRY_PTR next_hentry = NULL; /* Next element in linked list */
358  float rehash_factor;
359  unsigned int hash;
360  unsigned int est_size;
361  unsigned int size;
362  unsigned int i;
363 
364  /* Find an estimated size for hash table entries */
365  rehash_factor = 1.0f + ((float) ht->ncollisions / (float) ht->nentries);
366  if (CCI_MHT_REHASH_FACTOR > rehash_factor)
367  {
368  est_size = (unsigned int) (ht->size * CCI_MHT_REHASH_FACTOR);
369  }
370  else
371  {
372  est_size = (unsigned int) (ht->size * rehash_factor);
373  }
374 
375  est_size = cci_mht_calculate_htsize (est_size);
376 
377  /* Allocate a new vector to keep the estimated hash entries */
378  size = est_size * sizeof (CCI_HENTRY_PTR);
379  new_hvector = (CCI_HENTRY_PTR *) MALLOC (size);
380  if (new_hvector == NULL)
381  {
382  return CCI_ER_NO_MORE_MEMORY;
383  }
384 
385  /* Initialize all entries */
386  memset (new_hvector, 0x00, size);
387 
388  /* Now rehash the current entries onto the vector of hash entries table */
389  for (ht->ncollisions = 0, hvector = ht->table, i = 0; i < ht->size; hvector++, i++)
390  {
391  /* Go over each linked list */
392  for (hentry = *hvector; hentry != NULL; hentry = next_hentry)
393  {
394  next_hentry = hentry->next;
395 
396  hash = (*ht->hash_func) (hentry->key, est_size);
397  if (hash >= est_size)
398  {
399  hash %= est_size;
400  }
401 
402  /* Link the new entry with any previous elements */
403  hentry->next = new_hvector[hash];
404  if (hentry->next != NULL)
405  {
406  ht->ncollisions++;
407  }
408  new_hvector[hash] = hentry;
409  }
410  }
411 
412  /* Now move to new vector of entries */
413  FREE_MEM (ht->table);
414 
415  ht->table = new_hvector;
416  ht->size = est_size;
417  ht->rehash_at = (int) (est_size * CCI_MHT_REHASH_TRESHOLD);
418 
419  return CCI_ER_NO_ERROR;
420 }

422 /*
423  * cci_mht_destroy - destroy a hash table
424  * return: void
425  * ht(in/out): hash table
426  *
427  * Note: ht is set as a side effect
428  */
429 void
430 cci_mht_destroy (CCI_MHT_TABLE * ht, bool free_key, bool free_data)
431 {
432  CCI_HENTRY_PTR *hvector;
433  CCI_HENTRY_PTR hentry;
434  CCI_HENTRY_PTR next_hentry = NULL;
435  unsigned int i;
436 
437  assert (ht != NULL);
438 
439  for (hvector = ht->table, i = 0; i < ht->size; hvector++, i++)
440  {
441  /* Go over each linked list */
442  for (hentry = *hvector; hentry != NULL; hentry = next_hentry)
443  {
444  next_hentry = hentry->next;
445  if (free_key)
446  {
447  FREE (hentry->key);
448  }
449  if (free_data)
450  {
451  FREE (hentry->data);
452  }
453  FREE (hentry);
454  }
455  }
456 
457  while (ht->nprealloc_entries > 0)
458  {
459  hentry = ht->prealloc_entries;
460  ht->prealloc_entries = ht->prealloc_entries->next;
461  FREE_MEM (hentry);
462  ht->nprealloc_entries--;
463  }
464 
465  FREE_MEM (ht->table);
466  FREE_MEM (ht);
467 }

469 /*
470  * cci_mht_rem - remove a hash entry
471  * return: removed data
472  * ht(in): hash table
473  * key(in): hashing key
474  * free_key(in): flag to free key memory
475  * free_data(in): flag to free data memory
476  *
477  * Note: For each entry in hash table
478  */
479 void *
480 cci_mht_rem (CCI_MHT_TABLE * ht, void *key, bool free_key, bool free_data)
481 {
482  unsigned int hash;
483  CCI_HENTRY_PTR prev_hentry;
484  CCI_HENTRY_PTR hentry;
485  void *data = NULL;
486 
487  assert (ht != NULL && key != NULL);
488 
489  /*
490  * Hash the key and make sure that the return value is between 0 and size
491  * of hash table
492  */
493  hash = (*ht->hash_func) (key, ht->size);
494  if (hash >= ht->size)
495  {
496  hash %= ht->size;
497  }
498 
499  /* Now search the linked list.. Is there any entry with the given key ? */
500  for (hentry = ht->table[hash], prev_hentry = NULL; hentry != NULL; prev_hentry = hentry, hentry = hentry->next)
501  {
502  if (hentry->key == key || (*ht->cmp_func) (hentry->key, key))
503  {
504  data = hentry->data;
505 
506  /* Remove from double link list of active entries */
507  if (ht->act_head == ht->act_tail)
508  {
509  /* Single active element */
510  ht->act_head = ht->act_tail = NULL;
511  }
512  else if (ht->act_head == hentry)
513  {
514  /* Deleting from the head */
515  ht->act_head = hentry->act_next;
516  hentry->act_next->act_prev = NULL;
517  }
518  else if (ht->act_tail == hentry)
519  {
520  /* Deleting from the tail */
521  ht->act_tail = hentry->act_prev;
522  hentry->act_prev->act_next = NULL;
523  }
524  else
525  {
526  /* Deleting from the middle */
527  hentry->act_prev->act_next = hentry->act_next;
528  hentry->act_next->act_prev = hentry->act_prev;
529  }
530 
531  /* Remove from the hash */
532  if (prev_hentry != NULL)
533  {
534  prev_hentry->next = hentry->next;
535  ht->ncollisions--;
536  }
537  else if ((ht->table[hash] = hentry->next) != NULL)
538  {
539  ht->ncollisions--;
540  }
541  ht->nentries--;
542  /* Save the entry for future insertions */
543  ht->nprealloc_entries++;
544  hentry->next = ht->prealloc_entries;
545  ht->prealloc_entries = hentry;
546 
547  if (free_key)
548  {
549  FREE_MEM (hentry->key);
550  }
551 
552  if (free_data)
553  {
554  FREE_MEM (hentry->data);
555  return NULL;
556  }
557 
558  return data;
559  }
560  }
561 
562  return NULL;
563 }

602 /*
603  * cci_mht_put_internal - internal function for cci_mht_put(),
604  * cci_mht_put_new(), and cci_mht_put_data();
605  * insert an entry associating key with data
606  * return: Returns key if insertion was OK, otherwise, it returns NULL
607  * ht(in/out): hash table (set as a side effect)
608  * key(in): hashing key
609  * data(in): data associated with hashing key
610  * opt(in): options;
611  * CCI_MHT_OPT_DEFAULT - change data and the key as given of the hash
612  * entry; replace the old entry with the new
613  * one if there is an entry with the same key
614  * CCI_MHT_OPT_KEEP_KEY - change data but the key of the hash entry
615  * CCI_MHT_OPT_INSERT_ONLY - do not replace the existing hash entry
616  * even if there is an etnry with the same
617  * key
618  */
619 static void *
620 cci_mht_put_internal (CCI_MHT_TABLE * ht, void *key, void *data, CCI_MHT_PUT_OPT opt)
621 {
622  unsigned int hash;
623  CCI_HENTRY_PTR hentry;
624 
625  assert (ht != NULL && key != NULL);
626 
627  /*
628  * Hash the key and make sure that the return value is between 0 and size
629  * of hash table
630  */
631  hash = (*ht->hash_func) (key, ht->size);
632  if (hash >= ht->size)
633  {
634  hash %= ht->size;
635  }
636 
637  if (!(opt & CCI_MHT_OPT_INSERT_ONLY))
638  {
639  /* Now search the linked list.. Is there any entry with the given key ? */
640  for (hentry = ht->table[hash]; hentry != NULL; hentry = hentry->next)
641  {
642  if (hentry->key == key || (*ht->cmp_func) (hentry->key, key))
643  {
644  /* Replace the old data with the new one */
645  if (!(opt & CCI_MHT_OPT_KEEP_KEY))
646  {
647  hentry->key = key;
648  }
649  hentry->data = data;
650  return key;
651  }
652  }
653  }
654 
655  /* This is a new entry */
656  if (ht->nprealloc_entries > 0)
657  {
658  ht->nprealloc_entries--;
659  hentry = ht->prealloc_entries;
660  ht->prealloc_entries = ht->prealloc_entries->next;
661  }
662  else
663  {
664  hentry = (CCI_HENTRY_PTR) MALLOC (sizeof (CCI_HENTRY));
665  if (hentry == NULL)
666  {
667  return NULL;
668  }
669  }
