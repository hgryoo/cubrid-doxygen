506 void _MaxSizeLogAppender::roll()
507 {
508  std::stringstream newPath_stream;
509  newPath_stream << context.path << "." << getCurrDateTime();
510 
511  std::stringstream postfix_stream;
512  postfix_stream << "(" << currBackupCount++ << ")";
513 
514  std::string newPath = rename (newPath_stream.str().c_str(),
515  postfix_stream.str().c_str());
516  backupList.push_back (newPath);
517 
518  if (backupList.size() > (size_t) maxBackupCount)
519  {
520  std::string remove_path = backupList.front();
521  backupList.pop_front();
522 
523  int e = remove (remove_path.c_str());
524  if (e != 0)
525  {
526  perror ("remove");
527  }
528  }
529 }

687 void _Logger::logPrefix (CCI_LOG_LEVEL level)
688 {
689  struct tm cal;
690  time_t t;
691 
692  t = context.now.tv_sec;
693 
694  localtime_r ((const time_t *) &t, &cal);
695  cal.tm_year += 1900;
696  cal.tm_mon += 1;
697 
698  char buf[128];
699  unsigned long tid = gettid();
700  snprintf (buf, 128, "%d-%02d-%02d %02d:%02d:%02d.%03d [TID:%lu] [%5s]",
701  cal.tm_year, cal.tm_mon, cal.tm_mday, cal.tm_hour, cal.tm_min,
702  cal.tm_sec, (int) (context.now.tv_usec / 1000), tid,
703  cci_log_level_string[level]);
704 
705  write (buf);

727 _Logger *_LoggerManager::getLogger (const char *path)
728 {
