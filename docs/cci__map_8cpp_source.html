76 static Value
77 map_get_next_id (Map &map, Value &currValue)
78 {
79  do
80  {
81  currValue ++;
82  if (currValue < 0)
83  {
84  currValue = 1;
85  }
86  }
87  while (map.find (currValue) != map.end ());
88 
89  return currValue;

162 T_CCI_ERROR_CODE map_close_otc (T_CCI_CONN mapped_conn_id)
163 {
164  T_CCI_ERROR_CODE error;
165  int i;
166 
167  mutexConnection.lock ();
168 
169  IteratorMapConnection it = mapConnection.find (mapped_conn_id);
170  if (it == mapConnection.end())
171  {
172  error = CCI_ER_CON_HANDLE;
173  }
174  else
175  {
176  T_CON_HANDLE *connection;
177  T_REQ_HANDLE **statement_array;
178 
179  error = hm_get_connection_by_resolved_id (it->second, &connection);
180  if (error == CCI_ER_NO_ERROR && connection != NULL)
181  {
182  statement_array = connection->req_handle_table;
183  for (i = 0; statement_array && i < connection->max_req_handle; i++)
184  {
185  if (statement_array[i] != NULL
186  && statement_array[i]->mapped_stmt_id >= 0)
187  {
188  map_close_ots (statement_array[i]->mapped_stmt_id);
189  statement_array[i]->mapped_stmt_id = -1;
190  }
191  }
192  }
193 
194  mapConnection.erase (it);
195  error = CCI_ER_NO_ERROR;
196  }
197 
198 
199  mutexConnection.unlock ();
200 
201  return error;
202 }
