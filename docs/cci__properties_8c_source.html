100 static int
101 cci_url_get_bool (char *str, bool * value)
102 {
103  static const char *accepts[] = {
104  "true", "false", "on", "off", "yes", "no"
105  };
106  int i, dim;
107 
108  if (value == NULL)
109  {
110  return CCI_ER_INVALID_URL;
111  }
112 
113  dim = DIM (accepts);
114  for (i = 0; i < dim; i++)
115  {
116  if (strcasecmp (accepts[i], str) == 0)
117  {
118  *value = (i % 2) == 0;
119  return CCI_ER_NO_ERROR;
120  }
121  }
122 
123  return CCI_ER_INVALID_URL;
124 }

217 static int
218 cci_url_parse_properties (T_URL_PROPERTY props[], int len, char *properties)
219 {
220  char *token, *save_url = NULL;
221  int error = CCI_ER_NO_ERROR;
222 
223  if (props == NULL)
224  {
225  return CCI_ER_INVALID_URL;
226  }
227 
228  token = strtok_r (properties, "?&", &save_url);
229  while (token != NULL && error == CCI_ER_NO_ERROR)
230  {
231  char *name, *value, *save_property = NULL;
232  name = strtok_r (token, "=", &save_property);
233  value = strtok_r (NULL, "=", &save_property);
234  error = cci_url_set_properties (props, len, name, value);
235 
236  token = strtok_r (NULL, "&", &save_url);
237  }
238 
239  return error;
240 }

243 cci_url_set_althosts (T_CON_HANDLE * handle, char *data)
244 {
245  T_ALTER_HOST *hosts = handle->alter_hosts;
246  char *token, *save_data = NULL, *end;
247  int i, error = CCI_ER_NO_ERROR;
248 
249  memcpy (handle->alter_hosts[0].ip_addr, handle->ip_addr, 4);
250  handle->alter_hosts[0].port = handle->port;
251 
252  for (i = 1;; i++, data = NULL)
253  {
254  char *host, *port, *save_alter = NULL;
255  int v;
256 
257  if (i >= ALTER_HOST_MAX_SIZE)
258  {
259  return CCI_ER_INVALID_URL;
260  }
261 
262  token = strtok_r (data, ",", &save_data);
263  if (token == NULL)
264  {
265  break;
266  }
267

269  if (host == NULL)

274  if (error < 0)

300 static void
301 cci_shuffle_althosts (T_CON_HANDLE * handle)
302 {
303  struct timeval t;
304  int i, j;
305  double r;
306  struct drand48_data buf;
307  T_ALTER_HOST temp_host;
308 
309  gettimeofday (&t, NULL);
310 
311  /* tv_usec returned by gettimeofday on WINDOWS is millisec * 1000 and seeding it would result in generating an even
312  * random number at first. To avoid such a pattern in generating a random number, tv_usec/1000 is used on WINDOWS. */
313 #if defined (WINDOWS)
314  srand48_r (t.tv_usec / 1000, &buf);
315 #else /* WINDOWS */
316  srand48_r (t.tv_usec, &buf);
317 #endif /* !WINDOWS */
318 
319  /* Fisher-Yates shuffle */
320  for (i = handle->alter_host_count - 1; i > 0; i--)
321  {
322  drand48_r (&buf, &r);
323  j = (int) ((i + 1) * r);
324 
325  temp_host = handle->alter_hosts[j];
326  handle->alter_hosts[j] = handle->alter_hosts[i];
327  handle->alter_hosts[i] = temp_host;
328  }
329 }
