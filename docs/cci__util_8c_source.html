270 int
271 ut_str_to_date (char *str, T_CCI_DATE * value)
272 {
273  int error = 0;
274  int yr = 0, mon = 0, day = 0;
275  char *p = NULL;
276  char *end_p = NULL;
277  char delimiter = '\0';
278 
279  assert (value != NULL);
280 
281  if (str == NULL)
282  {
283  return CCI_ER_TYPE_CONVERSION;
284  }
285 
286  p = str;
287 
288  error = str_to_int32 (&yr, &end_p, p, 10);
289  if (error < 0)
290  {
291  return CCI_ER_TYPE_CONVERSION;
292  }
293 
294  delimiter = *end_p;
295  if (delimiter != '-' && delimiter != '/')
296  {
297  return CCI_ER_TYPE_CONVERSION;
298  }
299 
300  p = end_p + 1;
301 
302  error = str_to_int32 (&mon, &end_p, p, 10);
303  if (error < 0)
304  {
305  return CCI_ER_TYPE_CONVERSION;
306  }
307 
308  if (*end_p != delimiter)
309  {
310  return CCI_ER_TYPE_CONVERSION;
311  }
312 
313  p = end_p + 1;
314 
315  error = str_to_int32 (&day, &end_p, p, 10);
316  if (error < 0)
317  {
318  return CCI_ER_TYPE_CONVERSION;
319  }
320 
321  if (*end_p != 0 && !isspace ((int) *end_p))
322  {
323  return CCI_ER_TYPE_CONVERSION;
324  }
325 
326  memset (value, 0, sizeof (T_CCI_DATE));
327  value->yr = yr;
328  value->mon = mon;
329  value->day = day;
330  return 0;
331 }

451 ut_str_to_mtime (char *str, T_CCI_DATE * value)
452 {
453  int error = 0, offset = 0;
454  int hh = 0, mm = 0, ss = 0, ms = 0;
455  char *p = NULL;
456  char *end_p = NULL;
457  double ms_tmp = 0;
458 
459  assert (value != NULL);
460 
461  if (str == NULL)
462  {
463  return CCI_ER_TYPE_CONVERSION;
464  }
465 
466  p = str;
467 
468  error = str_to_int32 (&hh, &end_p, p, 10);
469  if (error < 0)
470  {
471  return CCI_ER_TYPE_CONVERSION;
472  }
473 
474  if (*end_p != ':')
475  {
476  return CCI_ER_TYPE_CONVERSION;
477  }
478 
479  p = end_p + 1;
480 
481  error = str_to_int32 (&mm, &end_p, p, 10);
482  if (error < 0)
483  {
484  return CCI_ER_TYPE_CONVERSION;
485  }
486 
487  if (*end_p != ':')
488  {
489  return CCI_ER_TYPE_CONVERSION;
490  }
491 
492  p = end_p + 1;
493 
494  error = str_to_int32 (&ss, &end_p, p, 10);
495  if (error < 0)
496  {
497  return CCI_ER_TYPE_CONVERSION;
498  }
499 
500  if (*end_p == '.')
501  {
502  p = end_p;

504

511  }
512  else
513  {
514  ms = 0;
515  }
516 
517  if (*end_p != 0 && !isspace ((int) *end_p))
518  {
519  return CCI_ER_TYPE_CONVERSION;
520  }
521 
522  if (*end_p != '\0')
523  {
524  p = end_p + 1;
525  offset = get_pm_offset (p, hh);
526  hh += offset;
527  }
528 
529  memset (value, 0, sizeof (T_CCI_DATE));
530  value->hh = hh;
531  value->mm = mm;
532  value->ss = ss;
533  value->ms = ms;

535  return 0;

777 void
778 ut_date_to_str (T_CCI_DATE * value, T_CCI_U_TYPE u_type, char *str, int size)
779 {
780  if (u_type == CCI_U_TYPE_DATE)
781  {
782  snprintf (str, size, "%04d-%02d-%02d", value->yr, value->mon, value->day);
783  }
784  else if (u_type == CCI_U_TYPE_TIME)
785  {
786  snprintf (str, size, "%02d:%02d:%02d", value->hh, value->mm, value->ss);
787  }
788  else if (u_type == CCI_U_TYPE_TIMESTAMP)
789  {
790  snprintf (str, size, "%04d-%02d-%02d %02d:%02d:%02d", value->yr, value->mon, value->day, value->hh, value->mm,
791  value->ss);
792  }
793  else
794  { /* u_type == CCI_U_TYPE_DATETIME */
795  snprintf (str, size, "%04d-%02d-%02d %02d:%02d:%02d.%03d", value->yr, value->mon, value->day, value->hh,
796  value->mm, value->ss, value->ms);
797  }
798 }

800 void
801 ut_date_tz_to_str (T_CCI_DATE_TZ * value, T_CCI_U_TYPE u_type, char *str, int size)
802 {
803  int len;
804 
805  if (u_type == CCI_U_TYPE_DATETIMETZ || u_type == CCI_U_TYPE_DATETIMELTZ || u_type == CCI_U_TYPE_TIMESTAMPTZ
806  || u_type == CCI_U_TYPE_TIMESTAMPLTZ)
807  {
808  int remain_size;
809  ut_date_to_str ((T_CCI_DATE *) value, u_type, str, size);
810 
811  len = strlen (str);
812  remain_size = size - len;
813  if (remain_size > 1)
814  {
815  str += len;
816  *str++ = ' ';
817  *str = '\0';
818  strncat (str, value->tz, remain_size - 1);
819  }
