32 /*
33  * Collapsable circular queue : this is similar to circular queue, with some particularities:
34  * - head value points to the first used element, tail points to the insert pointer
35  * - queue is empty when tail == head
36  * - elements of the queue may be flagged as free (consumed) from the middle of queue
37  * - when consuming or freeing the head (consume pointer), will collapse all free elements
38  * - consuming from middle or tail does not move tail/head pointers
39  * - the capacity of queue is fixed
40  * - the actual maximum size is (capacity - 1)
41  */
42 namespace mem
43 {
44  template <typename T>
45  class collapsable_circular_queue
46  {
47  private:
48  struct CCQ_SLOT
49  {
50  T value;
51  char flags;
52  };

55  {
56  CCQ_FREE = 0,
57  CCQ_USED
58  };

71  int size (void)
72  {
73  return (m_tail >= m_head) ? (m_tail - m_head) : (m_capacity + m_tail - m_head);
74  };

76  /* returns position in queue reserved */
77  T *produce (T &elem)
78  {
79  int pos;
80 
81  if (size () >= m_capacity - 1)
82  {
83  /* is full */
84  return NULL;
85  }

87 
88  assert (m_buffer[pos].flags == CCQ_FREE);

130  bool consume (T *elem, T *&last_used_elem)
131  {
132  CCQ_SLOT *slot = reinterpret_cast <CCQ_SLOT *> (elem);
133  int pos = (int) (slot - m_buffer);
134 
135  return consume (pos, last_used_elem);
136  };

138  /* deletes the last element of queue (this needs to be produced-undo without release the mutex) */
139  void undo_produce (T *elem)
140  {
141  CCQ_SLOT *slot = reinterpret_cast <CCQ_SLOT *> (elem);
142  int pos = (int) (slot - m_buffer);
143 
144  int last_element = (m_tail == 0) ? (m_capacity - 1) : (m_tail - 1);
145 
146  assert (pos = last_element);
147 
148  assert (m_buffer[pos].flags == CCQ_USED);
149 
150  m_buffer[pos].flags = CCQ_FREE;
151  m_tail = last_element;
152  };

156  {

172  /* returns count of collapsed positions and if last used element (or last before tail) */
173  bool consume (const int pos, T *&last_used_elem)
174  {
175  if (m_buffer[pos].flags != CCQ_USED)
176  {
177  assert (false);
178  return false;
179  }
180 
181  m_buffer[pos].flags = CCQ_FREE;
182 
183  if (pos == m_head)
184  {
185  int collapsed_count = 0;
186  int prev_used_pos;
187 
188  while (m_buffer[m_head].flags == CCQ_FREE)
189  {
190  prev_used_pos = m_head;
191  m_head = (m_head + 1) % m_capacity;
192  collapsed_count++;
193 
194  if (m_head == m_tail)
195  {
196  /* queue becomes free */
197  break;
198  }
199  }
200 
201  assert (m_head == m_tail || m_buffer[m_head].flags == CCQ_USED);
202 
203  last_used_elem = & (m_buffer[prev_used_pos].value);
204 
205  return true;
206  }
207 
208  /* do not collapse tail : is allowed only to advance */
209 
210  return false;
211  };
212
