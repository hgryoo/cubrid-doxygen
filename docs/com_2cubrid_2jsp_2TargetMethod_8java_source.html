89  private Class<?>[] classesFor(String args) throws ClassNotFoundException,
90  ExecuteException {
91  args = args.trim();
92  if (args.length() == 0) {
93  return new Class[0];
94  }
95  // Count semicolons occurences.
96  int semiColons = 0;
97  for (int i = 0; i < args.length(); i++) {
98  if (args.charAt(i) == ',') {
99  semiColons++;
100  }
101  }
102  Class<?>[] classes = new Class[semiColons + 1];
103 
104  int index = 0;
105  for (int i = 0; i < semiColons; i++) {
106  int sep = args.indexOf(',', index);
107  classes[i] = classFor(args.substring(index, sep).trim());
108  index = sep + 1;
109  }
110 
111  classes[semiColons] = classFor(args.substring(index).trim());
112  return classes;
113  }

115  private Class<?> classFor(String className) throws ClassNotFoundException,
116  ExecuteException {
117  int arrayIndex = className.indexOf("[]");
118  if (arrayIndex >= 0) {
119  String arrTag;
120  if (className.indexOf("[][]") >= 0) {
121  if (className.indexOf("[][][]") >= 0) {
122  throw new ExecuteException("Unsupport data type: "
123  + className);
124  }
125  arrTag = "[[";
126  } else {
127  arrTag = "[";
128  }
129  className = arrTag
130  + descriptorFor(className.substring(0, arrayIndex).trim());
131  }
132 
133  if (argClassMap.containsKey(className)) {
134  return argClassMap.get(className);
135  } else {
136  return getClass(className);
137  }
138  }
