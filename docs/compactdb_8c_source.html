92 /*
93  * compactdb - compactdb main routine
94  * return: 0 if successful, error code otherwise
95  * arg(in): a map of command line arguments
96  */
97 int
98 compactdb (UTIL_FUNCTION_ARG * arg)
99 {
100  UTIL_ARG_MAP *arg_map = arg->arg_map;
101  const char *exec_name = arg->command_name;
102  int error;
103  int status = 0;
104  const char *database_name;
105  bool verbose_flag = 0;
106 
107  database_name = utility_get_option_string_value (arg_map, OPTION_STRING_TABLE, 0);
108  verbose_flag = utility_get_option_bool_value (arg_map, COMPACT_VERBOSE_S);
109 
110  if (database_name == NULL || database_name[0] == '\0' || utility_get_option_string_table_size (arg_map) != 1)
111  {
112  compact_usage (arg->argv0);
113  return ER_GENERIC_ERROR;
114  }
115 
116  sysprm_set_force (prm_get_name (PRM_ID_JAVA_STORED_PROCEDURE), "no");
117 
118  AU_DISABLE_PASSWORDS ();
119  db_set_client_type (DB_CLIENT_TYPE_ADMIN_UTILITY);
120  if ((error = db_login ("DBA", NULL)) || (error = db_restart (arg->argv0, TRUE, database_name)))
121  {
122  PRINT_AND_LOG_ERR_MSG ("%s: %s.\n\n", exec_name, db_error_string (3));
123  status = error;
124  }
125  else
126  {
127  status = compactdb_start (verbose_flag);
128  if (status != 0)
129  {
130  util_log_write_errstr ("%s\n", db_error_string (3));
131  }
132  if ((error = db_shutdown ()))
133  {
134  PRINT_AND_LOG_ERR_MSG ("%s: %s.\n", exec_name, db_error_string (3));
135  status = error;
136  }
137  }
138  return status;
139 }

141 /*
142  * compactdb_start - compact database
143  * return: 0 if successful, error code otherwise
144  * verbose_flag(in)
145  */
146 static int
147 compactdb_start (bool verbose_flag)
148 {
149  LIST_MOPS *class_table = NULL;
150  int i;
151  MOBJ object = NULL;
152  HFID *hfid;
153  int status = 0;
154  THREAD_ENTRY *thread_p;
155 
156  /*
157  * Build class name table
158  */
159 
160  if (prm_get_integer_value (PRM_ID_COMPACTDB_PAGE_RECLAIM_ONLY) != 2)
161  {
162  class_table = locator_get_all_mops (sm_Root_class_mop, DB_FETCH_QUERY_READ, NULL);
163  if (class_table == NULL)
164  {
165  return 1; /* error */
166  }
167  }
168 
169  if (prm_get_integer_value (PRM_ID_COMPACTDB_PAGE_RECLAIM_ONLY) == 1)
170  {
171  goto phase2;
172  }
173  else if (prm_get_integer_value (PRM_ID_COMPACTDB_PAGE_RECLAIM_ONLY) == 2)
174  {
175  goto phase3;
176  }
177 
178  if (verbose_flag)
179  {
180  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_PASS1));
181  }
182 
183  thread_p = thread_get_thread_entry_info ();
184 
185  /*
186  * Dump the object definitions
187  */
188  disk_init ();
189  for (i = 0; i < class_table->num; i++)
190  {
191  if (!WS_IS_DELETED (class_table->mops[i]) && class_table->mops[i] != sm_Root_class_mop)
192  {
193  process_class (thread_p, class_table->mops[i], verbose_flag);
194  }
195  }
196  disk_final ();
197 
198  db_commit_transaction ();
199 
200  if (failed_objects != 0)
201  {
202  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_FAILED),
203  total_objects - failed_objects, total_objects);
204  status = 1;
205  /*
206  * TODO Processing should not continue in this case as we cannot be sure
207  * that all references to deleted objects have been set to NULL. Most of
208  * the code in the offline compactdb should be modified to check for
209  * error conditions.
210  */
211  }
212  else
213  {
214  if (verbose_flag)
215  {
216  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_PROCESSED),
217  total_objects);
218  }
219  }
220 
221 phase2:
222  if (verbose_flag)
223  {
224  printf ("\n");
225  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_PASS2));
226  }
227 
228  for (i = 0; i < class_table->num; i++)
229  {
230  ws_find (class_table->mops[i], &object);
231  if (object == NULL)
232  {
233  continue;
234  }
235 
236  if (verbose_flag)
237  {
238  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_CLASS),
239  sm_ch_name (object));
240  }
241  hfid = sm_ch_heap (object);
242  if (hfid->vfid.fileid == NULL_FILEID)
243  {
244  continue;
245  }
246  (void) heap_reclaim_addresses (hfid);
247 
248  }
249 
250 phase3:
251  catalog_reclaim_space (thread_p);
252  db_commit_transaction ();
253 
254  if (file_tracker_reclaim_marked_deleted (thread_p) != NO_ERROR)
255  {
256  /* how to handle error? */
257  ASSERT_ERROR ();
258  er_set (ER_NOTIFICATION_SEVERITY, ARG_FILE_LINE, ER_GENERIC_ERROR, 0);
259  }
260  db_commit_transaction ();
261 
262  /*
263  * Cleanup
264  */
265  locator_free_list_mops (class_table);
266 
267  return status;
268 }

271 /*
272  * process_class - dump objects for given class
273  * return: void
274  * cl_no(in): class table index
275  * verbose_flag(in)
276  */
277 static void
278 process_class (THREAD_ENTRY * thread_p, DB_OBJECT * class_, bool verbose_flag)
279 {
280  int i = 0;
281  SM_CLASS *class_ptr;
282  LC_COPYAREA *fetch_area; /* Area where objects are received */
283  HFID *hfid;
284  OID *class_oid;
285  OID last_oid;
286  LOCK lock = S_LOCK; /* Lock to acquire for the above purpose */
287  int nobjects, nfetched;
288  LC_COPYAREA_MANYOBJS *mobjs; /* Describe multiple objects in area */
289  LC_COPYAREA_ONEOBJ *obj; /* Describe on object in area */
290  RECDES recdes; /* Record descriptor */
291  DESC_OBJ *desc_obj; /* The object described by obj */
292 
293  class_objects = 0;
294 
295  /* Get the class data */
296  ws_find (class_, (MOBJ *) (&class_ptr));
297  if (class_ptr == NULL)
298  {
299  return;
300  }
301 
302  class_oid = ws_oid (class_);
303 
304  if (verbose_flag)
305  {
306  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_CLASS),
307  sm_ch_name ((MOBJ) class_ptr));
308  }
309 
310 #if defined(CUBRID_DEBUG)
311  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_OID), class_oid->volid,
312  class_oid->pageid, class_oid->slotid);
313 #endif
314 
315  /* Find the heap where the instances are stored */
316  hfid = sm_ch_heap ((MOBJ) class_ptr);
317  if (hfid->vfid.fileid == NULL_FILEID)
318  {
319  if (verbose_flag)
320  {
321  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_INSTANCES), 0);
322  }
323  return;
324  }
325 
326  /* Flush all the instances */
327 
328  if (locator_flush_all_instances (class_, DONT_DECACHE) != NO_ERROR)
329  {
330  if (verbose_flag)
331  {
332  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_INSTANCES), 0);
333  }
334  return;
335  }
336 
337  nobjects = 0;
338  nfetched = -1;
339  OID_SET_NULL (&last_oid);
340 
341  /* Now start fetching all the instances */
342  desc_obj = make_desc_obj (class_ptr);
343  while (nobjects != nfetched)
344  {
345  if (locator_fetch_all (hfid, &lock, LC_FETCH_MVCC_VERSION, class_oid, &nobjects, &nfetched, &last_oid,
346  &fetch_area) == NO_ERROR)
347  {
348  if (fetch_area != NULL)
349  {
350  mobjs = LC_MANYOBJS_PTR_IN_COPYAREA (fetch_area);
351  obj = LC_START_ONEOBJ_PTR_IN_COPYAREA (mobjs);
352 
353  for (i = 0; i < mobjs->num_objs; i++)
354  {
355  class_objects++;
356  total_objects++;
357  LC_RECDES_TO_GET_ONEOBJ (fetch_area, obj, &recdes);
358  if (desc_disk_to_obj (class_, class_ptr, &recdes, desc_obj) == NO_ERROR)
359  {
360  process_object (thread_p, desc_obj, &obj->oid, verbose_flag);
361  }
362  else
363  {
364  failed_objects++;
365  }
366 
367  obj = LC_NEXT_ONEOBJ_PTR_IN_COPYAREA (obj);
368  }
369  locator_free_copy_area (fetch_area);
370  }
371  else
372  {
373  /* No more objects */
374  break;
375  }
376  }
377  }
378  desc_free (desc_obj);
379 
380  /*
381  * Now that the class has been processed, we can reclaim space in the catalog
382  * and the schema.
383  * Might be able to ignore failed objects here if we're sure that's
384  * indicative of a corrupted object.
385  */
386  if (failed_objects == 0)
387  {
388  if (catalog_drop_old_representations (thread_p, class_oid) == NO_ERROR)
389  {
390  (void) sm_destroy_representations (class_);
391  }
392  }
393  /* else, should have some sort of warning message ? */
394 
395  if (verbose_flag)
396  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_INSTANCES), class_objects);
397 
398 }

400 /*
401  * process_object - process one object. update instance if needed
402  * return: void
403  * desc_obj(in): object data
404  * obj_oid(in): object oid
405  * recdes(in): record descriptor
406  * verbose_flag(in)
407  */
408 static void
409 process_object (THREAD_ENTRY * thread_p, DESC_OBJ * desc_obj, OID * obj_oid, bool verbose_flag)
410 {
411  SM_CLASS *class_ptr;
412  SM_ATTRIBUTE *attribute;
413  DB_VALUE *value;
414  OID *class_oid;
415  int v = 0;
416  int update_flag = 0;
417 
418  class_ptr = desc_obj->class_;
419  class_oid = ws_oid (desc_obj->classop);
420 
421 #if defined(CUBRID_DEBUG)
422  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_OID), obj_oid->volid,
423  obj_oid->pageid, obj_oid->slotid);
424 #endif
425 
426  attribute = class_ptr->attributes;
427  while (attribute)
428  {
429  value = &desc_obj->values[v++];
430  update_flag += process_value (thread_p, value);
431  attribute = (SM_ATTRIBUTE *) attribute->header.next;
432  }
433 
434  if ((desc_obj->updated_flag) || (update_flag))
435  {
436  if (verbose_flag)
437  {
438  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_UPDATING));
439  }
440  disk_update_instance (desc_obj->classop, desc_obj, obj_oid);
441  }
442 }

445 /*
446  * process_value - process one value
447  * return: whether the object should be updated.
448  * value(in): the value to process
449  */
450 static int
451 process_value (THREAD_ENTRY * thread_p, DB_VALUE * value)
452 {
453  int return_value = 0;
454 
455  switch (DB_VALUE_TYPE (value))
456  {
457  case DB_TYPE_OID:
458  case DB_TYPE_OBJECT:
459  {
460  OID *ref_oid;
461  SCAN_CODE scan_code;
462  HEAP_SCANCACHE scan_cache;
463 
464  if (DB_VALUE_TYPE (value) == DB_TYPE_OID)
465  {
466  ref_oid = db_get_oid (value);
467  }
468  else
469  {
470  ref_oid = WS_OID (db_get_object (value));
471  }
472 
473  if (OID_ISNULL (ref_oid))
474  {
475  break;
476  }
477 
478  heap_scancache_quick_start (&scan_cache);
479  scan_cache.mvcc_snapshot = logtb_get_mvcc_snapshot (thread_p);
480  scan_code = heap_get_visible_version (thread_p, ref_oid, NULL, NULL, &scan_cache, PEEK, NULL_CHN);
481  heap_scancache_end (thread_p, &scan_cache);
482 
483 #if defined(CUBRID_DEBUG)
484  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_REFOID), ref_oid->volid,
485  ref_oid->pageid, ref_oid->slotid, ref_class_oid.volid, ref_class_oid.pageid, ref_class_oid.slotid);
486 #endif
487 
488  if (scan_code != S_SUCCESS)
489  {
490  /* Set NULL link. */
491  OID_SET_NULL (ref_oid);
492  return_value = 1;
493  break;
494  }
495 
496  /* No updates */
497  break;
498  }
499 
500  case DB_TYPE_POINTER:
501  case DB_TYPE_SET:
502  case DB_TYPE_MULTISET:
503  case DB_TYPE_SEQUENCE:
504  {
505  return_value = process_set (thread_p, db_get_set (value));
506  break;
507  }
508 
509  case DB_TYPE_NULL:
510  case DB_TYPE_BLOB:
511  case DB_TYPE_CLOB:
512  default:
513  break;
514  }
515 
516  return return_value;
517 }

520 /*
521  * process_set - process one set
522  * return: whether the object should be updated.
523  * set(in): the set to process
524  */
525 static int
526 process_set (THREAD_ENTRY * thread_p, DB_SET * set)
527 {
528  SET_ITERATOR *it;
529  DB_VALUE *element_value;
530  int return_value = 0;
531 
532  it = set_iterate (set);
533  while ((element_value = set_iterator_value (it)) != NULL)
534  {
535  return_value += process_value (thread_p, element_value);
536  set_iterator_next (it);
537  }
538  set_iterator_free (it);
539 
540  return return_value;
541 }

560 /*
561  * disk_update_instance - update object instance
562  * return: number of processed instance. 0 is error.
563  * classop(in): class object
564  * obj(in): object instance
565  * oid(in): oid
566  */
567 static int
568 disk_update_instance (MOP classop, DESC_OBJ * obj, OID * oid)
569 {
570  HEAP_OPERATION_CONTEXT update_context;
571  HFID *hfid;
572  int save_newsize;
573  bool has_indexes;
574 
575  assert (oid != NULL);
576 
577  Diskrec->length = 0;
578  if (desc_obj_to_disk (obj, Diskrec, &has_indexes))
579  {
580  if (Diskrec->length >= 0) /* OID_ISTEMP */
581  {
582  return 0;
583  }
584  else
585  {
586  save_newsize = -Diskrec->length + DB_PAGESIZE;
587  /* make the record larger */
588  free_recdes (Diskrec);
589  Diskrec = alloc_recdes (save_newsize);
590  if (Diskrec == NULL)
591  {
592  return 0;
593  }
594  /* try one more time */
595  if (desc_obj_to_disk (obj, Diskrec, &has_indexes))
596  {
597  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_CANT_TRANSFORM));
598  return (0);
599  }
600  }
601  }
602 
603  hfid = sm_ch_heap ((MOBJ) (obj->class_));
604  if (HFID_IS_NULL (hfid))
605  {
606  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_NO_HEAP));
607  return (0);
608  }
609 
610  if (has_indexes)
611  {
612  update_indexes (WS_OID (classop), oid, Diskrec);
613  }
614 
615  heap_create_update_context (&update_context, hfid, oid, WS_OID (classop), Diskrec, NULL,
616  UPDATE_INPLACE_CURRENT_MVCCID);
617  if (heap_update_logical (NULL, &update_context) != NO_ERROR)
618  {
619  printf (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_COMPACTDB, COMPACTDB_MSG_CANT_UPDATE));
620  return (0);
621  }
622 
623  return 1;
624 }

678 /*
679  * update_indexes - update index for given OID
680  * return: NO_ERROR or error code
681  * class_oid(in): class oid
682  * obj_oid(in): object oid
683  * rec(in): RECDES instance having new value
684  */
685 static int
686 update_indexes (OID * class_oid, OID * obj_oid, RECDES * rec)
687 {
688  RECDES oldrec = { 0, 0, 0, NULL };
689  bool old_object;
690  int success;
691  HEAP_GET_CONTEXT context;
692  HEAP_SCANCACHE scan_cache;
693 
694  (void) heap_scancache_quick_start (&scan_cache);
695  heap_init_get_context (NULL, &context, obj_oid, class_oid, &oldrec, &scan_cache, COPY, NULL_CHN);
696 
697  old_object = (heap_get_last_version (NULL, &context) == S_SUCCESS);
698 
699  if (old_object)
700  {
701  /*
702  * 4th arg -> give up setting updated attr info
