153  {
154  (*class_list)[i] = NULL;
155  }
156

158  {

179  }
180  }

182  return status;
183

187  free (*class_list);
188  *class_list = NULL;

199 /*
200  * get_class_mops_from_file - Get a list of class mops from specified file
201  * return: error code
202  * input_filename(in): input file name
203  * class_list(out): pointer to returned list of class mops
204  * num_class_mops(out): pointer to returned number of mops
205  */
206 static int
207 get_class_mops_from_file (const char *input_filename, MOP ** class_list, int *num_class_mops)
208 {
209  int status = NO_ERROR;
210  int i = 0;
211  FILE *input_file;
212  char buffer[DB_MAX_IDENTIFIER_LENGTH];
213  char **class_names = NULL;
214  int num_class = 0;
215  int len = 0;
216  char *ptr = NULL;
217 
218  if (input_filename == NULL || class_list == NULL || num_class_mops == NULL)
219  {
220  return ER_FAILED;
221  }
222 
223  *class_list = NULL;
224  *num_class_mops = 0;
225  if (get_num_requested_class (input_filename, &num_class) != NO_ERROR)
226  {
227  return ER_FAILED;
228  }
229 
230  if (num_class == 0)
231  {
232  return ER_FAILED;
233  }
234 
235  input_file = fopen (input_filename, "r");
236  if (input_file == NULL)
237  {
238  perror (input_filename);
239  return ER_FAILED;
240  }
241 
242  class_names = (char **) malloc (DB_SIZEOF (char *) * num_class);
243  if (class_names == NULL)
244  {
245  return ER_FAILED;
246  }
247  for (i = 0; i < num_class; i++)
248  {
249  class_names[i] = NULL;
250  }
251 
252  for (i = 0; i < num_class; ++i)
253  {
254  if (fgets ((char *) buffer, DB_MAX_IDENTIFIER_LENGTH, input_file) == NULL)
255  {
256  status = ER_FAILED;
257  goto end;
258  }
259 
260  ptr = strchr (buffer, '\n');
261  if (ptr)
262  {
263  len = CAST_BUFLEN (ptr - buffer);
264  }
265  else
266  {
267  len = (int) strlen (buffer);
268  }
269 
270  if (len < 1)
271  {
272  status = ER_FAILED;
273  goto end;
274  }
275 
276  class_names[i] = (char *) malloc (DB_SIZEOF (char) * (len + 1));
277  if (class_names[i] == NULL)
278  {
279  status = ER_FAILED;
280  goto end;
281  }
282 
283  strncpy (class_names[i], buffer, len);
284  class_names[i][len] = 0;
285  }
286 
287  status = get_class_mops (class_names, num_class, class_list, num_class_mops);
288 
289 end:
290 
291  if (input_file)
292  {
293  fclose (input_file);
294  }
295 
296  if (class_names)
297  {
298  for (i = 0; i < num_class; i++)
299  {
300  if (class_names[i] != NULL)
301  {
302  free (class_names[i]);
303  class_names[i] = NULL;
304  }
305  }
306 
307  free (class_names);
308  class_names = NULL;
309  }
310 
311  return status;
312 }

1229 static int
1230 do_reclaim_class_addresses (const OID class_oid, char **class_name, bool * const any_class_can_be_referenced,
1231  bool * const correctly_processed, bool * const addresses_reclaimed,
1232  int *const error_while_processing)
1233 {

1235  DB_OBJECT *parent_mop = NULL;
1236  SM_CLASS *class_ = NULL;
1237  SM_CLASS *parent_class_ = NULL;
1238  int error_code = NO_ERROR;
1239  int skipped_error_code = NO_ERROR;
1240  bool do_abort_on_error = true;
1241  bool can_reclaim_addresses = true;
1242  LIST_MOPS *lmops = NULL;
1243  HFID *hfid = NULL;
1244  int is_class = 0;
1245 
1246  assert (!OID_ISNULL (&class_oid));
1247  assert (any_class_can_be_referenced != NULL);
1248  assert (correctly_processed != NULL);
1249  assert (addresses_reclaimed != NULL);
1250  assert (error_while_processing != NULL);
1251  assert (class_name != NULL);
1252 
1253  *correctly_processed = false;
1254  *addresses_reclaimed = false;
1255  *error_while_processing = NO_ERROR;
1256 
1257  error_code = db_commit_transaction ();
1258  if (error_code != NO_ERROR)
1259  {
1260  goto error_exit;
1261  }
1262 
1263  error_code = db_set_isolation (TRAN_READ_COMMITTED);
1264  if (error_code != NO_ERROR)
1265  {
1266  goto error_exit;
1267  }
1268 
1269  /*
1270  * Trying to force an ISX_LOCK on the root class. It somehow happens that
1271  * we are left with an IX_LOCK in the end...
1272  */
1273  if (locator_fetch_class (sm_Root_class_mop, DB_FETCH_QUERY_WRITE) == NULL)
1274  {
1275  error_code = ER_FAILED;
1276  goto error_exit;
1277  }
1278 
1279  class_mop = db_object ((OID *) (&class_oid));
1280  if (class_mop == NULL)
1281  {
1282  skipped_error_code = ER_FAILED;
1283  goto error_exit;
1284  }
1285 
1286  is_class = locator_is_class (class_mop, DB_FETCH_WRITE);
1287  if (is_class < 0)
1288  {
1289  skipped_error_code = is_class;
1290  goto error_exit;
1291  }
1292  if (!is_class)
1293  {
1294  skipped_error_code = ER_FAILED;
1295  goto error_exit;
1296  }
1297 
1298  /*
1299  * We need an SCH-M lock on the class to process as early as possible so that
1300  * other transactions don't add references to it in the schema.
1301  */
1302  class_ = (SM_CLASS *) locator_fetch_class (class_mop, DB_FETCH_WRITE);
1303  if (class_ == NULL)
1304  {
1305  assert (er_errid () != NO_ERROR);
1306  skipped_error_code = er_errid ();
1307  goto error_exit;
1308  }
1309 
1310  assert (*class_name == NULL);
1311  *class_name = strdup (sm_ch_name ((MOBJ) class_));
1312  if (*class_name == NULL)
1313  {
1314  error_code = ER_FAILED;
1315  goto error_exit;
1316  }
1317 
1318  if (class_->partition != NULL)
1319  {
1320  /*
1321  * If the current class is a partition of a partitioned class we need
1322  * to get its parent partitioned table and check for references to its
1323  * parent too. If table tbl has partition tbl__p__p0, a reference to tbl
1324  * can point to tbl__p__p0 instances too.
1325  */
1326  skipped_error_code = do_get_partition_parent (class_mop, &parent_mop);
1327  if (skipped_error_code != NO_ERROR)
1328  {
1329  goto error_exit;
1330  }
1331  if (parent_mop != NULL)
1332  {
1333  parent_class_ = (SM_CLASS *) locator_fetch_class (parent_mop, DB_FETCH_WRITE);
1334  if (parent_class_ == NULL)
1335  {
1336  assert (er_errid () != NO_ERROR);
1337  skipped_error_code = er_errid ();
1338  goto error_exit;
1339  }
1340  }
1341  }
1342 
1343  skipped_error_code = locator_flush_all_instances (class_mop, DECACHE);
1344  if (skipped_error_code != NO_ERROR)
1345  {
1346  goto error_exit;
1347  }
1348 
1349  if (class_->class_type != SM_CLASS_CT)
1350  {
1351  can_reclaim_addresses = false;
1352  }
1353  else
1354  {
1355  hfid = sm_ch_heap ((MOBJ) class_);
1356  if (HFID_IS_NULL (hfid))
1357  {
1358  can_reclaim_addresses = false;
1359  }
1360  }
1361 
1362  if (class_->flags & SM_CLASSFLAG_SYSTEM)
1363  {
1364  /*
1365  * It should be safe to process system classes also but we skip them for
1366  * now. Please note that class_instances_can_be_referenced () does not
1367  * check for references from system classes.
1368  * If this is ever changed please consider the impact of reusing system
1369  * objects OIDs.
