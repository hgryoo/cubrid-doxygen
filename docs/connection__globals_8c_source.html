155 /*
156  * css_get_required_conn_num_for_ha() - calculate the number
157  * of connections required for HA
158  * return: the number of connections required for HA
159  */
160 static int
161 css_get_required_conn_num_for_ha (void)
162 {
163  int required_conn_num = 0, num_of_nodes = 0;
164  char *ha_node_list_p = NULL;
165  char *ha_replica_list_p = NULL;
166  int curr_ha_mode;
167 
168 #if defined (SA_MODE)
169  curr_ha_mode = util_get_ha_mode_for_sa_utils ();
170 #else /* SA_MODE */
171  curr_ha_mode = HA_GET_MODE ();
172 #endif /* !SA_MODE */
173 
174  if (curr_ha_mode == HA_MODE_OFF)
175  {
176  return 0;
177  }
178 
179  ha_node_list_p = prm_get_string_value (PRM_ID_HA_NODE_LIST);
180  num_of_nodes = util_get_num_of_ha_nodes (ha_node_list_p);
181 
182  if (HA_GET_MODE () == HA_MODE_REPLICA)
183  {
184  /* one applylogdb for each node */
185  return num_of_nodes * 2;
186  }
187 
188  ha_replica_list_p = prm_get_string_value (PRM_ID_HA_REPLICA_LIST);
189  /* one copylogdb for each replica */
190  required_conn_num = util_get_num_of_ha_nodes (ha_replica_list_p);
191 
192  /* applylogdb and copylogdb for each node */
193  required_conn_num += (num_of_nodes - 1) * 3;
194 
195  return required_conn_num;
196 }
