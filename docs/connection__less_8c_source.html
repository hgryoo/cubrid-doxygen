37 /*
38  * css_make_eid() - create an eid which is a combination of the entry id and
39  * the request id
40  * return: enquiry id
41  * entry_id(in): entry id
42  * rid(in): request id
43  */
44 unsigned int
45 css_make_eid (unsigned short entry_id, unsigned short rid)
46 {
47  int top;
48 
49  top = entry_id;
50  return ((top << 16) | rid);

76 /*
77  * css_make_entry_id() - create an entry structure that will be queued for
78  * reuse
79  * return: entry id
80  * anchor(in): map entry anchor
81  */
82 static unsigned short
83 css_make_entry_id (CSS_MAP_ENTRY * anchor)
84 {
85  CSS_MAP_ENTRY *map_entry_p;
86  static unsigned short entry_id = 0;
87  unsigned short old_value;
88 
89  old_value = entry_id++;
90  if (!entry_id)
91  {
92  entry_id++;
93  }
94 
95  for (map_entry_p = anchor; map_entry_p; map_entry_p = map_entry_p->next)
96  {
97  if (entry_id == old_value)
98  {
99  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ERR_CSS_ENTRY_OVERRUN, 0);
100  }
101 
102  if (entry_id == map_entry_p->id)
103  {
104  entry_id++;
105  map_entry_p = anchor;
106  }
107  }
108 
109  return entry_id;
110 }

112 /*
113  * css_queue_connection() - connection onto the connection entry queue
114  * return: man entry if success, or NULL
115  * conn(in): connection
116  * host(in): host name to connec
117  * anchor(out): map entry anchor
118  */
119 CSS_MAP_ENTRY *
120 css_queue_connection (CSS_CONN_ENTRY * conn, const char *host, CSS_MAP_ENTRY ** anchor)
121 {
122  CSS_MAP_ENTRY *map_entry_p;
123 
124  if (conn == NULL)
125  {
126  return NULL;
127  }
128 
129  map_entry_p = (CSS_MAP_ENTRY *) malloc (sizeof (CSS_MAP_ENTRY));
130  if (map_entry_p != NULL)
131  {
132  if (host)
133  {
134  map_entry_p->key = (char *) malloc (strlen (host) + 1);
135  if (map_entry_p->key != NULL)
136  {
137  strcpy (map_entry_p->key, host);
138  }
139  }
140  else
141  {
142  map_entry_p->key = NULL;
143  }
144  map_entry_p->conn = conn;
145  map_entry_p->next = *anchor;
146  map_entry_p->id = css_make_entry_id (*anchor);
147  *anchor = map_entry_p;
148 
149  return (map_entry_p);
150  }
151 
152  return (NULL);
153 }

178 /*
179  * css_remove_queued_connection_by_entry() - remove the entry from our queue
180  * when a connection is "closed"
181  * return: void
182  * entry(in): entry to find
183  * anchor(in/out): map entry anchor
184  */
185 void
186 css_remove_queued_connection_by_entry (CSS_MAP_ENTRY * entry, CSS_MAP_ENTRY ** anchor)
187 {
188  CSS_MAP_ENTRY *map_entry_p, *prev_map_entry_p;
189 
190  for (map_entry_p = *anchor, prev_map_entry_p = NULL; map_entry_p;
191  prev_map_entry_p = map_entry_p, map_entry_p = map_entry_p->next)
192  {
193  if (entry == map_entry_p)
194  {
195  if (map_entry_p == *anchor)
196  {
197  *anchor = map_entry_p->next;
198  }
199  else
200  {
201  prev_map_entry_p->next = map_entry_p->next;
202  }
203  break;
204  }
205  }
206 
207  if (map_entry_p)
208  {
209  free_and_init (map_entry_p->key);
210  free_and_init (map_entry_p);
211  }
212 }
