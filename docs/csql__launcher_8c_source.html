43 /*
44  * utility_csql_usage() - display csql usage
45  */
46 static void
47 utility_csql_usage (void)
48 {
49  DSO_HANDLE util_sa_library;
50  CSQL_GET_MESSAGE csql_get_message;
51  const char *message;
52 
53  utility_load_library (&util_sa_library, LIB_UTIL_SA_NAME);
54  if (util_sa_library == NULL)
55  {
56  utility_load_print_error (stderr);
57  return;
58  }
59  utility_load_symbol (util_sa_library, (DSO_HANDLE *) (&csql_get_message), "csql_get_message");
60  if (csql_get_message == NULL)
61  {
62  utility_load_print_error (stderr);
63  return;
64  }
65  message = (*csql_get_message) (CSQL_MSG_USAGE);
66  fprintf (stderr, message, PRODUCT_STRING, UTIL_CSQL_NAME);
67 }

152  csql_arg.string_width = 0;

173  case CSQL_CS_MODE_S:
174  csql_arg.cs_mode = true;
175  break;
176 
177  case CSQL_USER_S:

184

191  util_hide_password (optarg);
192  break;
193 
194  case CSQL_ERROR_CONTINUE_S:
195  csql_arg.continue_on_error = true;
196  break;
197 
198  case CSQL_INPUT_FILE_S:
199  if (csql_arg.in_file_name != NULL)

219  csql_arg.single_line_execution = false;
220  break;
221 
222  case CSQL_COMMAND_S:
223  if (csql_arg.command != NULL)
224  {
225  free ((void *) csql_arg.command);
226  }
227  csql_arg.command = strdup (optarg);
228  break;
229 
230  case CSQL_LINE_OUTPUT_S:
231  csql_arg.line_output = true;
232  break;
233 
234  case CSQL_READ_ONLY_S:
235  csql_arg.read_only = true;
236  break;
237 
238  case CSQL_NO_AUTO_COMMIT_S:
239  csql_arg.auto_commit = false;
240  break;
241 
242  case CSQL_NO_PAGER_S:
243  csql_arg.nopager = true;

247  csql_arg.sysadm = true;

268 
269  case CSQL_NO_TRIGGER_ACTION_S:

272

274  csql_arg.plain_output = true;

278  csql_arg.skip_column_names = true;

283  goto exit_on_end;

285  default:

289 
290  if (argc - optind == 1)
291  {

306  {

308  csql_arg.plain_output = false;
309  }
310 
311  if (csql_arg.sysadm && (csql_arg.user_name == NULL || strcasecmp (csql_arg.user_name, "DBA")))
312  {

316

319  /* write_on_standby must come with sysadm */
320  goto print_usage;
321  }
322 
323  if (csql_arg.sa_mode && csql_arg.cs_mode)
324  {
325  /* Don't allow both at once. */
326  goto print_usage;
327  }
328  else if (explicit_single_line && csql_arg.single_line_execution == false)
329  {
330  /* Don't allow both at once. */
331  goto print_usage;
332  }
333  else if (csql_arg.sa_mode)
334  {
335  utility_load_library (&util_library, LIB_UTIL_SA_NAME);
336  }
337  else
338  {
339  utility_load_library (&util_library, LIB_UTIL_CS_NAME);
340  }
341 
342  if (util_library == NULL)
343  {
344  utility_load_print_error (stderr);
345  goto exit_on_error;
346  }
347 
348  utility_load_symbol (util_library, (DSO_HANDLE *) (&csql), "csql");
349  if (csql == NULL)
350  {
351  utility_load_print_error (stderr);
352  goto exit_on_error;
353  }
354 
355  error = (*csql) (argv[0], &csql_arg);
356 
357 exit_on_end:
358  if (csql_arg.user_name != NULL)
359  {
360  free ((void *) csql_arg.user_name);
361  }
362  if (csql_arg.passwd != NULL)
363  {
364  free ((void *) csql_arg.passwd);
365  }
366  if (csql_arg.in_file_name != NULL)
367  {
368  free ((void *) csql_arg.in_file_name);
369  }

371  {
372  free ((void *) csql_arg.out_file_name);
373  }
374  if (csql_arg.command != NULL)
375  {
376  free ((void *) csql_arg.command);
377  }
378 
379  return error;
380
