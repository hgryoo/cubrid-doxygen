128 /*
129  * csql_get_session_cmd_no() - find a session command
130  * return: SESSION_CMD number if success.
131  * if error, -1 is returned and csql_Error_code is set.
132  * input(in)
133  *
134  * Note:
135  * The search function succeed when there is only one entry which starts
136  * with the given string, or there is an entry which matches exactly.
137  */
138 int
139 csql_get_session_cmd_no (const char *input)
140 {
141  int i; /* loop counter */
142  int input_cmd_length; /* input command length */
143  int num_matches = 0; /* # of matched commands */
144  int matched_index = -1; /* last matched entry index */
145 
146  if (*input == '\0')
147  {
148  /* csql>; means csql>;xrun */
149  return S_CMD_XRUN;
150  }
151 
152  input_cmd_length = (int) strlen (input);
153  num_matches = 0;
154  matched_index = -1;
155  for (i = 0; i < (int) DIM (csql_Session_cmd_table); i++)
156  {
157  if (strncasecmp (input, csql_Session_cmd_table[i].text, input_cmd_length) == 0)
158  {
159  int ses_cmd_length;
160 
161  ses_cmd_length = (int) strlen (csql_Session_cmd_table[i].text);
162  if (ses_cmd_length == input_cmd_length)
163  {
164  return (csql_Session_cmd_table[i].cmd_no);
165  }
166  num_matches++;
167  matched_index = i;
168  }
169  }
170  if (num_matches != 1)
171  {
172  csql_Error_code = (num_matches > 1) ? CSQL_ERR_SESS_CMD_AMBIGUOUS : CSQL_ERR_SESS_CMD_NOT_FOUND;
173  return (-1);
174  }
175 #if defined (CS_MODE)
176  if (csql_Session_cmd_table[matched_index].flags & CMD_CHECK_CONNECT)
177  {
178  if (db_Connect_status != DB_CONNECTION_STATUS_CONNECTED)
179  {
180  csql_Error_code = CSQL_ERR_SQL_ERROR;
181  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OBJ_NO_CONNECT, 0);
182  return (-1);
183  }
184  }
185 #endif
186 
187  return (csql_Session_cmd_table[matched_index].cmd_no);
188 }

213 /*
214  * csql_help_schema() - display schema information for given class name
215  * return: none
216  * class_name(in)
217  */
218 void
219 csql_help_schema (const char *class_name)
220 {
221  char **line_ptr;
222  char class_title[2 * DB_MAX_IDENTIFIER_LENGTH + 2];
223  char fixed_class_name[DB_MAX_IDENTIFIER_LENGTH];
224  char *class_name_composed = NULL;
225  int composed_size, class_name_size;
226  class_description class_descr;
227 
228  if (class_name == NULL || class_name[0] == 0)
229  {
230  csql_Error_code = CSQL_ERR_CLASS_NAME_MISSED;
231  goto error;
232  }
233 
234  /* class name may be in Unicode decomposed form, in DB we store only composed form */
235  class_name_size = (int) strlen (class_name);
236  if (LANG_SYS_CODESET == INTL_CODESET_UTF8
237  && unicode_string_need_compose (class_name, class_name_size, &composed_size, lang_get_generic_unicode_norm ()))
238  {
239  bool is_composed = false;
240 
241  class_name_composed = (char *) malloc (composed_size + 1);
242  if (class_name_composed == NULL)
243  {
244  csql_Error_code = CSQL_ERR_NO_MORE_MEMORY;
245  goto error;
246  }
247 
248  unicode_compose_string (class_name, class_name_size, class_name_composed, &composed_size, &is_composed,
249  lang_get_generic_unicode_norm ());
250  class_name_composed[composed_size] = '\0';
251  assert (composed_size <= class_name_size);
252 
253  if (is_composed)
254  {
255  class_name = class_name_composed;
256  }
257  }
258 
259  if (strlen (class_name) >= DB_MAX_IDENTIFIER_LENGTH)
260  {
261  csql_Error_code = CSQL_ERR_TOO_LONG_LINE;
262  goto error;
263  }
264  else
265  {
266  strcpy (fixed_class_name, class_name);
267  /* check that both lower and upper case are not truncated */
268  if (intl_identifier_fix (fixed_class_name, -1, true) != NO_ERROR)
269  {
270  csql_Error_code = CSQL_ERR_TOO_LONG_LINE;
271  goto error;
272  }
273  class_name = fixed_class_name;
274  }
275 
276  if (class_descr.init (class_name) != NO_ERROR)
277  {
278  csql_Error_code = CSQL_ERR_SQL_ERROR;
279  goto error;
280  }
281 
282  snprintf (class_title, (2 * DB_MAX_IDENTIFIER_LENGTH + 2),
283  msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_CLASS_HEAD_TEXT),
284  class_descr.class_type);
285  APPEND_HEAD_LINE (class_title);
286  APPEND_MORE_LINE (5, class_descr.name);
287 
288  if (class_descr.supers != NULL)
289  {
290  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_SUPER_CLASS_HEAD_TEXT));
291  for (line_ptr = class_descr.supers; *line_ptr != NULL; line_ptr++)
292  {
293  APPEND_MORE_LINE (5, *line_ptr);
294  }
295  }
296 
297  if (class_descr.subs != NULL)
298  {
299  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_SUB_CLASS_HEAD_TEXT));
300  for (line_ptr = class_descr.subs; *line_ptr != NULL; line_ptr++)
301  {
302  APPEND_MORE_LINE (5, *line_ptr);
303  }
304  }
305 
306  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_ATTRIBUTE_HEAD_TEXT));
307  if (class_descr.attributes == NULL)
308  {
309  APPEND_MORE_LINE (5, msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_NONE_TEXT));
310  }
311  else
312  {
313  for (line_ptr = class_descr.attributes; *line_ptr != NULL; line_ptr++)
314  {
315  APPEND_MORE_LINE (5, *line_ptr);
316  }
317  }
318 
319  if (class_descr.class_attributes != NULL)
320  {
321  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL,
322  CSQL_HELP_CLASS_ATTRIBUTE_HEAD_TEXT));
323  for (line_ptr = class_descr.class_attributes; *line_ptr != NULL; line_ptr++)
324  {
325  APPEND_MORE_LINE (5, *line_ptr);
326  }
327  }
328 
329  if (class_descr.constraints != NULL)
330  {
331  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_CONSTRAINT_HEAD_TEXT));
332  for (line_ptr = class_descr.constraints; *line_ptr != NULL; line_ptr++)
333  {
334  APPEND_MORE_LINE (5, *line_ptr);
335  }
336  }
337 
338  if (class_descr.object_id != NULL)
339  {
340  APPEND_MORE_LINE (0, "");
341  APPEND_MORE_LINE (1, class_descr.object_id);
342  }
343 
344  if (class_descr.methods != NULL)
345  {
346  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_METHOD_HEAD_TEXT));
347  for (line_ptr = class_descr.methods; *line_ptr != NULL; line_ptr++)
348  {
349  APPEND_MORE_LINE (5, *line_ptr);
350  }
351  }
352 
353  if (class_descr.class_methods != NULL)
354  {
355  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_CLASS_METHOD_HEAD_TEXT));
356  for (line_ptr = class_descr.class_methods; *line_ptr != NULL; line_ptr++)
357  {
358  APPEND_MORE_LINE (5, *line_ptr);
359  }
360  }
361 
362  if (class_descr.resolutions != NULL)
363  {
364  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_RESOLUTION_HEAD_TEXT));
365  for (line_ptr = class_descr.resolutions; *line_ptr != NULL; line_ptr++)
366  {
367  APPEND_MORE_LINE (5, *line_ptr);
368  }
369  }
370 
371  if (class_descr.method_files != NULL)
372  {
373  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_METHFILE_HEAD_TEXT));
374  for (line_ptr = class_descr.method_files; *line_ptr != NULL; line_ptr++)
375  {
376  APPEND_MORE_LINE (5, *line_ptr);
377  }
378  }
379 
380  if (class_descr.query_spec != NULL)
381  {
382  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_QUERY_SPEC_HEAD_TEXT));
383  for (line_ptr = class_descr.query_spec; *line_ptr != NULL; line_ptr++)
384  {
385  APPEND_MORE_LINE (5, *line_ptr);
386  }
387  }
388 
389  if (!class_descr.triggers.empty ())
390  {
391  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_TRIGGER_HEAD_TEXT));
392 #if 0 //bSolo: temporary until evolve above gcc 4.4.7
393 /* *INDENT-OFF* */
394  for (auto it : class_descr.triggers)
395 /* *INDENT-ON* */
396  {
397  APPEND_MORE_LINE (5, it);
398  }
399 #else
400  for (auto it = class_descr.triggers.begin (); it != class_descr.triggers.end (); ++it)
401  {
402  APPEND_MORE_LINE (5, *it);
403  }
404 #endif
405  }
406 
407  if (!class_descr.partition.empty ())
408  {
409  APPEND_HEAD_LINE (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_PARTITION_HEAD_TEXT));
410 #if 0 //bSolo: temporary until evolve above gcc 4.4.7
411 /* *INDENT-OFF* */
412  for (auto it : class_descr.partition)
413 /* *INDENT-ON* */
414  {
415  APPEND_MORE_LINE (5, it);
416  }
417 #else
418  for (auto it = class_descr.partition.begin (); it != class_descr.partition.end (); ++it)
419  {
420  APPEND_MORE_LINE (5, *it);
421  }
422 #endif
423  }
424 
425  csql_display_more_lines (msgcat_message (MSGCAT_CATALOG_CSQL, MSGCAT_CSQL_SET_CSQL, CSQL_HELP_SCHEMA_TITLE_TEXT));
426 
427  csql_free_more_lines ();
428 
429  if (class_name_composed != NULL)
430  {
431  free_and_init (class_name_composed);
432  }
433  return;
434 
435 error:
436 
437  if (class_name_composed != NULL)
438  {
439  free_and_init (class_name_composed);
440  }
441 
442  if (csql_Error_code == CSQL_ERR_SQL_ERROR)
443  {
444  csql_display_csql_err (0, 0);
445  }
446  else
447  {
448  nonscr_display_error (csql_Scratch_text, SCRATCH_TEXT_LEN);
449  }
450  csql_free_more_lines ();
451 }
