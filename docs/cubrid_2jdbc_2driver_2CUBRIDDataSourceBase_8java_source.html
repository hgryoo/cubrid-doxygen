193  synchronized String getDataSourceID(String username) {
194  if (username == null) {
195  username = "";
196  }
197  if (dataSourceID == null) {
198  if (url != null) {
199  dataSourceID = url;
200  } else {
201  String host;
202  String hostName = ((serverName != null) ? serverName : "");
203  try {
204  host = InetAddress
205  .getByName(hostName)
206  .getHostAddress();
207  } catch (Exception e) {
208  host = hostName;
209  }
210  dataSourceID = "jdbc:cubrid:" + host + ":"
211  + portNumber + ":"
212  + ((databaseName != null) ? databaseName : "");
213  }
214  }
215  return (dataSourceID + ":" + username);
216  }
