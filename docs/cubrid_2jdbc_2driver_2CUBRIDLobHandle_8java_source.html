45  private void initLob() {
46  int pos = 0;
47 
48  if (packedLobHandle == null) {
49  throw new NullPointerException();
50  }
51 
52  pos += 4; // skip db_type
53 
54  lobSize = 0;
55  for (int i = pos; i < pos + 8; i++) {
56  lobSize <<= 8;
57  lobSize |= (packedLobHandle[i] & 0xff);
58  }
59  pos += 8; // lob_size
60 
61  int locatorSize = 0;
62  for (int i = pos; i < pos + 4; i++) {
63  locatorSize <<= 8;
64  locatorSize |= (packedLobHandle[i] & 0xff);
65  }
66  pos += 4; // locator_size
67 
68  locator = new String(packedLobHandle, pos, locatorSize - 1);
69  // remove terminating null character
70  }
