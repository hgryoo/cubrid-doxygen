110  void setOutParam(int index, int sqlType) throws UJciException {

113 
114  if (dbmsType == UConnection.DBMS_ORACLE

117  throw new UJciException(UErrorCode.ER_INVALID_ARGUMENT);

125  synchronized void writeParameter(UOutputBuffer outBuffer)
126  throws UJciException {
127  try {
128  for (int i = 0; i < number; i++) {
129  if (isSetDefaultValue(i) == false && values[i] == null) {
130  outBuffer.addByte(UUType.U_TYPE_NULL);
131  outBuffer.addNull();
132  } else {
133  outBuffer.addByte((byte) types[i]);
134  outBuffer.writeParameter(((byte) types[i]), values[i],
135  isSetDefaultValue(i));
136  }
137  }
138  } catch (IOException e) {

140  }
141  }

143  synchronized void flushLobStreams() {
144  for (int i = 0; i < number; i++) {
145  if (values[i] == null) {
146  continue;
147  } else if (values[i] instanceof CUBRIDBlob) {
148  ((CUBRIDBlob) values[i]).flushFlushableStreams();
149  } else if (values[i] instanceof CUBRIDClob) {
150  ((CUBRIDClob) values[i]).flushFlushableStreams();
151  }
152  }
