65  UInputBuffer(UTimedDataInputStream relatedI, UConnection con)
66  throws IOException, UJciException {
67  input = relatedI;
68  position = 0;
69  uconn = con;
70 
71  int readLen = 0;
72  int totalReadLen = 0;
73  byte[] headerData = new byte[8];
74 
75  while (totalReadLen < 8) {
76  readLen = input.read(headerData, totalReadLen, 8 - totalReadLen);
77  if (readLen == -1) {
78  throw uconn.createJciException(UErrorCode.ER_ILLEGAL_DATA_SIZE);
79  }
80  totalReadLen = totalReadLen + readLen;
81  }
82 
83  capacity = UJCIUtil.bytes2int(headerData, 0);
84  casinfo = new byte[CAS_INFO_SIZE];
85  System.arraycopy(headerData, 4, casinfo, 0, 4);
86  con.setCASInfo(casinfo);
87 
88  if (capacity <= 0) {
89  resCode = 0;
90  capacity = 0;
91  return;
92  }
93 
94  buffer = new byte[capacity];
95  readData();
96 
97  resCode = readInt();
98 
99  if (resCode < 0) {
100  int eCode = readInt();
101  String msg;
102 
103  if (con.isRenewedSessionId()) {
104  byte[] newSessionId = new byte[20];
105 
106  msg = readString(remainedCapacity() - newSessionId.length,
107  UJCIManager.sysCharsetName);
108  readBytes(newSessionId);
109  con.setNewSessionId (newSessionId);
110  } else {
111  msg = readString(remainedCapacity(),
112  UJCIManager.sysCharsetName);
113  }
114 
115  eCode = convertErrorByVersion(resCode, eCode);
116  throw uconn.createJciException(UErrorCode.ER_DBMS, resCode, eCode, msg);
117  }
118  }

267  if (year == 0 && month == 0 && day == 0) {
