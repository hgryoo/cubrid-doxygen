151 #define CHECK_1ARG_RETURN_EXPR(obj, expr) \
152  do { \
153  if((obj) == NULL) { \

159 #define CHECK_2ARGS_RETURN_EXPR(obj1, obj2, expr) \

163  return (expr); \
164  } \
165  } while (0)

173  } while (0)
