138 db_json_convert_string_and_call (const char *json_raw, size_t json_raw_length, Fn &&func, Args &&... args)
139 {
140  JSON_DOC *doc = NULL;

142 
143  error_code = db_json_get_json_from_str (json_raw, doc, json_raw_length);
144  if (error_code != NO_ERROR)
145  {
146  return error_code;
147  }
148 
149  error_code = func (doc, std::forward<Args> (args)...);
150  db_json_delete_doc (doc);
151  return error_code;
152 }
