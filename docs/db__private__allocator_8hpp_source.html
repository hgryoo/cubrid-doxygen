186 template<typename T>
187 inline typename db_private_allocator<T>::pointer
188 db_private_allocator<T>::allocate (size_type count)
189 {
190 #if defined (SERVER_MODE)
191  /* todo: this check takes a few operations. what is better? to allow this overhead or to just add assert and assume
192  * we catch all issues? */
193  if (m_heapid != m_thread_p->private_heap_id)
194  {
195  /* this is not something we should do! */
196  assert (false);
197 
198  HL_HEAPID save_heapid = db_private_set_heapid_to_thread (m_thread_p, m_heapid);
199  pointer p = reinterpret_cast<pointer> (db_private_alloc (m_thread_p, count * sizeof (T)));
200  (void) db_private_set_heapid_to_thread (m_thread_p, save_heapid);
201  return p;
202  }
203  else
204 #endif /* !SERVER_MODE */
205  {
206  return reinterpret_cast<pointer> (db_private_alloc (m_thread_p, count * sizeof (T)));
207  }
208 }

210 template<typename T>
211 inline void
212 db_private_allocator<T>::deallocate (pointer p, size_type UNUSED)
213 {
214 #if defined (SERVER_MODE)
215  if (m_heapid != m_thread_p->private_heap_id)
216  {
217  assert (false);
218  /* what am I gonna do on release mode? this is memory leak! */
219  HL_HEAPID save_heapid = db_private_set_heapid_to_thread (m_thread_p, m_heapid);
220  db_private_free (m_thread_p, p);
221  (void) db_private_set_heapid_to_thread (m_thread_p, save_heapid);
222  }
223  else
224 #endif /* !SERVER_MODE */
225  {
226  db_private_free (m_thread_p, p);
227  }
