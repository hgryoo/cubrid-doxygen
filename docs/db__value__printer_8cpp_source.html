53  // DB_VALUE of type DB_TYPE_BIT or DB_TYPE_VARBIT
54  void describe_bit_string (string_buffer &buf, const db_value *value, bool pad_byte)
55  {
56  unsigned char *bstring;
57  int nibble_length, nibbles, count;
58 
59  assert (value != NULL);
60 
61  bstring = (unsigned char *) db_get_string (value);
62  if (bstring == NULL)
63  {
64  return;
65  }
66 
67  nibble_length = ((db_get_string_length (value) + 3) / 4);
68  for (nibbles = 0, count = 0; nibbles < nibble_length - 1; count++, nibbles += 2)
69  {
70  buf ("%02x", bstring[count]);
71  }
72 
73  /* If we don't have a full byte on the end, print the nibble. */
74  if (nibbles < nibble_length)
75  {
76  if (pad_byte)
77  {
78  buf ("%02x", bstring[count]);
79  }
80  else
81  {
82  //use only the 1st hex digit
83  char tmp[3] = {0};
84  sprintf (tmp, "%1x", bstring[count]);
85  buf += tmp[0];
86  }
87  }
88  }

91  void describe_real (string_buffer &buf, double value, int precision)
92  {
93  char tbuf[24];
94 
95  snprintf (tbuf, sizeof (tbuf), db_value_printer::DECIMAL_FORMAT, precision, value);
96  if (strstr (tbuf, "Inf"))
97  {
98  snprintf (tbuf, sizeof (tbuf), db_value_printer::DECIMAL_FORMAT, precision, (value > 0 ? FLT_MAX : -FLT_MAX));
99  }
100  buf (tbuf);
101  }

266 void db_value_printer::describe_data (const db_value *value)
267 {
268  OID *oid = 0;
269  db_object *obj = 0;
270  db_monetary *money = 0;
271  DB_SET *set = 0;
272  db_elo *elo = 0;
273  DB_MIDXKEY *midxkey;
274  char *src, *pos, *end;
275  double d;
276  char line[1025];
277  char *json_body = NULL;
278 
279  if (DB_IS_NULL (value))
280  {
281  m_buf ("NULL");
282  }
283  switch (DB_VALUE_TYPE (value))
284  {
285  case DB_TYPE_INTEGER:
286  m_buf ("%d", db_get_int (value));
287  break;
288 
289  case DB_TYPE_BIGINT:
290  m_buf ("%lld", (long long) db_get_bigint (value));
291  break;
292 
293  case DB_TYPE_POINTER:
294  m_buf ("%p", db_get_pointer (value));
295  break;
296 
297  case DB_TYPE_SHORT:
298  m_buf ("%d", (int) db_get_short (value));
299  break;
300 
301  case DB_TYPE_ERROR:
302  m_buf ("%d", (int) db_get_error (value));
303  break;
304 
305  case DB_TYPE_FLOAT:
306  describe_real (m_buf, db_get_float (value), DB_FLOAT_DECIMAL_PRECISION);
307  break;
308 
309  case DB_TYPE_DOUBLE:
310  describe_real (m_buf, db_get_double (value), DB_DOUBLE_DECIMAL_PRECISION);
311  break;
312 
313  case DB_TYPE_NUMERIC:
314  m_buf ("%s", numeric_db_value_print (value, line));
315  break;
316 
317  case DB_TYPE_BIT:
318  case DB_TYPE_VARBIT:
319  describe_bit_string (m_buf, value, m_padding);
320  break;
321 
322  case DB_TYPE_CHAR:
323  case DB_TYPE_NCHAR:
324  case DB_TYPE_VARCHAR:
325  case DB_TYPE_VARNCHAR:
326  /* Copy string into buf providing for any embedded quotes. Strings may have embedded NULL characters and
327  * embedded quotes. None of the supported multibyte character codesets have a conflict between a quote
328  * character and the second byte of the multibyte character.
329  */
330  src = db_get_string (value);
331  end = src + db_get_string_size (value);
332  while (src < end)
333  {
334  /* Find the position of the next quote or the end of the string, whichever comes first. This loop is
335  * done in place of strchr in case the string has an embedded NULL.
336  */
337  for (pos = src; pos && pos < end && (*pos) != '\''; pos++)
338  ;
339 
340  /* If pos < end, then a quote was found. If so, copy the partial buffer and duplicate the quote */
341  if (pos < end)
342  {
343  m_buf.add_bytes (pos - src + 1, src);
344  m_buf += '\'';
345  }
346  /* If not, copy the remaining part of the buffer */
347  else
348  {
349  m_buf.add_bytes (end - src, src);
350  }
351 
352  /* advance src to just beyond the point where we left off */
353  src = pos + 1;
354  }
355  break;
356 
357  case DB_TYPE_OBJECT:
358 #if defined(SERVER_MODE)
359  assert (false);
360 #else //#if defined(SERVER_MODE)
361  obj = db_get_object (value);
362  if (obj == NULL)
363  {
364  break;
365  }
366  if (obj->is_vid)
367  {
368  DB_VALUE vobj;
369  vid_object_to_vobj (obj, &vobj);
370  describe_value (&vobj);
371  break;
372  }
373  oid = WS_OID (obj);
374  m_buf ("%d|%d|%d", int (oid->volid), int (oid->pageid), int (oid->slotid));
375 #endif //#if defined(SERVER_MODE)
376  break;
377  /* If we are on the server, fall thru to the oid case The value is probably nonsense, but that is safe to do.
378  * This case should simply not occur.
379  */
380 
381  case DB_TYPE_OID:
382  oid = (OID *) db_get_oid (value);
383  if (oid == NULL)
384  {
385  break;
386  }
387  m_buf ("%d|%d|%d", int (oid->volid), int (oid->pageid), int (oid->slotid));
388  break;
389 
390  case DB_TYPE_VOBJ:
391  m_buf ("vid:");
392  case DB_TYPE_SET:
393  case DB_TYPE_MULTISET:
394  case DB_TYPE_SEQUENCE:
395  set = db_get_set (value);
396  if (set != NULL)
397  {
398  describe_set ((const db_collection *) set);
399  }
400  else
401  {
402  m_buf ("NULL");
403  }
404  break;
405 
406  case DB_TYPE_JSON:
407  json_body = db_get_json_raw_body (value);
408  m_buf ("%s", json_body);
409  db_private_free (NULL, json_body);
410  break;
411  case DB_TYPE_MIDXKEY:
412  midxkey = db_get_midxkey (value);
413  if (midxkey != NULL)
414  {
415  describe_midxkey (midxkey);
416  }
417  else
418  {
419  m_buf ("NULL");
420  }
421  break;
422 
423  case DB_TYPE_BLOB:
424  case DB_TYPE_CLOB:
425  elo = db_get_elo (value);
426  if (elo != NULL)
427  {
428  if (elo->type == ELO_FBO)
429  {
430  assert (elo->locator != NULL);
431  m_buf ("%s", elo->locator);
432  }
433  else /* ELO_LO */
434  {
435  /* should not happen for now */
436  assert (0);
437  }
438  }
439  else
440  {
441  m_buf ("NULL");
442  }
443  break;
444 
445  /*
446  * This constant is necessary to fake out the db_?_to_string()
447  * routines that are expecting a buffer length. Since we assume
448  * that our buffer is big enough in this code, just pass something
449  * that ought to work for every case.
450  */
451 #define TOO_BIG_TO_MATTER 1024
452 
453  case DB_TYPE_TIME:
454  (void) db_time_to_string (line, TOO_BIG_TO_MATTER, db_get_time (value));
455  m_buf (line);
456  break;
457 
458  case DB_TYPE_TIMESTAMP:
459  (void) db_utime_to_string (line, TOO_BIG_TO_MATTER, db_get_timestamp (value));
460  m_buf (line);
461  break;
462  case DB_TYPE_TIMESTAMPLTZ:
463  (void) db_timestampltz_to_string (line, TOO_BIG_TO_MATTER, db_get_timestamp (value));
464  m_buf (line);
465  break;
466 
467  case DB_TYPE_TIMESTAMPTZ:
468  {
469  DB_TIMESTAMPTZ *ts_tz;
470 
471  ts_tz = db_get_timestamptz (value);
472  (void) db_timestamptz_to_string (line, TOO_BIG_TO_MATTER, & (ts_tz->timestamp), & (ts_tz->tz_id));
473  m_buf (line);
474  }
475  break;
476 
477  case DB_TYPE_DATETIME:
478  (void) db_datetime_to_string (line, TOO_BIG_TO_MATTER, db_get_datetime (value));
479  m_buf (line);
480  break;
481  case DB_TYPE_DATETIMELTZ:
482  (void) db_datetimeltz_to_string (line, TOO_BIG_TO_MATTER, db_get_datetime (value));
483  m_buf (line);
484  break;
485 
486  case DB_TYPE_DATETIMETZ:
487  {
488  DB_DATETIMETZ *dt_tz;
489 
490  dt_tz = db_get_datetimetz (value);
491  (void) db_datetimetz_to_string (line, TOO_BIG_TO_MATTER, & (dt_tz->datetime), & (dt_tz->tz_id));
492  m_buf (line);
493  }
494  break;
495 
496  case DB_TYPE_DATE:
497  (void) db_date_to_string (line, TOO_BIG_TO_MATTER, db_get_date (value));
498  m_buf (line);
499  break;
500 
501  case DB_TYPE_MONETARY:
502  money = db_get_monetary (value);
503  OR_MOVE_DOUBLE (&money->amount, &d);
504  describe_money (money);
505  break;
506 
507  case DB_TYPE_NULL:
508  /* Can't get here because the DB_IS_NULL test covers DB_TYPE_NULL */
509  break;
510 
511  case DB_TYPE_VARIABLE:
512  case DB_TYPE_SUB:
513  case DB_TYPE_DB_VALUE:
514  /* make sure line is NULL terminated, may not be necessary line[0] = '\0'; */
515  break;
516 
517  default:
518  /* NB: THERE MUST BE NO DEFAULT CASE HERE. ALL TYPES MUST BE HANDLED! */
519  assert (false);
520  break;
521  }
522 }
