453 dwb_make_wait_queue_entry (DWB_WAIT_QUEUE * wait_queue, void *data)
454 {
455  DWB_WAIT_QUEUE_ENTRY *wait_queue_entry;

458

474 
475  wait_queue_entry->data = data;
476  wait_queue_entry->next = NULL;

517  * dwb_block_disconnect_wait_queue_entry () - Disconnect entry from wait queue.
518  *
519  * return : Disconnected queue entry.
520  * wait_queue (in/out) : The wait queue.
521  * data (in): The data.
522  * Note: If data is NULL, then first entry will be disconnected.
523  */
524 STATIC_INLINE DWB_WAIT_QUEUE_ENTRY *
525 dwb_block_disconnect_wait_queue_entry (DWB_WAIT_QUEUE * wait_queue, void *data)
526 {
527  DWB_WAIT_QUEUE_ENTRY *wait_queue_entry = NULL, *prev_wait_queue_entry = NULL;
528 
529  assert (wait_queue != NULL);
530 
531  if (wait_queue->head == NULL)
532  {
533  return NULL;
534  }
535 
536  prev_wait_queue_entry = NULL;
537  wait_queue_entry = wait_queue->head;
538  if (data != NULL)
539  {
540  /* search for data */
541  while (wait_queue_entry != NULL)
542  {
543  if (wait_queue_entry->data == data)
544  {
545  break;
546  }
547  prev_wait_queue_entry = wait_queue_entry;
548  wait_queue_entry = wait_queue_entry->next;
549  }
550  }
551 
552  if (wait_queue_entry == NULL)
553  {
554  return NULL;
555  }
556 
557  if (prev_wait_queue_entry != NULL)
558  {
559  prev_wait_queue_entry->next = wait_queue_entry->next;
560  }
561  else
562  {
563  assert (wait_queue_entry == wait_queue->head);
564  wait_queue->head = wait_queue_entry->next;
565  }
566 
567  if (wait_queue_entry == wait_queue->tail)
568  {
569  wait_queue->tail = prev_wait_queue_entry;
570  }
571 
572  wait_queue_entry->next = NULL;
573  wait_queue->count--;
574 
575  return wait_queue_entry;
576 }

705 /*
706  * dwb_adjust_write_buffer_values () - Adjust double write buffer values.
707  *
708  * return : Error code.
709  * p_double_write_buffer_size (in/out) : Double write buffer size.
710  * p_num_blocks (in/out): The number of blocks.
711  *
712  * Note: The buffer size must be a multiple of 512 K. The number of blocks must be a power of 2.
713  */
714 STATIC_INLINE void
715 dwb_adjust_write_buffer_values (unsigned int *p_double_write_buffer_size, unsigned int *p_num_blocks)
716 {
717  unsigned int min_size;
718  unsigned int max_size;
719 
720  assert (p_double_write_buffer_size != NULL && p_num_blocks != NULL
721  && *p_double_write_buffer_size > 0 && *p_num_blocks > 0);
722 
723  min_size = DWB_MIN_SIZE;
724  max_size = DWB_MAX_SIZE;
725 
726  if (*p_double_write_buffer_size < min_size)
727  {
728  *p_double_write_buffer_size = min_size;
729  }
730  else if (*p_double_write_buffer_size > min_size)
731  {
732  if (*p_double_write_buffer_size > max_size)
733  {
734  *p_double_write_buffer_size = max_size;
735  }
736  else
737  {
738  /* find smallest number multiple of 512 k */
739  unsigned int limit1 = min_size;
740 
741  while (*p_double_write_buffer_size > limit1)
742  {
743  assert (limit1 <= DWB_MAX_SIZE);
744  if (limit1 == DWB_MAX_SIZE)
745  {
746  break;
747  }
748  limit1 = limit1 << 1;
749  }
750 
751  *p_double_write_buffer_size = limit1;
752  }
753  }
754 
755  min_size = DWB_MIN_BLOCKS;
756  max_size = DWB_MAX_BLOCKS;
757 
758  assert (*p_num_blocks >= min_size);
759 
760  if (*p_num_blocks > min_size)
761  {
762  if (*p_num_blocks > max_size)
763  {
764  *p_num_blocks = max_size;
765  }
766  else if (!IS_POWER_OF_2 (*p_num_blocks))
767  {
768  unsigned int num_blocks = *p_num_blocks;
769 
770  do
771  {
772  num_blocks = num_blocks & (num_blocks - 1);
773  }
774  while (!IS_POWER_OF_2 (num_blocks));
775 
776  *p_num_blocks = num_blocks << 1;
777 
778  assert (*p_num_blocks <= max_size);
779  }
780  }
781 }

783 /*
784  * dwb_starts_structure_modification () - Starts structure modifications.
785  *
786  * return : Error code
787  * thread_p (in): The thread entry.
788  * current_position_with_flags(out): The current position with flags.
789  *
790  * Note: This function must be called before changing structure of DWB.
791  */
792 STATIC_INLINE int
793 dwb_starts_structure_modification (THREAD_ENTRY * thread_p, UINT64 * current_position_with_flags)
794 {
795  UINT64 local_current_position_with_flags, new_position_with_flags, min_version;
796  unsigned int block_no;
797  int error_code = NO_ERROR;
798  unsigned int start_block_no, blocks_count;
799  DWB_BLOCK *helper_flush_block;
800 
801  assert (current_position_with_flags != NULL);
802 
803  do
804  {
805  local_current_position_with_flags = ATOMIC_INC_64 (&dwb_Global.position_with_flags, 0ULL);
806  if (DWB_IS_MODIFYING_STRUCTURE (local_current_position_with_flags))
807  {
808  /* Only one thread can change the structure */
809  return ER_FAILED;
810  }
811 
812  new_position_with_flags = DWB_STARTS_MODIFYING_STRUCTURE (local_current_position_with_flags);
813  /* Start structure modifications, the threads that want to flush afterwards, have to wait. */
814  }
815  while (!ATOMIC_CAS_64 (&dwb_Global.position_with_flags, local_current_position_with_flags, new_position_with_flags));
816 
817 #if defined(SERVER_MODE)
818  while (dwb_flush_block_daemon_is_running () || dwb_flush_block_helper_daemon_is_running ())
819  {
820  /* Can't modify structure while flush thread can access DWB. */
821  thread_sleep (20);
822  }
823 #endif
824 
825  /* Since we set the modify structure flag, I'm the only thread that access the DWB. */
826  helper_flush_block = dwb_Global.helper_flush_block;
827  if (helper_flush_block != NULL)
828  {
829  /* All remaining blocks are flushed by me. */
830  (void) ATOMIC_TAS_ADDR (&dwb_Global.helper_flush_block, (DWB_BLOCK *) NULL);
831  dwb_log ("Structure modification, needs to flush DWB block = %d having version %lld\n",
832  helper_flush_block->block_no, helper_flush_block->version);
833  }
834 
835  local_current_position_with_flags = ATOMIC_INC_64 (&dwb_Global.position_with_flags, 0ULL);
836 
837  /* Need to flush incomplete blocks, ordered by version. */
838  start_block_no = DWB_NUM_TOTAL_BLOCKS;
839  min_version = 0xffffffffffffffff;
840  blocks_count = 0;
841  for (block_no = 0; block_no < DWB_NUM_TOTAL_BLOCKS; block_no++)
842  {
843  if (DWB_IS_BLOCK_WRITE_STARTED (local_current_position_with_flags, block_no))
844  {
845  if (dwb_Global.blocks[block_no].version < min_version)
846  {
847  min_version = dwb_Global.blocks[block_no].version;
848  start_block_no = block_no;
849  }
850  blocks_count++;
851  }
852  }
853 
854  block_no = start_block_no;
855  while (blocks_count > 0)
856  {
857  if (DWB_IS_BLOCK_WRITE_STARTED (local_current_position_with_flags, block_no))
858  {
859  /* Flush all pages from current block */
860  error_code = dwb_flush_block (thread_p, &dwb_Global.blocks[block_no], &local_current_position_with_flags);
861  if (error_code != NO_ERROR)
862  {
863  /* Something wrong happened. */
864  dwb_log_error ("Can't flush block = %d having version %lld\n", block_no,
865  dwb_Global.blocks[block_no].version);
866 
867  return error_code;
868  }
869 
870  dwb_log_error ("DWB flushed %d block having version %lld\n", block_no, dwb_Global.blocks[block_no].version);
871  blocks_count--;
872  }
873 
874  block_no = (block_no + 1) % DWB_NUM_TOTAL_BLOCKS;
875  }
876 
877  local_current_position_with_flags = ATOMIC_INC_64 (&dwb_Global.position_with_flags, 0ULL);
878  assert (DWB_GET_BLOCK_STATUS (local_current_position_with_flags) == 0);
879 
880  *current_position_with_flags = local_current_position_with_flags;
881 
882  return NO_ERROR;
883 }

1489  slot->position_in_block);
1490  }
1491 #endif
1492  }
1493  }
1494 
1495  dwb_log ("Replace hash key (%d, %d), the new LSA=(%lld,%d), the old LSA = (%lld,%d)",
1496  vpid->volid, vpid->pageid, slot->lsa.pageid, slot->lsa.offset,
1497  slots_hash_entry->slot->lsa.pageid, slots_hash_entry->slot->lsa.offset);
1498  }

1502  slot->lsa.offset);
1503  }
1504 
1505  slots_hash_entry->slot = slot;
1506  pthread_mutex_unlock (&slots_hash_entry->mutex);

1508

1510 }
