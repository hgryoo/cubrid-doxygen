108 {

112 
113  if (e == CCI_ER_DBMS) {
114  facility_msg = "DBMS";
115  if (error) {
116  err_code = error->err_code;
117  err_msg = error->err_msg;
118  } else {
119  err_code = 0;
120  err_msg = "Unknown DBMS Error";
121  }
122  } else {
123  err_msg = get_error_msg(e);
124  err_code = e;
125 
126  if (e > -1000) {
127  facility_msg = "CCI";
128  } else if (e > -2000) {
129  facility_msg = "CAS";
130  } else if (e > -3000) {
131  facility_msg = "CLIENT";
132  } else {
133  facility_msg = "UNKNOWN";
134  }
135  }
136 
137  if (!err_msg) {
138  err_msg = "Unknown Error";
139  }
140 
141  sprintf(msg, "ERROR: %s, %d, %s", facility_msg, err_code, err_msg);
142  rb_raise(rb_eStandardError, msg);
143 }
