72  void er_message::swap (er_message &other)
73  {
74  ERMSG_LOG ("before_swap", *this);
75  ERMSG_LOG ("before_swap", other);
76 
77  std::swap (this->err_id, other.err_id);
78  std::swap (this->severity, other.severity);
79  std::swap (this->file_name, other.file_name);
80  std::swap (this->line_no, other.line_no);
81 
82  // msg_area
83  const std::size_t bufsize = sizeof (this->msg_buffer);
84  if (this->msg_area_size <= bufsize && other.msg_area_size <= bufsize)
85  {
86  assert (this->msg_area_size == bufsize && other.msg_area_size == bufsize);
87 
88  // swap buffer contexts
89  char aux_buffer[bufsize];
90  std::memcpy (aux_buffer, this->msg_buffer, bufsize);
91  std::memcpy (this->msg_buffer, other.msg_buffer, bufsize);
92  std::memcpy (other.msg_buffer, aux_buffer, bufsize);
93  }
94  else if (this->msg_area_size > bufsize && other.msg_area_size > bufsize)
95  {
96  /* swap pointers */
97  std::swap (this->msg_area, other.msg_area);
98  std::swap (this->msg_area_size, other.msg_area_size);
99  }
100  else if (this->msg_area_size <= bufsize)
101  {
102  assert (this->msg_area_size == bufsize);
103  assert (other.msg_area_size > bufsize);
104 
105  // copy this buffer to other
106  std::memcpy (other.msg_buffer, this->msg_buffer, bufsize);
107  // move msg_area pointer from other to this
108  this->msg_area = other.msg_area;
109  other.msg_area = other.msg_buffer;
110  // swap area size
111  std::swap (this->msg_area_size, other.msg_area_size);
112  }
113  else
114  {
115  assert (this->msg_area_size > bufsize);
116  assert (other.msg_area_size == bufsize);
117 
118  // copy other buffer to this
119  std::memcpy (this->msg_buffer, other.msg_buffer, bufsize);
120  // move msg_area pointer from this to other
121  other.msg_area = this->msg_area;
122  this->msg_area = this->msg_buffer;
123  // swap area size
124  std::swap (this->msg_area_size, other.msg_area_size);
125  }
126 
127  assert ((this->msg_area_size == bufsize) == (this->msg_area == this->msg_buffer));
128  assert ((other.msg_area_size == bufsize) == (other.msg_area == other.msg_buffer));
129  assert (this->msg_area != other.msg_area);
130 
131  // swap args, nargs
132  std::swap (this->args, other.args);
133  std::swap (this->nargs, other.nargs);
134 
135  ERMSG_LOG ("after_swap", *this);
136  ERMSG_LOG ("after_swap", other);
137  }

186  er_message::reserve_message_area (std::size_t size)
187  {

189  {
190  // no need to resize

192  }

194  ERMSG_LOG ("before_resize", *this);

196  std::size_t new_size = msg_area_size;
