392 /*
393  * er_event - Notify a error event to the handler
394  * return: none
395  */
396 static void
397 er_event (void)
398 {
399  int err_id, severity, nlevels, line_no;
400  const char *file_name, *msg;
401 
402  if (er_Event_pipe == NULL || er_Event_started == false)
403  {
404  return;
405  }
406 
407  /* Get the most detailed error message available */
408  er_all (&err_id, &severity, &nlevels, &line_no, &file_name, &msg);
409 
410 #if !defined(WINDOWS)
411  saved_Sig_handler = os_set_signal_handler (SIGPIPE, er_event_sigpipe_handler);
412 #endif /* not WINDOWS */
413  if (_setjmp (er_Event_jmp_buf) == 0)
414  {
415  fprintf (er_Event_pipe, "%d %s %s\n", err_id, ER_SEVERITY_STRING (severity), msg);
416  fflush (er_Event_pipe);
417  }
418  else
419  {
420  er_Event_started = false;
421  if (er_Hasalready_initiated)
422  {
423  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_EV_BROKEN_PIPE, 1, prm_get_string_value (PRM_ID_EVENT_HANDLER));
424  }
425  if (er_Event_pipe != NULL)
426  {
427  fclose (er_Event_pipe);
428  er_Event_pipe = NULL;
429  }
430  }
431 #if !defined(WINDOWS)
432  os_set_signal_handler (SIGPIPE, saved_Sig_handler);
433 #endif /* not WINDOWS */
434 }

440 er_event_init (void)
441 {
442  int error = NO_ERROR;
443  const char *msg;
444 
445 #if !defined(WINDOWS)
446  saved_Sig_handler = os_set_signal_handler (SIGPIPE, er_event_sigpipe_handler);
447 #endif /* not WINDOWS */
448  if (_setjmp (er_Event_jmp_buf) == 0)
449  {
450  er_Event_started = false;
451  if (er_Event_pipe != NULL)
452  {
453  if (er_Hasalready_initiated)
454  {
455  er_set (ER_NOTIFICATION_SEVERITY, ARG_FILE_LINE, ER_EV_STOPPED, 0);
456  }
457  msg = msgcat_message (MSGCAT_CATALOG_CUBRID, MSGCAT_SET_ERROR, -ER_EV_STOPPED);
458  fprintf (er_Event_pipe, "%d %s %s", ER_EV_STOPPED, ER_SEVERITY_STRING (ER_NOTIFICATION_SEVERITY),
459  (msg ? msg : "?"));
460 
461  fflush (er_Event_pipe);
462  pclose (er_Event_pipe);
463  er_Event_pipe = NULL;
464  }
465 
466  er_Event_pipe = popen (prm_get_string_value (PRM_ID_EVENT_HANDLER), "w");
467  if (er_Event_pipe != NULL)
468  {
469  if (er_Hasalready_initiated)
470  {
471  er_set (ER_NOTIFICATION_SEVERITY, ARG_FILE_LINE, ER_EV_STARTED, 0);
472  }
473  msg = msgcat_message (MSGCAT_CATALOG_CUBRID, MSGCAT_SET_ERROR, -ER_EV_STARTED);
474  fprintf (er_Event_pipe, "%d %s %s", ER_EV_STARTED, ER_SEVERITY_STRING (ER_NOTIFICATION_SEVERITY),
475  (msg ? msg : "?"));
476 
477  fflush (er_Event_pipe);
478  er_Event_started = true;
479  }
480  else
481  {
482  if (er_Hasalready_initiated)
483  {
484  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_EV_CONNECT_HANDLER, 1,
485  prm_get_string_value (PRM_ID_EVENT_HANDLER));
486  }
487  error = ER_EV_CONNECT_HANDLER;
488  }
489  }
490  else
491  {
492  if (er_Hasalready_initiated)
493  {
494  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_EV_BROKEN_PIPE, 1, prm_get_string_value (PRM_ID_EVENT_HANDLER));
495  }
496  if (er_Event_pipe != NULL)
497  {
498  fclose (er_Event_pipe);
499  er_Event_pipe = NULL;
500  }
501  error = ER_EV_BROKEN_PIPE;
502  }
503 #if !defined(WINDOWS)
504  os_set_signal_handler (SIGPIPE, saved_Sig_handler);
505 #endif /* not WINDOWS */
506 
507  return error;

651  * return: none
652  * msglog_filename(in): name of message log file
653  * exit_ask(in): define behavior when a sever error is found
654  *
655  * Note: This function initializes parameters that define the behavior
656  * of the message module (i.e., logging file , and fatal error
657  * exit condition). If the value of msglog_filename is NULL,
658  * error messages are logged/sent to PRM_ER_LOG_FILE. If that
659  * is null, then messages go to stderr. If msglog_filename
660  * is equal to /dev/null, error messages are not logged.
661  */
662 int
663 er_init (const char *msglog_filename, int exit_ask)
664 {

668

672  // do not reinitialize
673  return NO_ERROR;
674  }
675

677  {
678  er_final (ER_ALL_FINAL);
679  }

682  {
683  if (er_Cached_msg[i] && er_Cached_msg[i] != er_Builtin_msg[i])
684  {
685  free_and_init (er_Cached_msg[i]);
686  }
687  er_Cached_msg[i] = (char *) er_Builtin_msg[i];
688  }
689 
690  assert (DIM (er_Fmt_list) > abs (ER_LAST_ERROR));
691 
692  for (i = 0; i < abs (ER_LAST_ERROR); i++)
693  {
694  er_init_fmt (&er_Fmt_list[i]);
695  }
696 
697  msg_catd = msgcat_get_descriptor (MSGCAT_CATALOG_CUBRID);
698  if (msg_catd != NULL)
699  {
700  er_Fmt_msg_fail_count = 0;
701  for (i = 2; i < abs (ER_LAST_ERROR); i++)
702  {
703  msg = msgcat_gets (msg_catd, MSGCAT_SET_ERROR, i, NULL);
704  if (msg == NULL || msg[0] == '\0')
705  {
706  er_Fmt_msg_fail_count++;
707  continue;
708  }
709  else
710  {
711  if (er_create_fmt_msg (&er_Fmt_list[i], -i, msg) == NULL)
712  {
713  er_Fmt_msg_fail_count++;
714  }
715  }
716  }
717  }
718  else
719  {
720  er_Fmt_msg_fail_count = abs (ER_LAST_ERROR) - 2;
721  }
722

724  // quick-fix for reloading error manager issues. the broker generated cas client need to reload error manager with
725  // update file name parameter. however, csql and other utilities should keep their initial file name.
726  // we need to treat this case properly. for now, I just consider "sticky" initialization when a specific file name
727  // is provided as argument. error manager is not reloaded after.
728  // for cas case, error manager is first initialized without a specific filename, so it won't be "sticky". It is
729  // reloaded during boot_register_client.
730  er_Has_sticky_init = (msglog_filename != NULL);

734  er_Singleton_context_p = new context ();

742  case ER_EXIT_DONT_ASK:
743  case ER_NEVER_EXIT:
744  case ER_ABORT:
745  er_Exit_ask = exit_ask;
746  break;
747 
748  default:
749  er_Exit_ask = ER_EXIT_DEFAULT;
750  er_log_debug (ARG_FILE_LINE, er_Cached_msg[ER_LOG_ASK_VALUE], exit_ask,
751  ((er_Exit_ask == ER_EXIT_ASK) ? "ER_EXIT_ASK" : "ER_NEVER_EXIT"));

987

1017  else
1018  {
1019  fp = fopen (path, "w");

1368  *

1370  */

1372 er_set_internal (int severity, const char *file_name, const int line_no, int err_id, int num_args,

1381

1384  assert (err_id != ER_NUM_OVERFLOW);
1385  assert (err_id != ER_QPROC_OVERFLOW_COERCION);

1387  assert (err_id != ER_FK_CANT_DROP_CACHE_ATTR);
1388  assert (err_id != ER_SM_CATALOG_SPACE);

1390  assert (err_id != ER_CT_REPRCNT_OVERFLOW);

1393  {
1394  assert (false);
1395  er_Errid_not_initialized = err_id;
1396  return ER_FAILED;

1398 
1399  // *INDENT-OFF*
1400  context &tl_context = context::get_thread_local_context ();
1401  // *INDENT-ON*

1406  */

1408 
1409  memcpy (&ap, ap_ptr, sizeof (ap));
1410 
1411  // *INDENT-OFF*
1412  // should force error stacking? yes if:
1413  // 1. this is a notification and an error was already set
1414  // 2. current error is interrupted error.
1415  er_message &prev_err = tl_context.get_current_error_level ();
1416  // *INDENT-ON*
1417 
1418  if ((severity == ER_NOTIFICATION_SEVERITY && prev_err.err_id != NO_ERROR) || (prev_err.err_id == ER_INTERRUPTED))
1419  {
1420  tl_context.push_error_stack ();

1422  }
1423 
1424  // *INDENT-OFF*
1425  // get current error reference
1426  er_message &crt_error = tl_context.get_current_error_level ();
1427  // *INDENT-ON*
1428 
1429  /* Initialize the area... */
1430  crt_error.set_error (err_id, severity, file_name, line_no);
1431 
1432  /*
1433  * Get hold of the compiled format string for this message and get an
1434  * estimate of the size of the buffer required to print it.
1435  */
1436  er_fmt = er_find_fmt (err_id, num_args);
1437  if (er_fmt == NULL)
1438  {
1439  /*
1440  * Assumes that er_find_fmt() has already called er_emergency().
1441  */
1442  ret_val = ER_FAILED;
1443  goto end;
1444  }
1445 
1446  if (err_id >= ER_FAILED || err_id <= ER_LAST_ERROR)
1447  {

1449  err_id = ER_FAILED; /* invalid error id handling */
1450  }
1451 
1452  new_size = er_estimate_size (er_fmt, ap_ptr);
1453 
1454  /* Do we need to copy the OS error message? */
1455  if (os_error)
1456  {
1457  new_size += 4 + strlen (os_error);
1458  }
1459 
1460  /* Do any necessary allocation for the buffer. */
1461  crt_error.reserve_message_area (new_size + 1);
1462 
1463  /* And now format the silly thing. */
1464  if (er_vsprintf (&crt_error, er_fmt, ap_ptr) == ER_FAILED)
1465  {
1466  ret_val = ER_FAILED;
1467  goto end;
1468  }
1469  if (os_error != NULL)
1470  {
1471  strcat (crt_error.msg_area, "... ");
1472  strcat (crt_error.msg_area, os_error);
1473  }
1474 
1475  /* Call the logging function if any */
1476  if (severity <= prm_get_integer_value (PRM_ID_ER_LOG_LEVEL)
1477  && !(prm_get_bool_value (PRM_ID_ER_LOG_WARNING) == false && severity == ER_WARNING_SEVERITY)
1478  && er_Fnlog[severity] != NULL)
1479  {
1480  // *INDENT-OFF*
1481  // protect log file mutex
1482  std::unique_lock<std::mutex> log_file_lock (er_Log_file_mutex); // mutex is released on destructor
1483  // *INDENT-ON*
1484  (*er_Fnlog[severity]) (err_id);
1485 
1486  /* call stack dump */
1487  er_call_stack_dump_on_error (severity, err_id);
1488 
1489  /* event handler */
1490  er_notify_event_on_error (err_id);
1491 
1492  if (fp != NULL)
1493  {
1494  /* print file contents */
1495  if (fseek (fp, 0L, SEEK_SET) == 0)
1496  {
1497  char buf[MAX_LINE];
1498  while (fgets (buf, MAX_LINE, fp) != NULL)
1499  {
1500  fprintf (er_Msglog_fh, "%s", buf);
1501  }
1502  (void) fflush (er_Msglog_fh);
1503  }
1504  }
1505 
1506  log_file_lock.unlock ();
1507 
1508  if (er_Print_to_console && severity <= ER_ERROR_SEVERITY && crt_error.msg_area)
1509  {
1510  fprintf (stderr, "%s\n", crt_error.msg_area);
1511  }
1512  }
1513 
1514  /*
1515  * Do we want to stop the system on this error ... for debugging purposes?
1516  */
1517  if (prm_get_integer_value (PRM_ID_ER_STOP_ON_ERROR) == err_id)
1518  {
1519  er_stop_on_error ();
1520  }
1521 
1522  if (severity == ER_NOTIFICATION_SEVERITY)
1523  {
1524  crt_error.clear_error ();
1525  }
1526 
1527 end:
1528 
1529  if (need_stack_pop)
1530  {
1531  tl_context.pop_error_stack_and_destroy ();
1532  }
1533 
1534  return ret_val;
1535 }

1657  */
