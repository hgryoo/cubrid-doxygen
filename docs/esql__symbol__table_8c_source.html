99 /*
100  * pp_new_symbol() - Allocate and initialize a new symbol.
101  * return : new allocated symbol
102  * name(in): character string for the new symbol
103  * scope(in): nesting level of this symbol
104  */
105 SYMBOL *
106 pp_new_symbol (const char *name, int scope)
107 {
108  SYMBOL *sym;
109 
110  if (symbol_free_list == NULL)
111  {
112  sym = (SYMBOL *) es_ht_alloc_new_symbol (sizeof (SYMBOL));
113  }
114  else
115  {
116  sym = symbol_free_list;
117  symbol_free_list = symbol_free_list->next;
118  }
119 
120  ++syms_allocated;
121 
122  sym->name = name ? ((unsigned char *) strdup (name)) : NULL;
123  sym->level = scope;
124  sym->type = NULL;
125  sym->etype = NULL;
126  sym->next = NULL;
127 
128  return sym;
129 }

131 /*
132  * pp_discard_symbol() - Discard a single symbol structure and any attacked
133  * links and args.
134  * return : void
135  * sym(in): the symbol to be discarded
136  *
137  * note : the args field is recycled for initializers.
138  */
139 void
140 pp_discard_symbol (SYMBOL * sym)
141 {
142 
143  if (sym)
144  {
145  if (sym->args)
146  {
147  pp_discard_symbol (sym->args);
148  }
149  ++syms_deallocated;
150 
151  if (sym->name)
152  {
153  free_and_init (sym->name);
154  }
155 
156  if (sym->type)
157  {
158  if (IS_FUNCT (sym->type))
159  {
160  pp_discard_symbol_chain (sym->args);
161  }
162  pp_discard_link_chain (sym->type);
163  }
164 
165  if (syms_to_free_lists)
166  {
167  sym->next = symbol_free_list;
168  symbol_free_list = sym;
169  }
170  else
171  {
172  es_ht_free_symbol (sym);
173  }
174  }
175 
176 }

178 /*
179  * pp_discard_symbol_chain() - Discard an entire cross-linked chain of symbols.
180  * return : void
181  * sym(in): the head of a chain of symbols to discarded
182  */
183 void
184 pp_discard_symbol_chain (SYMBOL * sym)
185 {
186  SYMBOL *p = sym;
187 
188  while (sym)
189  {
190  p = sym->next;
191  pp_discard_symbol (sym);
192  sym = p;
193  }
194 }

196 /*
197  * pp_new_link() - Return a new link. It is initialized to zeros.
198  * return : LINK *
199  */
200 LINK *
201 pp_new_link (void)
202 {
203  LINK *p;
204  int i;
205 
206  if (link_free_list == NULL)
207  {
208  LINKCHUNK *new_chunk = (LINKCHUNK *) malloc (sizeof (LINKCHUNK));
209  if (new_chunk == NULL)
210  {
211  esql_yyverror (pp_get_msg (EX_MISC_SET, MSG_OUT_OF_MEMORY));
212  exit (1);
213  return NULL;
214  }
215 
216  new_chunk->next = link_chunks;
217  link_chunks = new_chunk;
218  link_free_list = new_chunk->link;
219 
220  /* This loop executes LCHUNK-1 times. the last (LCHUNK-th) link is taken care of after the loop. */
221  for (p = link_free_list, i = LCHUNK; --i > 0; ++p)
222  {
223  p->next = p + 1;
224  }
225 
226  p->next = NULL;
227  }
228 
229  p = link_free_list;
230  link_free_list = link_free_list->next;
231 
232  ++links_allocated;
233 
234  memset ((char *) p, 0, sizeof (LINK));
235  p->decl.s.is_long = 0;
236  p->decl.s.is_short = 0;
237  p->next = NULL;
238 
239  return p;
240 }

242 /*
243  * pp_discard_link_chain() - Discard all links in the chain.
244  * Nothing is removed from the structure table, however.
245  * return : void
246  * p(in): The head of a chain of links to be discarded.
247  */
248 void
249 pp_discard_link_chain (LINK * p)

261 /*
262  * pp_discard_link() - Discard a single link.
263  * return : void
264  * p(in): the link to be discarded
265  */
266 void
267 pp_discard_link (LINK * p)
268 {
269  if (IS_FUNCT (p))
270  {
271  pp_discard_symbol_chain (p->decl.d.args);
272  }
273 
274  if (IS_ARRAY (p) && p->decl.d.num_ele)
275  {
276  free_and_init (p->decl.d.num_ele);
277  }
278 
279  ++links_deallocated;
280 
281  p->next = link_free_list;
282  link_free_list = p;
283 }

285 /*
286  * pp_new_structdef() - Allocate a new structdef.
287  * return : STRUCTDEF *
288  * tag(in): the tagname for the struct
289  */
290 STRUCTDEF *
291 pp_new_structdef (const char *tag)
292 {
293  /*
294  * Structs, even "anonymous" ones, must always have tag names for the
295  * benefit of the symbol table manager, so if none was supplied at
296  * creation time we must manufacture one ourselves. These variables
297  * are used in the manufacture of those names.
298  */
299  static int anon_tag = 0;
300  char buf[20];
301 
302  STRUCTDEF *sdef;
303 
304  if (struct_free_list == NULL)
305  {
306  sdef = (STRUCTDEF *) es_ht_alloc_new_symbol (sizeof (STRUCTDEF));
307  }
308  else
309  {
310  sdef = struct_free_list;
311  struct_free_list = struct_free_list->next;
312  }
313 
314  ++sdefs_allocated;
315 
316  if (tag == NULL)
317  {
318  sprintf (buf, "$%d", anon_tag);
319  anon_tag++;
320  tag = buf;
321  }
322 
323  sdef->tag = (unsigned char *) strdup (tag);
324  sdef->type_string = (unsigned char *) "struct";
325  sdef->type = 1; /* struct */
326  sdef->fields = NULL;
327  sdef->next = NULL;
328  sdef->by_name = 0;
329 
330  return sdef;
331 }

333 /*
334  * pp_discard_structdef() - Discard a structdef and any attached fields,
335  * but don't discard any linked structure definitions.
336  * return : void
337  * sdef(in): the structdef to be discarded
338  */
339 void
340 pp_discard_structdef (STRUCTDEF * sdef)
341 {
342  if (sdef)
343  {
344  ++sdefs_deallocated;
345 
346  if (sdef->tag)
347  {
348  free_and_init (sdef->tag);
349  }
350  pp_discard_symbol_chain (sdef->fields);
351 
352  if (syms_to_free_lists)
353  {
354  sdef->next = struct_free_list;
355  struct_free_list = sdef;
356  }
357  else
358  {
359  es_ht_free_symbol ((void *) sdef);
360  }
361  }
362 }

381 /*
382  * pp_new_pseudo_def() - Allocate a new structdef that describes
383  * the declaration
384  * struct {
385  * long length;
386  * char array[n];
387  * };
388  * This is used during the conversion of 'varchar' declarations into
389  * anonymous structs.
390  *
391  * returns/side-effects: STRUCTDEF *
392  * type(in): the type of the pseudo def (bit, varbit, varchar)
393  * subscript(in) : an expression for the array bound of the pseudo def
394  */
395 STRUCTDEF *
396 pp_new_pseudo_def (SPECIFIER_NOUN type, const char *subscript)
397 {
398  STRUCTDEF *sdef;
399  SYMBOL *length_field, *array_field, *db_int32;
400 
401  pp_push_spec_scope ();
402 
403  array_field = pp_new_symbol (VARCHAR_ARRAY_NAME, pp_nesting_level);
404  pp_reset_current_type_spec ();
405  pp_add_type_noun (CHAR_);
406  pp_add_declarator (array_field, D_ARRAY);
407  if (type == N_VARCHAR)
408  {
409  array_field->etype->decl.d.num_ele = pp_strdup (subscript);
410  }
411  else
412  {
413  varstring new_subscript;
414  vs_new (&new_subscript);
415  vs_sprintf (&new_subscript, "((%s)+7)/8", subscript);
416  array_field->etype->decl.d.num_ele = pp_strdup (vs_str (&new_subscript));
417  vs_free (&new_subscript);
418  }
419  pp_add_spec_to_decl (pp_current_type_spec (), array_field);
420  pp_discard_link (pp_current_type_spec ());
421 
422  length_field = pp_new_symbol (VARCHAR_LENGTH_NAME, pp_nesting_level);
423  db_int32 = pp_findsym (pp_Symbol_table, (unsigned char *) "int");
424  pp_reset_current_type_spec ();
425  pp_add_typedefed_spec (db_int32->type);
426  pp_add_spec_to_decl (pp_current_type_spec (), length_field);
427  /*
428  * Don't use pp_discard_link(pp_current_spec()) here, since it came
429  * from a typedef.
430  */
431 
432  pp_pop_spec_scope ();
433 
434  length_field->next = array_field;
435 
436  sdef = pp_new_structdef (NULL);
437  sdef->fields = length_field;
438  pp_Struct_table->add_symbol (pp_Struct_table, sdef);
439 
440  return sdef;
441 }

443 /*
444  * pp_add_declarator() - Add a declarator link to the end of a chain, the head
445  * of which is pointed to by sym->type and the tail of which is pointed to
446  * by sym->etype. *head must be NULL when the chain is empty. Both
447  * pointers are modified as necessary.
448  * return : void
449  * sym(in) : the symbol whose type description is to be modified
450  * type(in) : the new type to be added to the declarator chain
451  */
452 void
453 pp_add_declarator (SYMBOL * sym, int type)
454 {
455  LINK *link;
456 
457  if (type == D_FUNCTION && sym->etype && IS_ARRAY (sym->etype))
458  {
459  esql_yyverror (pp_get_msg (EX_SYMBOL_SET, MSG_ILLEGAL_ARRAY));
460  pp_add_declarator (sym, D_POINTER);
461  }
462 
463  link = pp_new_link ();
464  if (link == NULL)
465  {
466  return;
467  }
468 
469  link->decl.d.dcl_type = (DCL_TYPE) type;
470 
471  if (sym->type == NULL)
472  {
473  sym->type = sym->etype = link;
474  }
475  else if (sym->etype != NULL)
476  {
477  sym->etype->next = link;
478  sym->etype = link;
479  }
480 
481  if (type == D_FUNCTION && sym->etype != NULL)
482  {
483  sym->args = NULL;
484  sym->etype->decl.d.args = NULL;
485  }
486 }

510 /*
511  * pp_clone_type() - Manufacture a clone of the type chain in the input symbol.
512  * The tdef bit in the copy is always cleared.
513  * return : A pointer to the cloned chain, NULL if there were no declarators
514  * to clone.
515  * tchain(in): the type chain to duplicate
516  * endp(in): pointer to the last node in the cloned chain
517  */
518 LINK *
519 pp_clone_type (LINK * tchain, LINK ** endp)
520 {
521  LINK *last = NULL, *head = NULL;
522 
523  for (; tchain; tchain = tchain->next)
524  {
525  if (head == NULL && last == NULL) /* 1st node in the chain */
526  {
527  head = last = pp_new_link ();
528  }
529  else /* Subsequent node */
530  {
531  last->next = pp_new_link ();
532  if (last->next == NULL)
533  {
534  return NULL;
535  }
536  last = last->next;
537  }
538 
539  memcpy ((char *) last, (char *) tchain, sizeof (*last));
540  if (IS_ARRAY (tchain) && tchain->decl.d.num_ele)
541  {
542  last->decl.d.num_ele = pp_strdup (tchain->decl.d.num_ele);
543  }
544  last->next = NULL;
545  last->tdef = NULL;
546 
547  if (IS_FUNCT (last))
548  {
549  SYMBOL *sym, **lastsym;
550  lastsym = &last->decl.d.args;
551  sym = last->decl.d.args;
552  while (sym)
553  {
554  *lastsym = pp_clone_symbol (sym);
555  lastsym = &(*lastsym)->next;
556  sym = sym->next;
557  }
558  }
559  }
560 
561  *endp = last;
562 
563  return head;
564 }

566 /*
567  * pp_the_same_type() - Return 1 if the types match, 0 if they don't.
568  * Ignore the storage class. If 'relax' is true and the array declarator
569  * is the first link in the chain, then a pointer is considered equivalent
570  * to an array.
571  * return : 1 if the types match, 0 if they don't.
572  * p1(in): a type chain
573  * p2(in): another type chain
574  * relax(in): true iff arrays should be considered equivalent to pointers
575  */
576 int
577 pp_the_same_type (LINK * p1, LINK * p2, int relax)
578 {
579  if (relax && IS_PTR_TYPE (p1) && IS_PTR_TYPE (p2))
580  {
581  p1 = p1->next;
582  p2 = p2->next;
583  }
584 
585  for (; p1 && p2; p1 = p1->next, p2 = p2->next)
586  {
587  if (p1->class_ != p2->class_)
588  {
589  return 0;
590  }
591 
592  if (p1->class_ == DECLARATOR)
593  {
594  if (p1->decl.d.dcl_type != p2->decl.d.dcl_type)
595  {
596  return 0;
597  }
598  }
599  else
600  {
601  if ((p1->decl.s.noun == p2->decl.s.noun) && (p1->decl.s.is_long == p2->decl.s.is_long)
602  && (p1->decl.s.is_short == p2->decl.s.is_short) && (p1->decl.s.is_unsigned == p2->decl.s.is_unsigned))
603  {
604  return (p1->decl.s.noun == N_STRUCTURE) ? p1->decl.s.val.v_struct == p2->decl.s.val.v_struct : 1;
605  }
606  return 0;
607  }
608  }
609 
610  esql_yyerror (pp_get_msg (EX_SYMBOL_SET, MSG_UNKNOWN_CLASS));
611  return 0;
612 }

614 /*
615  * pp_attr_str() - Return a string representing the interesting attributes
616  * in a specifier other than the noun and storage class.
617  * return : char *
618  * spec(in): a type link
619  */
620 char *
621 pp_attr_str (LINK * type)
622 {
623  static char str[32];
624 
625  assert (IS_SPECIFIER (type));
626 
627  str[0] = '\0';
628 
629  switch (type->decl.s.noun)
630  {
631  case N_CHR:
632  if (type->decl.s.is_unsigned)
633  {
634  strcpy (str, "unsigned ");
635  }
636  break;
637 
638  case N_INT:
639  if (type->decl.s.is_unsigned)
640  {
641  strcpy (str, "unsigned ");
642  }
643 
644  strcat (str, type->decl.s.is_long ? "long " : type->decl.s.is_short ? "short " : "int ");
645  break;
646 
647  case N_FLOAT:
648  if (type->decl.s.is_long)
649  {
650  strcpy (str, "long ");
651  }
652  break;
653 
654  default:
655  break;
656  }
657 
658  return str;
659 }

661 /*
662  * pp_type_str() - Return a string representing the type represented by
663  * the link chain.
664  * return : char *
665  * link(in): a type chain
666  */
667 const char *
668 pp_type_str (LINK * link)
669 {
670  static char target[80];
671  static char buf[256];
672 
673  target[0] = '\0';
674  buf[0] = '\0';
675 
676  if (link == NULL)
677  {
678  return "(NULL)";
679  }
680 
681  if (link->tdef)
682  {
683  strcpy (target, "tdef ");
684  }
685 
686  for (; link; link = link->next)
687  {
688  if (IS_DECLARATOR (link))
689  {
690  switch (link->decl.d.dcl_type)
691  {
692  case D_ARRAY:
693  strncpy (buf, "array of ", sizeof (buf));
694  break;
695  case D_POINTER:
696  strncpy (buf, "ptr to ", sizeof (buf));
697  break;
698  case D_FUNCTION:
699  strncpy (buf, "function returning ", sizeof (buf));
700  break;
701  default:
702  strncpy (buf, "BAD DECL", sizeof (buf));
703  break;
704  }
705  }
706  else
707  {
708  /* it's a specifier */
709  const char *noun_str;
710 
711  strncpy (buf, pp_attr_str (link), sizeof (buf));
712 
713  switch (link->decl.s.noun)
714  {
715  case N_VOID:
716  noun_str = "void";
717  break;
718  case N_CHR:
719  noun_str = "char";
720  break;
721  case N_INT:
722  noun_str = "int";
723  break;
724  case N_FLOAT:
725  noun_str = "float";
726  break;
727  case N_LABEL:
728  noun_str = "label";
729  break;
730  case N_STRUCTURE:
731  noun_str = (const char *) link->decl.s.val.v_struct->type_string;
732  break;
733  case N_VARCHAR:
734  noun_str = "varchar";
735  break;
736  case N_BIT:
737  noun_str = "bit";
738  break;
739  case N_VARBIT:
740  noun_str = "bit varying";
741  break;
742  default:
743  noun_str = "BAD NOUN";
744  break;
745  }
746  strncat (buf, noun_str, sizeof (buf) - strnlen (buf, sizeof (buf)));
747 
748  if (link->decl.s.noun == N_STRUCTURE)
749  {
750  strncat (target, buf, sizeof (target) - strnlen (target, sizeof (target)));
751  snprintf (buf, sizeof (buf), " %s",
752  (link->decl.s.val.v_struct->tag ? link->decl.s.val.v_struct->
753  tag : ((unsigned char *) "untagged")));
754  }
755  }
756 
757  strncat (target, buf, sizeof (target) - strnlen (target, sizeof (target)));
758  }
759 
760  return target;
761 }
