60 /*
61  * pp_init_whenever_scope() - Allocate a new whenever scope block,
62  * initializing it from the old scope.
63  * return : void
64  * scope(in) :
65  * old_scope(in) : A pointer to another scope block.
66  */
67 void
68 pp_init_whenever_scope (WHENEVER_SCOPE * scope, WHENEVER_SCOPE * old_scope)
69 {
70  if (old_scope == NULL)
71  {
72  old_scope = &default_whenever_scope_initializer;
73  }
74  /*
75  * note : this copies any existing pointers to names associated
76  * with GOTO and CALL actions, so you can't just blindly free_and_init()
77  * the pointers when the scope is exited.
78  */
79  *scope = *old_scope;
80 }
