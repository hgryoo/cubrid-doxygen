300 /*
301  * sort_spage_initialize () - Initialize a slotted page
302  * return: void
303  * pgptr(in): Pointer to slotted page
304  * slots_type(in): Flag which indicates the type of slots
305  * alignment(in): Type of alignment
306  *
307  * Note: A slotted page must be initialized before records are inserted on the
308  * page. The alignment indicates the valid offset where the records
309  * should be stored. This is a requirment for peeking records on pages
310  * according to their alignmnet restrictions.
311  */
312 static void
313 sort_spage_initialize (PAGE_PTR pgptr, INT16 slots_type, INT16 alignment)
314 {
315  SLOTTED_PAGE_HEADER *sphdr;
316 
317  sphdr = (SLOTTED_PAGE_HEADER *) pgptr;
318 
319  sphdr->nslots = 0;
320  sphdr->nrecs = 0;
321 
322 #if defined(CUBRID_DEBUG)
323  if (!spage_is_valid_anchor_type (slots_type))
324  {
325  (void) fprintf (stderr,
326  "sort_spage_initialize: **INTERFACE SYSTEM ERROR BAD value for slots_type = %d.\n"
327  " UNANCHORED_KEEP_SEQUENCE was assumed **\n", slots_type);
328  slots_type = UNANCHORED_KEEP_SEQUENCE;
329  }
330  if (!(alignment == CHAR_ALIGNMENT || alignment == SHORT_ALIGNMENT || alignment == INT_ALIGNMENT
331  || alignment == LONG_ALIGNMENT || alignment == FLOAT_ALIGNMENT || alignment == DOUBLE_ALIGNMENT))
332  {
333  (void) fprintf (stderr,
334  "sort_spage_initialize: **INTERFACE SYSTEM ERROR BAD value = %d"
335  " for alignment. %d was assumed\n. Alignment must be"
336  " either SIZEOF char, short, int, long, float, or double", alignment, MAX_ALIGNMENT);
337  alignment = MAX_ALIGNMENT;
338  }
339 #endif /* CUBRID_DEBUG */
340 
341  sphdr->anchor_flag = slots_type;
342  sphdr->tfree = DB_PAGESIZE - sizeof (SLOTTED_PAGE_HEADER);
343  sphdr->cfree = sphdr->tfree;
344  sphdr->foffset = sizeof (SLOTTED_PAGE_HEADER);
345  sphdr->alignment = alignment;
346  sphdr->waste_align = DB_WASTED_ALIGN (sphdr->foffset, alignment);
347 
348  if (sphdr->waste_align != 0)
349  {
350  sphdr->foffset += sphdr->waste_align;
351  sphdr->cfree -= sphdr->waste_align;
352  sphdr->tfree -= sphdr->waste_align;
353  }
354 }

392 /*
393  * sort_spage_compact () - Compact an slotted page
394  * return: NO_ERROR
395  * pgptr(in): Pointer to slotted page
396  *
397  * Note: Only the records are compacted, the slots are not compacted.
398  */
399 static int
400 sort_spage_compact (PAGE_PTR pgptr)
401 {
402  int i, j;
403  SLOTTED_PAGE_HEADER *sphdr;
404  SLOT *sptr;
405  SLOT **sortptr;
406  INT16 to_offset;
407  int ret = NO_ERROR;
408 
409  sphdr = (SLOTTED_PAGE_HEADER *) pgptr;
410 
411  sortptr = (SLOT **) calloc ((unsigned) sphdr->nrecs, sizeof (SLOT *));
412  if (sortptr == NULL)
413  {
414  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sphdr->nrecs * sizeof (SLOT *));
415  return ER_OUT_OF_VIRTUAL_MEMORY;
416  }
417 
418  /* Populate the array to sort and then sort it */
419  sptr = (SLOT *) ((char *) pgptr + DB_PAGESIZE - sizeof (SLOT));
420  for (j = 0, i = 0; i < sphdr->nslots; sptr--, i++)
421  {
422  if (sptr->roffset != NULL_OFFSET)
423  {
424  sortptr[j++] = sptr;
425  }
426  }
427 
428  qsort ((void *) sortptr, (size_t) sphdr->nrecs, (size_t) sizeof (SLOT *), sort_spage_offsetcmp);
429 
430  to_offset = sizeof (SLOTTED_PAGE_HEADER);
431  for (i = 0; i < sphdr->nrecs; i++)
432  {
433  /* Make sure that the offset is aligned */
434  to_offset = DB_ALIGN (to_offset, sphdr->alignment);
435  if (to_offset == sortptr[i]->roffset)
436  {
437  /* Record slot is already in place */
438  to_offset += sortptr[i]->rlength;
439  }
440  else
441  {
442  /* Move the record */
443  memmove ((char *) pgptr + to_offset, (char *) pgptr + sortptr[i]->roffset, sortptr[i]->rlength);
444  sortptr[i]->roffset = to_offset;
445  to_offset += sortptr[i]->rlength;
446  }
447  }
448 
449  /* Make sure that the next inserted record will be aligned */
450  to_offset = DB_ALIGN (to_offset, sphdr->alignment);
451 
452  sphdr->cfree = sphdr->tfree = (DB_PAGESIZE - to_offset - sphdr->nslots * sizeof (SLOT));
453  sphdr->foffset = to_offset;
454  free_and_init (sortptr);
455 
456  /* The page is set dirty somewhere else */
457 
458  return ret;
459 }

461 /*
462  * sort_spage_find_free () - Find a free area/slot where a record of the given length
463  * can be inserted onto the given slotted page
464  * return: A slot identifier or NULL_SLOTID
465  * pgptr(in): Pointer to slotted page
466  * sptr(out): Pointer to slotted page array pointer
467  * length(in): Length of area/record
468  * type(in): Type of record to be inserted
469  * space(out): Space used/defined
470  *
471  * Note: If there is not enough space on the page, an error condition is
472  * indicated and NULLSLOTID is returned.
473  */
474 static INT16
475 sort_spage_find_free (PAGE_PTR pgptr, SLOT ** sptr, INT16 length, INT16 type, INT16 * space)
476 {
477  SLOTTED_PAGE_HEADER *sphdr;
478  INT16 slotid;
479  INT16 waste;
480 
481  sphdr = (SLOTTED_PAGE_HEADER *) pgptr;
482 
483  /* Calculate the wasting space that this record will introduce. We need to take in consideration the wasting space
484  * when there is space saved */
485  waste = DB_WASTED_ALIGN (length, sphdr->alignment);
486  *space = length + waste;
487 
488  /* Quickly check for available space. We may need to check again if a slot is created (instead of reused) */
489  if (*space > sphdr->tfree)
490  {
491  *sptr = NULL;
492  *space = 0;
493  return NULL_SLOTID;
494  }
495 
496  /* Find a free slot. Try to reuse an unused slotid, instead of allocating a new one */
497 
498  *sptr = (SLOT *) ((char *) pgptr + DB_PAGESIZE - sizeof (SLOT));
499 
500  if (sphdr->nslots == sphdr->nrecs)
501  {
502  slotid = sphdr->nslots;
503  *sptr -= slotid;
504  }
505  else
506  {
507  for (slotid = 0; slotid < sphdr->nslots; (*sptr)--, slotid++)
508  {
509  if ((*sptr)->rtype == REC_HOME || (*sptr)->rtype == REC_BIGONE)
510  {
511  ; /* go ahead */
512  }
513  else
514  { /* impossible case */
515  assert (false);
516  break;
517  }
518  }
519  }
520 
521  /* Make sure that there is enough space for the record and the slot */
522 
523  assert (slotid <= sphdr->nslots);
524 
525  if (slotid >= sphdr->nslots)
526  {
527  *space += sizeof (SLOT);
528  /* We are allocating a new slotid. Check for available space again */
529  if (*space > sphdr->tfree)
530  {
531  *sptr = NULL;
532  *space = 0;
533  return NULL_SLOTID;
534  }
535  else if (*space > sphdr->cfree && sort_spage_compact (pgptr) != NO_ERROR)
536  {
537  *sptr = NULL;
538  *space = 0;
539  return NULL_SLOTID;
540  }
541  /* Adjust the number of slots */
542  sphdr->nslots++;
543  }
544  else
545  {
546  /* We already know that there is total space available since the slot is reused and the space was check above */
547  if (*space > sphdr->cfree && sort_spage_compact (pgptr) != NO_ERROR)
548  {
549  *sptr = NULL;
550  *space = 0;
551  return NULL_SLOTID;
552  }
553  }
554 
555  /* Now separate an empty area for the record */
556  (*sptr)->roffset = sphdr->foffset;
557  (*sptr)->rlength = length;
558  (*sptr)->rtype = type;
559 
560  /* Adjust the header */
561  sphdr->nrecs++;
562  sphdr->tfree -= *space;
563  sphdr->cfree -= *space;
564  sphdr->foffset += length + waste;
565  sphdr->waste_align += waste;
566 
567  /* The page is set dirty somewhere else */
568 
569  return slotid;
570 }

572 /*
573  * sort_spage_insert () - Insert a record onto the given slotted page
574  * return: A slot identifier
575  * pgptr(in): Pointer to slotted page
576  * recdes(in): Pointer to a record descriptor
577  *
578  * Note: If the record does not fit on the page, an error condition is
579  * indicated and NULL_SLOTID is returned.
580  */
581 static INT16
582 sort_spage_insert (PAGE_PTR pgptr, RECDES * recdes)
583 {
584  SLOTTED_PAGE_HEADER *sphdr;
585  SLOT *sptr;
586  INT16 slotid;
587  INT16 used_space;
588 
589  if (recdes->length > SORT_MAXREC_LENGTH)
590  {
591  return NULL_SLOTID;
592  }
593 
594  assert (recdes->type == REC_HOME || recdes->type == REC_BIGONE);
595 
596  slotid = sort_spage_find_free (pgptr, &sptr, recdes->length, recdes->type, &used_space);
597  if (slotid != NULL_SLOTID)
598  {
599  /* Find the free slot and insert the record */
600  memcpy (((char *) pgptr + sptr->roffset), recdes->data, recdes->length);
601

603  sphdr = (SLOTTED_PAGE_HEADER *) pgptr;

609 /*
610  * sort_spage_get_record () - Get specific record
611  * return: S_SUCCESS, S_DOESNT_FIT, S_DOESNT_EXIST
612  * pgptr(in): Pointer to slotted page
613  * slotid(in): Slot identifier of current record
614  * recdes(in): Pointer to a record descriptor
615  * peek_p(in): Indicates whether the record is going to be copied or peeked
616  *
617  * Note: When ispeeking is PEEK, the desired available record is peeked onto
618  * the page. The address of the record descriptor is set to the portion
619  * of the buffer where the record is stored. Peeking a record should be
620  * executed with caution since the slotted module may decide to move
621  * the record around. In general, no other operation should be executed
622  * on the page until the peeking of the record is done. The page should
623  * be fixed and locked to avoid any funny behavior. RECORD should NEVER
624  * be MODIFIED DIRECTLY. Only reads should be performed, otherwise
625  * header information and other records may be corrupted.
626  *
627  * When ispeeking is COPY, the desired available record is
628  * read onto the area pointed by the record descriptor. If the record
629  * does not fit in such an area, the length of the record is returned
630  * as a negative value in recdes->length and an error is indicated in the
631  * return value.
632  */
633 static SCAN_CODE
634 sort_spage_get_record (PAGE_PTR pgptr, INT16 slotid, RECDES * recdes, bool peek_p)
635 {
636  SLOTTED_PAGE_HEADER *sphdr;
637  SLOT *sptr;
638 
639  sphdr = (SLOTTED_PAGE_HEADER *) pgptr;
640 
641  sptr = (SLOT *) ((char *) pgptr + DB_PAGESIZE - sizeof (SLOT));
642 
643  sptr -= slotid;
644 
645  if (slotid < 0 || slotid >= sphdr->nslots || sptr->roffset == NULL_OFFSET)
646  {
647  recdes->length = 0;
648  return S_DOESNT_EXIST;
649  }
650 
651  /*
652  * If peeking, the address of the data in the descriptor is set to the
653  * address of the record in the buffer. Otherwise, the record is copied
654  * onto the area specified by the descriptor
655  */
656  if (peek_p == PEEK)
657  {
658  recdes->area_size = -1;
659  recdes->data = (char *) pgptr + sptr->roffset;
660  }
661  else
662  {
663  /* copy the record */
664 
665  if (sptr->rlength > recdes->area_size)
666  {
667  /*
668  * DOES NOT FIT
669  * Give a hint to the user of the needed length. Hint is given as a
670  * negative value
671  */
672  recdes->length = -sptr->rlength;
673  return S_DOESNT_FIT;
674  }
675 
676  memcpy (recdes->data, (char *) pgptr + sptr->roffset, sptr->rlength);
677  }
678 
679  recdes->length = sptr->rlength;
680  recdes->type = sptr->rtype;
681 
682  return S_SUCCESS;
683 }

868 /*
869  * sort_run_find () - Finds the longest ascending or descending run it can
870  * return:
871  * source(in):
872  * top(in):
873  * st_p(in):
874  * limit(in):
875  * compare(in):
876  * comp_arg(in):
877  * option(in):
878  *
879  * Note: Flip descending run, and assign RUN start and stop
880  */
881 static void
882 sort_run_find (char **source, long *top, SORT_STACK * st_p, long limit, SORT_CMP_FUNC * compare, void *comp_arg,
883  SORT_DUP_OPTION option)
884 {
885  char **start;
886  char **stop;
887  char **next_stop;
888  char **limit_p;
889  SRUN *srun_p;
890 
891  char **dup;
892  char **non_dup;
893  int dup_num;
894  int cmp;
895  bool increasing_order;
896 
897  /* init new SRUN */
898  st_p->top++;
899  srun_p = &(st_p->srun[st_p->top]);
900  srun_p->tree_depth = 1;
901  srun_p->low_high = 'H';
902  srun_p->start = *top;
903 
904  if (*top >= (limit - 1))
905  {
906  /* degenerate run length 1. Must go ahead and compare with length 2, because we may need to flip them */
907  srun_p->stop = limit - 1;
908  *top = limit;
909 
910  return;
911  }
912 
913  start = &source[*top];
914  stop = start + 1;
915  next_stop = stop + 1;
916  limit_p = &source[limit];
917 
918  dup_num = 0;
919 
920  /* have a non-trivial run of length 2 or more */
921  cmp = (*compare) (start, stop, comp_arg);
922  if (cmp > 0)
923  {
924  increasing_order = false; /* mark as non-increasing order run */
925 
926  while (next_stop < limit_p && ((cmp = (*compare) (stop, next_stop, comp_arg)) >= 0))
927  {
928  /* mark duplicate as NULL */
929  SORT_CHECK_DUPLICATE (stop, next_stop); /* increase dup_num */
930 
931  stop = next_stop;
932  next_stop = next_stop + 1;
933  }
934  }
935  else
936  {
937  increasing_order = true; /* mark as increasing order run */
938 
939  /* mark duplicate as NULL */
940  SORT_CHECK_DUPLICATE (start, stop); /* increase dup_num */
941 
942  /* build increasing order run */
943  while (next_stop < limit_p && ((cmp = (*compare) (stop, next_stop, comp_arg)) <= 0))
944  {
945  /* mark duplicate as NULL */
946  SORT_CHECK_DUPLICATE (stop, next_stop); /* increase dup_num */
947 
948  stop = next_stop;
949  next_stop = next_stop + 1;
950  }
951  }
952 
953  /* eliminate duplicates; right-shift slots */
954  if (dup_num)
955  {
956  dup = stop - 1;
957  for (non_dup = dup - 1; non_dup >= start; dup--, non_dup--)
958  {
959  /* find duplicated value slot */
960  if (*dup == NULL)
961  {
962  /* find previous non-duplicated value slot */
963  for (; non_dup >= start; non_dup--)
964  {
965  /* move non-duplicated value slot to duplicated value slot */
966  if (*non_dup != NULL)
967  {
968  *dup = *non_dup;
969  *non_dup = NULL;
970  break;
971  }
972  }
973  }
974  }
975  }
976 
977  /* change non-increasing order run to increasing order run */
978  if (increasing_order != true)
979  {
980  sort_run_flip (start + dup_num, stop);
981  }
982 
983  *top += CAST_BUFLEN (stop - start); /* advance to last visited */
984  srun_p->start += dup_num;
985  srun_p->stop = *top;
986 
987  (*top)++; /* advance to next unvisited element */
988 
989  return;
990 }

992 /*
993  * sort_run_merge () - Merges two runs from source to dest, updateing dest_top
994  * return:
995  * low(in):
996  * high(in):
997  * st_p(in):
998  * compare(in):
999  * comp_arg(in):
1000  * option(in):
1001  */
1002 static void
1003 sort_run_merge (char **low, char **high, SORT_STACK * st_p, SORT_CMP_FUNC * compare, void *comp_arg,
1004  SORT_DUP_OPTION option)
1005 {
1006  char dest_low_high;
1007  char **left_start, **right_start;
1008  char **left_stop, **right_stop;
1009  char **dest_ptr;
1010  SRUN *left_srun_p, *right_srun_p;
1011  int cmp;
1012  int dup_num;
1013 
1014  do
1015  {
1016  /* STEP 1: initialize */
1017  left_srun_p = &(st_p->srun[st_p->top - 1]);
1018  right_srun_p = &(st_p->srun[st_p->top]);
1019 
1020  left_srun_p->tree_depth++;
1021 
1022  if (left_srun_p->low_high == 'L')
1023  {
1024  left_start = &low[left_srun_p->start];
1025  left_stop = &low[left_srun_p->stop];
1026  }
1027  else
1028  {
1029  left_start = &high[left_srun_p->start];
1030  left_stop = &high[left_srun_p->stop];
1031  }
1032 
1033  if (right_srun_p->low_high == 'L')
1034  {
1035  right_start = &low[right_srun_p->start];
1036  right_stop = &low[right_srun_p->stop];
1037  }
1038  else
1039  {
1040  right_start = &high[right_srun_p->start];
1041  right_stop = &high[right_srun_p->stop];
1042  }
1043 
1044  dup_num = 0;
1045 
1046  /* STEP 2: check CON conditions srun follows ascending order. if (left_max < right_min) do FORWARD-CON. we use
1047  * '<' instead of '<=' */
1048  cmp = (*compare) (left_stop, right_start, comp_arg);
1049  if (cmp < 0)
1050  {
1051  /* con == TRUE */
1052  dest_low_high = right_srun_p->low_high;
1053 
1054  if (left_srun_p->low_high == right_srun_p->low_high && left_srun_p->stop + 1 == right_srun_p->start)
1055  {
1056  ;
1057  }
1058  else
1059  {
1060  /* move LEFT to RIGHT's current PART */
1061  if (right_srun_p->low_high == 'L')
1062  {
1063  dest_ptr = &low[right_srun_p->start - 1];
1064  }
1065  else
1066  {
1067  dest_ptr = &high[right_srun_p->start - 1];
1068  }
1069 
1070  while (left_stop >= left_start)
1071  {
1072  /* copy LEFT */
1073  *dest_ptr-- = *left_stop--;
1074  }
1075  }
1076  }
1077  else
1078  {
1079  /* con == FALSE do the actual merge; right-shift merge slots */
1080  if (right_srun_p->low_high == 'L')
1081  {
1082  dest_low_high = 'H';
1083  dest_ptr = &high[right_srun_p->stop];
1084  }
1085  else
1086  {
1087  dest_low_high = 'L';
1088  dest_ptr = &low[right_srun_p->stop];
1089  }
1090 
1091  while (left_stop >= left_start && right_stop >= right_start)
1092  {
1093  cmp = (*compare) (left_stop, right_stop, comp_arg);
1094  if (cmp == 0)
1095  {
1096  /* eliminate duplicate */
1097  if (option == SORT_DUP)
1098  {
1099  sort_append (left_stop, right_stop);
1100  }
1101  dup_num++;
1102 
1103  *dest_ptr-- = *right_stop--;
1104  left_stop--;
1105  }
1106  else if (cmp > 0)
1107  {
1108  *dest_ptr-- = *left_stop--;
1109  }
1110  else
1111  {
1112  *dest_ptr-- = *right_stop--;
1113  }
1114  }
1115 
1116  while (left_stop >= left_start)
1117  {
1118  /* copy the rest of LEFT */
1119  *dest_ptr-- = *left_stop--;
1120  }
1121  while (right_stop >= right_start)
1122  {
1123  /* copy the rest of RIGHT */
1124  *dest_ptr-- = *right_stop--;
1125  }
1126  }
1127 
1128  /* STEP 3: reconfig SORT_STACK */
1129  st_p->top--;
1130  left_srun_p->low_high = dest_low_high;
1131  left_srun_p->start = right_srun_p->start - (left_srun_p->stop - left_srun_p->start + 1) + dup_num;
1132  left_srun_p->stop = right_srun_p->stop;
1133 
1134  }
1135  while ((st_p->top >= 1) /* may need to merge */
1136  && (st_p->srun[st_p->top - 1].tree_depth == st_p->srun[st_p->top].tree_depth));
1137 
1138  return;
1139 }

1141 /*
1142  * sort_run_sort () - An implementation of a run-sort algorithm
1143  * return: pointer to the sorted area
1144  * thread_p(in):
1145  * sort_param(in): sort parameters
1146  * base(in): pointer to the element at the base of the table
1147  * limit(in): numrecs of before current sort
1148  * sort_numrecs(in): numrecs of after privious sort
1149  * otherbase(in): pointer to alternate area suffecient to store base-limit
1150  * srun_limit(in): numrecs of after current sort
1151  *
1152  * Note: This sorts files by successive merging of runs.
1153  *
1154  * This has the advantage of being liner on sorted or reversed data,
1155  * and being order N log base k N, where k is the average length of a
1156  * run. Note that k must be at least 2, so the worst case is N log2 N.
1157  *
1158  * Overall, since long runs are common, this beats the pants off
1159  * quick-sort.
1160  *
1161  * This could be sped up a bit by looking for N runs, and sorting these
1162  * into lists of concatennatable runs.
1163  */
1164 static char **
1165 sort_run_sort (THREAD_ENTRY * thread_p, SORT_PARAM * sort_param, char **base, long limit, long sort_numrecs,
1166  char **otherbase, long *srun_limit)
1167 {
1168  SORT_CMP_FUNC *compare;
1169  void *comp_arg;
1170  SORT_DUP_OPTION option;
1171  char **src, **dest, **result;
1172  SORT_STACK sr_stack, *st_p;
1173  long src_top = 0;
1174  int cnt;
1175 
1176  assert_release (limit == *srun_limit);
1177 
1178  /* exclude already sorted items */
1179  limit -= sort_numrecs;
1180 
1181  if (limit == 0 || (limit == 1 && sort_numrecs == 0))
1182  {
1183  return base;
1184  }
1185 
1186  /* init */
1187  compare = sort_param->cmp_fn;
1188  comp_arg = sort_param->cmp_arg;
1189  option = sort_param->option;
1190 
1191  src = base;
1192  dest = otherbase;
1193  result = NULL;
1194 
1195  st_p = &sr_stack;
1196  st_p->top = -1;
1197 
1198  cnt = (int) (log10 (ceil ((double) limit / 2.0)) / log10 (2.0)) + 2;
1199  if (sort_numrecs)
1200  {
1201  /* reserve space for already found srun */
1202  cnt += 1;
1203  }
1204 
1205  st_p->srun = (SRUN *) db_private_alloc (NULL, cnt * sizeof (SRUN));
1206  if (st_p->srun == NULL)
1207  {
1208  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, cnt * sizeof (SRUN));
1209  return NULL;
1210  }
1211 
1212  do
1213  {
1214  sort_run_find (src, &src_top, st_p, limit, compare, comp_arg, option);
1215  if (src_top < limit)
1216  {
1217  sort_run_find (src, &src_top, st_p, limit, compare, comp_arg, option);
1218  }
1219 
1220  while ((st_p->top >= 1) /* may need to merge */
1221  && ((src_top >= limit) /* case 1: final merge stage */
1222  || ((src_top < limit) /* case 2: non-final merge stage */
1223  && (st_p->srun[st_p->top - 1].tree_depth == st_p->srun[st_p->top].tree_depth))))
1224  {
1225  sort_run_merge (dest, src, st_p, compare, comp_arg, option);
1226  }
1227  }
1228  while (src_top < limit);
1229 
1230  if (sort_numrecs > 0)
1231  {
1232  /* finally, merge with already found srun */
1233  SRUN *srun_p;
1234 
1235  srun_p = &(st_p->srun[++(st_p->top)]);
1236  srun_p->tree_depth = 1;
1237  srun_p->low_high = 'H';
1238  srun_p->start = limit;
1239  srun_p->stop = limit + sort_numrecs - 1;
1240 
1241  sort_run_merge (dest, src, st_p, compare, comp_arg, option);
1242  }
1243 
1244 #if defined(CUBRID_DEBUG)
1245  if (limit != src_top)
1246  {
1247  printf ("Inconsistent sort test d), %ld %ld\n", limit, src_top);
1248  }
1249 #endif /* CUBRID_DEBUG */
1250 
1251  /* include already sorted items */
1252  limit += sort_numrecs;
1253 
1254  /* save limit of non-duplicated value slot */
1255  *srun_limit = limit - st_p->srun[0].start;
1256 
1257  /* move base pointer */
1258  result = base + st_p->srun[0].start;
1259 
1260  if (st_p->srun[0].low_high == 'L')
1261  {
