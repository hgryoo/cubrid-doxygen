54 /*
55  * A Table of Prime Numbers
56  *
57  * Some prime numbers which were picked randomly. This table contains more
58  * smaller prime numbers. For example, between 1000 and 2000, we include
59  * only prime numbers that are farther than 50 units. And above 2000, we
60  * include prime numbers that are farther in 100 units.
61  *
62  * NOTE: if x is a prime number, the n is prime if
63  * X**(n-1) mod n == 1
64  */
65 
66 #define NPRIMES 100

89 /*
90  * fh_calculate_htsize - find a good size for a hash table
91  * return: htsize..
92  * htsize(in): Desired hash table size
93  * Note:
94  * A prime number is the best size for a hash table, unfortunately prime
95  * numbers are difficult to find. If the given htsize, falls among the
96  * prime numbers known by this module, a close prime number to
97  * the given htsize value is returned, otherwise, a power of two
98  * is returned.
99  */
100 static int
101 fh_calculate_htsize (int htsize)
102 {
103  int left, right, middle; /* Indices for binary search */
104 
105  /* Can we get a prime number ? */
106  if (htsize > fh_Primes[NPRIMES - 1])
107  {
108  /* Use a prower of two */
109  if (!ISPOWER2 (htsize))
110  {
111  /* Turn off some bits but the left most one */
112  while (!ISPOWER2 (htsize))
113  {
114  htsize = htsize & (htsize - 1);
115  }
116  htsize = htsize << 1;
117  }
118  }
119  else
120  {
121  /* We can assign a primary number, ... Binary search */
122  for (left = 0, right = NPRIMES - 1; left <= right;)
123  {
124  /* get the middle record */
125  middle = CEIL_PTVDIV ((left + right), 2);
126  if (htsize == fh_Primes[middle])
127  {
128  break;
129  }
130  else if (htsize > fh_Primes[middle])
131  {
132  left = middle + 1;
133  }
134  else
135  {
136  right = middle - 1;
137  }
138  }
139  htsize = fh_Primes[middle];
140  }
141  return htsize;
142 }

145 /*
146  * fh_create - create a new file hash table
147  * return: hash table handle
148  * name(in): Name of hash table
149  * est_size(in): Estimated number of entries
150  * page_size(in): Number of bytes per cached hash entry page
151  * cached_pages(in): Number of entry pages to cache
152  * hash_filename(in): pathname of the hash file
153  * key_type(in): Type of key
154  * data_size(in): Number of bytes of data per entry
155  * hfun(in): Hash function
156  * cmpfun(): Key comparison function
157  * Note:
158  * The estimated number of entries for the hash table is adjusted in order
159  * to obtain certain favorable circumstances when the table is created.
160  * hfun must return an integer between 0 and table_size - 1. cmpfun must
161  * return true if key1 == key2, otherwise, false.
162  */
163 FH_TABLE *
164 fh_create (const char *name, int est_size, int page_size, int cached_pages, const char *hash_filename,
165  FH_KEY_TYPE key_type, int data_size, HASH_FUNC hfun, CMP_FUNC cmpfun)
166 {
167  FH_TABLE *ht; /* Hash table information */
168  FH_PAGE_HDR *pg_hdr; /* Entries of hash table */
169  int size;
170  int n;
171  int entry_size;
172 
173  OR_PUT_NULL_OID (&null_oid);
174  null_int = -1;
175 
176  /* Validate arguments */
177  switch (key_type)
178  {
179  case FH_OID_KEY:
180  entry_size = data_size + offsetof (FH_INFO, fh_oidk_data) + offsetof (FH_ENTRY, info);
181  break;
182  case FH_INT_KEY:
183  entry_size = data_size + offsetof (FH_INFO, fh_intk_data) + offsetof (FH_ENTRY, info);
184  break;
185  default:
186  /*
187  * this should be calling er_set
188  * fprintf(stderr, "Invalid key type\n");
189  */
190  return NULL;
191  }
192 
193  if (page_size < entry_size)
194  {
195  /*
196  * should be calling er_set
197  * fprintf(stderr, "Invalid page_size\n");
198  */
199  return NULL;
200  }
201 
202  /* Allocate the header information for hash table */
203  if ((ht = (FH_TABLE *) malloc (DB_SIZEOF (FH_TABLE))) == NULL)
204  {
205  return NULL;
206  }
207  memset (ht, 0, DB_SIZEOF (FH_TABLE));
208 
209  /* Get a good number of entries for hash table */
210  if (est_size <= 0)
211  {
212  est_size = 2;
213  }
214  est_size = fh_calculate_htsize (est_size);
215 
216  /* Open the hash file */
217  ht->hash_filename = (char *) malloc (PATH_MAX);
218  if (ht->hash_filename == NULL)
219  {
220  fh_destroy (ht);
221  return NULL;
222  }
223  if (!hash_filename || hash_filename[0] == '\0')
224  {
225  tmpnam (ht->hash_filename);
226  }
227  else
228  {
229  strcpy (ht->hash_filename, hash_filename);
230  }
231 #if defined(SOLARIS) || defined(AIX) || (defined(I386) && defined(LINUX)) || (defined(HPUX) && (_LFS64_LARGEFILE == 1))
232  ht->fd = open64 (ht->hash_filename, O_RDWR | O_CREAT | O_TRUNC, 0600);
233 #else
234  ht->fd = open (ht->hash_filename, O_RDWR | O_CREAT | O_TRUNC, 0600);
235 #endif
236  if (ht->fd < 0)
237  {
238  perror (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_MIGDB, MIGDB_MSG_FH_HASH_FILENAME));
239  fh_destroy (ht);
240  return NULL;
241  }
242 
243  /* Allocate the page bitmap */
244  ht->page_size = page_size;
245  ht->data_size = data_size;
246  ht->entry_size = entry_size;
247  ht->entries_per_page = page_size / entry_size;
248  ht->overflow = est_size + 1;
249  ht->bitmap_size = (ht->overflow / ht->entries_per_page + 7) / 8;
250  ht->bitmap = (char *) malloc (ht->bitmap_size);
251  if (ht->bitmap == NULL)
252  {
253  fh_destroy (ht);
254  return NULL;
255  }
256  memset (ht->bitmap, 0, ht->bitmap_size);
257 
258  /* Allocate the cached page headers */
259  size = cached_pages * DB_SIZEOF (FH_PAGE_HDR);
260  if ((pg_hdr = (FH_PAGE_HDR *) malloc (size)) == NULL)
261  {
262  fh_destroy (ht);
263  return NULL;
264  }
265  ht->pg_hdr = ht->pg_hdr_alloc = pg_hdr;
266  memset (ht->pg_hdr, 0, size);
267 
268  /*
269  * Allocate the cached pages and fill in the page headers and
270  * initialize each of the hash entries
271  */
272  for (n = 0; n < cached_pages; n++, pg_hdr++)
273  {
274  pg_hdr->next = pg_hdr + 1;
275  pg_hdr->prev = pg_hdr - 1;
276  if ((pg_hdr->fh_entries = (FH_ENTRY *) malloc (page_size)) == NULL)
277  {
278  fh_destroy (ht);
279  return NULL;
280  }
281  memset ((char *) pg_hdr->fh_entries, 0, page_size);
282  pg_hdr->page = INVALID_FILE_POS;
283  }
284  pg_hdr--;
285  pg_hdr->next = NULL;
286  ht->pg_hdr->prev = NULL;
287 
288  ht->pg_hdr_last = pg_hdr;
289  ht->pg_hdr_free = ht->pg_hdr;
290  ht->hfun = hfun;
291  ht->cmpfun = cmpfun;
292  ht->name = name;
293  ht->size = est_size;
294  ht->cached_pages = cached_pages;
295  ht->nentries = 0;
296  ht->ncollisions = 0;
297  ht->key_type = key_type;
298 
299  return ht;
300 }

303 /*
304  * fh_destroy - destroy the given hash table
305  * return: void
306  * ht(out): Hash table (set as a side effect)
307  */
308 void
309 fh_destroy (FH_TABLE * ht)
310 {
311  FH_PAGE_HDR *pg_hdr; /* Entries of hash table */
312 
313  for (pg_hdr = ht->pg_hdr; pg_hdr; pg_hdr = pg_hdr->next)
314  {
315  if (pg_hdr->fh_entries)
316  {
317  free_and_init (pg_hdr->fh_entries);
318  }
319  }
320  if (ht->pg_hdr_alloc)
321  {
322  free_and_init (ht->pg_hdr_alloc);
323  }
324 
325  if (ht->fd > 0)
326  {
327  close (ht->fd);
328  unlink (ht->hash_filename);
329  }
330  if (ht->hash_filename)
331  {
332  free_and_init (ht->hash_filename);
333  }
334  if (ht->bitmap)
335  {
336  free_and_init (ht->bitmap);
337  }
338  free_and_init (ht);
339  return;
340 }

343 /*
344  * fh_get - Find the entry in hash table whose key is the value of the
345  * given key and return the data associated with the key.
346  * return: NO_ERROR or error code
347  * ht(in): Hash table
348  * key(in): Hashing key
349  * data(out): pointer to data or NULL if not found
350  */
351 int
352 fh_get (FH_TABLE * ht, FH_KEY key, FH_DATA * data)
353 {
354  int hash;
355  FH_FILE_POS pos;
356  int page_no;
357  int entry_no;
358  FH_ENTRY *entry;
359  FH_PAGE_HDR *pg_hdr;
360  char *ptr;
361 
362  /*
363  * Hash the key and make sure that the return value is between 0 and size
364  * of hash table
365  */
366  hash = (*ht->hfun) (key, ht->size);
367  if (hash < 0)
368  {
369  hash = -hash;
370  }
371  if (hash >= ht->size)
372  {
373  hash = hash % ht->size;
374  }
375 
376  /* Determine the page & entry in the hash file */
377  pos = hash;
378 
379  while (pos != INVALID_FILE_POS)
380  {
381  page_no = pos / ht->entries_per_page;
382  entry_no = pos % ht->entries_per_page;
383 
384  /* fetch the page */
385  if ((pg_hdr = fh_fetch_page (ht, page_no)) == NULL)
386  {
387  *data = NULL;
388  return ER_GENERIC_ERROR;
389  }
390 
391  /* Determine if the entry has the key */
392  ptr = (char *) pg_hdr->fh_entries;
393  ptr += entry_no * ht->entry_size;
394  entry = (FH_ENTRY *) ptr;
395  switch (ht->key_type)
396  {
397  case FH_OID_KEY:
398  /* If the entry is null, the key doesn't exist */
399  if ((*ht->cmpfun) (&entry->info.fh_oidk_key, &null_oid))
400  {
401  *data = NULL;
402  return ER_GENERIC_ERROR;
403  }
404  if ((*ht->cmpfun) (&entry->info.fh_oidk_key, key))
405  {
406  *data = &entry->info.fh_oidk_data;
407  return NO_ERROR;
408  }
409  break;
410  case FH_INT_KEY:
411  /* If the entry is null, the key doesn't exist */
412  if ((*ht->cmpfun) (&entry->info.fh_intk_key, &null_int))
413  {
414  *data = NULL;
415  return ER_GENERIC_ERROR;
416  }
417  if ((*ht->cmpfun) (&entry->info.fh_intk_key, key))
418  {
419  *data = &entry->info.fh_intk_data;
420  return NO_ERROR;
421  }
422  break;
423  default:
424  *data = NULL;
425  return ER_GENERIC_ERROR;
426  }
427  pos = entry->next;
428  }
429  *data = NULL;
430  return ER_GENERIC_ERROR;
431 }

434 /*
435  * fh_put - insert an entry into a hash table
436  * return: NO_ERROR or ER_GENERIC_ERROR
437  * ht(in/out): Hash table
438  * key(in): Hashing key
439  * data(in): Data associated with hashing key
440  * Note:
441  * If the key already exists, the new entry replaces the old one.
442  *
443  * The key and data are copied.
444  * The user must be aware, that changes to the key and value are
445  * not reflected in the hash table.
446  */
447 int
448 fh_put (FH_TABLE * ht, FH_KEY key, FH_DATA data)
449 {
450  int hash;
451  FH_FILE_POS pos;
452  int page_no;
453  int entry_no;
454  FH_ENTRY *entry = NULL;
455  FH_PAGE_HDR *pg_hdr;
456  char *ptr;
457 
458  /*
459  * Hash the key and make sure that the return value is between 0 and size
460  * of hash table
461  */
462  hash = (*ht->hfun) (key, ht->size);
463  if (hash < 0)
464  {
465  hash = -hash;
466  }
467  if (hash >= ht->size)
468  {
469  hash = hash % ht->size;
470  }
471 
472  /* Determine the page & entry in the hash file */
473  pos = hash;
474 
475  do
476  {
477  page_no = pos / ht->entries_per_page;
478  entry_no = pos % ht->entries_per_page;
479 
480  /* fetch the page */
481  if ((pg_hdr = fh_fetch_page (ht, page_no)) == NULL)
482  {
483  return ER_GENERIC_ERROR;
484  }
485 
486  /* Determine if the entry has the key */
487  ptr = (char *) pg_hdr->fh_entries;
488  ptr += entry_no * ht->entry_size;
489  entry = (FH_ENTRY *) ptr;
490  switch (ht->key_type)
491  {
492  case FH_OID_KEY:
493  /* If the entry is null, the key doesn't exist */
494  if ((*ht->cmpfun) (&entry->info.fh_oidk_key, &null_oid))
495  {
496  goto use_entry;
497  }
498  if ((*ht->cmpfun) (&entry->info.fh_oidk_key, key))
499  {
500  memcpy (&entry->info.fh_oidk_data, (char *) data, ht->data_size);
501  return NO_ERROR;
502  }
503  break;
504  case FH_INT_KEY:
505  if ((*ht->cmpfun) (&entry->info.fh_intk_key, &null_int))
506  {
507  goto use_entry;
508  }
509  if ((*ht->cmpfun) (&entry->info.fh_intk_key, key))
510  {
511  memcpy (&entry->info.fh_intk_data, (char *) data, ht->data_size);
512  return NO_ERROR;
513  }
514  break;
515  default:
516  return ER_GENERIC_ERROR;
517  }
518  pos = entry->next;
519  }
520  while (pos != INVALID_FILE_POS);
521 
522  /* This is a new entry, and the last entry examined is not available */
523  entry->next = ht->overflow;
524  /* fetch the overflow page */
525  page_no = ht->overflow / ht->entries_per_page;
526  if ((pg_hdr = fh_fetch_page (ht, page_no)) == NULL)
527  {
528  return ER_GENERIC_ERROR;
529  }
530  entry_no = ht->overflow % ht->entries_per_page;
531  ptr = (char *) pg_hdr->fh_entries;
532  ptr += entry_no * ht->entry_size;
533  entry = (FH_ENTRY *) ptr;
534  ++ht->overflow;
535  ++ht->ncollisions;
536 
537  /* This is a new entry, and the last entry examined is available */
538 use_entry:
539  ++ht->nentries;
540  switch (ht->key_type)
541  {
542  case FH_OID_KEY:
543  entry->info.fh_oidk_key.volid = ((OID *) key)->volid;
544  entry->info.fh_oidk_key.pageid = ((OID *) key)->pageid;
545  entry->info.fh_oidk_key.slotid = ((OID *) key)->slotid;
546  memcpy (&entry->info.fh_oidk_data, (char *) data, ht->data_size);
547  break;
548  case FH_INT_KEY:
549  entry->info.fh_intk_key = *(int *) key;
550  memcpy (&entry->info.fh_intk_data, (char *) data, ht->data_size);
551  break;
552  default:
553  return ER_GENERIC_ERROR;
554  }
555  entry->next = INVALID_FILE_POS;
556  return NO_ERROR;
557 }

560 /*
561  * fh_fetch_page - fetch a page from the hash file to the cache
562  * return: page header pointer if the page exists, or if there is enough
563  * disk space to create a new page. otherwise, it returns NULL.
564  * ht(in/out): Hash table
565  * page(in): Page desired
566  *
567  * Note:
568  * Determine if the page is already in the cache. If so, then
569  * return its page header. Otherwise, select the least recently
570  * used page, write it the the hash file and fetch the desired
571  * page. The page found or fetched becomes the most recently
572  * used page.
573  */
574 static FH_PAGE_HDR *
575 fh_fetch_page (FH_TABLE * ht, int page)
576 {
577  FH_PAGE_HDR *pg_hdr;
578 
579  /* Search for the page in the cache */
580  for (pg_hdr = ht->pg_hdr; pg_hdr; pg_hdr = pg_hdr->next)
581  {
582  /* The page is in the cache. */
583  if (pg_hdr->page == (unsigned int) page)
584  {
585  break;
586  }
587  }
588 
589  /* If the page is not in the cache, get it from the hash file */
590  if (pg_hdr == NULL)
591  {
592  if ((pg_hdr = fh_read_page (ht, page)) == NULL)
593  {
594  return NULL;
595  }
596  }
597 
598  /*
599  * If the page is the least recently used, make the previous page
600  * the least recently used.
601  */
602  if ((ht->pg_hdr_last == pg_hdr) && (ht->cached_pages > 1))
603  {
604  ht->pg_hdr_last = pg_hdr->prev;
605  }
606  /*
607  * If the page is the most recently used, do nothing.
608  * Otherwise, make the page the most recently used.
609  */
610  if (ht->pg_hdr != pg_hdr)
611  {
612  pg_hdr->prev->next = pg_hdr->next;
613  if (pg_hdr->next)
614  {
615  pg_hdr->next->prev = pg_hdr->prev;
616  }
617  pg_hdr->next = ht->pg_hdr;
618  ht->pg_hdr->prev = pg_hdr;
619  pg_hdr->prev = NULL;
620  ht->pg_hdr = pg_hdr;
621  }
622  return pg_hdr;
623 }

626 /*
627  * fh_read_page - read a page from the hash file to the cache
628  * return: page header pointer if the page exists, or if there is enough
629  * disk space to create a new page. otherwise, it returns NULL.
630  * ht(in/out): Hash table
631  * page(in): Page desired
632  *
633  * Note:
634  * Determine if a free page is in the cache. If so, use it.
635  * Otherwise, write the least recently used page to the hash
636  * file and use its page slot to read in the desired page.
637  * If the desired page has never been written, initialize it to
638  * contain all unused entries.
639  */
640 static FH_PAGE_HDR *
641 fh_read_page (FH_TABLE * ht, int page)
642 {
643  FH_PAGE_HDR *pg_hdr;
644 #if defined(SOLARIS) || defined(AIX) || (defined(HPUX) && (_LFS64_LARGEFILE == 1))
645  off64_t offset;
646 #elif defined(I386) && defined(LINUX)
647  __off64_t offset;
648 #elif defined(WINDOWS)
649  __int64 offset;
650 #else
651  int offset;
652 #endif
653  int n;
654 
655  /*
656  * If a free page exists, use it. Since pages are not freed, free
657  * pages only exist after initialization until they are used. Thus
658  * the next page after the first free page is the next free page.
659  */
660  if (ht->pg_hdr_free)
661  {
662  pg_hdr = ht->pg_hdr_free;
663  ht->pg_hdr_free = ht->pg_hdr_free->next;
664  }
665  /* Use the least recently page slot. Write out the page first. */
666  else
667  {
668  if (fh_write_page (ht, ht->pg_hdr_last) == NULL)
669  {
670  return NULL;
671  }
672  pg_hdr = ht->pg_hdr_last;
673  }
674 
675  /* If the page has never been written, initialize a new page */
676  if (fh_bittest (ht, page) == 0)
677  {
678  FH_ENTRY *entry;
679  int i;
680  char *ptr;
681 
682  for (i = 0, entry = pg_hdr->fh_entries; i < ht->entries_per_page; ++i)
683  {
684  switch (ht->key_type)
685  {
686  case FH_OID_KEY:
687  COPY_OID (&entry->info.fh_oidk_key, &null_oid);
688  break;
689  case FH_INT_KEY:
690  entry->info.fh_intk_key = null_int;
691  break;
692  default:
693  return NULL;
694  }
695  entry->next = INVALID_FILE_POS;
696  ptr = (char *) entry;
697  ptr += ht->entry_size;
698  entry = (FH_ENTRY *) ptr;
699  }

788  int bit;

796 /*
797  * fh_bittest - test a bit corresponding to a page in the hash file page bitmap
798  * return: 1/0
799  * ht(in): Hash table
800  * page(in): Page to test for
801  */
802 static int
803 fh_bittest (FH_TABLE * ht, int page)
804 {
805  int byte;
806  int bit;
807 
808  byte = page / 8;
809  if (byte + 1 > ht->bitmap_size)
810  {
811  ht->bitmap = (char *) realloc (ht->bitmap, byte + 1);
812  if (ht->bitmap == NULL)
813  {
814  perror ("SYSTEM ERROR");
815  exit (1);
816  }
817  memset (&ht->bitmap[ht->bitmap_size], 0, byte + 1 - ht->bitmap_size);
818  ht->bitmap_size = byte + 1;
819  return 0;
820  }
821  bit = page % 8;
822  return ht->bitmap[byte] & 1 << bit;
823 }
