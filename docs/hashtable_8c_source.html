123 int
124 hashtable_insert(hashtable *table, void *key, void *dat, int overwrite)
125 {
126  unsigned hash;
127  hash_node *node, *add;
128 
129  ASSERT(table != NULL);
130 
131  hash = table->key_hash(key);
132 
133  for (node = table->table[hash % table->size]; node; node = node->next)
134  if (hash == node->hash && table->key_cmp(key, node->key) == 0) {
135  if (overwrite == 0)
136  return 1;
137  if (table->key_del)
138  table->key_del(node->key);
139  if (table->dat_del)
140  table->dat_del(node->dat);
141  node->key = key;
142  node->dat = dat;
143  return 0;
144  }
145 
146  if ((add = MALLOC(sizeof(*add))) == NULL)
147  return -1;
148  add->key = key;
149  add->dat = dat;
150  add->hash = hash;
151  add->prev = NULL;
152 
153  hash %= table->size;
154  add->next = table->table[hash];
155  if (table->table[hash])
156  table->table[hash]->prev = add;
157  table->table[hash] = add;
158  table->count++;
159 
160  return 0;
161 }

163 int
164 hashtable_probe(hashtable *table, void *key, void **dat)
165 {
166  unsigned hash, mhash;
167  hash_node *node, *prev, *add;
168  void *tmp;
169 
170  ASSERT(table != NULL);
171  ASSERT(dat != NULL);
172 
173  hash = table->key_hash(key);
174  mhash = hash % table->size;
175 
176  prev = NULL;
177  node = table->table[mhash];
178  for (; node; prev = node, node = node->next)
179  if (hash == node->hash && table->key_cmp(key, node->key) == 0)
180  break;
181  if (node) {
182  if (prev) { /* Tranpose. */
183  SWAP(prev->key, node->key, tmp);
184  SWAP(prev->dat, node->dat, tmp);
185  SWAP(prev->hash, node->hash, hash);
186  node = prev;
187  }
188  *dat = node->dat;
189  return 0;
190  }
191 
192  if ((add = MALLOC(sizeof(*add))) == NULL)
193  return -1;
194  add->key = key;
195  add->dat = *dat;
196  add->hash = hash;
197  add->prev = NULL;
198 
199  add->next = table->table[mhash];
200  if (table->table[mhash])
201  table->table[mhash]->prev = add;
202  table->table[mhash] = add;
203  table->count++;
204  return 1;
205 }

207 void *
208 hashtable_search(hashtable *table, const void *key)
209 {
210  unsigned hash;
211  hash_node *node, *prev;
212  void *tmp;
213 
214  ASSERT(table != NULL);
215 
216  hash = table->key_hash(key);
217  prev = NULL;
218  node = table->table[hash % table->size];
219  for (; node; prev = node, node = node->next)
220  if (hash == node->hash && table->key_cmp(key, node->key) == 0)
221  break;
222  if (node) {
223  if (prev) {
224  /*
225  * Tranpose. This typically offers better performance than move-to-
226  * front, but requires a fairly large number of accesses to
227  * take a randomly ordered chain and re-arrange it to nearly
228  * optimal. According to [Gonnet 1984] it may take Big-Omega(n^2)
229  * to come within 1+epsilon of the final state.
230  */
231  SWAP(prev->key, node->key, tmp);
232  SWAP(prev->dat, node->dat, tmp);
233  SWAP(prev->hash, node->hash, hash);
234  node = prev;
235  }
236  /* Node was already at front of list. */
237  return node->dat;
238  }
239  return NULL;
240 }

254 int
255 hashtable_remove(hashtable *table, const void *key, int del)
256 {
257  hash_node *node, *prev;
258  unsigned hash, mhash;
259 
260  ASSERT(table != NULL);
261 
262  hash = table->key_hash(key);
263  mhash = hash % table->size;
264 
265  prev = NULL;
266  node = table->table[mhash];
267  for (; node; prev = node, node = node->next)
268  if (hash == node->hash && table->key_cmp(key, node->key) == 0)
269  break;
270  if (node == NULL)
271  return -1;
272 
273  if (prev)
274  prev->next = node->next;
275  else
276  table->table[mhash] = node->next;
277 
278  if (node->next)
279  node->next->prev = prev;
280 
281  if (del) {
282  if (table->key_del)
283  table->key_del(node->key);
284  if (table->dat_del)
285  table->dat_del(node->dat);
286  }

292 void
293 hashtable_empty(hashtable *table, int del)
294 {
295  hash_node *node, *next;
296  hash_node **tbl;
297  unsigned slot, size;
298 
299  ASSERT(table != NULL);
300 
301  tbl = table->table;
302  size = table->size;
303 
304  for (slot = 0; slot < size; slot++) {
305  if ((node = tbl[slot]) != NULL) {
306  tbl[slot] = NULL;
307  do {
308  next = node->next;
309  if (del) {
310  if (table->key_del)
311  table->key_del(node->key);
312  if (table->dat_del)
313  table->dat_del(node->dat);
314  }
315  FREE(node);
316  } while ((node = next) != NULL);
317  }
318  }
319 
320  table->count = 0;
321 }

368 int
369 hashtable_resize(hashtable *table, unsigned size)
370 {
371  hash_node **ntable;
372  hash_node *node, *next;
373  unsigned i, hash;
374 
375  ASSERT(table != NULL);
376  ASSERT(size != 0);
377 
378  if (table->size == size)
379  return 0;
380 
381  if ((ntable = MALLOC(size * sizeof(hash_node *))) == NULL)
382  return -1;
383 
384  for (i = 0; i < size; i++)
385  ntable[i] = NULL;
386 
387  /*
388  * This way of resizing completely reverses(!) the effects of the trans-
389  * positions that we have been doing in probes and lookups. Hopefully
390  * resizing the hashtable is something that is done rarely or not at all,
391  * so this won't make too much difference.
392  */
393  for (i = 0; i < table->size; i++) {
394  for (node = table->table[i]; node; node = next) {
395  next = node->next;
396  hash = node->hash % size;
397  node->next = ntable[hash];
398  node->prev = NULL;
399  if (ntable[hash])
400  ntable[hash]->prev = node;
401  ntable[hash] = node;
402  }
403  }
404 
405  FREE(table->table);
406  table->table = ntable;
407  table->size = size;
408 
409  return 0;
410 }
