179 int
180 hb_tree_insert(hb_tree *tree, void *key, void *dat, int overwrite)
181 {
182  int rv = 0;
183  hb_node *node, *parent = NULL, *q = NULL;
184 
185  ASSERT(tree != NULL);
186 
187  node = tree->root;
188  while (node) {
189  rv = tree->key_cmp(key, node->key);
190  if (rv < 0)
191  parent = node, node = node->llink;
192  else if (rv > 0)
193  parent = node, node = node->rlink;
194  else {
195  if (overwrite == 0)
196  return 1;
197  if (tree->key_del)
198  tree->key_del(node->key);
199  if (tree->dat_del)
200  tree->dat_del(node->dat);
201  node->key = key;
202  node->dat = dat;
203  return 0;
204  }
205  if (parent->bal)
206  q = parent;
207  }
208 
209  if ((node = node_new(key, dat)) == NULL)
210  return -1;
211  if ((node->parent = parent) == NULL) {
212  tree->root = node;
213  ASSERT(tree->count == 0);
214  tree->count = 1;
215  return 0;
216  }
217  if (rv < 0)
218  parent->llink = node;
219  else
220  parent->rlink = node;
221 
222  while (parent != q) {
223  parent->bal = (parent->rlink == node) * 2 - 1;
224  node = parent;
225  parent = node->parent;
226  }
227  if (q) {
228  if (q->llink == node) {
229  if (--q->bal == -2) {
230  if (q->llink->bal > 0)
231  rot_left(tree, q->llink);
232  rot_right(tree, q);
233  }
234  } else {
235  if (++q->bal == +2) {
236  if (q->rlink->bal < 0)
237  rot_right(tree, q->rlink);
238  rot_left(tree, q);
239  }
240  }
241  }
242  tree->count++;
243  return 0;
244 }

315 int
316 hb_tree_remove(hb_tree *tree, const void *key, int del)
317 {
318  int rv, left;
319  hb_node *node, *out, *parent = NULL;
320  void *tmp;
321 
322  ASSERT(tree != NULL);
323 
324  node = tree->root;
325  while (node) {
326  rv = tree->key_cmp(key, node->key);
327  if (rv == 0)
328  break;
329  parent = node;
330  node = rv < 0 ? node->llink : node->rlink;
331  }
332  if (node == NULL)
333  return -1;
334 
335  if (node->llink && node->rlink) {
336  for (out = node->rlink; out->llink; out = out->llink)
337  /* void */;
338  SWAP(node->key, out->key, tmp);
339  SWAP(node->dat, out->dat, tmp);
340  node = out;
341  parent = out->parent;
342  }
343 
344  out = node->llink ? node->llink : node->rlink;
345  FREE_NODE(node);
346  if (out)
347  out->parent = parent;
348  if (parent == NULL) {
349  tree->root = out;
350  tree->count--;
351  return 0;
352  }
353 
354  left = parent->llink == node;
355  if (left)
356  parent->llink = out;
357  else
358  parent->rlink = out;
359 
360  for (;;) {
361  if (left) {
362  if (++parent->bal == 0) {
363  node = parent;
364  goto higher;
365  }
366  if (parent->bal == +2) {
367  ASSERT(parent->rlink != NULL);
368  if (parent->rlink->bal < 0) {
369  rot_right(tree, parent->rlink);
370  rot_left(tree, parent);
371  } else {
372  ASSERT(parent->rlink->rlink != NULL);
373  if (rot_left(tree, parent) == 0)
374  break;
375  }
376  } else {
377  break;
378  }
379  } else {
380  if (--parent->bal == 0) {
381  node = parent;
382  goto higher;
383  }
384  if (parent->bal == -2) {
385  ASSERT(parent->llink != NULL);
386  if (parent->llink->bal > 0) {
387  rot_left(tree, parent->llink);
388  rot_right(tree, parent);
389  } else {
390  ASSERT(parent->llink->llink != NULL);
391  if (rot_right(tree, parent) == 0)
392  break;
393  }
394  } else {
395  break;
396  }
397  }
398 
399  /* Only get here on double rotations or single rotations that changed
400  * subtree height - in either event, `parent->parent' is positioned
401  * where `parent' was positioned before any rotations. */
402  node = parent->parent;
403 higher:
404  if ((parent = node->parent) == NULL)
405  break;
406  left = parent->llink == node;
407  }
408  tree->count--;
409  return 0;
410 }

604 /*
605  * rot_left(T, B):
606  *
607  * / /
608  * B D
609  * / \ / \
610  * A D ==> B E
611  * / \ / \
612  * C E A C
613  *
614  */
615 static int
616 rot_left(hb_tree *tree, hb_node *node)
617 {
618  int hc;
619  hb_node *rlink, *parent;
620 
621  ASSERT(tree != NULL);
622  ASSERT(node != NULL);
623  ASSERT(node->rlink != NULL);
624 
625  rlink = node->rlink;
626  node->rlink = rlink->llink;
627  if (rlink->llink)
628  rlink->llink->parent = node;
629  parent = node->parent;
630  rlink->parent = parent;
631  if (parent) {
632  if (parent->llink == node)
633  parent->llink = rlink;
634  else
635  parent->rlink = rlink;
636  } else {
637  tree->root = rlink;
638  }
639  rlink->llink = node;
640  node->parent = rlink;
641 
642  hc = rlink->bal != 0;
643  node->bal -= 1 + MAX(rlink->bal, 0);
644  rlink->bal -= 1 - MIN(node->bal, 0);
645  return hc;
646 }
647 
648 /*
649  * rot_right(T, D):
650  *
651  * / /
652  * D B
653  * / \ / \
654  * B E ==> A D
655  * / \ / \
656  * A C C E
657  *
658  */
659 static int
660 rot_right(hb_tree *tree, hb_node *node)
661 {
662  int hc;
663  hb_node *llink, *parent;
664 
665  ASSERT(tree != NULL);
666  ASSERT(node != NULL);
667  ASSERT(node->llink != NULL);
668 
669  llink = node->llink;
670  node->llink = llink->rlink;
671  if (llink->rlink)
672  llink->rlink->parent = node;
673  parent = node->parent;
674  llink->parent = parent;
675  if (parent) {
676  if (parent->llink == node)
677  parent->llink = llink;
678  else
679  parent->rlink = llink;
680  } else {
681  tree->root = llink;
682  }
683  llink->rlink = node;
684  node->parent = llink;
685 
686  hc = llink->bal != 0;
687  node->bal += 1 - MIN(llink->bal, 0);
688  llink->bal += 1 + MAX(node->bal, 0);
689  return hc;
690 }
