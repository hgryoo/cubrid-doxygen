303 /* HEAP_PAGE_VACUUM_STATUS -
304  * Heap page attribute used to predict when page is no longer going to need
305  * another vacuum actions. This allows page deallocations without risking
306  * future access from vacuum workers.
307  *
308  * There are three possible states.
309  *
310  * Initial state is HEAP_PAGE_VACUUM_NONE (empty page). This state can be
311  * reached again after vacuuming in state VACUUM_PAGE_VACUUM_ONCE.
312  *
313  * After one MVCC op, state becomes HEAP_PAGE_VACUUM_ONCE.
314  *
315  * After a second MVCC op without vacuum, state becomes
316  * HEAP_PAGE_VACUUM_UNKNOWN, since future access from vacuum workers becomes
317  * unpredictable. Trying to keep track of required number vacuum accesses is
318  * not trivial and requires too much effort to handle cases that are not
319  * necessarily common.
320  *
321  * HEAP_PAGE_VACUUM_UNKNOWN can be changed only one way: When heap page max
322  * MVCCID is older than vacuum data oldest MVCCID, it means all vacuum
323  * required for that page was executed. Next MVCC op will convert the state
324  * to HEAP_PAGE_VACUUM_ONCE.
325  */
326 typedef enum
327 {
328  HEAP_PAGE_VACUUM_NONE, /* Heap page is completely vacuumed. */
329  HEAP_PAGE_VACUUM_ONCE, /* Heap page requires one vacuum action. */
330  HEAP_PAGE_VACUUM_UNKNOWN /* Heap page requires an unknown number of vacuum actions. */
331 } HEAP_PAGE_VACUUM_STATUS;
