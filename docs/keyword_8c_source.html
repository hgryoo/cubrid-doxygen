525 static KEYWORD_RECORD *pt_find_keyword (const char *text);
526 static int keyword_cmp (const void *k1, const void *k2);
527 
528 
529 static int
530 keyword_cmp (const void *k1, const void *k2)
531 {
532  return strcmp (((KEYWORD_RECORD *) k1)->keyword, ((KEYWORD_RECORD *) k2)->keyword);
533 }
534 
535 /*
536  * pt_find_keyword () -
537  * return: keyword record corresponding to keyword text
538  * text(in): text to test
539  */
540 static KEYWORD_RECORD *
541 pt_find_keyword (const char *text)
542 {
543  static bool keyword_sorted = false;
544  KEYWORD_RECORD *result_key;
545  KEYWORD_RECORD dummy;
546 
547  if (keyword_sorted == false)
548  {
549  qsort (keywords, (sizeof (keywords) / sizeof (keywords[0])), sizeof (keywords[0]), keyword_cmp);
550  keyword_sorted = true;
551  }
552 
553  if (!text)
554  {
555  return NULL;
556  }
557 
558  if (strlen (text) >= MAX_KEYWORD_SIZE)
559  {
560  return NULL;
561  }
562 
563  intl_identifier_upper (text, dummy.keyword);
564 
565  result_key =
566  (KEYWORD_RECORD *) bsearch (&dummy, keywords, (sizeof (keywords) / sizeof (keywords[0])), sizeof (KEYWORD_RECORD),
567  keyword_cmp);
568 
569  return result_key;
