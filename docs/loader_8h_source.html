37 /*
38  * These are the "types" of strings that the lexer recognizes. The
39  * loader can specialize on each type.
40  * These values are used to set up a vector of type setting functions, based
41  * on information about each attribute parsed in the %CLASS line.
42  * The setter functions are invoked using the enumerated type as an index into
43  * the function vector. This gives us a significant saving when processing
44  * values in the instance line, over the previous loader.
45  */
46

55  LDR_FLOAT, /* Reals specified with C 'f' or 'F' notation */
56  LDR_OID, /* Object references */
