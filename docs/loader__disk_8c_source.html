62 /*
63  * alloc_recdes - allocates a record descriptor (for storing objects)
64  * and initializes the fields.
65  * return: record
66  * length(in): desired length
67  */
68 static RECDES *
69 alloc_recdes (int length)
70 {
71  RECDES *rec;
72 
73  rec = (RECDES *) malloc (sizeof (RECDES) + length);
74  if (rec != NULL)
75  {
76  rec->area_size = length;
77  rec->length = 0;
78  rec->type = 0;
79  rec->data = ((char *) rec) + sizeof (RECDES);
80  }
81  else
82  {
83  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_LDR_MEMORY_ERROR, 0);
84  }
85  return (rec);
86 }

135 /*
136  * estimate_object_size - estimate the size of object
137  * return: estimated size of given object, negative value if object has
138  * variable length attribute
139  * class(in): class structure to examine
140  */
141 static int
142 estimate_object_size (SM_CLASS * class_, int *offset_size_ptr)
143 {
144  SM_ATTRIBUTE *att;
145  int size, exact, bits;
146 
147  *offset_size_ptr = OR_BYTE_SIZE; /* 1byte */
148 
149  exact = 1;
150 re_check:
151  size = OR_NON_MVCC_HEADER_SIZE;
152  if (class_ != NULL)
153  {
154  size += class_->fixed_size + OR_VAR_TABLE_SIZE_INTERNAL (class_->variable_count, *offset_size_ptr);
155  for (att = class_->attributes; att != NULL; att = (SM_ATTRIBUTE *) att->header.next)
156  {
157  if (att->type->id == DB_TYPE_STRING)
158  {
159  if (att->domain->precision <= 0)
160  {
161  exact = 0; /* have to guess */
162  }
163  else
164  {
165  size += att->domain->precision;
166  bits = att->domain->precision & 3;
167  if (bits)
168  {
169  size += 4 - bits;
170  }
171  }
172  }
173  else if (att->type->variable_p)
174  {
175  exact = 0;
176  }
177  }
178  }
179 
180  /*
181  * if we couldn't make an exact estimate, return a negative number indicating
182  * the minimum size to expect
183  */
184  if (*offset_size_ptr == OR_BYTE_SIZE && size > OR_MAX_BYTE)
185  {
186  *offset_size_ptr = OR_SHORT_SIZE;
187  goto re_check;
188  }
189  if (*offset_size_ptr == OR_SHORT_SIZE && size > OR_MAX_SHORT)
190  {
191  *offset_size_ptr = BIG_VAR_OFFSET_SIZE;
192  goto re_check;
193  }
194 
195  if (!exact)
196  size = -size;
197 
198  return (size);
199 }

202 /*
203  * get_class_heap - This locates or creates a heap file for a class.
204  * return: heap file of the class
205  * classop(in): class object
206  * class(in): class structure
207  */
208 static HFID *
209 get_class_heap (MOP classop, SM_CLASS * class_)
210 {
211  HFID *hfid;
212  OID *class_oid;
213 
214  hfid = sm_ch_heap ((MOBJ) class_);
215  if (HFID_IS_NULL (hfid))
216  {
217  /* could also accomplish this by creating a single instance */
218  /*
219  * make sure the class is fetched for update so that it will be
220  * marked dirty and stored with the new heap
221  */
222  if (au_fetch_class (classop, &class_, AU_FETCH_UPDATE, AU_INSERT) != NO_ERROR)
223  {
224  hfid = NULL;
225  }
226  else
227  {
228  const bool reuse_oid = (class_->flags & SM_CLASSFLAG_REUSE_OID) ? true : false;
229 
230  class_oid = ws_oid (classop);
231  if (OID_ISTEMP (class_oid))
232  { /* not defined function */
233  class_oid = locator_assign_permanent_oid (classop);
234  }
235  if (xheap_create (NULL, hfid, class_oid, reuse_oid) != NO_ERROR)
236  {
237  hfid = NULL;
238  }
239  }
240  }
241  return (hfid);
242 }

246  * disk_reserve_instance - This is used to reserve space for an instance that
247  * was referenced in the load file but has not yet been defined.
248  * return: NO_ERROR if successful, error code otherwise
249  * classop(in): class

253 disk_reserve_instance (MOP classop, OID * oid)
254 {

290 /*
291  * update_indexes - Updates the indexes for an instance that is being inserted.
292  * return: NO_ERROR if successful, error code otherwise
293  * class_oid(in): class oid
294  * obj_oid(in): instance oid
295  * rec(in): record with new object
296  * Note:
297  * Updates the indexes for an instance that is being inserted.
298  * This only calls btree_insert because for use in the loader, we will never
299  * be updating existing instances. When the loader encounters a forward
300  * reference to an instance, it will use heap_reserve_address but will
301  * not attempt to store a default instance at that address.
302  * This means that there will never be an existing entry in the btree when
303  * we finally get around to storing the actual object contents. If this
304  * situation ever changes, then we will need to call btree_update here
305  * instead. This will require the current key value which must be
306  * extracted from the hf record being replaced.
307  * See locator_force to see how this can be done.
308  *
309  * Moral, avoid writing the instance record until absolutely necessary,
310  * and then do it only once.
311  */
312 static int
313 update_indexes (OID * class_oid, OID * obj_oid, RECDES * rec)
314 {
315  int error = NO_ERROR;
316 
317  if (locator_add_or_remove_index (NULL, rec, obj_oid, class_oid, true, SINGLE_ROW_INSERT, (HEAP_SCANCACHE *) NULL,
318  /* ejin: for replication... */
319  false, /* 7th arg -> data or schema */
320  false, NULL, /* 8th arg -> make repl. log or not */
321  NULL) != NO_ERROR)
322  {
323  assert (er_errid () != NO_ERROR);
324  error = er_errid ();
325  }
326 
327  return error;
328 }

331 /*
332  * disk_insert_instance - This inserts a new object in the database given a
333  * "descriptor" for the object.
334  * return: NO_ERROR if successful, error code otherwise
335  * classop(in): class object
336  * obj(in): object description
337  * oid(in): oid of the destination
338  */
339 int
340 disk_insert_instance (MOP classop, DESC_OBJ * obj, OID * oid)
341 {
342  int error = NO_ERROR;
343  HFID *hfid;
344  int newsize;
345  bool has_indexes;
346 
347  Diskrec->length = 0;
348  if (desc_obj_to_disk (obj, Diskrec, &has_indexes))
349  {
350  /* make the record larger */
351  newsize = -Diskrec->length + DB_PAGESIZE;
352  free_recdes (Diskrec);
353  Diskrec = alloc_recdes (newsize);
354  /* try one more time */
355  if (Diskrec == NULL || desc_obj_to_disk (obj, Diskrec, &has_indexes) != 0)
356  {
357  error = ER_LDR_CANT_TRANSFORM;
358  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, error, 0);
359  }
360  }
361 
362  if (!error)
363  {
364  hfid = get_class_heap (classop, obj->class_);
365  if (hfid == NULL)
366  {
367  assert (er_errid () != NO_ERROR);
368  error = er_errid ();
369  }
370  else
371  {
372  HEAP_OPERATION_CONTEXT context;
373 
374  /* set up context */
375  heap_create_insert_context (&context, hfid, WS_OID (classop), Diskrec, NULL, false);
376 
377  /* oid is set here as a side effect */
378  if (heap_insert_logical (NULL, &context) != NO_ERROR)
379  {
380  assert (er_errid () != NO_ERROR);
381  error = er_errid ();
382  }
383  else if (has_indexes)
384  {
385  COPY_OID (oid, &context.res_oid);
386  error = update_indexes (WS_OID (classop), oid, Diskrec);
387  }
388  else
389  {
390  COPY_OID (oid, &context.res_oid);
391  }
392  }
393  }
394  return (error);
395 }
