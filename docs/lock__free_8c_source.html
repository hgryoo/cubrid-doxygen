339 {

343  assert (entry->tran_system != NULL);
344 
345  edesc = entry->tran_system->entry_desc;

347  {
348  void *curr = entry->retired_list, *next = NULL;
349  while (curr != NULL)
350  {
351  next = (void *) OF_GET_PTR_DEREF (curr, edesc->of_local_next);
352  if (edesc->f_uninit != NULL)
353  {
354  edesc->f_uninit (curr);
355  }
356  edesc->f_free (curr);
357  curr = next;
358  }
359 
360  entry->retired_list = NULL;
361  }

563 /*
564  * lf_stack_pop () - pop an entry off a lock free stack
565  * returns: entry or NULL on empty stack
566  * top(in/out): top of stack
567  * edesc(in): descriptor for entry
568  *
569  * NOTE: This function is vulnerable to an ABA problem in the following
570  * scenario:
571  * (1) Thread_1 executes POP and is suspended exactly after MARK
572  * (2) Thread_2 executes: t := POP, POP, PUSH (t)
573  * (3) Thread_1 is continued; CAS will succeed but prev is pointing to the
574  * wrong address
575  */
576 void *
577 lf_stack_pop (void **top, LF_ENTRY_DESCRIPTOR * edesc)
578 {
579  void *rtop = NULL, *prev = NULL;
580  assert (top != NULL && edesc != NULL);
581 
582  while (true)
583  {
584  rtop = *top;
585  if (rtop == NULL)
586  {
587  /* at this time the stack is empty */
588  return NULL;
589  }
590 
591  prev = OF_GET_PTR_DEREF (rtop, edesc->of_local_next); /* MARK */
592  if (ATOMIC_CAS_ADDR (top, rtop, prev))
593  {
594  /* clear link */
595  OF_GET_PTR_DEREF (rtop, edesc->of_local_next) = NULL;
596 
597  /* return success */
598  return rtop;
599  }
600  }
601 
602  /* impossible case */
603  assert (false);
604  return NULL;
605 }

607 /*
608  * lf_freelist_alloc_block () - allocate a block of entries in the freelist
609  * returns: error code or NO_ERROR
610  * freelist(in): freelist to allocate to
611  */
612 static int
613 lf_freelist_alloc_block (LF_FREELIST * freelist)
614 {
615  void *head = NULL, *tail = NULL, *new_entry = NULL, *top = NULL;
616  LF_ENTRY_DESCRIPTOR *edesc;
617  int i;
618 
619  assert (freelist != NULL && freelist->entry_desc != NULL);
620  edesc = freelist->entry_desc;
621 
622  /* allocate a block */
623  for (i = 0; i < freelist->block_size; i++)
624  {
625  new_entry = edesc->f_alloc ();
626  if (new_entry == NULL)
627  {
628  /* we use a decoy size since we don't know it */
629  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, (size_t) 1);
630  return ER_OUT_OF_VIRTUAL_MEMORY;
631  }
632 
633  /* of_prev of new entry points to tail; new entry becomes tail */
634  OF_GET_PTR_DEREF (new_entry, edesc->of_local_next) = tail;
635  tail = new_entry;
636 
637  /* store first entry as head */
638  if (head == NULL)
639  {
640  head = new_entry;
641  }
642  }
643 
644  /* append block to freelist */
645  do
646  {
647  top = VOLATILE_ACCESS (freelist->available, void *);
648  OF_GET_PTR_DEREF (head, edesc->of_local_next) = top;
649  }
650  while (!ATOMIC_CAS_ADDR (&freelist->available, top, tail));
651 
652  /* increment allocated count */
653  ATOMIC_INC_32 (&freelist->alloc_cnt, freelist->block_size);
654  ATOMIC_INC_32 (&freelist->available_cnt, freelist->block_size);
655 
656  /* operation successful, block appended */
657  return NO_ERROR;
658 }

754 {

757  bool local_tran = false;
758

760  assert (freelist != NULL);
761  assert (freelist->entry_desc != NULL);
762

764

766 
767  /* first check temporary entry */
768  if (tran_entry->temp_entry != NULL)
769  {
770  entry = tran_entry->temp_entry;
771  tran_entry->temp_entry = NULL;
772  OF_GET_PTR_DEREF (entry, edesc->of_next) = NULL;
773 
774  LF_UNITTEST_INC (&lf_claims_temp, 1);
775  LF_UNITTEST_INC (&lf_temps, -1);
776  return entry;
777  }
778 
779  /* We need a transaction. Careful: we cannot increment transaction ID! */
780  if (tran_entry->transaction_id == LF_NULL_TRANSACTION_ID)
781  {
782  local_tran = true;
783  lf_tran_start_with_mb (tran_entry, false);
784  }
785

787  if (LF_TRAN_CLEANUP_NECESSARY (tran_entry))

790  {
791  if (local_tran)
792  {
793  lf_tran_end_with_mb (tran_entry);
794  }
795  return NULL;
796  }
797  }

814  {

816  }

818  }

829  }

833  * time; this behavior is acceptable given that the freelist has a _low_ enough value of block_size; it sure

850  assert (false);

852  {

854  }
855  return NULL;
856 }

926  * lf_freelist_transport () - transport local retired entries to freelist
927  * returns: error code or NO_ERROR
928  * tran_entry(in): transaction entry to use
929  * freelist(in): freelist
930  */
931 int
932 lf_freelist_transport (LF_TRAN_ENTRY * tran_entry, LF_FREELIST * freelist)
933 {
934  LF_TRAN_SYSTEM *tran_system;
935  LF_ENTRY_DESCRIPTOR *edesc;
936  UINT64 min_tran_id = 0;
937  int transported_count = 0;
938  void *list = NULL, *list_trailer = NULL;
939  void *aval_first = NULL, *aval_last = NULL;
940  void *old_head;
941 
942  assert (freelist != NULL);
943  assert (tran_entry != NULL);
944  tran_system = tran_entry->tran_system;
945  assert (tran_system != NULL);
946  edesc = freelist->entry_desc;
947  assert (edesc != NULL);
948 
949  /* check if cleanup is actually necessary */
950  min_tran_id = tran_system->min_active_transaction_id;
951  if (min_tran_id <= tran_entry->last_cleanup_id)
952  {
953  /* nothing to do */
954  return NO_ERROR;
955  }
956 
957  /* walk private list and unlink old entries */
958  for (list = tran_entry->retired_list; list != NULL; list = OF_GET_PTR_DEREF (list, edesc->of_local_next))
959  {
960 
961  if (aval_first == NULL)
962  {
963  UINT64 *del_id = (UINT64 *) OF_GET_PTR (list, edesc->of_del_tran_id);
964  assert (del_id != NULL);
965 
966  if (*del_id < min_tran_id)
967  {
968  /* found first reusable entry - since list is ordered by descending transaction id, entries that follow
969  * are also reusable */
970  aval_first = list;
971  aval_last = list;
972 
973  /* uninit the reusable entry */
974  if (edesc->f_uninit != NULL)
975  {
976  edesc->f_uninit (list);
977  }
978 
979  transported_count = 1;
980  }
981  else
982  {
983  /* save trailer */
984  list_trailer = list;
985  }
986  }
987  else
988  {
989  /* continue until we get to tail */
990  aval_last = list;
991 
992  /* uninit the reusable entry */
993  if (edesc->f_uninit != NULL)
994  {
995  edesc->f_uninit (list);
996  }
997 
998  transported_count++;
999  }
1000  }
1001 
1002  if (aval_first != NULL)
1003  {
1004  if (list_trailer != NULL)
1005  {
1006  /* unlink found part of list */
1007  OF_GET_PTR_DEREF (list_trailer, edesc->of_local_next) = NULL;
1008  }
1009  else
1010  {
1011  /* whole list can be reused */
1012  tran_entry->retired_list = NULL;
1013  }
1014 
1015  /* make sure we don't append an unlinked sublist */
1016  MEMORY_BARRIER ();
1017 
1018  /* link part of list to available */
1019  do
1020  {
1021  old_head = VOLATILE_ACCESS (freelist->available, void *);
1022  OF_GET_PTR_DEREF (aval_last, edesc->of_local_next) = old_head;
1023  }
1024  while (!ATOMIC_CAS_ADDR (&freelist->available, old_head, aval_first));
1025 
1026  /* update counters */
1027  ATOMIC_INC_32 (&freelist->available_cnt, transported_count);
1028  ATOMIC_INC_32 (&freelist->retired_cnt, -transported_count);
1029 
1030  LF_UNITTEST_INC (&lf_transports, transported_count);
1031  }
1032 
1033  /* register cleanup */
1034  tran_entry->last_cleanup_id = min_tran_id;
1035 
1036  /* all ok */
1037  return NO_ERROR;
1038 }

1915
