421 /*
422  * log_2pc_make_global_tran_id - Make a global transaction identifier for
423  * 2pc purposes
424  *
425  * return:
426  *
427  * tranid(in): Transaction identifier
428  *
429  * NOTE:Build a global transaction identifier based on the host
430  * identifier, the process identifier and the transaction
431  * identifier.
432  */
433 static int
434 log_2pc_make_global_tran_id (TRANID tranid)
435 {
436  unsigned char *ptr;
437  unsigned int hash;
438  unsigned int value;
439  unsigned int unsig_gtrid;
440  int gtrid;
441  unsigned int i;
442 
443  unsig_gtrid = 0;
444 
445  /* HASH THE HOST IDENTIFIER INTO ONE BYTE */
446  value = css_gethostid ();
447  hash = 0;
448  ptr = (unsigned char *) &value;
449  for (i = 0; i < sizeof (value); i++)
450  {
451  hash = (hash << 5) - hash + *ptr++;
452  }
453  /* Don't use more than one byte */
454  /* set the MSB to zero */
455  unsig_gtrid = unsig_gtrid + ((hash % UCHAR_MAX) & 0x7F);
456 
457  /* HASH THE PROCESS IDENTIFIER INTO ONE BYTE */
458  value = (unsigned int) getpid ();
459  hash = 0;
460  ptr = (unsigned char *) &value;
461  for (i = 0; i < sizeof (value); i++)
462  {
463  hash = (hash << 5) - hash + *ptr++;
464  }
465  /* Don't use more than one byte */
466  unsig_gtrid = (unsig_gtrid << 8) + (hash % UCHAR_MAX);
467 
468 
469  /* FOLD the TRANSACTION IDENTIFIER INTO TWO */
470  hash = *(unsigned short *) &tranid;
471  hash = (hash << 5) - hash + *((unsigned short *) &tranid + 1);
472 
473  /* Don't use more than two byte */
474  unsig_gtrid = (unsig_gtrid << 16) + (hash % SHRT_MAX);
475 
476 
477  /*
478  * Make sure that another 2PC transaction does not have this identifier.
479  * If there is one, subtract one and check again. Note that the identifier
480  * may be duplicated in a remote site, we do not have control about this.
481  */
482 
483  gtrid = (int) unsig_gtrid;
484 
485  value = 1;
486  if (gtrid == LOG_2PC_NULL_GTRID)
487  {
488  gtrid--;
489  }
490 
491  while (log_2pc_check_duplicate_global_tran_id (gtrid))
492  {
493  gtrid--;
494 
495  if (gtrid == LOG_2PC_NULL_GTRID)
496  {
497  gtrid--;
498  }
499  }
500 
501  return (int) gtrid;
502 }

534 /*
535  * log_2pc_commit_first_phase
536  *
537  * return:
538  *
539  * tdes(in):
540  *
541  * Note:
542  */
543 static int
544 log_2pc_commit_first_phase (THREAD_ENTRY * thread_p, LOG_TDES * tdes, LOG_2PC_EXECUTE execute_2pc_type, bool * decision)
545 {
546  int i;
547 
548  /* Start the first phase of 2PC. Prepare to commit or voting phase */
549  if (tdes->state == TRAN_ACTIVE)
550  {
551  /*
552  * Start prepare to commit at coordinator site
553  * Note: that the transient classnames table entries should have
554  * already been taken care of.
555  */
556 
557  /*
558  * Release share locks types and record that the 2PC has started.
559  * NOTE: that log_2pc_append_start flushes the log and change the state
560  * of the transaction 2PC collecting votes
561  */
562 
563  /*
564  * Start the 2PC for this coordinator
565  */
566  log_2pc_append_start (thread_p, tdes);
567 
568  if (execute_2pc_type == LOG_2PC_EXECUTE_FULL)
569  {
570  /*
571  * This is the coordinatoor root
572  */
573  lock_unlock_all_shared_get_all_exclusive (thread_p, NULL);
574  }
575 
576  /* Initialize the Acknowledgement vector to 0 */
577  i = sizeof (int) * tdes->coord->num_particps;
578 
579  tdes->coord->ack_received = (int *) malloc (i);
580  if (tdes->coord->ack_received == NULL)
581  {
582  /* Out of memory */
583  logpb_fatal_error (thread_p, true, ARG_FILE_LINE, "log_2pc_commit");
584  return ER_OUT_OF_VIRTUAL_MEMORY;
585  }
586 
587  for (i = 0; i < tdes->coord->num_particps; i++)
588  {
589  tdes->coord->ack_received[i] = false;
590  }
591 
592  *decision = log_2pc_send_prepare (tdes->gtrid, tdes->coord->num_particps, tdes->coord->block_particps_ids);
593  }
594  else
595  {
596  /*
597  * We are not in the right mode for the prepare to commit phase
598  */
599  *decision = false;
600 
601  return ER_FAILED;
602  }
603 
604  return NO_ERROR;
605 }

613  *
614  * Note:
615  */
616 static TRAN_STATE
617 log_2pc_commit_second_phase (THREAD_ENTRY * thread_p, LOG_TDES * tdes, bool * decision)
618 {
619  TRAN_STATE state;
620 
621  if (*decision == true)
622  {
623  /*
624  * DECISION is COMMIT
625  * The coordinator and all participants of distributed transaction
626  * have agreed to commit the transaction. The commit decision is
627  * forced to the log to find out the decision in the case of a crash.
628  */
629 
630  log_2pc_append_decision (thread_p, tdes, LOG_2PC_COMMIT_DECISION);
631 
632  /*
633  * The transaction has been declared as 2PC commit. We could execute the
634  * LOCAL COMMIT AND THE REMOTE COMMITS IN PARALLEL, however our
635  * communication subsystem does not support asynchronous communication
636  * types. The commitment of the participants is done after the local
637  * commitment is completed.
638  */
639 
640  /* Save the state.. so it can be reverted to the 2pc state .. */
641  state = tdes->state;
642  /* 2PC protocol does not support RETAIN LOCK */
643  (void) log_commit_local (thread_p, tdes, false, false);
644 
645  tdes->state = state; /* Revert to 2PC state... */
646  /*
647  * If the following function fails, the transaction will be dangling
648  * and we need to retry sending the decision at another point.
649  * We have already decided and log the decision in the log file.
650  */
651  (void) log_2pc_send_commit_decision (tdes->gtrid, tdes->coord->num_particps, tdes->coord->ack_received,
652  tdes->coord->block_particps_ids);
653  /* Check if all the acknowledgments have been received */
654  state = log_complete_for_2pc (thread_p, tdes, LOG_COMMIT, LOG_NEED_NEWTRID);
655  }
656  else
657  {
658  /*
659  * DECISION is ABORT
660  * The coordinator and/or some of the participants of distributed
661  * transaction could not agree to commit the transaction. The abort
662  * decision is logged. We do not need to forced since the default is
663  * abort. It does not matter whether this is a root coordinator or not
664  * the current site has decide to abort.
665  */
666 
667  /*
668  * If the transaction is active and there are not acknowledgments
669  * needed, the abort for the distributed transaction was decided
670  * without using the 2PC
671  */
672 
673  if (tdes->state != TRAN_ACTIVE || tdes->coord->ack_received != NULL)
674  {
675  log_2pc_append_decision (thread_p, tdes, LOG_2PC_ABORT_DECISION);
676  }
677 
678  /*
679  * The transaction has been declared as 2PC abort. We could execute the
680  * LOCAL ABORT AND THE REMOTE ABORTS IN PARALLEL, however our
681  * communication subsystem does not support asynchronous communication
682  * types. The abort of the participants is done after the local abort
683  * is completed.
684  */
685 
686  /* Save the state.. so it can be reverted to the 2pc state .. */
687  state = tdes->state;
688  /* 2PC protocol does not support RETAIN LOCK */
689  (void) log_abort_local (thread_p, tdes, false);
690 
691  if (tdes->state == TRAN_UNACTIVE_ABORTED)
692  {
693  tdes->state = state; /* Revert to 2PC state... */
694  }
695  /*
696  * Execute the abort at participants sites at this time.
697  */
698  if (tdes->coord->ack_received)
699  {
700  /*
701  * Current site was also a coordinator site (of course not the root
702  * coordinator). Thus, we need to collect acknowledgments.
703  *
704  * If the following function fails, the transaction will be dangling
705  * and we need to retry sending the decision at another point.
706  * We have already decided and log the decision in the log file.
707  */
708  (void) log_2pc_send_abort_decision (tdes->gtrid, tdes->coord->num_particps, tdes->coord->ack_received,
709  tdes->coord->block_particps_ids, true);
710  }
711  else
712  {
713  /*
714  * Abort was decided without using the 2PC protocol at this site.
715  * That is, the participants are not prepare to commit). Therefore,
716  * there is no need to collect acknowledgments.
717  *
718  * If the following function fails, the transaction will be dangling
719  * and we need to retry sending the decision at another point.
720  * We have already decided and log the decision in the log file.
721  */
722  (void) log_2pc_send_abort_decision (tdes->gtrid, tdes->coord->num_particps, tdes->coord->ack_received,
723  tdes->coord->block_particps_ids, false);
724  }
725  /* Check if all the acknowledgments have been received */
726  state = log_complete_for_2pc (thread_p, tdes, LOG_ABORT, LOG_NEED_NEWTRID);
727  }
728 
729  return state;
730 }

732 /*
733  * log_2pc_commit - Follow two phase commit protocol (coordinator's algorithm)
734  *
735  * return: TRAN_STATE
736  *
737  * tdes(in): State structure of transaction of the log record
738  * execute_2pc_type(in): Phase of two phase commit to execute
739  * decision(in/out): Wheater coordinator and its participants agree to
740  * commit or abort the transaction
741  *
742  */
743 TRAN_STATE
744 log_2pc_commit (THREAD_ENTRY * thread_p, LOG_TDES * tdes, LOG_2PC_EXECUTE execute_2pc_type, bool * decision)
745 {
746  TRAN_STATE state;
747 
748  if (tdes->gtrid == LOG_2PC_NULL_GTRID)
749  {
750  TR_TABLE_CS_ENTER (thread_p);
751  tdes->gtrid = log_2pc_make_global_tran_id (tdes->trid);
752  TR_TABLE_CS_EXIT (thread_p);
753  }
754 
755  /*
756  * PHASE I of 2PC: Guarantee commitment (i.e., coordinator and participants
757  * are Voting)
758  */
759 
760  /* If the transaction is ready to commit, the first phase of the 2PC has already been completed, so skip it here */
761 
762  if (execute_2pc_type == LOG_2PC_EXECUTE_FULL || execute_2pc_type == LOG_2PC_EXECUTE_PREPARE)
763  {
764  if (log_2pc_commit_first_phase (thread_p, tdes, execute_2pc_type, decision) != NO_ERROR)
765  {
766  return tdes->state;
767  }
768  }
769  else
770  {
771  /*
772  * We are currently not executing the first phase of 2PC. The decsion is
773  * already known
774  */
775  if (execute_2pc_type == LOG_2PC_EXECUTE_COMMIT_DECISION)
776  {
777  *decision = true;
778  }
779  else
780  {
781  *decision = false;
782  }
783  }
784 
785  /*
786  * PHASE II of 2PC: Inform decsion to participants (i.e., either commit or
787  * abort)
788  */
789  if (execute_2pc_type != LOG_2PC_EXECUTE_PREPARE || *decision == false)
790  {
791  state = log_2pc_commit_second_phase (thread_p, tdes, decision);
792  }
793  else
794  {
795  state = tdes->state;
796  }
797 
798  return state;
799 }

1328  * previously aborted, and the coordinator is notified of such

1523  LK_ACQUIRED_LOCKS acq_locks; /* List of acquired locks before the system crash */
