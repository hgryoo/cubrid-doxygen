153 /*
154  * css_master_timeout()
155  * return: 0 if css_Master_timeout time is expired,
156  * otherwise 1
157  *
158  * Note:
159  * This is to handle the case when nothing is happening. We will check the
160  * process status to see if the server is still running. If not, then we
161  * remove our entry.
162  */
163 static int
164 css_master_timeout (void)
165 {
166 #if !defined(WINDOWS)
167  int rv;
168  SOCKET_QUEUE_ENTRY *temp;
169 #endif
170  struct timeval timeout;
171 
172  /* check for timeout */
173  if (css_Master_timeout && time ((time_t *) (&timeout.tv_sec)) != (time_t) (-1)
174  && css_Master_timeout->tv_sec < timeout.tv_sec)
175  {
176  return (0);
177  }
178 
179 #if !defined(WINDOWS)
180  /* On the WINDOWS, it is not clear if we will ever have spawned child processes, at least initially. There don't
181  * appear to be any similarly named "wait" functions in the MSVC runtime library. */
182  rv = pthread_mutex_lock (&css_Master_socket_anchor_lock);
183  for (temp = css_Master_socket_anchor; temp; temp = temp->next)
184  {
185  if (kill (temp->pid, 0) && errno == ESRCH)
186  {
187 #if defined(DEBUG)
188  if (css_Active_server_count > 0)
189  {
190  css_Active_server_count--;
191  }
192 #endif
193 #if !defined(WINDOWS)
194  if (temp->ha_mode)
195  {
196  hb_cleanup_conn_and_start_process (temp->conn_ptr, temp->fd);
197  }
198  else
199 #endif
200  {
201  css_remove_entry_by_conn (temp->conn_ptr, &css_Master_socket_anchor);
202  }
203  break;
204  }
205  }
206  pthread_mutex_unlock (&css_Master_socket_anchor_lock);
207 #endif
208  return (1);
209 }

318 /*
319  * css_accept_new_request() - Accepts a connect request from a new server
320  * return: none
321  * conn(in)
322  * rid(in)
323  * server_name(in)
324  * server_name_length(in)
325  */
326 static void
327 css_accept_new_request (CSS_CONN_ENTRY * conn, unsigned short rid, char *server_name, int server_name_length)
328 {
329  char *datagram;
330  int datagram_length;
331  SOCKET server_fd = INVALID_SOCKET;
332  int length;
333  CSS_CONN_ENTRY *datagram_conn;
334  SOCKET_QUEUE_ENTRY *entry;
335 
336  datagram = NULL;
337  datagram_length = 0;
338  css_accept_server_request (conn, SERVER_REQUEST_ACCEPTED);

340  {
341  if (datagram != NULL && css_tcp_master_datagram (datagram, &server_fd))
342  {
343  datagram_conn = css_make_conn (server_fd);
344 #if defined(DEBUG)
345  css_Active_server_count++;
346 #endif
347  css_add_request_to_socket_queue (datagram_conn, false, server_name, server_fd, READ_WRITE, 0,
348  &css_Master_socket_anchor);
349  length = (int) strlen (server_name) + 1;
350  if (length < server_name_length)
351  {
352  entry = css_return_entry_of_server (server_name, css_Master_socket_anchor);
353  if (entry != NULL)
354  {
355  server_name += length;
356  entry->version_string = (char *) malloc (strlen (server_name) + 1);
357  if (entry->version_string != NULL)
358  {
359  strcpy (entry->version_string, server_name);
360  server_name += strlen (entry->version_string) + 1;
361 
362  entry->env_var = (char *) malloc (strlen (server_name) + 1);
363  if (entry->env_var != NULL)
364  {
365  strcpy (entry->env_var, server_name);
366  }
367 
368  server_name += strlen (server_name) + 1;
369 
370  entry->pid = atoi (server_name);
371  }
372  else
373  {
374  entry->env_var = NULL;
375  }
376  }
377  }
378  }
379  }
380  if (datagram)
381  {
382  free_and_init (datagram);
383  }

436  * conn(in)
437  * rid(in)
438  *
439  * Note: This will allow us to pass fds for future requests to the server.

441 static void
442 css_register_new_server (CSS_CONN_ENTRY * conn, unsigned short rid)
443 {
444  int name_length;
445  char *server_name = NULL;
446  SOCKET_QUEUE_ENTRY *entry;
447 
448  /* read server name */
449  if (css_receive_data (conn, rid, &server_name, &name_length, -1) == NO_ERRORS)
450  {
451  entry = css_return_entry_of_server (server_name, css_Master_socket_anchor);
452  if (entry != NULL)
453  {
454  if (IS_INVALID_SOCKET (entry->fd))
455  {
456  /* accept a server that was auto-started */
457  css_accept_old_request (conn, rid, entry, server_name, name_length);
458  }
459  else
460  {
461  /* reject a server with a duplicate name */
462  css_reject_server_request (conn, SERVER_ALREADY_EXISTS);
463  }
464  }
465  else
466  {
467 #if defined(WINDOWS)
468  css_accept_server_request (conn, SERVER_REQUEST_ACCEPTED);
469 #if defined(DEBUG)
470  css_Active_server_count++;
471 #endif
472  css_add_request_to_socket_queue (conn, false, server_name, conn->fd, READ_WRITE, 0,
473  &css_Master_socket_anchor);
474 
475 #else /* ! WINDOWS */
476  /* accept a request from a new server */
477  css_accept_new_request (conn, rid, server_name, name_length);
478 #endif /* ! WINDOWS */
479  }
480  }
481  if (server_name != NULL)
482  {
483  free_and_init (server_name);
484  }
485 
486 #if !defined(WINDOWS)
487  /* WINDOWS wants to keep this conn--it is the main connection */
488  css_free_conn (conn);
489 #endif /* ! WINDOWS */
490 }

493  * css_register_new_server2()
494  * return: none
495  * conn(in)

510

561  {
562  strcpy (entry->version_string, recv_data);

618 /*
619  * css_send_to_existing_server() - Sends a new client request to
620  * an existing server
621  * return: none
622  * conn(in)
623  * rid(in)
624  *
625  * Note:
626  * There are two ways this can go. First we locate the requsted server on
627  * our connection list and then see if it was registered with an explicit
628  * port id. If not, we attempt to pass the open client fd to the server
629  * using the "old" style connection protocol.
630  *
631  * If the server has instead given us an explicit port id, we tell the
632  * client to use the new style protocol and establish their own connection
633  * to the server.
634  */
635 static void
636 css_send_to_existing_server (CSS_CONN_ENTRY * conn, unsigned short rid, CSS_SERVER_REQUEST request)
637 {
638  SOCKET_QUEUE_ENTRY *temp;
639  char *server_name = NULL;
640  int name_length, buffer;
641 
642  name_length = 1024;
643  if (css_receive_data (conn, rid, &server_name, &name_length, -1) == NO_ERRORS && server_name != NULL)
644  {
645  temp = css_return_entry_of_server (server_name, css_Master_socket_anchor);
646  if (temp != NULL
647 #if !defined(WINDOWS)
648  && (temp->ha_mode == false || hb_is_deactivation_started () == false)
649 #endif /* !WINDOWS */
650  )
651  {
652  if (temp->port_id == -1)
653  {
654  /* use old style connection */
655  if (IS_INVALID_SOCKET (temp->fd))
656  {
657  css_reject_client_request (conn, rid, SERVER_STARTED);
658  free_and_init (server_name);
659  css_free_conn (conn);
660  return;
661  }
662  else
663  {
664 #if !defined(WINDOWS)
665  if (hb_is_hang_process (temp->fd))
666  {
667  css_reject_client_request (conn, rid, SERVER_HANG);
668  free_and_init (server_name);
669  css_free_conn (conn);
670  return;
671  }
672 #endif
673  if (css_send_new_request_to_server (temp->fd, conn->fd, rid, request))
