297  {
298  required_size = 0;
299 
300  /* if HA mode server */
301  if (IS_MASTER_CONN_NAME_HA_SERVER (temp->name))
302  {
303  required_size += strlen (HA_SERVER_FORMAT_STRING);
304  }
305  else
306  {
307  required_size += strlen (SERVER_FORMAT_STRING);
308  }
309  required_size += strlen (temp->name);
310  if (temp->version_string != NULL)
311  {
312  required_size += strlen (temp->version_string);
313  }
314  required_size += 5; /* length of pid string */
315 
316  bufsize += required_size;
317 
318  if (buffer == NULL)
319  {
320  buffer = (char *) malloc (bufsize * sizeof (char));
321  if (buffer == NULL)
322  {
323  goto error_return;
324  }
325  buffer[0] = '\0';
326  }
327  else
328  {
329  char *oldbuffer = buffer; /* save pointer in case realloc fails */
330  buffer = (char *) realloc (buffer, bufsize * sizeof (char));
331  if (buffer == NULL)
332  {
333  free_and_init (oldbuffer);
334  goto error_return;
335  }
336  }
337 
338  /* if HA mode server */
339  if (IS_MASTER_CONN_NAME_HA_SERVER (temp->name))
340  {
341  snprintf (buffer + strlen (buffer), required_size, HA_SERVER_FORMAT_STRING, temp->name + 1,
342  (temp->version_string == NULL ? "?" : temp->version_string), temp->pid);
343  }
344  else
345  {
346  snprintf (buffer + strlen (buffer), required_size, SERVER_FORMAT_STRING, temp->name,
347  (temp->version_string == NULL ? "?" : temp->version_string), temp->pid);
348 
349  }
350  }
351  }
352 
353  if (buffer == NULL)
354  {
355  goto error_return;
356  }
357 
358  if (css_send_data (conn, request_id, buffer, strlen (buffer) + 1) != NO_ERRORS)
359  {
360  css_cleanup_info_connection (conn);
361  }
362 
363  free_and_init (buffer);
364  return;
365 
366 error_return:
367  if (css_send_data (conn, request_id, "\0", 1) != NO_ERRORS)
368  {
369  css_cleanup_info_connection (conn);
370  }
371 
372  if (buffer != NULL)
373  {
374  free_and_init (buffer);
375  }
376  return;
377 }
