119 /*
120  * mht_1str_pseudo_key() - hash string key into pseudo integer key
121  * return: pseudo integer key
122  * key(in): string key to hash
123  * key_size(in): size of key or -1 when unknown
124  *
125  * Note: It generates a pseudo integer key based on Gosling's emacs hash
126  * function.
127  */
128 static unsigned int
129 mht_1str_pseudo_key (const void *key, int key_size)
130 {
131  unsigned const char *byte_p = (unsigned char *) key;
132  unsigned int pseudo_key;
133 
134  assert (key != NULL);
135 
136  for (pseudo_key = 0;; byte_p++)
137  {
138  if (key_size == -1)
139  {
140  if (!(*byte_p))
141  {
142  break;
143  }
144  }
145  else
146  {
147  if (key_size <= 0)
148  {
149  break;
150  }
151  }
152 
153  pseudo_key = (pseudo_key << 5) - pseudo_key + *byte_p;
154 
155  if (key_size > 0)
156  {
157  key_size--;
158  }
159  }
160 
161  return pseudo_key;
162 }

165  * mht_2str_pseudo_key() - hash string key into pseudo integer key
166  * return: pseudo integer key
167  * key(in): string key to hash
168  * key_size(in): size of key or -1 when unknown
169  *
170  * Note: It generates a pseudo integer key based on hash function from
171  * Aho, Sethi, and Ullman's dragon book; pp. 436.
172  * This function can handles strings when they are binary different.
173  * For collation dependent function use 'mht2str'.
174  */

176 mht_2str_pseudo_key (const void *key, int key_size)
177 {

179  unsigned int pseudo_key;
180  unsigned int i;
181 
182  assert (key != NULL);
183 
184  for (pseudo_key = 0;; byte_p++)
185  {
186  if (key_size == -1)
187  {
188  if (!(*byte_p))
189  {
190  break;

192  }
193  else
194  {
195  if (key_size <= 0)
196  {
197  break;
198  }
199  }
200 
201  pseudo_key = (pseudo_key << 4) + *byte_p;
202  i = pseudo_key & 0xf0000000;
203  if (i != 0)

205  pseudo_key ^= i >> 24;
206  pseudo_key ^= i;
207  }
208

210  {
211  key_size--;
212  }
213  }
214 
215  return pseudo_key;
216 }

269  * key(in): string key to hash
270  * key_size(in): size of key or -1 when unknown

272  * Note: It generates four values between 0 and 255, concatenate them,

274  * Based on Fast Hashing of Variable-Length Text Strings
275  * by Peter K. Pearson Communications of the ACM, June 1990.
276  */
277 static unsigned int
278 mht_4str_pseudo_key (const void *key, int key_size)
279 {
280  /* a permutation of values 0 to 255 */

300  211, 16, 9, 14, 47, 60, 197, 26, 75, 40,
301  65, 230, 39, 212, 125, 114, 195, 64, 121, 190,
302  31, 108, 53, 202, 59, 88, 177, 150, 23, 4,
303  237, 34, 179, 112, 233, 110, 15, 156, 165, 122,
304  43, 136, 33, 70, 7, 52, 93, 210, 163, 160,
305  89, 30, 255, 204, 21, 42, 27, 184, 145, 246,
306  247, 100, 205, 130, 147, 208, 201, 206, 239, 252,
307  133, 218, 11, 232, 1, 0
308  };
309  unsigned int byte1, byte2, byte3, byte4;
310  unsigned const char *byte_p = (unsigned char *) key;
311 
312  assert (key != NULL);
313 
314  byte1 = tbl[*byte_p];
315  byte2 = tbl[(unsigned int) (*byte_p + 1) % 256];
316  byte3 = tbl[(unsigned int) (*byte_p + 2) % 256];
317  byte4 = tbl[(unsigned int) (*byte_p + 3) % 256];
318 
319  if (key_size == -1)
320  {
321  if (*byte_p)
322  {
323  byte_p++;
324  }
325  }
326  else if (key_size > 0)
327  {
328  byte_p++;
329  key_size--;
330  }
331 
332  for (;; byte_p++)
333  {
334  if (key_size == -1)
335  {
336  if (!(*byte_p))
337  {
338  break;
339  }
340  }
341  else
342  {
343  if (key_size <= 0)
344  {
345  break;
346  }
347  }
348 
349  /*
350  * Each of the following hash values,
351  * generates a value between 0 and 255
352  */
353  byte1 = tbl[byte1 ^ *byte_p];
354  byte2 = tbl[byte2 ^ *byte_p];
355  byte3 = tbl[byte3 ^ *byte_p];
356  byte4 = tbl[byte4 ^ *byte_p];
357 
358  if (key_size > 0)
359  {
360  key_size--;
361  }
362  }
363 
364  /* Concatenate all the values */
365  return (byte1 | (byte2 << 8) | (byte3 << 16) | (byte4 << 24));
366 }

368 /*
369  * mht_5str_pseudo_key() - hash string key into pseudo integer key
370  * return: pseudo integer key
371  * key(in): string key to hash
372  * key_size(in): size of key or -1 when unknown
373  *
374  * Note: Based on hash method reported by Diniel J. Bernstein.
375  */
376 static unsigned int
377 mht_5str_pseudo_key (const void *key, int key_size)
378 {
379  unsigned int hash = 5381;
380  int i = 0;
381  char *k;
382 
383  assert (key != NULL);
384  k = (char *) key;
385 
386  if (key_size == -1)
387  {
388  int c;
389  while ((c = *(k + i++)) != 0)
390  {
391  hash = ((hash << 5) + hash) + c; /* hash * 33 + c */
392  }
393  }
394  else
395  {
396  for (; i < key_size; i++)
397  {
398  hash = ((hash << 5) + hash) + *(k + i);
399  }
400  }
401 
402  hash += ~(hash << 9);
403  hash ^= ((hash >> 14) | (i << 18)); /* >>> */
404  hash += (hash << 4);
405  hash ^= ((hash >> 10) | (i << 22)); /* >>> */
406 
407  return hash;
408 }

576 /*

578  * return: hash value
579  * key(in): pointer to DB_VALUE key to hash
580  * ht_size(in): size of hash table
581  */

583 mht_valhash (const void *key, const unsigned int ht_size)
584 {
585  unsigned int hash = 0;
586  const DB_VALUE *val = (const DB_VALUE *) key;
587  int t_n;
588  DB_VALUE t_val;
589 
590  if (key != NULL)
591  {
592  switch (db_value_type (val))
593  {
594  case DB_TYPE_NULL:
595  hash = 0;
596  break;
597  case DB_TYPE_INTEGER:
598  hash = (unsigned int) db_get_int (val);
599  break;
600  case DB_TYPE_SHORT:
601  hash = (unsigned int) db_get_short (val);
602  break;
603  case DB_TYPE_BIGINT:
604  {
605  DB_BIGINT bigint;
606  unsigned int x, y;
607 
608  bigint = db_get_bigint (val);
609  x = bigint >> 32;
610  y = (unsigned int) bigint;
611  hash = x ^ y;
612  break;
613  }
614  case DB_TYPE_FLOAT:
615  hash = (unsigned int) db_get_float (val);
616  break;
617  case DB_TYPE_DOUBLE:
618  hash = (unsigned int) db_get_double (val);
619  break;
620  case DB_TYPE_NUMERIC:
621  hash = mht_1str_pseudo_key (db_get_numeric (val), -1);
622  break;
623  case DB_TYPE_CHAR:
624  case DB_TYPE_NCHAR:
625  case DB_TYPE_VARCHAR:
626  case DB_TYPE_VARNCHAR:
627  hash = mht_1str_pseudo_key (db_get_string (val), db_get_string_size (val));
628  break;
629  case DB_TYPE_BIT:
630  case DB_TYPE_VARBIT:
631  hash = mht_1str_pseudo_key (db_get_bit (val, &t_n), -1);
632  break;
633  case DB_TYPE_TIME:
634  {
635  unsigned int *time = db_get_time (val);
636  hash = (unsigned int) (*time);
637  break;
638  }
639  case DB_TYPE_TIMESTAMP:
640  case DB_TYPE_TIMESTAMPLTZ:
641  {
642  DB_TIMESTAMP *time_st = db_get_timestamp (val);
643  hash = (unsigned int) (*time_st);
644  break;
645  }
646  case DB_TYPE_TIMESTAMPTZ:
647  {
648  DB_TIMESTAMPTZ *ts_tz = db_get_timestamptz (val);
649  hash = (unsigned int) (ts_tz->timestamp);
650  }
651  break;
652  case DB_TYPE_DATETIME:
653  case DB_TYPE_DATETIMELTZ:
654  {
655  DB_DATETIME *datetime;
656  datetime = db_get_datetime (val);
657  hash = (unsigned int) (datetime->date ^ datetime->time);
658  }
659  break;
660  case DB_TYPE_DATETIMETZ:
661  {
662  DB_DATETIMETZ *dt_tz;
663  dt_tz = db_get_datetimetz (val);
664  hash = (unsigned int) (dt_tz->datetime.date ^ dt_tz->datetime.time);
665  }
666  break;
667  case DB_TYPE_DATE:
668  {
669  DB_DATE *date = db_get_date (val);
670  hash = (unsigned int) (*date);
671  break;
672  }
673  case DB_TYPE_MONETARY:
674  {
675  DB_MONETARY *mon = db_get_monetary (val);
676  hash = (unsigned int) mon->amount;
677  break;
678  }
679  case DB_TYPE_SET:
680  case DB_TYPE_MULTISET:
681  case DB_TYPE_SEQUENCE:
682  db_make_null (&t_val);
683  {
684  DB_SET *set;
685  set = db_get_set (val);
686  if (set_get_element (set, 0, &t_val) == NO_ERROR)
687  {
688  hash = mht_valhash (&t_val, ht_size);
689  (void) pr_clear_value (&t_val);
690  t_n = set_size (set);
691  if ((t_n > 0) && set_get_element (set, t_n - 1, &t_val) == NO_ERROR)
692  {
693  hash += mht_valhash (&t_val, ht_size);
694  (void) pr_clear_value (&t_val);
695  }
696  }
697  }
698  break;
699  case DB_TYPE_OBJECT:
700  hash = GET_PTR_FOR_HASH (db_get_object (val));
701  break;
702  case DB_TYPE_OID:
703  hash = (unsigned int) OID_PSEUDO_KEY (db_get_oid (val));
704  break;
705  case DB_TYPE_MIDXKEY:
706  db_make_null (&t_val);
707  {
708  DB_MIDXKEY *midxkey;
709  midxkey = db_get_midxkey (val);
710  if (pr_midxkey_get_element_nocopy (midxkey, 0, &t_val, NULL, NULL) == NO_ERROR)
711  {

818 /*

820  * return: adjusted hash table size
821  * ht_size(in): given hash table size
822  *

824  * size for a hash table, unfortunately prime numbers are
825  * difficult to found. If the given htsize, falls among the
826  * prime numbers known by this module, a close prime number to
827  * the given htsize value is returned, otherwise, a power of two
828  * is returned.
829  */
830 #define NPRIMES 170

848  20507, 21313, 22123, 23131, 24133, 25147, 26153, 27179, 28181, 29123
849 };
850 
851 static unsigned int
852 mht_calculate_htsize (unsigned int ht_size)
853 {
854  int left, right, middle; /* indices for binary search */
855 
856  if (ht_size > mht_Primes[NPRIMES - 1])
857  {
858  /* get a power of two */
859  if (!((ht_size & (ht_size - 1)) == 0))
860  {
861  /* Turn off some bits but the left most one */
862  while (!(ht_size & (ht_size - 1)))
863  {
864  ht_size &= (ht_size - 1);
865  }
866  ht_size <<= 1;
867  }
868  }
869  else
870  {
871  /* we can assign a primary number; binary search */
872  for (middle = 0, left = 0, right = NPRIMES - 1; left <= right;)
873  {
874  middle = CEIL_PTVDIV ((left + right), 2);
875  if (ht_size == mht_Primes[middle])
876  {
877  break;
878  }
879  else if (ht_size > mht_Primes[middle])
880  {
881  left = middle + 1;
882  }
883  else
884  {
885  right = middle - 1;
886  }
887  }
888  /* If we didn't find the size, get the larger size and not the small one */
889  if (ht_size > mht_Primes[middle] && middle < (NPRIMES - 1))
890  {
891  middle++;
892  }
893  ht_size = mht_Primes[middle];
894  }
895 
896  return ht_size;
897 }

906  *
907  * Note: Create a new hash table. The estimated number of entries for
908  * the hash table may be adjusted (to a prime number) in order to
909  * obtain certain favorable circumstances when the table is
910  * created, when the table is almost full and there are at least
911  * 5% of collisions. 'hash_func' must return an integer between 0 and
912  * table_size - 1. 'cmp_func' must return TRUE if key1 = key2,
913  * otherwise, FALSE.
914  */
915 MHT_TABLE *
916 mht_create (const char *name, int est_size, unsigned int (*hash_func) (const void *key, unsigned int ht_size),
917  int (*cmp_func) (const void *key1, const void *key2))
918 {

920  HENTRY_PTR *hvector; /* Entries of hash table */
921  unsigned int ht_estsize;
922  size_t size;
923 
924  assert (hash_func != NULL && cmp_func != NULL);
925 
926  /* Get a good number of entries for hash table */

928  {

939

990 /*
991  * mht_rehash - rehash all entires of a hash table
992  * return: error code
993  * ht(in/out): hash table to rehash
994  *
995  * Note: Expand a hash table. All entries are rehashed onto a larger
996  * hash table of entries.
997  */
998 static int
999 mht_rehash (MHT_TABLE * ht)
1000 {
1001  HENTRY_PTR *new_hvector; /* New entries of hash table */
1002  HENTRY_PTR *hvector; /* Entries of hash table */
1003  HENTRY_PTR hentry; /* A hash table entry. linked list */
1004  HENTRY_PTR next_hentry = NULL; /* Next element in linked list */
1005  float rehash_factor;
1006  unsigned int hash;
1007  unsigned int est_size;
1008  size_t size;
1009  unsigned int i;
1010 
1011  /* Find an estimated size for hash table entries */
1012 
1013  rehash_factor = (float) (1.0 + ((float) ht->ncollisions / (float) ht->nentries));
1014  if (MHT_REHASH_FACTOR > rehash_factor)
1015  {
1016  est_size = (unsigned int) (ht->size * MHT_REHASH_FACTOR);
1017  }
1018  else
1019  {
1020  est_size = (unsigned int) (ht->size * rehash_factor);
1021  }
1022 
1023  est_size = mht_calculate_htsize (est_size);
1024 
1025  /* Allocate a new vector to keep the estimated hash entries */
1026  size = est_size * DB_SIZEOF (*hvector);
1027  new_hvector = (HENTRY_PTR *) malloc (size);
1028  if (new_hvector == NULL)
1029  {
1030  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, size);
1031  return ER_OUT_OF_VIRTUAL_MEMORY;
1032  }
1033 
1034  /* Initialize all entries */
1035  memset (new_hvector, 0x00, size);
1036 
1037  /* Now rehash the current entries onto the vector of hash entries table */
1038  for (ht->ncollisions = 0, hvector = ht->table, i = 0; i < ht->size; hvector++, i++)
1039  {
1040  /* Go over each linked list */
1041  for (hentry = *hvector; hentry != NULL; hentry = next_hentry)
1042  {
1043  next_hentry = hentry->next;
1044 
1045  hash = (*ht->hash_func) (hentry->key, est_size);
1046  if (hash >= est_size)
1047  {
1048  hash %= est_size;
1049  }
1050 
1051  /* Link the new entry with any previous elements */
1052  hentry->next = new_hvector[hash];
1053  if (hentry->next != NULL)
1054  {
1055  ht->ncollisions++;
1056  }
1057  new_hvector[hash] = hentry;
1058  }
1059  }
1060 
1061  /* Now move to new vector of entries */
1062  free_and_init (ht->table);
1063 
1064  ht->table = new_hvector;
1065  ht->size = est_size;
1066  ht->rehash_at = (int) (est_size * MHT_REHASH_TRESHOLD);
1067 
1068  return NO_ERROR;
1069 }
