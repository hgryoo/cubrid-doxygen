62 /*
63  * MESSAGE CATALOG FILE FORMAT.
64  *
65  * The NetBSD/FreeBSD message catalog format is similar to the format used by
66  * Svr4 systems. The differences are:
67  * * fixed byte order (big endian)
68  * * fixed data field sizes
69  *
70  * A message catalog contains four data types: a catalog header, one
71  * or more set headers, one or more message headers, and one or more
72  * text strings.
73  *
74  * NOTE: some changes were done to the initial code. we can no longer use the ntypes.h. and if nltypes.h is included
75  * there will be conflicts... so just rename everything.
76  */

130 cub_nl_catd
131 cub_catopen (const char *name, int type)
132 {
133  int spcleft, saverr;
134  char path[PATH_MAX];
135  char *nlspath, *lang, *base, *cptr, *pathP, *tmpptr;
136  char *cptr1, *plang, *pter, *pcode;
137  struct stat sbuf;
138 
139  if (name == NULL || *name == '\0')
140  {
141  errno = EINVAL;
142  return NULL;
143  }
144 
145  /* is it absolute path ? if yes, load immediately */
146  if (strchr (name, '/') != NULL)
147  {
148  return load_msgcat (name);
149  }
150 
151  if (type == CUB_NL_CAT_LOCALE)
152  {
153  lang = setlocale (LC_MESSAGES, NULL);
154  }
155  else
156  {
157  lang = getenv ("CHARSET");
158  }
159 
160  if (lang == NULL || *lang == '\0' || strlen (lang) > ENCODING_LEN
161  || (lang[0] == '.' && (lang[1] == '\0' || (lang[1] == '.' && lang[2] == '\0'))) || strchr (lang, '/') != NULL)
162  {
163  lang = (char *) "C";
164  }
165 
166  plang = cptr1 = strdup (lang);
167  if (cptr1 == NULL)
168  {
169  return NULL;
170  }
171 
172  cptr = strchr (cptr1, '@');
173  if (cptr != NULL)
174  {
175  *cptr = '\0';
176  }
177 
178  pter = pcode = (char *) "";
179  cptr = strchr (cptr1, '_');
180  if (cptr != NULL)
181  {
182  *cptr++ = '\0';
183  pter = cptr1 = cptr;
184  }
185 
186  cptr = strchr (cptr1, '.');
187  if (cptr != NULL)
188  {
189  *cptr++ = '\0';
190  pcode = cptr;
191  }
192 
193  nlspath = getenv ("NLSPATH");
194  if (nlspath == NULL)
195  {
196  nlspath = (char *) DEFAULT_NLS_PATH;
197  }
198 
199  base = cptr = strdup (nlspath);
200  if (cptr == NULL)
201  {
202  saverr = errno;
203  free (plang);
204  errno = saverr;
205  return NULL;
206  }
207 
208  while ((nlspath = strsep (&cptr, ":")) != NULL)
209  {
210  pathP = path;
211  if (*nlspath)
212  {
213  for (; *nlspath; ++nlspath)
214  {
215  if (*nlspath == '%')
216  {
217  switch (*(nlspath + 1))
218  {
219  case 'l':
220  tmpptr = plang;
221  break;
222  case 't':
223  tmpptr = pter;
224  break;
225  case 'c':
226  tmpptr = pcode;
227  break;
228  case 'L':
229  tmpptr = lang;
230  break;
231  case 'N':
232  tmpptr = (char *) name;
233  break;
234  case '%':
235  ++nlspath;
236  /* fallthrough */
237  default:
238  if (pathP - path >= PATH_MAX - 1)
239  {
240  goto too_long;
241  }
242  *(pathP++) = *nlspath;
243  continue;
244  }
245  ++nlspath;
246 
247  put_tmpptr:
248  spcleft = PATH_MAX - (CAST_STRLEN (pathP - path)) - 1;
249  if (strlcpy (pathP, tmpptr, spcleft) >= (size_t) spcleft)
250  {
251  too_long:
252  free (plang);
253  free (base);
254  errno = ENAMETOOLONG;
255  return NULL;
256  }
257  pathP += strlen (tmpptr);
258  }
259  else
260  {
261  if (pathP - path >= PATH_MAX - 1)
262  {
263  goto too_long;
264  }
265  *(pathP++) = *nlspath;
266  }
267  }
268  *pathP = '\0';
269  if (stat (path, &sbuf) == 0)
270  {
271  free (plang);
272  free (base);
273  return load_msgcat (path);
274  }
275  }
276  else
277  {
278  tmpptr = (char *) name;
279  --nlspath;
280  goto put_tmpptr;
281  }
282  }
283 
284  free (plang);
285  free (base);
286  errno = ENOENT;
287 
288  return NULL;
289 }

291 char *
292 cub_catgets (cub_nl_catd catd, int set_id, int msg_id, const char *s)
293 {
294  struct nls_cat_hdr *cat_hdr;
295  struct nls_set_hdr *set_hdr;
296  struct nls_msg_hdr *msg_hdr;
297  int l, u, i, r;
298 
299  if (catd == NULL)
300  {
301  errno = EBADF;
302  /* LINTED interface problem */
303  return (char *) s;
304  }
305 
306  cat_hdr = (struct nls_cat_hdr *) catd->_data;
307  set_hdr = (struct nls_set_hdr *) (void *) ((char *) catd->_data + sizeof (struct nls_cat_hdr));
308 
309  /* binary search, see knuth algorithm b */
310  l = 0;
311  u = ntohl ((UINT32) cat_hdr->_nsets) - 1;
312  while (l <= u)
313  {
314  i = (l + u) / 2;
315  r = set_id - ntohl ((UINT32) set_hdr[i]._setno);
316 
317  if (r == 0)
318  {
319  msg_hdr =
320  (struct nls_msg_hdr *) (void *) ((char *) catd->_data + sizeof (struct nls_cat_hdr) +
321  ntohl ((UINT32) cat_hdr->_msg_hdr_offset));
322 
323  l = ntohl ((UINT32) set_hdr[i]._index);
324  u = l + ntohl ((UINT32) set_hdr[i]._nmsgs) - 1;
325  while (l <= u)
326  {
327  i = (l + u) / 2;
328  r = msg_id - ntohl ((UINT32) msg_hdr[i]._msgno);
329  if (r == 0)
330  {
331  return ((char *) catd->_data + sizeof (struct nls_cat_hdr) +
332  ntohl ((UINT32) cat_hdr->_msg_txt_offset) + ntohl ((UINT32) msg_hdr[i]._offset));
333  }
334  else if (r < 0)
335  {
336  u = i - 1;
337  }
338  else
339  {
340  l = i + 1;
341  }
342  }
343 
344  /* not found */
345  goto notfound;
346 
347  }
348  else if (r < 0)
349  {
350  u = i - 1;
351  }
352  else
353  {
354  l = i + 1;
355  }
356  }
357 
358 notfound:
359  /* not found */
360  errno = ENOMSG;
361  /* LINTED interface problem */
362  return (char *) s;
363 }

388 static cub_nl_catd
389 load_msgcat (const char *path)
390 {
391  struct stat st;
392  cub_nl_catd catd;
393  void *data = NULL;
394 #if defined(WINDOWS)
395  HANDLE map_handle;
396  HANDLE file_handle;
397 #else
398  int fd;
399 #endif
400 
401  /* XXX: path != NULL? */
402 
403 #if defined(WINDOWS)
404  file_handle = CreateFile (path, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
405  if (file_handle == NULL)
406  {
407  return NULL;
408  }
409  if (stat (path, &st) != 0)
410  {
411  CloseHandle (file_handle);
412  return NULL;
413  }
414 
415  map_handle = CreateFileMapping ((HANDLE) file_handle, NULL, PAGE_READONLY, 0, (DWORD) st.st_size, NULL);
416  if (map_handle != NULL)
417  {
418  data = MapViewOfFile (map_handle, FILE_MAP_READ, 0, 0, 0);
419  }
420 
421  if (data == NULL)
422  {
423  return NULL;
424  }
425 #else
426  fd = open (path, O_RDONLY);
427  if (fd == -1)
428  {
429  return NULL;
430  }
431 
432  if (fstat (fd, &st) != 0)
433  {
434  close (fd);
435  return NULL;
436  }
437 
438  data = mmap (0, (size_t) st.st_size, PROT_READ, MAP_SHARED, fd, (off_t) 0);
439  close (fd);
440 
441  if (data == MAP_FAILED)
442  {
443  return NULL;
444  }
445 #endif
446 
447  if (ntohl ((UINT32) ((struct nls_cat_hdr *) data)->_magic) != NLS_MAGIC)
448  {
449 #if defined(WINDOWS)
450  UnmapViewOfFile (data);
451  CloseHandle (map_handle);
452 #else
453  munmap (data, (size_t) st.st_size);
454 #endif
455  errno = EINVAL;
456  return NULL;
457  }
458 
459  catd = (cub_nl_catd) malloc (sizeof (*catd));
460  if (catd == NULL)
461  {
462 #if defined(WINDOWS)
463  UnmapViewOfFile (data);
464  CloseHandle (map_handle);
465 #else
466  munmap (data, (size_t) st.st_size);
467 #endif
468  return NULL;
469  }
470 
471  catd->_data = data;
472  catd->_size = (int) st.st_size;
473 #if defined(WINDOWS)
474  catd->map_handle = map_handle;
475 #endif
476  return catd;
477 }
