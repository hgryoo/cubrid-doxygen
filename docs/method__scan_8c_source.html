81 /*
82  * xs_open_va_scan () -
83  * return: int
84  * scan_buf(in) : Value array buffer
85  * scanbuf_type(in) : Method signature list
86  */
87 static int
88 method_open_value_array_scan (METHOD_SCAN_BUFFER * scan_buffer_p)
89 {
90  int num_methods;
91 
92  num_methods = scan_buffer_p->s.method_ctl.method_sig_list->num_methods;
93  if (num_methods <= 0)
94  {
95  num_methods = MAX_XS_SCANBUF_DBVALS; /* for safe-guard */
96  }
97 
98  scan_buffer_p->dbval_list = (QPROC_DB_VALUE_LIST) malloc (sizeof (scan_buffer_p->dbval_list[0]) * num_methods);
99 
100  if (scan_buffer_p->dbval_list == NULL)
101  {
102  return ER_FAILED;
103  }
104 
105 #ifdef SERVER_MODE
106  scan_buffer_p->vacomm_buffer = method_initialize_vacomm_buffer ();
107  if (scan_buffer_p->vacomm_buffer == NULL)
108  {
109  return ER_FAILED;
110  }
111 #endif /* SERVER_MODE */
112 
113  return NO_ERROR;
114 }

133 /*
134  * method_scan_next_value_array () -
135  * return: int
136  * scan_buf(in) : Value array buffer
137  * val_list(in) :
138  */
139 static SCAN_CODE
140 method_scan_next_value_array (METHOD_SCAN_BUFFER * scan_buffer_p, VAL_LIST * value_list_p)
141 {
142  SCAN_CODE scan_result = S_SUCCESS;
143  QPROC_DB_VALUE_LIST dbval_list;
144  int n;
145 
146  dbval_list = scan_buffer_p->dbval_list;
147  for (n = 0; n < value_list_p->val_cnt; n++)
148  {
149  dbval_list->next = dbval_list + 1;
150  dbval_list++;
151  }
152 
153  scan_buffer_p->dbval_list[value_list_p->val_cnt - 1].next = NULL;
154  value_list_p->valp = scan_buffer_p->dbval_list;
155 
156  return scan_result;
157 }

159 /*
160  * method_open_scan () -
161  * return: int
162  * scan_buf(in) : Value array buffer
163  * list_id(in) :
164  * method_sig_list(in): Method signature list
165  */
166 int
167 method_open_scan (THREAD_ENTRY * thread_p, METHOD_SCAN_BUFFER * scan_buffer_p, QFILE_LIST_ID * list_id_p,
168  METHOD_SIG_LIST * method_sig_list_p)
169 {
170  int error;
171  METHOD_INFO *method_ctl_p;
172 
173  method_ctl_p = &scan_buffer_p->s.method_ctl;
174 
175  method_ctl_p->list_id = list_id_p;
176  method_ctl_p->method_sig_list = method_sig_list_p;
177 
178  error = method_open_value_array_scan (scan_buffer_p);
179  if (error != NO_ERROR)
180  {
181  return error;
182  }
183 
184  error = method_invoke (thread_p, scan_buffer_p);
185  if (error != NO_ERROR)
186  {
187  (void) method_close_value_array_scan (scan_buffer_p);
188  return error;
189  }
190 
191  return NO_ERROR;
192 }

194 /*
195  * method_close_scan () -
196  * return: int
197  * scan_buf(in) : Value array buffer
198  */
199 int
200 method_close_scan (THREAD_ENTRY * thread_p, METHOD_SCAN_BUFFER * scan_buffer_p)
201 {
202 #ifdef SERVER_MODE
203  VACOMM_BUFFER *vacomm_buffer_p;
204 
205  /*
206  * If the method scan is being closed before the client is done, the status could be zero (1st buffer not received)
207  * or METHOD_SUCCESS (last buffer not received). */
208 
209  vacomm_buffer_p = scan_buffer_p->vacomm_buffer;
210 
211  if ((vacomm_buffer_p) && ((vacomm_buffer_p->status == 0) || (vacomm_buffer_p->status == METHOD_SUCCESS)))
212  {
213  vacomm_buffer_p->action = VACOMM_BUFFER_ABORT;
214 
215  do
216  {
217  (void) method_receive_results (thread_p, scan_buffer_p);
218  }
219  while (vacomm_buffer_p->status == METHOD_SUCCESS);
220  }
221 #else /* SERVER_MODE */
222  method_clear_scan_buffer (scan_buffer_p);
223 #endif /* SERVER_MODE */
224 
225  return method_close_value_array_scan (scan_buffer_p);
226 }

234 SCAN_CODE
235 method_scan_next (THREAD_ENTRY * thread_p, METHOD_SCAN_BUFFER * scan_buffer_p, VAL_LIST * value_list_p)
236 {
237  SCAN_CODE scan_result;
238 
239  scan_result = method_receive_results (thread_p, scan_buffer_p);
240 
241  if (scan_result == S_SUCCESS)
242  {
243  value_list_p->val_cnt = scan_buffer_p->s.method_ctl.method_sig_list->num_methods;
244  scan_result = method_scan_next_value_array (scan_buffer_p, value_list_p);
245  }
246 
247  return scan_result;
248 }

260 static int
261 method_invoke_from_stand_alone (METHOD_SCAN_BUFFER * scan_buffer_p)
262 {
263  int meth_num;
264  int val_cnt;
265  int i;
266  METHOD_INFO *method_ctl_p;
267  METHOD_SIG_LIST *method_sig_list;
268  METHOD_SIG *meth_sig;
269 
270  method_ctl_p = &scan_buffer_p->s.method_ctl;
271  method_sig_list = method_ctl_p->method_sig_list;
272  scan_buffer_p->crs_id.buffer = NULL;
273 
274  val_cnt = 0;
275  meth_sig = method_sig_list->method_sig;
276 
277  for (meth_num = 0; meth_num < method_sig_list->num_methods; meth_num++)
278  {
279  val_cnt += meth_sig->num_method_args + 1;
280  meth_sig = meth_sig->next;
281  }
282 
283  if (method_ctl_p->list_id->type_list.type_cnt > val_cnt)
284  {
285  val_cnt = method_ctl_p->list_id->type_list.type_cnt;
286  }
287 
288  scan_buffer_p->val_cnt = val_cnt;
289  scan_buffer_p->vallist = (DB_VALUE *) malloc (sizeof (DB_VALUE) * val_cnt);
290 
291  if ((val_cnt > 0) && (scan_buffer_p->vallist == NULL))
292  {
293  return ER_FAILED;
294  }
295  /*
296  * Make sure that these containers get initialized with meaningful
297  * bits. It's possible to wind up in method_clear_scan_buffer() without ever
298  * having actually received any method results, and if that happens
299  * we'll be trying to db_value_clear() junk unless we initialize things
300  * here.
301  */
302  for (i = 0; i < val_cnt; i++)
303  {
304  db_make_null (&scan_buffer_p->vallist[i]);
305  }
306 
307  scan_buffer_p->valptrs = (DB_VALUE **) malloc (sizeof (DB_VALUE *) * (val_cnt + 1));
308 
309  if (scan_buffer_p->valptrs == NULL)
310  {
311  method_clear_scan_buffer (scan_buffer_p);
312  return ER_FAILED;
313  }
314 
315  scan_buffer_p->oid_cols = (int *) malloc (sizeof (int) * method_sig_list->num_methods);
316 
317  if (scan_buffer_p->oid_cols == NULL)
318  {
319  method_clear_scan_buffer (scan_buffer_p);
320  return ER_FAILED;
321  }
322 
323  meth_sig = method_sig_list->method_sig;
324  for (meth_num = 0; meth_num < method_sig_list->num_methods; meth_num++)
325  {
326  scan_buffer_p->oid_cols[meth_num] = meth_sig->method_arg_pos[0];
327  meth_sig = meth_sig->next;
328  }
329 
330  if (!cursor_open (&scan_buffer_p->crs_id, method_ctl_p->list_id, false, false))
331  {
332  method_clear_scan_buffer (scan_buffer_p);
333  return ER_FAILED;
334  }
335 
336  /* tfile_vfid pointer as query id for method scan */
337  scan_buffer_p->crs_id.query_id = (QUERY_ID) method_ctl_p->list_id->tfile_vfid;
338 
339  cursor_set_oid_columns (&scan_buffer_p->crs_id, scan_buffer_p->oid_cols, method_sig_list->num_methods);
340 
341  return NO_ERROR;
342 }

360 #ifdef SERVER_MODE
361 static SCAN_CODE
362 method_receive_results_for_server (THREAD_ENTRY * thread_p, METHOD_SCAN_BUFFER * scan_buffer_p)
363 {
364  QPROC_DB_VALUE_LIST dbval_list;
365  int meth_num, i;
366  METHOD_SIG *meth_sig_p;
367  SCAN_CODE result;
368  DB_VALUE *dbval_p;
369  METHOD_SIG_LIST *method_sig_list_p;
370 
371  method_sig_list_p = scan_buffer_p->s.method_ctl.method_sig_list;
372  meth_sig_p = method_sig_list_p->method_sig;
373  dbval_list = scan_buffer_p->dbval_list;
374 
375  for (meth_num = 0; meth_num < method_sig_list_p->num_methods; ++meth_num)
376  {
377  dbval_p = (DB_VALUE *) malloc (sizeof (DB_VALUE));
378  dbval_list->val = dbval_p;
379 
380  if (dbval_p == NULL)
381  {
382  result = S_ERROR;
383  }
384  else
385  {
386  db_make_null (dbval_p);
387  result = method_receive_value (thread_p, dbval_p, scan_buffer_p->vacomm_buffer);
388  }
389 
390  if (result != S_SUCCESS)
391  {
392  for (i = 0; i <= meth_num; i++)
393  {
394  if (scan_buffer_p->dbval_list[i].val)
395  {
396  db_value_clear (scan_buffer_p->dbval_list[i].val);
397  free_and_init (scan_buffer_p->dbval_list[i].val);
398  }
399  }
400 
401  if (result == S_ERROR)
402  {
403  scan_buffer_p->vacomm_buffer->status = METHOD_ERROR;
404  }
405 
406  return result;
407  }
408 
409  dbval_list++;
410  meth_sig_p = meth_sig_p->next;
411  }
412 
413  return S_SUCCESS;
414 }

418 static SCAN_CODE
419 method_receive_results_for_stand_alone (METHOD_SCAN_BUFFER * scan_buffer_p)
420 {
421  QPROC_DB_VALUE_LIST dbval_list;
422  int meth_num;
423  METHOD_SIG *meth_sig;
424  int crs_result;
425  int num_args;
426  int pos;
427  int arg;
428  int turn_on_auth = 1;
429  DB_VALUE val;
430  int error = NO_ERROR;
431  HL_HEAPID save_pri_heap_id;
432  METHOD_SIG_LIST *method_sig_list;
433 
434  method_sig_list = scan_buffer_p->s.method_ctl.method_sig_list;
435 
436  EXIT_SERVER_IN_METHOD_CALL (save_pri_heap_id);
437 
438  crs_result = cursor_next_tuple (&scan_buffer_p->crs_id);
439  if (crs_result == DB_CURSOR_SUCCESS)
440  {
441  DB_VALUE *ptr;
442  int i;
443 
444  /* Since we may be calling this in a loop, we need to clear out the vallist to avoid leaking old pointers. */
445  for (i = 0, ptr = scan_buffer_p->vallist; ptr && i < scan_buffer_p->val_cnt; i++, ptr++)
446  {
447  pr_clear_value (ptr);
448  }
449 
450  if (cursor_get_tuple_value_list (&scan_buffer_p->crs_id,
451  scan_buffer_p->s.method_ctl.list_id->type_list.type_cnt,
452  scan_buffer_p->vallist) != NO_ERROR)
453  {
454  method_clear_scan_buffer (scan_buffer_p);
455  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
456  return S_ERROR;
457  }
458 
459  meth_sig = method_sig_list->method_sig;
460  dbval_list = scan_buffer_p->dbval_list;
461 
462  for (meth_num = 0; meth_num < method_sig_list->num_methods; meth_num++)
463  {
464  /* The first position # is for the object ID */
465  num_args = meth_sig->num_method_args + 1;
466  for (arg = 0; arg < num_args; ++arg)
467  {
468  pos = meth_sig->method_arg_pos[arg];
469  scan_buffer_p->valptrs[arg] = &scan_buffer_p->vallist[pos];
470  }
471 
472  scan_buffer_p->valptrs[num_args] = NULL;
473  db_make_null (&val);
474 
475  if (meth_sig->class_name != NULL)
476  {
477  /* Don't call the method if the object is NULL or it has been deleted. A method call on a NULL object is
478  * NULL. */
479  if (!DB_IS_NULL (scan_buffer_p->valptrs[0]))
480  {
481  error = db_is_any_class (db_get_object (scan_buffer_p->valptrs[0]));
482  if (error == 0)
483  {
484  error = db_is_instance (db_get_object (scan_buffer_p->valptrs[0]));
485  }
486  }
487  if (error == ER_HEAP_UNKNOWN_OBJECT)
488  {
489  error = NO_ERROR;
490  }
491  else if (error > 0)
492  {
493  /* methods must run with authorization turned on and database modifications turned off. */
494  turn_on_auth = 0;
495  AU_ENABLE (turn_on_auth);
496  db_disable_modification ();
497  ++method_Num_method_jsp_calls;
498  error = obj_send_array (db_get_object (scan_buffer_p->valptrs[0]), meth_sig->method_name, &val,
499  &scan_buffer_p->valptrs[1]);
500  --method_Num_method_jsp_calls;
501  db_enable_modification ();
502  AU_DISABLE (turn_on_auth);
503  }
504  }
505  else
506  {
507  /* java stored procedure call */
508  turn_on_auth = 0;
509  AU_ENABLE (turn_on_auth);
510  db_disable_modification ();
511  ++method_Num_method_jsp_calls;
512  error = jsp_call_from_server (&val, scan_buffer_p->valptrs, meth_sig->method_name,
513  meth_sig->num_method_args);
514  --method_Num_method_jsp_calls;
515  db_enable_modification ();
516  AU_DISABLE (turn_on_auth);
517  }
518 
519  if (error != NO_ERROR)
520  {
521  method_clear_scan_buffer (scan_buffer_p);
522  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
523  return S_ERROR;
524  }
525 
526  if (DB_VALUE_TYPE (&val) == DB_TYPE_ERROR)
527  {
528  if (er_errid () == NO_ERROR)
529  {
530  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_GENERIC_ERROR, 1);
531  }
532  method_clear_scan_buffer (scan_buffer_p);
533  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
534  return S_ERROR;
535  }
536 
537  dbval_list->val = regu_dbval_db_alloc ();
538 
539  if (dbval_list->val == NULL)
540  {
541  pr_clear_value (&val);
542  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
543  return S_ERROR;
544  }
545 
546  /* Don't forget to translate any OBJECTS to OIDs. */
547  if (DB_VALUE_DOMAIN_TYPE (&val) == DB_TYPE_OBJECT)
548  {
549  db_make_oid (dbval_list->val, ws_oid (db_get_object (&val)));
550  }
551  else if (db_value_clone (&val, dbval_list->val) != NO_ERROR)
552  {
553  pr_clear_value (&val);
554  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
555  return S_ERROR;
556  }
557  pr_clear_value (&val);
558 
559  dbval_list++;
560  meth_sig = meth_sig->next;
561  }
562 
563  /* enter server after all jsp functions are finished */
564  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
565  return S_SUCCESS;
566  }
567  else if (crs_result == DB_CURSOR_END)
568  {
569  method_clear_scan_buffer (scan_buffer_p);
570  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
571  return S_END;
572  }
573  else
574  {
575  method_clear_scan_buffer (scan_buffer_p);
576  ENTER_SERVER_IN_METHOD_CALL (save_pri_heap_id);
577  return (SCAN_CODE) crs_result;
578  }
579 }

644 /*
645  * method_initialize_vacomm_buffer () - Initializes the method comm buffer
646  * return:
647  */
648 static VACOMM_BUFFER *
649 method_initialize_vacomm_buffer (void)
650 {
651  VACOMM_BUFFER *vacomm_buffer;
652 
653  vacomm_buffer = (VACOMM_BUFFER *) malloc (sizeof (VACOMM_BUFFER));
654  if (vacomm_buffer == NULL)
655  {
656  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_OUT_OF_VIRTUAL_MEMORY, 1, sizeof (VACOMM_BUFFER));
657  return NULL;
658  }
659 
660  vacomm_buffer->length = 0;
661  vacomm_buffer->status = 0;
662  vacomm_buffer->action = VACOMM_BUFFER_SEND;
