20 //
21 // monitor_transaction.hpp - interface for managing separate sheets of statistics for active transactions.
22 //
23 // cubrid monitoring module collects by default so called global statistics. in special circumstances, separate set
24 // of statistics may be tracked for one or more transactions.
25 //
26 // this interface provides statistics that can collect to separate transaction sheets automatically.
27 //
28 // to keep memory footprint as reduced as possible, statistics extend the number of sheets dynamically, based on
29 // requirements. in most workloads, no or just few extra sheets will be created.
30 //
31 // IMPORTANT - sheets are reused in order to avoid allocating more than necessary. as consequence, a reused sheet
32 // will inherit values from previous usage. the correct way of inspecting the execution of a transaction
33 // or thread is to fetch two snapshots, once at the beginning and once at the end and do the difference
34 // between snapshots.
35 //

152  template <class S>
153  void
154  transaction_statistic<S>::extend (std::size_t to)
155  {
156  assert (to > 0);
157 
158  std::unique_lock<std::mutex> ulock (m_extend_mutex);
159 
160  if (to > transaction_sheet_manager::MAX_SHEETS)
161  {
162  // to be on safe side
163  assert (false);
164  to = transaction_sheet_manager::MAX_SHEETS;
165  }
166 
167  if (to <= m_sheet_stats_count)
168  {
169  // already extended
170  return;
171  }
172 
173  // allocate new buffer
174  statistic_type *new_collectors = new statistic_type[to];
175  if (m_sheet_stats_count != 0)
176  {
177  // copy old buffer
178  std::memcpy (new_collectors, m_sheet_stats, m_sheet_stats_count * sizeof (statistic_type));
179  }
180 
181  // delete old buffer
182  delete [] m_sheet_stats;
183 
184  // update buffer
185  m_sheet_stats = new_collectors;
186  m_sheet_stats_count = to;
187  }

191  transaction_statistic<S>::fetch (statistic_value *destination, fetch_mode mode /* = FETCH_GLOBAL */) const
192  {
193  if (mode == FETCH_GLOBAL)
194  {
195  m_global_stat.fetch (destination);
196  }
197  else
198  {

201  if (sheet == transaction_sheet_manager::INVALID_TRANSACTION_SHEET)
