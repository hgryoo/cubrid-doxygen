59  /*
60  * write API for stream:
61  * 1. reserve position (with pointer in bip_buffer)
62  * 2. call write action
63  * 3. commit the reserve position
64  * the write_function is expected to return error code (negative value) or number of written bytes (positive)
65  */
66  int multi_thread_stream::write (const size_t byte_count, write_func_t &write_action)
67  {
68  int err = NO_ERROR;
69  stream_reserve_context *reserve_context = NULL;
70  char *ptr;
71  stream_position reserved_pos;
72  int written_bytes;
73 
74  ptr = reserve_with_buffer (byte_count, reserve_context);
75  if (ptr == NULL)
76  {
77  err = ER_FAILED;
78  return err;
79  }
80  reserved_pos = reserve_context->start_pos;
81 
82  written_bytes = write_action (reserved_pos, ptr, byte_count);
83 
84  reserve_context->written_bytes = written_bytes;
85 
86  err = commit_append (reserve_context);
87 
88  if (written_bytes < 0)
89  {
90  err = ER_FAILED;
91  }
92 
93  return (err < 0) ? err : written_bytes;
94  }

96  /*
97  * read API:
98  * 1. acquire pointer to data range
99  * 2. if actual range is same as requested continue with (7)
100  * 3. (actual range < requested range) allocate new local memory buffer for requested amount
101  * 4. copy partial range to local buffer
102  * 5. acquire pointer to rest of data
103  * 6. copy the remaining range to continuation of local buffer
104  * 7. execute read action using local buffer or initial provided pointer
105  * 8. release latch read (in case local buffer not used)
106  */
107  int multi_thread_stream::read (const stream_position first_pos, const size_t byte_count, read_func_t &read_action)
108  {
109  int err = NO_ERROR;
110  char *ptr;
111  size_t actual_read_bytes = 0;
112  char *local_buffer = NULL;
113  mem::buffer_latch_read_id read_latch_page_idx;
114  int read_bytes;
115 
116  ptr = get_data_from_pos (first_pos, byte_count, actual_read_bytes, read_latch_page_idx);
117  if (ptr == NULL)
118  {
119  err = ER_FAILED;
120  return err;
121  }
122 
123  /* the start of requested stream position may be located at the end of bip_buffer, use local buffer */
124  if (actual_read_bytes < byte_count)
125  {
126  stream_position next_pos = first_pos + actual_read_bytes;
127  size_t next_actual_read_bytes = 0;
128  local_buffer = new char[byte_count];
129 
130  memcpy (local_buffer, ptr, actual_read_bytes);
131  unlatch_read_data (read_latch_page_idx);
132 
133  ptr = get_data_from_pos (next_pos, byte_count - actual_read_bytes, next_actual_read_bytes,
134  read_latch_page_idx);
135  if (ptr == NULL || next_actual_read_bytes != byte_count - actual_read_bytes)
136  {
137  err = ER_FAILED;
138 
139  if (ptr != NULL)
140  {
141  unlatch_read_data (read_latch_page_idx);
142  }
143 
144  delete [] local_buffer;
145  return err;
146  }
147 
148  assert (next_actual_read_bytes + actual_read_bytes == byte_count);
149 
150  memcpy (local_buffer + actual_read_bytes, ptr, next_actual_read_bytes);
151 
152  unlatch_read_data (read_latch_page_idx);
153 
154  ptr = local_buffer;
155  }
156 
157  read_bytes = read_action (ptr, byte_count);
158 
159  if (ptr != local_buffer)
160  {
161  err = unlatch_read_data (read_latch_page_idx);
162  }
163  else
164  {
165  delete [] local_buffer;
166  }
167 
168  return (err < 0) ? err : read_bytes;
169  }

171  /*
172  * read_serial API:
173  * 1. wait for data to be committed for the whole requested range
174  * 2. acquire pointer to data range
175  * 3. if actual range is same as requested continue with (8)
176  * 4. (actual range < requested range) allocate new local memory buffer for requested amount
177  * 5. copy partial range to local buffer
178  * 6. acquire pointer to rest of data
179  * 7. copy the remaining range to continuation of local buffer
180  * 8. execute read_prepare_action action using local buffer or initial provided pointer
181  * this is expected to read a header of data and return the size of payload
182  * 9. release latch read (in case local buffer not used)
183  * 10. wait for an amount of payload of data to be produced (committed)
184  */
185  int multi_thread_stream::read_serial (const size_t amount, read_prepare_func_t &read_prepare_action)
186  {
187  int err = NO_ERROR;
188  char *ptr;
189  size_t actual_read_bytes = 0;
190  char *local_buffer = NULL;
191  mem::buffer_latch_read_id read_latch_page_idx;
192  stream_position to_read_pos;
193  stream_position trail_pos;
194  size_t payload_size;
195  int read_bytes;
196 
197  /* wait for stream to receive data */
198  if (m_read_position + amount > m_last_committed_pos)
199  {
200  err = wait_for_data (amount, STREAM_DONT_SKIP);
201 
202  /* fetch is expect to return NO_ERROR upon completion of fetch with success */
203  if (err != NO_ERROR)
204  {
205  return err;
206  }
207  }
208 
209  to_read_pos = m_read_position;
210 
211  ptr = get_data_from_pos (to_read_pos, amount, actual_read_bytes, read_latch_page_idx);
212  if (ptr == NULL)
213  {
214  err = ER_FAILED;
215  return err;
216  }
217 
218  /* the start of requested stream position may be located at the end of bip_buffer, use local buffer */
219  if (actual_read_bytes < amount)
220  {
221  stream_position next_pos = to_read_pos + actual_read_bytes;
222  size_t next_actual_read_bytes = 0;
223  local_buffer = new char[amount];
224 
225  memcpy (local_buffer, ptr, actual_read_bytes);
226  unlatch_read_data (read_latch_page_idx);
227 
228  ptr = get_data_from_pos (next_pos, amount - actual_read_bytes, next_actual_read_bytes,
229  read_latch_page_idx);
230  if (ptr == NULL || next_actual_read_bytes != amount - actual_read_bytes)
231  {
232  err = ER_FAILED;
233 
234  if (ptr != NULL)
235  {
236  unlatch_read_data (read_latch_page_idx);
237  }
238 
239  delete [] local_buffer;
240  return err;
241  }
242 
243  assert (next_actual_read_bytes + actual_read_bytes == amount);
244 
245  memcpy (local_buffer + actual_read_bytes, ptr, next_actual_read_bytes);
246 
247  unlatch_read_data (read_latch_page_idx);
248 
249  ptr = local_buffer;
250  }
251 
252  trail_pos = to_read_pos + amount;
253  read_bytes = read_prepare_action (trail_pos, ptr, amount, payload_size);
254 
255  if (ptr != local_buffer)
256  {
257  unlatch_read_data (read_latch_page_idx);
258  }
259  else
260  {
261  delete [] local_buffer;
262  }
263 
264  err = wait_for_data (amount + payload_size, STREAM_SKIP);
265 
266  return (err < 0) ? err : read_bytes;
267  }

293  /*
294  * Commit phase of a stream append (all operations are protected by same mutex) :
295  * 1. queue->consume : marks the reserved context as completed and if the context is in the head
296  * of queue, it collapses all elements until a non-completed context is reached (or the queue is emptied);
297  * 2. (only if queue is collapsed) buffer->commit : advances the commit pointer of the bip_buffer
298  * 3. notifies the m_ready_pos_handler handler that new amount is ready to be read
299  * (only if threshold is reached compared to previous notified position)
300  */
301  int multi_thread_stream::commit_append (stream_reserve_context *reserve_context)
302  {
303  stream_reserve_context *last_used_context = NULL;
304  bool collapsed_reserve;
305  char *ptr_commit;
306  int err = NO_ERROR;
307  stream_position new_completed_position = reserve_context->start_pos + reserve_context->reserved_amount;
308  bool signal_data_ready = false;
309 
310  std::unique_lock<std::mutex> ulock (m_buffer_mutex);
311 
312  collapsed_reserve = m_reserved_positions.consume (reserve_context, last_used_context);
313  if (collapsed_reserve)
314  {
315  assert (last_used_context != NULL);
316 
317  assert (new_completed_position <= last_used_context->start_pos + last_used_context->reserved_amount);
318  new_completed_position = last_used_context->start_pos + last_used_context->reserved_amount;
319 
320  ptr_commit = last_used_context->ptr + last_used_context->reserved_amount;
321  m_bip_buffer.commit (ptr_commit);
322 
323  assert (new_completed_position > m_last_committed_pos);
324  m_last_committed_pos = new_completed_position;
325  }
326  if (m_last_committed_pos >= m_serial_read_wait_pos)
327  {
328  signal_data_ready = true;
329  }
330 
331  ulock.unlock ();
332 
333  if (signal_data_ready)
334  {
335  m_serial_read_cv.notify_one ();
336  }
337 
338  /* notify readers of the new completed position */
339  if (m_ready_pos_handler && new_completed_position > m_last_notified_committed_pos + m_trigger_min_to_read_size)
340  {
341  stream_position save_last_notified_commited_pos = m_last_notified_committed_pos;
342  size_t committed_bytes = new_completed_position - m_last_notified_committed_pos;
343 
344  err = m_ready_pos_handler (save_last_notified_commited_pos, committed_bytes);
345  }
346 
347  return err;
348  }

350  /*
351  * Reserve phase of append:
352  * 1. queue->produce : tries adds slot in queue (further data is filled later);
353  * it repeats this steps until successful
354  * 2. buffer->reserve : tries to reserve the amount in buffer;
355  * if unsuccessful, reverts the first step (still under mutex), release the mutex and restarts from (1)
356  * (this allows other threads to unlatch reads, advance commit pointer - to unblock the situation)
357  * 3. increase the m_append_position of stream
358  * 4. init/fills-in reserve context data (reserved position, reserved pointer, amount)
359  * 5. release mutex
360  * 6. if needed, notifies m_filled_stream_handler that too much data was appended since drop position
361  * drop_position is a logical position in past of stream; this is set by an aggregator of "interested" clients
362  * of stream data and instructs the stream up to which position is safe to drop the data;
363  * the interested clients may be : normal readers or the readers which saves the data to disk (for persistence)
364  */
365  char *multi_thread_stream::reserve_with_buffer (const size_t amount, stream_reserve_context *&reserved_context)
366  {
367  char *ptr = NULL;
368 
369  assert (amount > 0);
370 
371  m_buffer_mutex.lock ();
372  while (1)
373  {
374  reserved_context = m_reserved_positions.produce ();
375  if (reserved_context == NULL)
376  {
377  /* this may happen due to a very slow committee, which may cause to fill up the queue */
378  m_buffer_mutex.unlock ();
379  std::this_thread::sleep_for (std::chrono::microseconds (100));
380  m_stat_reserve_queue_spins++;
381  m_buffer_mutex.lock ();
382  continue;
383  }
384 
385  ptr = (char *) m_bip_buffer.reserve (amount);
386  if (ptr != NULL)
387  {
388  break;
389  }
390  m_reserved_positions.undo_produce (reserved_context);
391 
392  m_buffer_mutex.unlock ();
393  std::this_thread::sleep_for (std::chrono::microseconds (100));
394  m_stat_reserve_buffer_spins++;
395  m_buffer_mutex.lock ();
396  }
397 
398  assert (reserved_context != NULL);
399 
400  reserved_context->start_pos = m_append_position;
401 
402  m_append_position += amount;
403 
404  /* set pointer under mutex lock, a "commit" call may collapse until this slot */
405  reserved_context->ptr = ptr;
406  reserved_context->reserved_amount = amount;
407  reserved_context->written_bytes = 0;
408 
409  float stream_fill = stream_fill_factor ();
410  stream_position drop_pos = m_last_dropable_pos;
411  size_t produced_since_drop = m_append_position - m_last_dropable_pos;
412 
413  m_buffer_mutex.unlock ();
414 
415  /* notify that stream content needs to be saved, otherwise it may be overwritten in bip_buffer */
416  if (m_filled_stream_handler && stream_fill >= 1.0f)
417  {
418  m_filled_stream_handler (drop_pos, produced_since_drop);
419  }
420 
421  return ptr;
422  }

424  /*
425  * This is used in context of serial reading and performs an action when not enough data is committed.
426  * The action may be : block (until data is committed) or actively fetch data (from disk or other on-demand producer)
427  * skip_mode argument indicates which action to take after data becomes available.
428  */
429  int multi_thread_stream::wait_for_data (const size_t amount, const STREAM_SKIP_MODE skip_mode)
430  {
431  int err = NO_ERROR;
432 
433  if (m_read_position + amount <= m_last_committed_pos)
434  {
435  if (skip_mode == STREAM_SKIP)
436  {
437  m_read_position += amount;
438  }
439  return NO_ERROR;
440  }
441 
442  m_stat_read_not_enough_data_cnt++;
443 
444  std::unique_lock<std::mutex> local_lock (m_buffer_mutex);
445  m_serial_read_wait_pos = m_read_position + amount;
446  m_serial_read_cv.wait (local_lock,
447  [&] { return m_is_stopped || m_last_committed_pos >= m_serial_read_wait_pos; });
448  m_serial_read_wait_pos = std::numeric_limits<stream_position>::max ();
449  local_lock.unlock ();
450 
451  if (m_is_stopped)
452  {
453  err = ER_STREAM_NO_MORE_DATA;
454  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_STREAM_NO_MORE_DATA, 3, this->name ().c_str (), m_read_position, amount);
455  return err;
456  }
457 
458  assert (m_read_position + amount <= m_last_committed_pos);
459 
460  if (skip_mode == STREAM_SKIP)
461  {
462  assert (err == NO_ERROR);
463  m_read_position += amount;
464  }
465 
466  return err;
467  }

469  /*
470  * starts a range read from a position in stream
471  * this must be in the same scope with a unlatch_read_data call (to make sure the read latch is released)
472  * if portion of range is still in buffer, it reads the available amount, returning the actual read amount and
473  * a read latch id
474  * if no range can be found in bip_buffer, the stream_io is used to provide the range from disk
475  * TODO: read_latch_page_idx type will change to account of reading from stream_file source.
476  *
477  * 1. if needed, wait for data to be committed
478  * 2. if part of range is not in bip_buffer, use stream_io to get range
479  * 3. acquire mutex
480  * 4. bip_buffer.get_read_ranges : get available readable ranges from bip_buffer
481  * 5. convert logical range into physical range
482  * 6. try to read-latch the read range in bip_buffer
483  * 7. release mutex
484  */
485  char *multi_thread_stream::get_data_from_pos (const stream_position &req_start_pos, const size_t amount,
486  size_t &actual_read_bytes, mem::buffer_latch_read_id &read_latch_page_idx)
487  {
488  int err = NO_ERROR;
489  char *ptr = NULL;
490 
491  assert (amount > 0);
492 
493  if (req_start_pos + amount > m_last_committed_pos)
494  {
495  m_stat_read_not_enough_data_cnt++;
496  /* not yet produced */
497  assert (false);
498  err = ER_STREAM_NO_MORE_DATA;
499  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_STREAM_NO_MORE_DATA, 3, this->name ().c_str (), m_read_position,
500  amount);
501 
502  return NULL;
503  }
504 
505  if (req_start_pos < m_oldest_readable_position)
506  {
507  m_stat_read_not_in_buffer_cnt++;
508  /* not in buffer anymore request it from stream_io */
509  /* TODO[arnia] */
510  // m_io->read (req_start_pos, buf, amount);
511  return NULL;
512  }
513 
514  std::unique_lock<std::mutex> ulock (m_buffer_mutex);
515  /* update mapped positions according to buffer pointers */
516  const char *ptr_trail_a;
517  const char *ptr_trail_b;
518  size_t amount_trail_a, amount_trail_b, total_in_buffer;
519 
520  m_bip_buffer.get_read_ranges (ptr_trail_b, amount_trail_b, ptr_trail_a, amount_trail_a);
521  if (amount_trail_a == 0 && amount_trail_b == 0)
522  {
523  /* no readable regions */
524  m_stat_read_no_readable_pos_cnt++;
525  return NULL;
526  }
527 
528  total_in_buffer = amount_trail_b + amount_trail_a;
529 
530  m_oldest_readable_position = m_last_committed_pos - total_in_buffer;
531 
532  if (req_start_pos < m_oldest_readable_position)
533  {
534  m_stat_read_not_in_buffer_cnt++;
535  /* not in buffer anymore request it from stream_io */
536  /* TODO: */
537  // m_io->read (req_start_pos, buf, amount);
538  return NULL;
539  }
540 
541  if (m_last_committed_pos - req_start_pos <= amount_trail_b)
542  {
543  /* the area may be found in trail of region B
544  * this includes case when B does not exit - from start of buffer) */
545  ptr = (char *) ptr_trail_b + amount_trail_b - (m_last_committed_pos - req_start_pos);
546  assert (ptr_trail_b + amount_trail_b > ptr);
547  actual_read_bytes = MIN ((int) amount, ptr_trail_b + amount_trail_b - ptr);
548  }
549  else
550  {
551  ptr = (char *) ptr_trail_a + amount_trail_a - (m_last_committed_pos - req_start_pos - amount_trail_b);
552  assert (ptr_trail_a + amount_trail_a > ptr);
553  actual_read_bytes = MIN ((int) amount, ptr_trail_a + amount_trail_a - ptr);
554  }
555 
556  err = m_bip_buffer.start_read (ptr, actual_read_bytes, read_latch_page_idx);
557  if (err != NO_ERROR)
558  {
559  m_stat_read_buffer_failed_cnt++;
560  return NULL;
561  }
562 
563  return ptr;
564  }
