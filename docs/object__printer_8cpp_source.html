45 void object_printer::describe_comment (const char *comment)
46 {
47  db_value comment_value;
48 
49  assert (comment != NULL);
50 
51  db_make_null (&comment_value);
52  db_make_string_by_const_str (&comment_value, comment);
53 
54  m_buf ("COMMENT ");
55  if (comment != NULL && comment[0] != '\0')
56  {

69 void object_printer::describe_partition_parts (const sm_partition &parts, class_description::type prt_type)
70 {
71  DB_VALUE ele;
72  int setsize, i;
73  db_value_printer obj_print (m_buf);
74 
75  db_make_null (&ele);
76 
77  m_buf ("PARTITION ");
78  describe_identifier (parts.pname, prt_type);
79 
80  switch (parts.partition_type)
81  {
82  case PT_PARTITION_HASH:
83  break;
84  case PT_PARTITION_RANGE:
85  m_buf (" VALUES LESS THAN ");
86  if (!set_get_element (parts.values, 1, &ele))
87  {
88  /* 0:MIN, 1: MAX */
89  if (DB_IS_NULL (&ele))
90  {
91  m_buf ("MAXVALUE");
92  }
93  else
94  {
95  m_buf ("(");
96  obj_print.describe_value (&ele);
97  m_buf (")");
98  }
99  }
100  break;
101  case PT_PARTITION_LIST:
102  m_buf (" VALUES IN (");
103  setsize = set_size (parts.values);
104  for (i = 0; i < setsize; i++)
105  {
106  if (i > 0)
107  {
108  m_buf (", ");
109  }
110  if (set_get_element (parts.values, i, &ele) == NO_ERROR)
111  {
112  obj_print.describe_value (&ele);
113  }
114  }
115  m_buf (")");
116  break;
117  }
118 
119  if (parts.comment != NULL && parts.comment[0] != '\0')
120  {
121  m_buf (" ");
122  describe_comment (parts.comment);
123  }
124 
125  pr_clear_value (&ele);
126 }

155 /*
156  * obj_print_describe_domain() - Describe the domain of an attribute
157  * return: advanced buffer pointer
158  * parser(in) :
159  * buffer(in) : current buffer pointer
160  * domain(in) : domain structure to describe
161  * prt_type(in): the print type: csql schema or show create table
162  * force_print_collation(in): true if collation is printed no matter system
163  * collation
164  *
165  */
166 //--------------------------------------------------------------------------------
167 void object_printer::describe_domain (/*const*/tp_domain &domain, class_description::type prt_type,
168  bool force_print_collation)
169 {
170  TP_DOMAIN *temp_domain;
171  char temp_buffer[27];
172  int precision = 0, idx, count;
173  int has_collation;
174 
175  /* filter first, usually not necessary but this is visible */
176  sm_filter_domain (&domain, NULL);
177 
178  for (temp_domain = &domain; temp_domain != NULL; temp_domain = temp_domain->next)
179  {
180  has_collation = 0;
181  switch (TP_DOMAIN_TYPE (temp_domain))
182  {
183  case DB_TYPE_INTEGER:
184  case DB_TYPE_BIGINT:
185  case DB_TYPE_FLOAT:
186  case DB_TYPE_DOUBLE:
187  case DB_TYPE_BLOB:
188  case DB_TYPE_CLOB:
189  case DB_TYPE_TIME:
190  case DB_TYPE_TIMESTAMP:
191  case DB_TYPE_TIMESTAMPTZ:
192  case DB_TYPE_TIMESTAMPLTZ:
193  case DB_TYPE_DATETIME:
194  case DB_TYPE_DATETIMETZ:
195  case DB_TYPE_DATETIMELTZ:
196  case DB_TYPE_DATE:
197  case DB_TYPE_MONETARY:
198  case DB_TYPE_SUB:
199  case DB_TYPE_POINTER:
200  case DB_TYPE_ERROR:
201  case DB_TYPE_SHORT:
202  case DB_TYPE_VOBJ:
203  case DB_TYPE_OID:
204  case DB_TYPE_NULL:
205  case DB_TYPE_VARIABLE:
206  case DB_TYPE_DB_VALUE:
207  strcpy (temp_buffer, temp_domain->type->name);
208  m_buf ("%s", ustr_upper (temp_buffer));
209  break;
210 
211  case DB_TYPE_OBJECT:
212  if (temp_domain->class_mop != NULL)
213  {
214  describe_identifier (sm_get_ch_name (temp_domain->class_mop), prt_type);
215  }
216  else
217  {
218  m_buf ("%s", temp_domain->type->name);
219  }
220  break;
221 
222  case DB_TYPE_VARCHAR:
223  has_collation = 1;
224  if (temp_domain->precision == TP_FLOATING_PRECISION_VALUE)
225  {
226  m_buf ("STRING");
227  break;
228  }
229  /* fall through */
230  case DB_TYPE_CHAR:
231  case DB_TYPE_NCHAR:
232  case DB_TYPE_VARNCHAR:
233  has_collation = 1;
234  case DB_TYPE_BIT:
235  case DB_TYPE_VARBIT:
236  strcpy (temp_buffer, temp_domain->type->name);
237  if (temp_domain->precision == TP_FLOATING_PRECISION_VALUE)
238  {
239  precision = DB_MAX_STRING_LENGTH;
240  }
241  else
242  {
243  precision = temp_domain->precision;
244  }
245  m_buf ("%s(%d)", ustr_upper (temp_buffer), precision);
246  break;
247 
248  case DB_TYPE_JSON:
249  strcpy (temp_buffer, temp_domain->type->name);
250  ustr_upper (temp_buffer);
251  if (temp_domain->json_validator != NULL)
252  {
253  m_buf ("%s(\'%s\')", temp_buffer, db_json_get_schema_raw_from_validator (temp_domain->json_validator));
254  }
255  else
256  {
257  m_buf (temp_buffer);
258  }
259  break;
260 
261  case DB_TYPE_NUMERIC:
262  strcpy (temp_buffer, temp_domain->type->name);
263  m_buf ("%s(%d,%d)", ustr_upper (temp_buffer), temp_domain->precision, temp_domain->scale);
264  break;
265 
266  case DB_TYPE_SET:
267  case DB_TYPE_MULTISET:
268  case DB_TYPE_SEQUENCE:
269  strcpy (temp_buffer, temp_domain->type->name);
270  ustr_upper (temp_buffer);
271  m_buf ("%s OF ", temp_buffer);
272  if (temp_domain->setdomain != NULL)
273  {
274  if (temp_domain->setdomain->next != NULL && prt_type == class_description::SHOW_CREATE_TABLE)
275  {
276  m_buf ("(");
277  describe_domain (*temp_domain->setdomain, prt_type, force_print_collation);
278  m_buf (")");
279  }
280  else
281  {
282  describe_domain (*temp_domain->setdomain, prt_type, force_print_collation);
283  }
284  }
285  break;
286 
287  case DB_TYPE_ENUMERATION:
288  has_collation = 1;
289  strcpy (temp_buffer, temp_domain->type->name);
290  m_buf ("%s(", ustr_upper (temp_buffer));
291  count = DOM_GET_ENUM_ELEMS_COUNT (temp_domain);
292  for (idx = 1; idx <= count; idx++)
293  {
294  if (idx > 1)
295  {
296  m_buf (", ");
297  }
298  m_buf ("'");
299  m_buf.add_bytes (DB_GET_ENUM_ELEM_STRING_SIZE (&DOM_GET_ENUM_ELEM (temp_domain, idx)),
300  DB_GET_ENUM_ELEM_STRING (&DOM_GET_ENUM_ELEM (temp_domain, idx)));
301  m_buf ("'");
302  }
303  m_buf (")");
304  break;
305 
306  default:
307  break;
308  }
309 
310  if (has_collation && (force_print_collation || temp_domain->collation_id != LANG_SYS_COLLATION))
311  {
312  m_buf (" COLLATE %s", lang_get_collation_name (temp_domain->collation_id));
313  }
314  if (temp_domain->next != NULL)
315  {
316  m_buf (", ");
317  }
318  }
319 }

331 void object_printer::describe_argument (const sm_method_argument &argument, class_description::type prt_type)
332 {

335  /* method and its arguments do not inherit collation from class, collation printing is not enforced */

348 /*
349  * describe_method() - Describes the definition of a method in a class
350  * return: advanced buffer pointer
351  * parser(in) : current buffer pointer
352  * op(in) : class with method
353  * method_p(in) : method to describe
354  * prt_type(in): the print type: csql schema or show create table
355  *
356  */
357 //--------------------------------------------------------------------------------
358 void object_printer::describe_method (const struct db_object &op, const sm_method &method,
359  class_description::type prt_type)
360 {
361  SM_METHOD_SIGNATURE *signature_p;
362 
363  /* assume for the moment that there can only be one signature, simplifies the output */
364  describe_identifier (method.header.name, prt_type);
365 
366  signature_p = method.signatures;
367  if (signature_p == NULL)
368  {
369  m_buf ("()");
370  }
371  else
372  {
373  m_buf ("(");
374  describe_signature (*signature_p, prt_type);
375  m_buf (") ");
376 
377  if (signature_p->value != NULL)
378  {
379  /* make this look more like the actual definition instead strcpy(line, "returns "); line += strlen(line); */
380  describe_argument (*signature_p->value, prt_type);
381  }
382  if (signature_p->function_name != NULL)
383  {
384  m_buf (" FUNCTION ");
385  describe_identifier (signature_p->function_name, prt_type);
386  }
387  }
388 
389  /* add the inheritance source */
390  if (method.class_mop != NULL && method.class_mop != &op)
391  {
392 
393  m_buf ("(from ");
394  describe_identifier (sm_get_ch_name (method.class_mop), prt_type);
395  m_buf (")");
396  }
397 }

399 /*
400  * obj_print_describe_signature() - Describes a method signature
401  * return: advanced buffer pointer
402  * parser(in) :
403  * buffer(in) : current buffer pointer
404  * signature_p(in) : signature to describe
405  * prt_type(in): the print type: csql schema or show create table
406  *
407  */
408 //--------------------------------------------------------------------------------
409 void object_printer::describe_signature (const sm_method_signature &signature, class_description::type prt_type)
410 {
411  SM_METHOD_ARGUMENT *argument_p;
412  int i;
413 
414  for (i = 1; i <= signature.num_args; i++)
415  {
416  for (argument_p = signature.args; argument_p != NULL && argument_p->index != i; argument_p=argument_p->next)
417  ;
418 
419  if (argument_p != NULL)
420  {
421  describe_argument (*argument_p, prt_type);
422  }
423  else
424  {
425  m_buf ("??");
426  }
427 
428  if (i < signature.num_args)
429  {
430  m_buf (", ");
431  }
432  }
433 }

437 /*
438  * obj_print_describe_attribute() - Describes the definition of an attribute
439  * in a class
440  * return: advanced bufbuffer pointer
441  * class_p(in) : class being examined
442  * parser(in) :
443  * attribute_p(in) : attribute of the class
444  * is_inherited(in) : is the attribute inherited
445  * prt_type(in): the print type: csql schema or show create table
446  * obj_print_describe_attribute(in):
447  *
448  */
449 //--------------------------------------------------------------------------------
450 void object_printer::describe_attribute (const struct db_object &cls, const sm_attribute &attribute,
451  bool is_inherited, class_description::type prt_type, bool force_print_collation)
452 {
453  char str_buf[NUMERIC_MAX_STRING_SIZE];
454  db_value_printer printer (m_buf);
455 
456  if (prt_type == class_description::CSQL_SCHEMA_COMMAND)
457  {
458  m_buf ("%-20s ", attribute.header.name);
459  }
460  else
461  {
462  /* prt_type == OBJ_PRINT_SHOW_CREATE_TABLE */
463  describe_identifier (attribute.header.name, prt_type);
464  m_buf (" ");
465  }
466 
467  /* could filter here but do in describe_domain */
468  describe_domain (*attribute.domain, prt_type, force_print_collation);
469 
470  if (attribute.header.name_space == ID_SHARED_ATTRIBUTE)
471  {
472  m_buf (" SHARED ");
473  if (!DB_IS_NULL (&attribute.default_value.value))
474  {
475  printer.describe_value (&attribute.default_value.value);
476  }
477  }
478  else if (attribute.header.name_space == ID_ATTRIBUTE)
479  {
480  if (attribute.flags & SM_ATTFLAG_AUTO_INCREMENT)
481  {
482  m_buf (" AUTO_INCREMENT ");
483 
484  assert (is_inherited || attribute.auto_increment != NULL);
485 
486  if (prt_type == class_description::SHOW_CREATE_TABLE)
487  {
488  DB_VALUE min_val, inc_val;
489  char buff[DB_MAX_NUMERIC_PRECISION * 2 + 4];
490  int offset;
491 
492  assert (attribute.auto_increment != NULL);
493 
494  db_make_null (&min_val);
495  db_make_null (&inc_val);
496 
497  if (db_get (attribute.auto_increment, "min_val", &min_val) != NO_ERROR)
498  {
499  return;
500  }
501 
502  if (db_get (attribute.auto_increment, "increment_val", &inc_val) != NO_ERROR)
503  {
504  pr_clear_value (&min_val);
505  return;
506  }
507 
508  offset = snprintf (buff, DB_MAX_NUMERIC_PRECISION + 3, "(%s, ",
509  numeric_db_value_print (&min_val, str_buf));
510  snprintf (buff + offset, DB_MAX_NUMERIC_PRECISION + 1, "%s)", numeric_db_value_print (&inc_val, str_buf));
511  m_buf (buff);
512 
513  pr_clear_value (&min_val);
514  pr_clear_value (&inc_val);
515  }
516  }
517 
518  if (!DB_IS_NULL (&attribute.default_value.value)
519  || attribute.default_value.default_expr.default_expr_type != DB_DEFAULT_NONE)
520  {
521  const char *default_expr_type_str;
522 
523  m_buf (" DEFAULT ");
524 
525  if (attribute.default_value.default_expr.default_expr_op == T_TO_CHAR)
526  {
527  m_buf ("TO_CHAR(");
528  }
529 
530  default_expr_type_str = db_default_expression_string (attribute.default_value.default_expr.default_expr_type);
531  if (default_expr_type_str != NULL)
532  {
533  m_buf ("%s", default_expr_type_str);
534  }
535  else
536  {
537  assert (attribute.default_value.default_expr.default_expr_op == NULL_DEFAULT_EXPRESSION_OPERATOR);
538  printer.describe_value (&attribute.default_value.value);
539  }
540 
541  if (attribute.default_value.default_expr.default_expr_op == T_TO_CHAR)
542  {
543  if (attribute.default_value.default_expr.default_expr_format)
544  {
545  m_buf (", \'%s\'", attribute.default_value.default_expr.default_expr_format);
546  }
547  m_buf (")");
548  }
549  }
550 
551  if (attribute.on_update_default_expr != DB_DEFAULT_NONE)
552  {
553  const char *default_expr_type_str;
554 
555  m_buf (" ON UPDATE ");
556  default_expr_type_str = db_default_expression_string (attribute.on_update_default_expr);
557  m_buf ("%s", default_expr_type_str);
558  }
559  }
560  else if (attribute.header.name_space == ID_CLASS_ATTRIBUTE)
561  {
562  if (!DB_IS_NULL (&attribute.default_value.value))
563  {
564  m_buf (" VALUE ");
565  printer.describe_value (&attribute.default_value.value);
566  }
567  }
568 
569  if (attribute.flags & SM_ATTFLAG_NON_NULL)
570  {
571  m_buf (" NOT NULL");
572  }
573 
574  if (attribute.class_mop != NULL && attribute.class_mop != &cls)
575  {
576  m_buf (" /* from ");
577  describe_identifier (sm_get_ch_name (attribute.class_mop), prt_type);
578  m_buf (" */");
579  }
580 
581  if (attribute.comment != NULL && attribute.comment[0] != '\0')
582  {
583  m_buf (" ");
584  describe_comment (attribute.comment);
585  }
586 }

588 /*
589  * obj_print_describe_constraint() - Describes the definition of an attribute
590  * in a class
591  * return: advanced buffer pointer
592  * parser(in) :
593  * class_p(in) : class being examined
594  * constraint_p(in) :
595  * prt_type(in): the print type: csql schema or show create table
596  *
597  */
598 //--------------------------------------------------------------------------------
599 void object_printer::describe_constraint (const sm_class &cls, const sm_class_constraint &constraint,
600  class_description::type prt_type)
601 {
602  SM_ATTRIBUTE **attribute_p;
603  const int *asc_desc;
604  const int *prefix_length;
605  int k, n_attrs = 0;
606 
607  if (prt_type == class_description::CSQL_SCHEMA_COMMAND)
608  {
609  switch (constraint.type)
610  {
611  case SM_CONSTRAINT_INDEX:
612  m_buf ("INDEX ");
613  break;
614  case SM_CONSTRAINT_UNIQUE:
615  m_buf ("UNIQUE ");
616  break;
617  case SM_CONSTRAINT_REVERSE_INDEX:
618  m_buf ("REVERSE INDEX ");
619  break;
620  case SM_CONSTRAINT_REVERSE_UNIQUE:
621  m_buf ("REVERSE UNIQUE ");
622  break;
623  case SM_CONSTRAINT_PRIMARY_KEY:
624  m_buf ("PRIMARY KEY ");
625  break;
626  case SM_CONSTRAINT_FOREIGN_KEY:
627  m_buf ("FOREIGN KEY ");
628  break;
629  default:
630  m_buf ("CONSTRAINT ");
631  break;
632  }
633 
634  m_buf ("%s ON %s (", constraint.name, sm_ch_name ((MOBJ) (&cls)));
635  asc_desc = NULL; /* init */
636  if (!SM_IS_CONSTRAINT_REVERSE_INDEX_FAMILY (constraint.type))
637  {
638  asc_desc = constraint.asc_desc;
639  }
640  }
641  else
642  {
643  /* prt_type == OBJ_PRINT_SHOW_CREATE_TABLE */
644  switch (constraint.type)
645  {
646  case SM_CONSTRAINT_INDEX:
647  case SM_CONSTRAINT_REVERSE_INDEX:
648  m_buf (" INDEX ");
649  describe_identifier (constraint.name, prt_type);
650  break;
651  case SM_CONSTRAINT_UNIQUE:
652  case SM_CONSTRAINT_REVERSE_UNIQUE:
653  m_buf (" CONSTRAINT ");
654  describe_identifier (constraint.name, prt_type);
655  m_buf (" UNIQUE KEY ");
656  break;
657  case SM_CONSTRAINT_PRIMARY_KEY:
658  m_buf (" CONSTRAINT ");
659  describe_identifier (constraint.name, prt_type);
660  m_buf (" PRIMARY KEY ");
661  break;
662  case SM_CONSTRAINT_FOREIGN_KEY:
663  m_buf (" CONSTRAINT ");
664  describe_identifier (constraint.name, prt_type);
665  m_buf (" FOREIGN KEY ");
666  break;
667  default:
668  assert (false);
669  break;
670  }
671 
672  m_buf (" (");
673  asc_desc = constraint.asc_desc;
674  }
675 
676  prefix_length = constraint.attrs_prefix_length;
677 
678  /* If the index is a function index, then the corresponding expression is printed at the right position in the
679  * attribute list. Since the expression is not part of the attribute list, when searching through that list we must
680  * check if the position of the expression is reached and then print it. */
681  k = 0;
682  if (constraint.func_index_info)
683  {
684  n_attrs = constraint.func_index_info->attr_index_start + 1;
685  }
686  else
687  {
688  for (attribute_p = constraint.attributes; *attribute_p; attribute_p++)
689  {
690  n_attrs++;
691  }
692  }
693 
694  for (attribute_p = constraint.attributes; k < n_attrs; attribute_p++)
695  {
696  if (constraint.func_index_info && k == constraint.func_index_info->col_id)
697  {
698  if (k > 0)
699  {
700  m_buf (", ");
701  }
702  m_buf ("%s", constraint.func_index_info->expr_str);
703  if (constraint.func_index_info->fi_domain->is_desc)
704  {
705  m_buf (" DESC");
706  }
707  k++;
708  }
709  if (k == n_attrs)
710  {
711  break;
712  }
713  if (k > 0)
714  {
715  m_buf (", ");
716  }
717  describe_identifier ((*attribute_p)->header.name, prt_type);
718 
719  if (prefix_length)
720  {
721  if (*prefix_length != -1)
722  {
723  m_buf ("(%d)", *prefix_length);
724  }
725 
726  prefix_length++;
727  }
728 
729  if (asc_desc)
