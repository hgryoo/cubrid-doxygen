61 /************************************************************************
62 * name: ReallocImproved
63 * arguments:
64 * dest - destination memory pointer
65 * destSize - total size of dest
66 * usedSize - used memory size of dest
67 * allocSize - realloc size
68 * returns/side-effects:
69 * destSize increased
70 * description:
71 * if remain size(destSize - usedSize) < alloc size,
72 * realloc a bunch of bytes
73 * NOTE:
74 ************************************************************************/
75 PUBLIC ERR_CODE
76 ReallocImproved (char **dest, int *destSize, int usedSize, int allocSize)
77 {
78  int remainDestSize;
79 
80  remainDestSize = *destSize - usedSize;
81 
82  /* MUST reserve one byte for extra */
83  while (remainDestSize <= allocSize)
84  {
85  if (*dest == NULL)
86  {
87  *dest = UT_ALLOC (UNIT_MEMORY_SIZE);
88  if (*dest == NULL)
89  {
90  *destSize = 0;
91  return -1;
92  }
93  memset (*dest, 0, UNIT_MEMORY_SIZE);
94  *destSize = UNIT_MEMORY_SIZE;
95  remainDestSize = UNIT_MEMORY_SIZE;
96 
97  }
98  else
99  {
100  *dest = (char *) UT_REALLOC (*dest, *destSize + UNIT_MEMORY_SIZE);
101  if (*dest == NULL)
102  {
103  *destSize = 0;
104  return -1;
105  }
106  memset (*dest + *destSize, 0, UNIT_MEMORY_SIZE);
107 
108  *destSize += UNIT_MEMORY_SIZE;
109  remainDestSize += UNIT_MEMORY_SIZE;
110  }
111  }
112  return 0;
113 }

115 /************************************************************************
116 * name: MemcatImproved
117 * arguments:
118 * dest - destination pointer
119 * usedSize - total alloc size = used size + remain size
120 * src - source memory pointer,
121 * srcSize - source memory size
122 * returns/side-effects:
123 * OK, ERROR
124 * description:
125 * Not string copy.
126 * So this can copy memory stream including null char(\0)
127 * NOTE:
128 * Using this function,
129 * initialize dest to null, usedSize to zero, remainSize to zero
130 * At least, dest to NULL -> InisStr()
131 ************************************************************************/
132 PUBLIC ERR_CODE
133 MemcatImproved (D_STRING * dest, char *src, int srcSize)
134 {
135 
136  if (dest->value == NULL)
137  {
138  dest->totalSize = 0;
139  dest->usedSize = 0;
140  }
141  if (ReallocImproved (&(dest->value), &(dest->totalSize), dest->usedSize, srcSize) < 0)
142  {
143  return -1;
144  }
145  memcpy (dest->value + dest->usedSize, src, srcSize);
146 
147  dest->usedSize += srcSize;
148 
149  return 0;
150 }

383 /************************************************************************
384 * name: element_from_setstring
385 * arguments:
386 * returns/side-effects:
387 * 0 false - no more data, 1 ture - value ok
388 * description:
389 * setstring means set delimited by DELIMITER
390 * eg) "1;;2;;5"
391 * NOTE:
392 ************************************************************************/
393 PUBLIC int
394 element_from_setstring (char **current, char *buf)
395 {
396  char *pt;
397 
398  if (*current == NULL || **current == '\0')
399  {
400  buf[0] = '\0';
401  return 0;
402  }
403 
404 
405  pt = strstr (*current, UT_SET_DELIMITER);
406  if (pt != NULL)
407  {
408  strncpy (buf, *current, pt - *current + 1);
409  buf[pt - *current] = '\0';
410  *current = pt + strlen (UT_SET_DELIMITER);
411  }
412  else
413  {
414  strcpy (buf, *current);
415  *current = *current + strlen (*current);
416  }
417 
418 
419  return 1;

427 ListTailAdd (ST_LIST * head, void *key, void *val, ERR_CODE (*assignFunc) (ST_LIST *, void *, void *))
428 {
429  ST_LIST *newNode;
430  ST_LIST *temp;
431  ERR_CODE retval;
432

434  if (newNode == NULL)
435  return -1;
436 
437  retval = (*assignFunc) (newNode, (void *) key, (void *) val);
438  if (retval < 0)
439  {
440  UT_FREE (newNode);
441  return -1;
442  }
443 
444  /* find tail node */
445  for (temp = head;; temp = temp->next)
446  {
447  if (temp->next == NULL)
448  break;
449  }
450 
451  temp->next = newNode;
452  newNode->next = NULL;
453 
454  return 0;

719 /************************************************************************
720  * name: is_oidstr
721  * arguments:
722  * returns/side-effects:
723  * description:
724  * string oid .
725  * NOTE:
726  * .
727  ************************************************************************/
728 PUBLIC short
729 is_oidstr (char *str)
730 {
731  char *pt;
732  char state;
733 
734 
735  for (pt = str, state = 0; *pt != '\0'; ++pt)
736  {
737  switch (state)
738  {
739  case 0: // start
740  if (*pt == '@')
741  state = 1;
742  else if (*pt == ' ' || *pt == '\t')
743  break;
744  else
745  return _FALSE_;
746 
747  break;
748 
749  case 1: // first digit of page id
750  case 3: // first digit of slot id
751  case 5: // first digit of volume id
752  if (*pt >= '0' && *pt <= '9')
753  ++state;
754  else
755  {
756  return _FALSE_;
757  }
758  break;
759 
760  case 2: // page id
761  case 4: // slot id
762  if (*pt == '|')
763  ++state;
764  else if (*pt < '0' || *pt > '9')
765  {
766  return _FALSE_;
767  }
768  break;
769 
770  case 6:
771  if (*pt == ' ' || *pt == '\t')
772  state = 7;
773  else if (*pt < '0' || *pt > '9')
774  {
775  return _FALSE_;
776  }
777 
778  break;
779  case 7:
780  if (*pt == ' ' || *pt == '\t')
781  break;
782  else
783  return _FALSE_;
784 
785  break;
786 
787  default:
788  return _FALSE_;
789  }
790  }
791 
792  if (state != 6 && state != 7)
793  {
794  return _FALSE_;
795  }
796 
797  return _TRUE_;
798 }

801 /************************************************************************
802  * name: replace_oid
803  * arguments:
804  * returns/side-effects:
805  * oid parameter return.
806  * description:
807  * NOTE:
808  * parameter number 1 base .
809  ************************************************************************/
810 PUBLIC int
811 replace_oid (char *sql_text, char **org_param_pos_pt, char **oid_param_pos_pt, char **oid_param_val_pt)
812 {
813  char *oid_buf = NULL;
814  char oid_param_pos[256];
815  char org_param_pos[256];
816  char oid_param_val[1024];
817  char buf[126];
818  char *pt;
819  char *pt_tmp;
820  char current_param_pos = 0;
821  char oid_param_num = 0;
822 
823  org_param_pos[0] = '\0';
824  oid_param_pos[0] = '\0';
825  oid_param_val[0] = '\0';
826 
827  for (pt = sql_text; *pt != '\0'; ++pt)
828  {
829  switch (*pt)
830  {
831  case '?':
832  ++current_param_pos;
833  sprintf (buf, "%d", current_param_pos);
834  add_element_to_setstring (org_param_pos, buf);
835  break;
836 
837  case '\'':
838  // find the matched string marker
839  pt_tmp = pt;
840  while (pt_tmp = strchr (pt_tmp + 1, '\''))
841  {
842  if (pt_tmp == NULL || *(pt_tmp - 1) != '\\')
843  break;
844 
845  }
846  if (pt_tmp == NULL)
847  break;
848 
849  oid_buf = UT_MAKE_STRING (pt + 1, (int) (pt_tmp - pt - 1));
850  odbc_trim (oid_buf);
851 
852  if (is_oidstr (oid_buf) == _TRUE_)
853  {
854  ++current_param_pos;
855 
856  ++oid_param_num;
857 
858  sprintf (buf, "%d", current_param_pos);
859  add_element_to_setstring (oid_param_pos, buf);
860 
861  sprintf (buf, "%s", oid_buf);
862  add_element_to_setstring (oid_param_val, buf);
863 
864  // replace oid string value to parameter marker
865  *pt = '?';
866  memcpy (pt + 1, pt_tmp + 1, strlen (pt_tmp + 1) + 1); // include Null terminator
867  }
868 
869  NA_FREE (oid_buf);
870  break;
871  }
872  }
873 
874  if (org_param_pos_pt != NULL)
875  *org_param_pos_pt = UT_MAKE_STRING (org_param_pos, -1);
876  if (oid_param_pos_pt != NULL)
877  *oid_param_pos_pt = UT_MAKE_STRING (oid_param_pos, -1);
878  if (oid_param_val_pt != NULL)
879  *oid_param_val_pt = UT_MAKE_STRING (oid_param_val, -1);
880 
881  return oid_param_num;
882 
883 }

886 /************************************************************************
887  * name: odbc_trim
888  * arguments:
889  * returns/side-effects:
890  * description:
891  * . (str )
892  * NOTE:
893  ************************************************************************/
894 PUBLIC char *
895 odbc_trim (char *str)
896 {
897  char *p;
898  char *s;
899 
900  if (str == NULL)
901  return (str);
