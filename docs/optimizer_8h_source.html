72 /*
73  * The PRM_OPTIMIZATION_LEVEL parameter actually encodes (at least) two
74  * different pieces of information: the actual level of optimization
75  * desired, and whether we're supposed to dump a readable version of the
76  * plan. These macros encapsulate the representation decisions for that
77  * information.

80 #define OPTIMIZATION_ENABLED(level) (OPT_LEVEL(level) != 0)

82 #define SIMPLE_DUMP(level) ((level) & 0x100)
83 #define DETAILED_DUMP(level) ((level) & 0x200)
