203  do \
204  { \
205  assert ((p_scan_id) != NULL); \
206  if ((p_scan_id)->type == S_INDX_SCAN) \
207  { \
208  INDX_SCAN_ID * p_idx_scan_id = &(p_scan_id)->s.isid; \
209  if ((FILTER_INFO *) (p_range_filter) != NULL) \
210  { \
211  scan_init_filter_info ((p_range_filter), \
212  &(p_idx_scan_id)->range_pred, \
213  &(p_idx_scan_id)->range_attrs, \
214  (p_scan_id)->val_list, (p_scan_id)->vd, \
215  &(p_idx_scan_id)->cls_oid, 0, NULL, \
216  &(p_idx_scan_id)->num_vstr, \
217  (p_idx_scan_id)->vstr_ids); \
218  } \
219  if ((FILTER_INFO *) (p_key_filter) != NULL) \
220  { \
221  scan_init_filter_info ((p_key_filter), &(p_idx_scan_id)->key_pred, \

224  &(p_idx_scan_id)->cls_oid, \
225  (p_idx_scan_id)->bt_num_attrs, \
226  (p_idx_scan_id)->bt_attr_ids, \
227  &(p_idx_scan_id)->num_vstr, \
228  (p_idx_scan_id)->vstr_ids); \
229  } \
230  if ((FILTER_INFO *) (p_data_filter) != NULL) \
231  { \
232  scan_init_filter_info ((p_data_filter), \
233  &(p_idx_scan_id)->scan_pred, \
234  &(p_idx_scan_id)->pred_attrs, \

240  else if ((p_scan_id)->type == S_HEAP_SCAN) \

252  } \
253  if ((FILTER_INFO *) (p_range_filter) != NULL) \
254  { \
255  memset (p_range_filter, 0, sizeof (FILTER_INFO)); \
256  } \
257  if ((FILTER_INFO *) (p_key_filter) != NULL) \

261  } \
262  } \
