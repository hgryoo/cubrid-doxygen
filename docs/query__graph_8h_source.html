257 struct qo_node
258 {
259  /*
260  * The environment in which this Node is embedded, and to which it
261  * refers for other important information.
262  */
263  QO_ENV *env;
264 
265  /*
266  * The PT_NODE from env->pt_tree that gave rise to this graph
267  * node. entity_spec holds all of the interesting information about
268  * class oids, etc. which we need to communicate with the statistics
269  * manager.
270  */
271  PT_NODE *entity_spec;
272 
273  /*
274  * The segments (and their equivalence classes) that cover (i.e.,
275  * emanate from) this node.
276  */
277  BITSET segs;
278  BITSET eqclasses;
279 
280  /*
281  * The partition (see below) to which this node belongs.
282  */
283  QO_PARTITION *partition;
284 
285  /*
286  * If non-NULL, oid_seg is a segment corresponding to the (virtual)
287  * oid attribute of a class. It is often necessary to retrieve the
288  * oid of an object along with other of its attributes (in order to,
289  * for example, supply enough information for update queries, or for
290  * certain kinds of joins), and it is convenient to be able to treat
291  * this the same as any other kind of attribute.
292  */
293  QO_SEGMENT *oid_seg;
294 
295  /*
296  * The set of all nodes that this node MAY be dependent on if it is
297  * a derived table that is correlated to this level. It will be all
298  * nodes with an idx less than this node. This is coarser than
299  * perhaps necessary, but makes the implementation considerably
300  * cleaner.
301  */
302  BITSET dep_set;
303 
304  /*
305  * The set of sargs that apply to this node. This is an implicit
306  * conjunction; elements produced from a scan of this node will
307  * satisfy all sargs in the set. selectivity is the product of the
308  * selectivities of the sargs in the set.
309  */
310  BITSET sargs;
311  double selectivity;
312 
313  /*
314  * The set of all subqueries that must be evaluated whenever a new
315  * row is produced from this node.
316  */
317  BITSET subqueries;
318 
319 
320  QO_CLASS_INFO *info;
321 
322  /*
323  * Summaries of size information from the various constituent
324  * classes. ncard is the total number of objects represented by this
325  * node, while tcard is the total number of disk pages occupied by
326  * those objects. These silly names are historical artifacts that
327  * correspond to the names in the 197? IBM report on query
328  * optimization in System R.
329  */
330  unsigned long int ncard;
331  unsigned long int tcard;
332 
333  /*
334  * The nominal name of this node, used pretty much only for dumping
335  * debugging information. Right now, we just pick the name of the
336  * first class in the list of classes and subclasses underlying this
337  * node (see above).
338  */
339  const char *class_name;
340 
341  /*
342  * The ordinal id of this node; this is the id used to identify this
343  * node in various bitsets.
344  */
345  int idx;
346 
347  /*
348  * The relative id of this node in partition; this is the id used to
349  * identify this node in join_info vector.
350  */
351  int rel_idx;
352 
353  /*
354  * Indexes can be viewed from a variety of perspectives. Each class
355  * associated with this node has a collection of indexes which are
356  * possible candidates for use in an index scan. For scans across
357  * the class hierarchy, it is necessary to find compatible indexes
358  * at each class in the hierarchy. This hierarchical grouping is
359  * maintained in QO_NODE_INDEX.
360  */
361  QO_NODE_INDEX *indexes;
362  QO_USING_INDEX *using_index; /* indexes specified in USING INDEX clause */
363  /* NULL if no USING INDEX clause in the query */
364 
365  BITSET outer_dep_set; /* outer join dependency; to preserve join sequence */
366  PT_HINT_ENUM hint; /* hint comment contained in given */
367  bool sargable; /* whether sargs are applicable to this node */
368  bool sort_limit_candidate; /* whether this node is a candidate for a SORT_LIMIT plan */
369 };

562  /*

570  /* The env in which this term is embedded. */

572 
573  /*
574  * The nodes referenced by this term (i.e., the heads of all segments
575  * used in this term).
576  *
577  * If this term is a dependent term, this is the set of all nodes on
578  * which the dependent node depends, PLUS the dependent node.
579  */
580  BITSET nodes;
581 
582  /*
583  * The set of segments involved in the expression that gives rise to
584  * this term.
585  */
586  BITSET segments;
587 
588  /*
589  * The selectivity of this term (i.e., the fraction of candidates
590  * that we expect to satisfy the term) when it is not used as an
591  * index. If T is the cartesian product of all nodes referenced by
592  * the term, selectivity is the fraction of the rows in T that
593  * (are expected to) satisfy the term.
594  */
595  double selectivity;
596 
597  /*
598  * The "flavor" of this term. This is determined by analysis of the
599  * segment or where-clause disjunct that gives rise to the term.
600  */
601  QO_TERMCLASS term_class;
602 
603  /*
604  * The rank of this term. used for the same selectivity
605  */
606  int rank;
607 
608  /*
609  * The expression that gave rise to this term. This is either a
610  * conjunct from the conjunctive normal form of the query's search
611  * condition, or a "manufactured" term spawned by a path term.
612  */
613  PT_NODE *pt_expr;
614 
615  /*
616  * The set of all correlated subqueries appearing in this term.
617  */
618  BITSET subqueries;
619 
620  /*
621  * -1 == NO_JOIN iff this term is not suitable as a join predicate
622  * Currently only applicable to path terms.
623  */
624  JOIN_TYPE join_type;
625 
626  /*
627  * can_use_index is non-zero if this term can be implemented with an
628  * index under appropriate circumstances, i.e., one or both sides are
629  * local names, and each side is "independent" of the other (no node
630  * contributes segments to both sides).
631  *
632  * if can_use_index is non-zero, the first "can_use_index" entries in
633  * index_seg hold the segments that can use an index.
634  */
635  int can_use_index;
636  QO_SEGMENT *index_seg[2];
637 
638  /*
639  * The two segments involved in a path join term. It's relatively
640  * important to know which one is the (virtual) oid segment and which
641  * is the real segment.
642  */
643  QO_SEGMENT *seg;
644  QO_SEGMENT *oid_seg;
645 
646  /*
647  * The head and tail nodes joined by this term if it is a join term.
648  * This is redundant, but it is convenient to cache this information
649  * here since it tends to get accessed a lot.
650  *
651  * If the term is a dependency link term, tail identifies the
652  * dependent node.
653  */
654  QO_NODE *head;
