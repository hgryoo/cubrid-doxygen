241  * return: source domain of the given expression

272  pt_set_domain_class_list (result, entity_names, entity_names->info.name.virt_object);

395 /*
396  * pt_get_select_list () - PT_NODE *, a pointer to query's select list
397  * NULL if query is a 'SELECT *' query
398  * return:
399  * parser(in): handle to parser used to process & derive query
400  * query(out): abstract syntax tree form of a SELECT expression
401  */
402 PT_NODE *
403 pt_get_select_list (PARSER_CONTEXT * parser, PT_NODE * query)
404 {
405  PT_NODE *list, *attr;
406  PT_NODE *arg1, *arg2, *attr1, *attr2, *select_list, *col, *next;
407  int cnt1, cnt2;
408  PT_TYPE_ENUM common_type;
409 
410  if (query == NULL)
411  {
412  return NULL;
413  }
414 
415  switch (query->node_type)
416  {
417  default:
418  return NULL;
419 
420  case PT_SELECT:
421  list = query->info.query.q.select.list;
422  if (list == NULL)
423  {
424  return NULL;
425  }
426 
427  /* return the first row of PT_NODE_LIST */
428  if (list->node_type == PT_NODE_LIST)
429  {
430  return list->info.node_list.list;
431  }
432 
433  if (list->node_type == PT_VALUE && list->type_enum == PT_TYPE_STAR)
434  {
435  return NULL;
436  }
437 
438  for (attr = list; attr; attr = attr->next)
439  {
440  if (attr->node_type == PT_NAME && attr->type_enum == PT_TYPE_STAR)
441  {
442  /* found "class_name.*" */
443  return NULL;
444  }
445  }
446 
447  return list;
448 
449  case PT_DIFFERENCE:
450  case PT_INTERSECTION:
451  case PT_UNION:
452  select_list = query->info.query.q.union_.select_list;
453 
454  assert (query->parser_id == parser->id);
455  if (select_list && select_list->parser_id != parser->id)
456  {
457  /*
458  * Union PT_NODE keeps select_list as reference
459  * this case means, this parser copy other parsers tree
460  * but union.info.select_list points old reference
461  *
462  * this function can free & realloc select_list->data_type
463  * so, to prevent modifying (other parser's) original
464  * tree, deep copy select_list in this parser's context
465  */
466  select_list = parser_copy_tree_list (parser, select_list);
467  query->info.query.q.union_.select_list = select_list;
468  }
469 
470  arg1 = pt_get_select_list (parser, query->info.query.q.union_.arg1);
471  arg2 = pt_get_select_list (parser, query->info.query.q.union_.arg2);
472 
473  if (select_list == NULL)
474  {
475  /* if arg1 or arg2 is null, there need be resolved firstly. */
476  if (arg1 == NULL || arg2 == NULL)
477  {
478  return NULL;
479  }
480 
481  cnt1 = pt_length_of_select_list (arg1, EXCLUDE_HIDDEN_COLUMNS);
482  cnt2 = pt_length_of_select_list (arg2, EXCLUDE_HIDDEN_COLUMNS);
483 
484  if (cnt1 != cnt2)
485  {
486  PT_ERRORmf2 (parser, arg1, MSGCAT_SET_PARSER_SEMANTIC, MSGCAT_SEMANTIC_ARITY_MISMATCH, cnt1, cnt2);
487  return NULL;
488  }
489 
490  select_list = parser_copy_tree_list (parser, arg1);
491  if (cnt1 != pt_length_of_select_list (arg1, INCLUDE_HIDDEN_COLUMNS))
492  {
493  /* hidden column is included. get rid of it */
494  for (col = select_list; col && col->next; col = next)
495  {
496  next = col->next;
497  if (next->is_hidden_column)
498  {
499  parser_free_tree (parser, next);
500  col->next = NULL;
501  break;
502  }
503  }
504  }
505  }
506 
507  for (col = select_list, attr1 = arg1, attr2 = arg2; col && attr1 && attr2;
508  col = col->next, attr1 = attr1->next, attr2 = attr2->next)
509  {
510  /* preserve the ENUM type as common type between two ENUMs in UNION, DIFFERENCE and INTERSECTION. The ENUM
511  * domains already have been verified and if they are different an error is generated. */
512  if (attr1->type_enum == attr2->type_enum && attr1->type_enum == PT_TYPE_ENUMERATION)
513  {
514  common_type = PT_TYPE_ENUMERATION;
515  }
516  else
517  {
518  common_type = pt_common_type (attr1->type_enum, attr2->type_enum);
519  }
520 
521  if (col->type_enum == PT_TYPE_NA || col->type_enum == PT_TYPE_NULL)
522  {
523  db_make_null (&col->info.value.db_value);
524  }
525 
526  if ((attr2->type_enum == PT_TYPE_NONE) || (attr2->type_enum == PT_TYPE_NA)
527  || (attr2->type_enum == PT_TYPE_NULL))
528  {
529  /* convert type to that of non-null */
530  if (col->type_enum == PT_TYPE_NA && col->alias_print == NULL)
531  {
532  col->alias_print = pt_append_string (parser, NULL, "na");
533  }
534  else if (col->type_enum == PT_TYPE_NULL && col->alias_print == NULL)
535  {
536  col->alias_print = pt_append_string (parser, NULL, "null");
537  }
538  col->type_enum = attr1->type_enum;
539  if (col->data_type)
540  {
541  parser_free_tree (parser, col->data_type);
542  }
543  col->data_type = parser_copy_tree_list (parser, attr1->data_type);
544 
545  }
546  else if ((attr1->type_enum == PT_TYPE_NONE) || (attr1->type_enum == PT_TYPE_NA)
547  || (attr1->type_enum == PT_TYPE_NULL))
548  {
549  /* convert type to that of non-null */
550  if (col->type_enum == PT_TYPE_NA && col->alias_print == NULL)
551  {
552  col->alias_print = pt_append_string (parser, NULL, "na");
553  }
554  else if (col->type_enum == PT_TYPE_NULL && col->alias_print == NULL)
555  {
556  col->alias_print = pt_append_string (parser, NULL, "null");
557  }
558  col->type_enum = attr2->type_enum;
559  if (col->data_type)
560  {
561  parser_free_tree (parser, col->data_type);
562  }
563  col->data_type = parser_copy_tree_list (parser, attr2->data_type);
564  }
565  else if (PT_IS_PARAMETERIZED_TYPE (common_type)
566  || (common_type != PT_TYPE_NONE && col->type_enum != common_type))
567  {
568  col->type_enum = common_type;
569 
570  if (col->data_type)
571  {
572  parser_free_tree (parser, col->data_type);
573  col->data_type = NULL;
574  }
575  if (PT_IS_COMPLEX_TYPE (common_type))
576  {
577  col->data_type = parser_copy_tree_list (parser, attr2->data_type);
578  }
579  }
580  }
581 
582  if (select_list != query->info.query.q.union_.select_list)
583  {
584  parser_free_tree (parser, query->info.query.q.union_.select_list);
585  }
586 
587  query->info.query.q.union_.select_list = select_list;
588 
589  return select_list;
590  }
