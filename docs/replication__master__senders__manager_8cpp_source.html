155  void master_senders_manager::master_senders_supervisor_task::execute (cubthread::entry &context)
156  {
157 #if defined (SERVER_MODE)
158  static unsigned int check_conn_delay_counter = 0;
159  bool promoted_to_write = false;
160 
161  if (check_conn_delay_counter >
162  SUPERVISOR_DAEMON_CHECK_CONN_MS / SUPERVISOR_DAEMON_DELAY_MS)
163  {
164  std::vector <cubstream::transfer_sender *>::iterator it;
165 
166  rwlock_read_lock (&master_senders_lock);
167  for (it = master_server_stream_senders.begin (); it != master_server_stream_senders.end ();)
168  {
169  if (! (*it)->get_channel ().is_connection_alive ())
170  {
171  if (!promoted_to_write)
172  {
173  rwlock_read_unlock (&master_senders_lock);
174 
175  rwlock_write_lock (&master_senders_lock);
176  it = master_server_stream_senders.begin ();
177 
178  promoted_to_write = true;
179  }
180  else
181  {
182  it = master_server_stream_senders.erase (it);
183  }
184  }
185  else

187  ++it;
188  }
189  }
190  if (!promoted_to_write)
191  {
192  rwlock_read_unlock (&master_senders_lock);
193  }
194  else
195  {
196  rwlock_write_unlock (&master_senders_lock);
197  }
198  check_conn_delay_counter = 0;
199  }
200 
201  check_conn_delay_counter++;
