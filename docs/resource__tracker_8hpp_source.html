20 //
21 // resource_tracker.hpp - interface to track and debug resource usage (allocations, page fixes)
22 //
23 // the purpose of this interface is to track resource usage and detect possible issues:
24 //
25 // 1. resource leaks (at the end of the scope, used resources are not freed/retired)
26 // 2. resource over-usage (a resource exceeds a maximum allowed of usage).
27 // 3. invalid free/retire.
28 //
29 // how to use:
30 //
31 // 1. specialize your tracker based on resource type:
32 //
33 // using my_tracker_type = cubbase::resource_tracker<my_resource_type>;
34 //
35 // note - my_resource_type must be comparable; resource_tracker uses internally a std::map specialized by
36 // provided type and std::less of same type.
37 // todo - template for compare function
38 //
39 // 2. create an instance of your tracker.
40 //
41 // my_tracker_type my_tracker ("My Tracker", true, MAX_RESOURCES, "my res", MAX_REUSE);
42 //
43 // 3. start tracking:
44 //
45 // my_tracker.push_track ();
46 //
47 // 4. track resource; resource over-usage may be detected:
48 //
49 // my_tracker.increment (__FILE__, __LINE__, my_res_to_track, use_count /* default 1 */)
50 //
51 // 5. un-track resource; resource invalid free/retire may be detected:
52 //
53 // my_tracker.decrement (my_res_to_track, use_count /* default 1 */);
54 //
55 // 6. stop tracking; resource leaks may be detected:
56 //
57 // my_tracker.pop_track ();
58 //

201  template <typename Res>
202  void
203  resource_tracker<Res>::increment (const char *filename, const int line, const res_type &res,
204  unsigned use_count /* = 1 */)
205  {
206  if (!m_enabled || m_is_aborted)
207  {
208  return;
209  }
210 
211  if (m_tracked_stack.empty ())
212  {
213  // not active
214  return;
215  }
216 
217  // given use_count cannot exceed max reuse
218  restrack_assert (use_count <= m_max_reuse);
219 
220  // get current level map
221  map_type &map = get_current_map ();
222 
223  // add resource to map or increment the use count of existing entry
224  //
225  // std::map::try_emplace returns a pair <it_insert_or_found, did_insert>
226  // it_insert_or_found = did_insert ? iterator to inserted : iterator to existing
227  // it_insert_or_found is an iterator to pair <res_type, resource_tracker_item>
228  //
229  // as of c++17 we could just do:
230  // auto inserted = map.try_emplace (res, filename, line, use_count);
231  //
232  // however, for now we have to use next piece of code I got from
233  // https://en.cppreference.com/w/cpp/container/map/emplace
234  auto inserted = map.emplace (std::piecewise_construct, std::forward_as_tuple (res),
235  std::forward_as_tuple (filename, line, use_count));
236  if (inserted.second)
237  {
238  // did insert
239  if (++m_resource_count > m_max_resources)
240  {
241  // max size reached. abort the tracker
242  abort ();
243  return;
244  }
245  }
246  else
247  {
248  // already exists
249  // increment use_count
250  inserted.first->second.m_reuse_count += use_count;
251 
252  restrack_assert (inserted.first->second.m_reuse_count <= m_max_reuse);
253  }
254  }

256  template <typename Res>
257  void
258  resource_tracker<Res>::decrement (const res_type &res, unsigned use_count /* = 1 */)
259  {
260  if (!m_enabled || m_is_aborted)
261  {
262  return;
263  }
264 
265  if (m_tracked_stack.empty ())
266  {
267  // not active
268  return;
269  }
270 
271  map_type &map = get_current_map ();
272 
273  // note - currently resources from different levels cannot be mixed. one level cannot free resources from another
274  // level... if we want to do that, I don't think the stacking has any use anymore. we can just keep only
275  // one level and that's that.
276 
277  // std::map::find returns a pair to <res_type, resource_tracker_item>
278  auto tracked = map.find (res);
279  if (tracked == map.end ())
280  {
281  // not found; invalid free
282  restrack_assert (false);
283  }
284  else
285  {
286  // decrement
287  if (use_count > tracked->second.m_reuse_count)
288  {
289  // more than expected; invalid free
290  restrack_assert (false);
291  map.erase (tracked);
292  m_resource_count--;
293  }
294  else
295  {
296  tracked->second.m_reuse_count -= use_count;
297  if (tracked->second.m_reuse_count == 0)
298  {
299  // remove from map
300  map.erase (tracked);
301  m_resource_count--;
302  }
303  }
304  }
305  }

400  template <typename Res>
401  void
402  resource_tracker<Res>::pop_track (void)
403  {
404  if (!m_enabled)
405  {
406  return;
407  }
408 
409  if (m_tracked_stack.empty ())
410  {
411  restrack_assert (false);
412  return;
413  }
414 
415  // get current level map
416  map_type &map = m_tracked_stack.front ();
417  if (!map.empty ())
418  {
419  if (!m_is_aborted)
420  {
421  // resources are leaked
422  dump ();
423  restrack_assert (false);
424  }
425  else
426  {
427  // tracker was aborted; we don't know if there are really leaks
428  }
429  // remove resources
430  m_resource_count -= map.size ();
431  }
432  // remove level from stack
433  m_tracked_stack.pop_front ();
434 
435  // if no levels, resource count should be zero
436  restrack_assert (!m_tracked_stack.empty () || m_resource_count == 0);
437  }
