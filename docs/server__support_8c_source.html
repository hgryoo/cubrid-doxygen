387 /*
388  * css_master_thread() - Master thread, accept/process master process's request
389  * return:
390  * arg(in):
391  */
392 THREAD_RET_T THREAD_CALLING_CONVENTION
393 css_master_thread (void)
394 {
395  int r, run_code = 1, status = 0, nfds;
396  struct pollfd po[] = { {0, 0, 0}, {0, 0, 0} };
397 
398  while (run_code)
399  {
400  /* check if socket has error or client is down */
401  if (!IS_INVALID_SOCKET (css_Pipe_to_master) && css_check_conn (css_Master_conn) < 0)
402  {
403  css_shutdown_conn (css_Master_conn);
404  css_Pipe_to_master = INVALID_SOCKET;
405  }
406 
407  /* clear the pollfd each time before poll */
408  nfds = 0;
409  po[0].fd = -1;
410  po[0].events = 0;
411 
412  if (!IS_INVALID_SOCKET (css_Pipe_to_master))
413  {
414  po[0].fd = css_Pipe_to_master;
415  po[0].events = POLLIN;
416  nfds = 1;
417  }
418 #if defined(WINDOWS)
419  if (!IS_INVALID_SOCKET (css_Server_connection_socket))
420  {
421  po[1].fd = css_Server_connection_socket;
422  po[1].events = POLLIN;
423  nfds = 2;
424  }
425 #endif /* WINDOWS */
426 
427  /* select() sets timeout value to 0 or waited time */
428  r = poll (po, nfds, (prm_get_integer_value (PRM_ID_TCP_CONNECTION_TIMEOUT) * 1000));
429  if (r > 0 && (IS_INVALID_SOCKET (css_Pipe_to_master) || !(po[0].revents & POLLIN))
430 #if defined(WINDOWS)
431  && (IS_INVALID_SOCKET (css_Server_connection_socket) || !(po[1].revents & POLLIN))
432 #endif /* WINDOWS */
433  )
434  {
435  continue;
436  }
437 
438  if (r < 0)
439  {
440  if (!IS_INVALID_SOCKET (css_Pipe_to_master)
441 #if defined(WINDOWS)
442  && ioctlsocket (css_Pipe_to_master, FIONREAD, (u_long *) (&status)) == SockError
443 #else /* WINDOWS */
444  && fcntl (css_Pipe_to_master, F_GETFL, status) == SockError
445 #endif /* WINDOWS */
446  )
447  {
448  css_close_connection_to_master ();
449  break;
450  }
451  }
452  else if (r > 0)
453  {
454  if (!IS_INVALID_SOCKET (css_Pipe_to_master) && (po[0].revents & POLLIN))
455  {
456  run_code = css_process_master_request (css_Pipe_to_master);
457  if (run_code == -1)
458  {
459  css_close_connection_to_master ();
460  /* shutdown message received */
461  run_code = (!HA_DISABLED ())? 0 : 1;
462  }
463 
464  if (run_code == 0 && !HA_DISABLED ())
465  {
466  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_HB_PROCESS_EVENT, 2,
467  "Disconnected with the cub_master and will shut itself down", "");
468  }
469  }
470 #if !defined(WINDOWS)
471  else
472  {
473  break;
474  }
475 
476 #else /* !WINDOWS */
477  if (!IS_INVALID_SOCKET (css_Server_connection_socket) && (po[1].revents & POLLIN))
478  {
479  css_process_new_connection_request ();
480  }
481 #endif /* !WINDOWS */
482  }
483 
484  if (run_code)
485  {
486  if (IS_INVALID_SOCKET (css_Pipe_to_master))
487  {
488  css_reestablish_connection_to_master ();
489  }
490  }
491  else
492  {
493  break;
494  }
495  }
496 
497  css_set_shutdown_timeout (prm_get_integer_value (PRM_ID_SHUTDOWN_WAIT_TIME_IN_SECS));
498 
499 #if defined(WINDOWS)
500  return 0;
501 #else /* WINDOWS */
502  return NULL;
503 #endif /* WINDOWS */
504 }

529  * return:
530  * master_fd(in):
531  * read_fd_var(in):
532  * exception_fd_var(in):
533  */
534 static int
535 css_process_master_request (SOCKET master_fd)
536 {
537  int request, r;
538 
539  r = 1;
540  request = (int) css_get_master_request (master_fd);
541 
542  switch (request)
543  {
544  case SERVER_START_NEW_CLIENT:
545  css_process_new_client (master_fd);
546  break;
547 
548  case SERVER_START_SHUTDOWN:
549  css_process_shutdown_request (master_fd);
550  r = 0;
551  break;
552 
553  case SERVER_STOP_SHUTDOWN:
554  case SERVER_SHUTDOWN_IMMEDIATE:
555  case SERVER_START_TRACING:
556  case SERVER_STOP_TRACING:
557  case SERVER_HALT_EXECUTION:
558  case SERVER_RESUME_EXECUTION:
559  case SERVER_REGISTER_HA_PROCESS:
560  break;
561  case SERVER_GET_HA_MODE:

563  break;
564 #if !defined(WINDOWS)
565  case SERVER_CHANGE_HA_MODE:

569  css_process_get_eof_request (master_fd);

573  break;
574  case SERVER_CONNECT_NEW_SLAVE:
575  css_process_new_slave (master_fd);
576  break;
577 #endif
578  default:
579  /* master do not respond */
580  r = -1;
581  break;
582  }
583

1033 /*
1034  * css_connection_handler_thread () - Accept/process request from one client
1035  * return:
1036  * arg(in):
1037  *
1038  * Note: One server thread per one client
1039  */
1040 static int
1041 css_connection_handler_thread (THREAD_ENTRY * thread_p, CSS_CONN_ENTRY * conn)
1042 {
1043  int n, type, rv, status;
1044  volatile int conn_status;
1045  int css_peer_alive_timeout, poll_timeout;
1046  int max_num_loop, num_loop;
1047  SOCKET fd;
1048  struct pollfd po[1] = { {0, 0, 0} };
1049 
1050  if (thread_p == NULL)
1051  {
1052  thread_p = thread_get_thread_entry_info ();
1053  }
1054 
1055  fd = conn->fd;
1056 
1057  pthread_mutex_unlock (&thread_p->tran_index_lock);
1058 
1059  thread_p->type = TT_SERVER; /* server thread */
1060 
1061  css_peer_alive_timeout = 5000;
1062  poll_timeout = 100;
1063  max_num_loop = css_peer_alive_timeout / poll_timeout;
1064  num_loop = 0;
1065 
1066  status = NO_ERRORS;
1067  /* check if socket has error or client is down */
1068  while (thread_p->shutdown == false && conn->stop_talk == false)
1069  {
1070  /* check the connection */
1071  conn_status = conn->status;
1072  if (conn_status == CONN_CLOSING)
1073  {
1074  /* There's an interesting race condition among client, worker thread and connection handler.
1075  * Please find CBRD-21375 for detail and also see sboot_notify_unregister_client.
1076  *
1077  * We have to synchronize here with worker thread which may be in sboot_notify_unregister_client
1078  * to let it have a chance to send reply to client.
1079  */
1080  rmutex_lock (thread_p, &conn->rmutex);
1081 
1082  conn_status = conn->status;
1083 
1084  rmutex_unlock (thread_p, &conn->rmutex);
1085  }
1086 
1087  if (conn_status != CONN_OPEN)
1088  {
1089  er_log_debug (ARG_FILE_LINE, "css_connection_handler_thread: conn->status (%d) is not CONN_OPEN.",
1090  conn_status);
1091  status = CONNECTION_CLOSED;
1092  break;
1093  }
1094 
1095  po[0].fd = fd;
1096  po[0].events = POLLIN;
1097  po[0].revents = 0;
1098  n = poll (po, 1, poll_timeout);
1099  if (n == 0)
1100  {
1101  if (num_loop < max_num_loop)
1102  {
1103  num_loop++;
1104  continue;
1105  }
1106  num_loop = 0;
1107 
1108 #if !defined (WINDOWS)
1109  /* 0 means it timed out and no fd is changed. */
1110  if (CHECK_CLIENT_IS_ALIVE ())
1111  {
1112  if (css_peer_alive (fd, css_peer_alive_timeout) == false)
1113  {
1114  er_log_debug (ARG_FILE_LINE, "css_connection_handler_thread: css_peer_alive() error\n");
1115  status = CONNECTION_CLOSED;
1116  break;
1117  }
1118  }
1119 
1120  /* check server's HA state */
1121  if (ha_Server_state == HA_SERVER_STATE_TO_BE_STANDBY && conn->in_transaction == false
1122  && css_count_transaction_worker_threads (thread_p, conn->get_tran_index (), conn->client_id) == 0)
1123  {
1124  status = REQUEST_REFUSED;
1125  break;
1126  }
1127 #endif /* !WINDOWS */
1128  continue;
1129  }
1130  else if (n < 0)
1131  {
1132  num_loop = 0;
1133 
1134  if (errno == EINTR)
1135  {
1136  continue;
1137  }
1138  else
1139  {
1140  er_log_debug (ARG_FILE_LINE, "css_connection_handler_thread: select() error\n");
1141  status = ERROR_ON_READ;
1142  break;
1143  }
1144  }
1145  else
1146  {
1147  num_loop = 0;
1148 
1149  if (po[0].revents & POLLERR || po[0].revents & POLLHUP)
1150  {
1151  status = ERROR_ON_READ;
1152  break;
1153  }
1154 
1155  /* read command/data/etc request from socket, and enqueue it to appr. queue */
1156  status = css_read_and_queue (conn, &type);
1157  if (status != NO_ERRORS)
1158  {
1159  er_log_debug (ARG_FILE_LINE, "css_connection_handler_thread: css_read_and_queue() error\n");
1160  break;
1161  }
1162  else
1163  {
1164  /* if new command request has arrived, make new job and add it to job queue */
1165  if (type == COMMAND_TYPE)
1166  {
1167  // push new task
1168  css_push_server_task (*conn);
1169  }
1170  }
1171  }
1172  }
1173 
1174  /* check the connection and call connection error handler */
1175  if (status != NO_ERRORS || css_check_conn (conn) != NO_ERROR)
1176  {
1177  er_log_debug (ARG_FILE_LINE,
1178  "css_connection_handler_thread: status %d conn { status %d transaction_id %d "
1179  "db_error %d stop_talk %d stop_phase %d }\n", status, conn->status, conn->get_tran_index (),
1180  conn->db_error, conn->stop_talk, conn->stop_phase);
1181  rv = pthread_mutex_lock (&thread_p->tran_index_lock);
1182  (*css_Connection_error_handler) (thread_p, conn);
1183  }
1184  else
1185  {
1186  assert (thread_p->shutdown == true || conn->stop_talk == true);
1187  }
1188 
1189  return 0;
1190 }

1255  * initiated by the client.
1256  *

1258  * check if a request is actually sent on the socket. If data was sent
1259  * (not a request), then just return and the scheduler will wake up the
1260  * thread that is blocking for data.

1290  }
1291  }

1303  css_set_thread_info (&thread_ref, -1, 0, local_tran_index, -1);

1307  pthread_mutex_unlock (&thread_ref.tran_index_lock);
1308
