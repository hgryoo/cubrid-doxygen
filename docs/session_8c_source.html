288 /*
289  * session_state_uninit () - uninitialize a session state
290  * returns: error code or NO_ERROR
291  * st(in): state to uninitialize
292  */
293 static int
294 session_state_uninit (void *st)
295 {
296  SESSION_STATE *session = (SESSION_STATE *) st;
297  SESSION_VARIABLE *vcurent = NULL, *vnext = NULL;
298  PREPARED_STATEMENT *pcurent = NULL, *pnext = NULL;
299  THREAD_ENTRY *thread_p = thread_get_thread_entry_info ();
300  SESSION_QUERY_ENTRY *qcurent = NULL, *qnext = NULL;
301  int cnt = 0;
302 
303  if (session == NULL)
304  {
305  return NO_ERROR;
306  }
307 #if defined (SESSION_DEBUG)
308  er_log_debug (ARG_FILE_LINE, "session_free_session %u\n", session->id);
309 #endif /* SESSION_DEBUG */
310 
311  /* free session variables */
312  vcurent = session->session_variables;
313  while (vcurent != NULL)
314  {
315  vnext = vcurent->next;
316  free_session_variable (vcurent);
317  vcurent = vnext;
318  }
319 
320  /* free session statements */
321  pcurent = session->statements;
322  while (pcurent != NULL)
323  {
324  pnext = pcurent->next;
325  session_free_prepared_statement (pcurent);
326  pcurent = pnext;
327  }
328  /* free holdable queries */
329  qcurent = session->queries;
330  while (qcurent)
331  {
332  qnext = qcurent->next;
333  qcurent->next = NULL;
334  session_free_sentry_data (thread_p, qcurent);
335  free_and_init (qcurent);
336  qcurent = qnext;
337  cnt++;
338  }
339 
340  if (session->session_parameters)
341  {
342  sysprm_free_session_parameters (&session->session_parameters);
343  }
344 
345  (void) pgbuf_release_private_lru (thread_p, session->private_lru_index);
346  session->private_lru_index = -1;
347 
348 #if defined (SESSION_DEBUG)
349  er_log_debug (ARG_FILE_LINE, "session_free_session closed %d queries for %d\n", cnt, session->id);
350 #endif /* SESSION_DEBUG */
351 
352  pr_clear_value (&session->cur_insert_id);
353  pr_clear_value (&session->last_insert_id);
354 
355  if (session->trace_stats != NULL)
356  {
357  free_and_init (session->trace_stats);
358  }
359 
360  if (session->plan_string != NULL)
361  {
362  free_and_init (session->plan_string);
363  }
364 
365  return NO_ERROR;
366 }

624 /*
625  * session_state_create () - Create a sessions state with the specified id
626  * return: NO_ERROR or error code
627  * session_id (in) : the session id
628  *
629  * Note: This function creates and adds a sessions state object to the
630  * sessions state memory hash. This function should be called when a
631  * session starts.
632  */
633 int
634 session_state_create (THREAD_ENTRY * thread_p, SESSION_ID * id)
635 {
636  LF_TRAN_ENTRY *t_entry = thread_get_tran_entry (thread_p, THREAD_TS_SESSIONS);
637  SESSION_STATE *session_p = NULL;
638  SESSION_ID next_session_id;
639  int ret;
640 
641  assert (id != NULL);
642 
643 #if defined (SERVER_MODE)
644  if (thread_p && thread_p->conn_entry && thread_p->conn_entry->session_p)
645  {
646  SESSION_ID old_id = thread_p->conn_entry->session_id;
647 
648  /* session_check_session should clear session_p, right? add safe-guard if necessary. */
649 
650  assert (thread_p->conn_entry->session_p->id == old_id);
651 
652  ret = lf_hash_find (t_entry, &sessions.sessions_table, (void *) &old_id, (void **) &session_p);
653  if (ret != NO_ERROR)
654  {
655  assert (false);
656  return ER_FAILED;
657  }
658  if (session_p == NULL)
659  {
660  thread_p->conn_entry->session_id = DB_EMPTY_SESSION;
661  thread_p->conn_entry->session_p = NULL;
662  }
663  else
664  {
665  assert (session_p == thread_p->conn_entry->session_p);
666 
667 #if !defined(NDEBUG)
668  session_state_verify_ref_count (thread_p, session_p);
669 #endif
670  thread_p->conn_entry->session_id = DB_EMPTY_SESSION;
671  thread_p->conn_entry->session_p = NULL;
672  session_state_decrease_ref_count (thread_p, session_p);
673 
674  logtb_set_current_user_active (thread_p, false);
675  pthread_mutex_unlock (&session_p->mutex);
676  }
677  }
678 #endif
679 
680  /* create search key */
681  next_session_id = ATOMIC_INC_32 (&sessions.last_session_id, 1);
682  *id = next_session_id;
683 
684  /* insert new entry into hash table */
685  ret = lf_hash_insert (t_entry, &sessions.sessions_table, (void *) id, (void **) &session_p, NULL);
686  if (ret != NO_ERROR)
687  {
688  return ret;
689  }
690  else if (session_p == NULL)
691  {
692  /* should not happen */
693  assert (false);
694  return ER_FAILED;
695  }
696 
697  /* inserted key might have been incremented; if last_session_id was not modified in the meantime, store the new value
698  */
699  ATOMIC_CAS_32 (&sessions.last_session_id, next_session_id, *id);
700 
701  /* initialize session active time */
702  session_p->active_time = time (NULL);
703 
704 #if defined (SERVER_MODE)
705 #if !defined (NDEBUG)
706  (void) session_state_verify_ref_count (thread_p, session_p);
707 #endif
708  /* increase reference count of new session_p */
709  session_state_increase_ref_count (thread_p, session_p);
710 
711  session_p->private_lru_index = pgbuf_assign_private_lru (thread_p, false, (int) session_p->id);
712  /* set as thread session */
713  session_set_conn_entry_data (thread_p, session_p);
714 
715  logtb_set_current_user_active (thread_p, true);
716 #endif
717 
718  /* done with the entry */
719  pthread_mutex_unlock (&session_p->mutex);
720 
721 #if defined (SESSION_DEBUG)
722  /* debug logging */
723  er_log_debug (ARG_FILE_LINE, "adding session with id %u\n", *id);
724  if (prm_get_bool_value (PRM_ID_ER_LOG_DEBUG) == true)
725  {
726  LF_HASH_TABLE_ITERATOR it;

905 /*
906  * session_remove_expired_sessions () - remove expired sessions
907  * return : NO_ERROR or error code
908  */
909 static int
910 session_remove_expired_sessions (THREAD_ENTRY * thread_p)
911 {
912 #define EXPIRED_SESSION_BUFFER_SIZE 1024
913  LF_TRAN_ENTRY *t_entry = thread_get_tran_entry (thread_p, THREAD_TS_SESSIONS);
914  LF_HASH_TABLE_ITERATOR it;
915  SESSION_STATE *state = NULL;
916  int err = NO_ERROR, success = 0;
917  bool is_expired = false;
918  SESSION_INFO active_sessions;
919  SESSION_ID expired_sid_buffer[EXPIRED_SESSION_BUFFER_SIZE];
920  int n_expired_sids = 0;
921  int sid_index;
922  bool finished = false;
923 
924  active_sessions.count = -1;
925  active_sessions.session_ids = NULL;
926 
927  /* Loop until all expired sessions are removed.
928  * NOTE: We cannot call lf_hash_delete while iterating... lf_hash_delete may have to retry, which also resets the
929  * lock-free transaction. And resetting lock-free transaction can break our iterator.
930  */
931  while (!finished)
932  {
933  lf_hash_create_iterator (&it, t_entry, &sessions.sessions_table);
934  while (true)
935  {
936  state = (SESSION_STATE *) lf_hash_iterate (&it);
937  if (state == NULL)
938  {
939  finished = true;
940  break;
941  }
942 
943  /* iterate next. the mutex lock of the current state will be released */
944  if (session_check_timeout (state, &active_sessions, &is_expired) != NO_ERROR)
945  {
946  pthread_mutex_unlock (&state->mutex);
947  lf_tran_end_with_mb (t_entry);
948  err = ER_FAILED;
949  goto exit_on_end;
950  }
951 
952  if (is_expired)
953  {
954  expired_sid_buffer[n_expired_sids++] = state->id;
955  if (n_expired_sids == EXPIRED_SESSION_BUFFER_SIZE)
956  {
957  /* No more room in buffer */
958 
959  /* Interrupt iteration. */
960  /* Free current entry mutex. */
961  pthread_mutex_unlock (&state->mutex);
962  /* End lock-free transaction started by iterator. */
963  lf_tran_end_with_mb (t_entry);
964  break;
965  }
966  }
967  }
968 
969  /* Remove expired sessions. */
970  for (sid_index = 0; sid_index < n_expired_sids; sid_index++)
971  {
972  err = lf_hash_delete (t_entry, &sessions.sessions_table, &expired_sid_buffer[sid_index], &success);
973  if (err != NO_ERROR)
974  {
975  /* I don't think we can expect errors. */
976  assert (false);
977  goto exit_on_end;
978  }
979  if (!success)
980  {
981  /* we don't have clear operations on this hash table, this shouldn't happen */
982  assert_release (false);
983  err = ER_FAILED;
984  goto exit_on_end;
985  }
986  }
987  n_expired_sids = 0;
988  }
989 
990 exit_on_end:
991  if (active_sessions.session_ids != NULL)
992  {
993  assert (active_sessions.count > 0);
994  free_and_init (active_sessions.session_ids);
995  }
996 
997  return err;
998 
999 #undef EXPIRED_SESSION_BUFFER_SIZE
1000 }

1716  if (stmt_p == NULL)

1786  {

1788  state_p->statements = stmt_p;
1789  }
1790  }

1797
