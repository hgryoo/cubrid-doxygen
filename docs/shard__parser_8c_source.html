63 SP_PARSER_CTX *
64 sp_create_parser (const char *sql_stmt)
65 {
66  SP_PARSER_CTX *parser_p = (SP_PARSER_CTX *) malloc (sizeof (SP_PARSER_CTX));
67  if (parser_p == NULL)
68  {
69  PROXY_LOG (PROXY_LOG_MODE_ERROR, "Not enough virtual memory. failed to malloc parser context. " "(size:%d). \n",
70  sizeof (SP_PARSER_CTX));
71  return NULL;
72  }
73 
74  parser_p->sql_stmt = strdup (sql_stmt);
75  if (parser_p->sql_stmt == NULL)
76  {
77  free (parser_p);
78  PROXY_LOG (PROXY_LOG_MODE_ERROR, "Not enough virtual memory. failed to strdup sql statement. " "(sql_stmt:[%s]).",
79  sql_stmt);
80  return NULL;
81  }
82  parser_p->is_select = false;
83  parser_p->bind_count = 0;
84  parser_p->cursor.pos = parser_p->sql_stmt;
85  parser_p->cursor.token = TT_NONE;
86  parser_p->prv_cursor.pos = parser_p->sql_stmt;
87  parser_p->prv_cursor.token = TT_NONE;
88  parser_p->operator_ = TT_NONE;
89  sp_init_praser_hint_list (&parser_p->list_a);
90  sp_init_praser_hint_list (&parser_p->list_t[BT_STATIC]);
91  sp_init_praser_hint_list (&parser_p->list_t[BT_DYNAMIC]);
92 
93  return parser_p;
94 }

203 bool
204 sp_is_pair_token (SP_TOKEN start_token, SP_TOKEN end_token)
205 {
206  switch (start_token)
207  {
208  case TT_NONE:
209  case TT_WHITESPACE:
210  case TT_NEWLINE:
211  case TT_C_COMMENT_END:
212  case TT_ASSIGN_OP:
213  case TT_IN_OP:
214  case TT_RIGHT_BRAKET:
215  case TT_BIND_CHAR:
216  case TT_SINGLE_QUOTED:
217  case TT_DOUBLE_QUOTED:
218  return start_token == end_token;
219  case TT_LEFT_BRAKET:
220  return end_token == TT_RIGHT_BRAKET;
221  case TT_CSQL_COMMENT:
222  case TT_CPP_COMMENT:
223  return end_token == TT_NEWLINE;
224  case TT_C_COMMENT:
225  case TT_HINT:
226  case TT_SHARD_HINT:
227  return end_token == TT_C_COMMENT_END;
228  }
229  return false;
230 }

271 static int
272 sp_make_sp_value (SP_VALUE * value_p, const char *pos, int length)
273 {
274  SP_TOKEN token;
275  char *p = (char *) left_trim (pos);
276  while (isspace (p[length - 1]))
277  {
278  length--;
279  }
280 
281  sp_get_token_type (p, &token);
282  if (token == TT_SINGLE_QUOTED || token == TT_DOUBLE_QUOTED)
283  {
284  return sp_make_string_sp_value (value_p, p + 1, length - 2);
285  }
286  else
287  {
288  return sp_make_int_sp_value_from_string (value_p, p, length);
289  }
290 }

292 static int
293 sp_make_string_sp_value (SP_VALUE * value_p, const char *pos, int length)
294 {
295  if (length <= 0)
296  {
297  return ER_SP_INVALID_HINT;
298  }
299 
300  if (length > SP_VALUE_INIT_SIZE)
301  {
302  value_p->string.value_ex = (char *) malloc (sizeof (char) * (length + 1));
303  if (value_p->string.value_ex == NULL)
304  {
305  return ER_SP_OUT_OF_MEMORY;
306  }
307  value_p->string.value = value_p->string.value_ex;
308  }
309  memcpy (value_p->string.value, pos, length);
310  value_p->type = VT_STRING;
311  value_p->string.length = length;
312  value_p->string.value[length] = '\0';
313  return NO_ERROR;
314 }

361  }
362

364  hint_p->next_t[BT_STATIC] = NULL;
365  hint_p->next_t[BT_DYNAMIC] = NULL;
366  hint_p->hint_type = HT_NONE;
367  hint_p->bind_type = BT_STATIC;
368  hint_p->bind_position = -1 /* INVALID POSITION */ ;
369  return hint_p;

426 static int
427 sp_parse_sql_internal (SP_PARSER_CTX * parser_p)
428 {
429  int error = NO_ERROR;
430  SP_TOKEN token_type = TT_NONE;
431 
432  while (*parser_p->cursor.pos)
433  {
434  parser_p->cursor.pos = sp_get_token_type (parser_p->cursor.pos, &token_type);
435  if (token_type == TT_NONE)
436  {
437  continue;
438  }
439 
440  if (parser_p->cursor.token == TT_NONE)
441  {
442  if (sp_is_start_token (token_type) == false)
443  {
444  return ER_SP_INVALID_SYNTAX;
445  }
446 
447  parser_p->cursor.token = token_type;
448 
449  error = sp_process_token (parser_p, token_type);
450  if (error != NO_ERROR)
451  {
452  return error;
453  }
454 
455  if (!sp_is_exist_pair_token (parser_p->cursor.token))
456  {
457  sp_copy_cursor_to_prv (parser_p);
458  }
459  }
460  else if (sp_is_pair_token (parser_p->cursor.token, token_type))
461  {
462  sp_copy_cursor_to_prv (parser_p);
463  }
464  }
465 
466  if (parser_p->cursor.token == TT_CSQL_COMMENT || parser_p->cursor.token == TT_CPP_COMMENT)
467  {
468  parser_p->cursor.token = TT_NONE;
469  }
470 
471  return NO_ERROR;
472 }

489 const char *
490 sp_get_token_type (const char *sql, SP_TOKEN * token)
491 {
492  const char *p = sql;
493 
494  *token = TT_NONE;
495 
496  switch (*(p++))
497  {
498  case '\'':
499  *token = TT_SINGLE_QUOTED;
500  break;
501  case '"':
502  *token = TT_DOUBLE_QUOTED;
503  break;
504  case '\\':
505  /* escape char */
506  p += 1;
507  break;
508  case '/':
509  switch (*p)
510  {
511  case '/':
512  p += 1;
513  *token = TT_CPP_COMMENT;
514  break;
515  case '*':
516  if (*(p + 1) == '+')
517  {
518  p += 2;
519  *token = TT_HINT;
520  }
521  else
522  {
523  p += 1;
524  *token = TT_C_COMMENT;
525  }
526  break;
527  }
528  break;
529  case '-':
530  if (*p == '-')
531  {
532  p += 1;
533  *token = TT_CSQL_COMMENT;
534  }
535  break;
536  case '*':
537  if (*p == '/')
538  {
539  p += 1;
540  *token = TT_C_COMMENT_END;
541  }
542  break;
543  case '=':
544  *token = TT_ASSIGN_OP;
545  break;
546  case '\n':
547  *token = TT_NEWLINE;
548  break;
549  case ' ':
550  case '\f':
551  case '\r':
552  case '\t':
553  case '\v':
554  *token = TT_WHITESPACE;
555  if (tolower (*p) == 'i' && tolower (*(p + 1)) == 'n' && (isspace (*(p + 2)) || *(p + 2) == '('))
556  {
557  p += 3;
558  *token = TT_IN_OP;
559  }
560  break;
561  case '?':
562  *token = TT_BIND_CHAR;
563  break;
564  case '(':
565  *token = TT_LEFT_BRAKET;
566  break;
567  case ')':
568  *token = TT_RIGHT_BRAKET;
569  break;
570  }
571 
572  return p;
573 }

575 const char *
576 sp_get_hint_type (const char *sql, SP_HINT_TYPE * hint_type)
577 {
578  sql = left_trim (sql);
579 
580  *hint_type = HT_NONE;
581  if (strncasecmp (sql, "shard_", 6) != 0)
582  {
583  return sql;
584  }
585 
586  sql += 6;
587  if (strncasecmp (sql, "key", 3) == 0)
588  {
589  sql += 3;
590  *hint_type = HT_KEY;
591  }
592  else if (strncasecmp (sql, "val", 3) == 0)
593  {
594  sql += 3;
595  *hint_type = HT_VAL;
596  }
597  else if (strncasecmp (sql, "all", 3) == 0)
598  {
599  sql += 3;
600  *hint_type = HT_ALL;
601  }
602  else if (strncasecmp (sql, "id", 2) == 0)
603  {
604  sql += 2;
605  *hint_type = HT_ID;
606  }
607  return sql;
608 }

610 const char *
611 sp_get_hint_arg (const char *sql, SP_PARSER_HINT * hint_p, int *error)
612 {
613  SP_TOKEN start_token;
614  SP_TOKEN end_token;
615  const char *p, *q;
616 
617  *error = ER_SP_INVALID_HINT;
618 
619  sql = left_trim (sql);
620  sp_get_token_type (sql, &start_token);
621 
622  if (start_token != TT_LEFT_BRAKET)
623  {
624  if (hint_p->hint_type != HT_VAL && hint_p->hint_type != HT_ID)
625  {
626  *error = NO_ERROR;
627  }
628  return sql;
629  }
630 
631  sql = left_trim (sql + 1);
632  q = p = sql;
633 
634  while (*p)
635  {
636  q = sp_get_token_type (p, &end_token);
637  if (sp_is_pair_token (start_token, end_token))
638  {
639  *error = sp_make_sp_value (&hint_p->arg, sql, p - sql);
640  return q;
641  }
642  p = q;
643  }
644 
645  *error = ER_SP_INVALID_HINT;
646  return sql;
647 }

666 static int
667 sp_process_token (SP_PARSER_CTX * parser_p, SP_TOKEN token_type)
668 {
669  int error = NO_ERROR;
670  SP_PARSER_HINT *hint_p = NULL;
671  SP_HINT_TYPE hint_type = HT_NONE;
672 
673  switch (token_type)
674  {
675  case TT_BIND_CHAR:
676  parser_p->bind_count++;
677  break;
678  case TT_HINT:
679  parser_p->cursor.pos = sp_get_hint_type (parser_p->cursor.pos, &hint_type);
680  if (hint_type == HT_NONE)
681  {
682  return NO_ERROR;
683  }
684  parser_p->cursor.token = TT_SHARD_HINT;
685 
686  hint_p = sp_create_parser_hint ();
687  if (hint_p == NULL)
688  {
689  error = ER_SP_OUT_OF_MEMORY;
690  goto PARSER_ERROR;
691  }
692  hint_p->hint_type = hint_type;
693 
694  parser_p->cursor.pos = sp_get_hint_arg (parser_p->cursor.pos, hint_p, &error);
695  if (error != NO_ERROR)
696  {
697  goto PARSER_ERROR;
698  }
699 
700  parser_p->cursor.pos = sp_check_end_of_hint (parser_p->cursor.pos, &error);
701  if (error != NO_ERROR)
702  {
703  goto PARSER_ERROR;
704  }
705 
706  error = sp_is_valid_hint (parser_p, hint_p);
707  if (error != NO_ERROR)
708  {
709  goto PARSER_ERROR;
710  }
711 
712  if (hint_p->hint_type == HT_KEY)
713  {
714  error = sp_get_bind_type_and_value (parser_p, hint_p);
715  if (error != NO_ERROR)
716  {
717  goto PARSER_ERROR;
718  }
719  }
720 
721  if (parser_p->operator_ == TT_ASSIGN_OP)
722  {
723  parser_p->operator_ = TT_NONE;
724  }
725 
726  sp_append_parser_hint_to_ctx (parser_p, hint_p);
727  break;
728  case TT_IN_OP:
729  case TT_ASSIGN_OP:
730  parser_p->operator_ = token_type;
