168 static void
169 proxy_log_write_internal (int level, char *svc_code, bool do_flush, const char *fmt, va_list ap)
170 {
171  char buf[PROXY_LOG_BUFFER_SIZE], *p;
172  int write_len, remain, n;
173 
174  p = buf;
175  remain = PROXY_LOG_BUFFER_SIZE;
176 
177  n = ut_time_string (p, NULL);
178  remain -= n;
179  p += n;
180 
181  if (remain > 0)
182  {
183  if (svc_code == NULL)
184  {
185  n = snprintf (p, remain, " [%s] ", proxy_log_level_str[level]);
186  }
187  else
188  {
189  n = snprintf (p, remain, " [%s][%s] ", proxy_log_level_str[level], svc_code);
190  }
191 
192  if (n < 0)
193  {
194  n = 0;
195  }
196  else if (n >= remain)
197  {
198  n = (remain - 1);
199  }
200 
201  remain -= n;
202  p += n;
203 
204  if (remain > 0)
205  {
206  n = vsnprintf (p, remain, fmt, ap);
207  if (n < 0)
208  {
209  n = 0;
210  }
211  else if (n >= remain)
212  {
213  n = (remain - 1);
214  }
215 
216  remain -= n;
217  p += n;
218  }
219  }
220 
221  write_len = MIN ((int) (p - buf), PROXY_LOG_BUFFER_SIZE);
222  fwrite (buf, write_len, 1, log_fp);
223  fputc ('\n', log_fp);
224 
225  if (do_flush == true)
226  {
227  fflush (log_fp);
228  }
229 }

233 {

255  va_start (ap, fmt);
