808  * return: error code

811 static int
812 init_db_attribute_list (SHOWSTMT_METADATA * md)
813 {
814  int i;
815  DB_DOMAIN *domain;

817 
818  if (md == NULL)

826  if (domain == NULL)
827  {
828  goto on_error;
829  }

836  }
837  att->domain = domain;
838  att->auto_increment = NULL;
839 
840  if (attrs == NULL)
841  {
842  attrs = att;
843  }
844  else
845  {
846  att->order_link = attrs;
847  attrs = att;
848  }
849  }
850 
851  md->showstmt_attrs = attrs;
852  return NO_ERROR;
853 
854 on_error:

859  att->order_link = NULL;

861  classobj_free_attribute (att);
862  }
863
