237 /*
238  * showstmt_next_scan () - scan values from different show statment.
239  * return: NO_ERROR, or ER_code
240  * thread_p(in):
241  * s_id(in):
242  */
243 SCAN_CODE
244 showstmt_next_scan (THREAD_ENTRY * thread_p, SCAN_ID * s_id)
245 {
246  SHOWSTMT_SCAN_ID *stsidp = &s_id->s.stsid;
247  SHOWSTMT_TYPE show_type = stsidp->show_type;
248  NEXT_SCAN_FUNC next_func = NULL;
249  SCAN_CODE code;
250  int i;
251 
252  assert (show_type == show_Requests[show_type].show_type);
253 
254  next_func = show_Requests[show_type].next_func;
255  if (next_func == NULL)
256  {
257  return S_END;
258  }
259 
260  /* free values which need be cleared */
261  for (i = 0; i < stsidp->out_cnt; i++)
262  {
263  pr_clear_value (stsidp->out_values[i]);
264  }
265 
266  code = (*next_func) (thread_p, stsidp->cursor++, stsidp->out_values, stsidp->out_cnt, stsidp->ctx);
267  return code;
268 }

322 /*
323  * showstmt_alloc_array_context () - init context for db_values arrays
324  * return: NO_ERROR, or ER_code
325  * thread_p(in):
326  * num_total(in):
327  * num_col(in):
328  */
329 SHOWSTMT_ARRAY_CONTEXT *
330 showstmt_alloc_array_context (THREAD_ENTRY * thread_p, int num_total, int num_cols)
331 {
332  SHOWSTMT_ARRAY_CONTEXT *ctx;
333 
334  ctx = (SHOWSTMT_ARRAY_CONTEXT *) db_private_alloc (thread_p, sizeof (SHOWSTMT_ARRAY_CONTEXT));
335  if (ctx == NULL)
336  {
337  return NULL;
338  }
339 
340  ctx->num_used = 0;
341  ctx->num_cols = num_cols;
342  ctx->num_total = num_total;
343  ctx->tuples = (DB_VALUE **) db_private_alloc (thread_p, sizeof (DB_VALUE *) * num_total);
344  if (ctx->tuples == NULL)
345  {
346  goto on_error;
347  }
348 
349  memset (ctx->tuples, 0, sizeof (DB_VALUE *) * num_total);
350  return ctx;
351 
352 on_error:
353  if (ctx != NULL)
354  {
355  db_private_free (thread_p, ctx);
356  }
357  return NULL;
358 }

360 /*
361  * showstmt_free_array_context () - free context for db_values arrays
362  * return: NO_ERROR, or ER_code
363  * thread_p(in):
364  * ctx(in):
365  */
366 void
367 showstmt_free_array_context (THREAD_ENTRY * thread_p, SHOWSTMT_ARRAY_CONTEXT * ctx)
368 {
369  int i, j;
370  DB_VALUE *vals;
371 
372  assert (ctx != NULL);
373 
374  for (i = 0; i < ctx->num_used; i++)
375  {
376  vals = ctx->tuples[i];
377  for (j = 0; j < ctx->num_cols; j++)
378  {
379  pr_clear_value (&vals[j]);
380  }
381 
382  db_private_free (thread_p, vals);
383  }
384 
385  db_private_free (thread_p, ctx->tuples);
386  db_private_free (thread_p, ctx);
387 }

389 /*
390  * showstmt_alloc_tuple_in_context () - alloc and return next tuple from context
391  * return: tuple pointer
392  * thread_p(in):
393  * ctx(in):
394  */
395 DB_VALUE *
396 showstmt_alloc_tuple_in_context (THREAD_ENTRY * thread_p, SHOWSTMT_ARRAY_CONTEXT * ctx)
397 {
398  int i, num_new_total;
399  DB_VALUE **new_tuples = NULL;
400  DB_VALUE *vals = NULL;
401 
402  if (ctx->num_used == ctx->num_total)
403  {
404  num_new_total = (int) (ctx->num_total * 1.5 + 1);
405  new_tuples = (DB_VALUE **) db_private_realloc (thread_p, ctx->tuples, sizeof (DB_VALUE *) * num_new_total);
406  if (new_tuples == NULL)
407  {
408  return NULL;
409  }
410 
411  memset (new_tuples + ctx->num_total, 0, sizeof (DB_VALUE *) * (num_new_total - ctx->num_total));
412 
413  ctx->tuples = new_tuples;
414  ctx->num_total = num_new_total;
415  }
416 
417  vals = (DB_VALUE *) db_private_alloc (thread_p, sizeof (DB_VALUE) * ctx->num_cols);
418  if (vals == NULL)
419  {
420  return NULL;
421  }
422  for (i = 0; i < ctx->num_cols; i++)
423  {
424  db_make_null (&vals[i]);
425  }
426 
427  ctx->tuples[ctx->num_used++] = vals;
428  return vals;
429 }

431 /*
432  * showstmt_array_next_scan () - next scan function for array
433  * return: NO_ERROR, or ER_code
434  * thread_p(in):
435  * cursor(in):
436  * out_values(in/out):
437  * out_cnt(in):
438  * ptr(in):
439  */
440 static SCAN_CODE
441 showstmt_array_next_scan (THREAD_ENTRY * thread_p, int cursor, DB_VALUE ** out_values, int out_cnt, void *ptr)
442 {
443  SHOWSTMT_ARRAY_CONTEXT *ctx = (SHOWSTMT_ARRAY_CONTEXT *) ptr;
444  DB_VALUE *vals = NULL;
445  int i;
446 
447  if (ctx == NULL || cursor < 0 || cursor >= ctx->num_used)
448  {
449  return S_END;
450  }
451 
452  assert (out_cnt == ctx->num_cols);
453 
454  vals = ctx->tuples[cursor];
455 
456  for (i = 0; i < ctx->num_cols; i++)
457  {
458  pr_clone_value (&vals[i], out_values[i]);
459  }
460 
461  return S_SUCCESS;
462 }
