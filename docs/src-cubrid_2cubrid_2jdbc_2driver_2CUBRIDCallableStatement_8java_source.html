604  private void doRegisterOutParameter(int index, int sqlType) throws SQLException {
605  checkIsOpen();
606  synchronized (u_stmt) {

609  }
610 
611  checkBindError();
612  }

614  private void beforeGetValue(int index) throws SQLException {
615  if (index < 0 || index > u_stmt.getParameterCount()) {
616  throw con.createCUBRIDException(CUBRIDJDBCErrorCode.invalid_index, null);
617  }
618 
619  synchronized (u_stmt) {
620  u_stmt.fetch();
621  error = u_stmt.getRecentError();
622  }
623 
624  switch (error.getErrorCode()) {
625  case UErrorCode.ER_NO_ERROR:
626  break;
627  default:
628  throw con.createCUBRIDException(error);
629  }
630  }
631 
632  void checkGetXXXError() throws SQLException {
633  switch (error.getErrorCode()) {
634  case UErrorCode.ER_NO_ERROR:
635  was_null = false;
636  break;
637  case UErrorCode.ER_WAS_NULL:
638  was_null = true;
639  break;
640  default:
641  throw con.createCUBRIDException(error);
642  }
643  }
