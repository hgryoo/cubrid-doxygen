106  static byte getObjArrBaseDBtype(Object values) {
107  if (values instanceof String[])
108  return UUType.U_TYPE_VARCHAR;
109  else if (values instanceof Byte[])
110  return UUType.U_TYPE_SHORT;
111  else if (values instanceof byte[][])
112  return UUType.U_TYPE_VARBIT;
113  else if (values instanceof Boolean[])
114  return UUType.U_TYPE_BIT;
115  else if (values instanceof Short[])
116  return UUType.U_TYPE_SHORT;
117  else if (values instanceof Integer[])
118  return UUType.U_TYPE_INT;
119  else if (values instanceof Long[])
120  return UUType.U_TYPE_BIGINT;
121  else if (values instanceof Double[])
122  return UUType.U_TYPE_DOUBLE;
123  else if (values instanceof Float[])
124  return UUType.U_TYPE_FLOAT;
125  else if (values instanceof BigDecimal[])
126  return UUType.U_TYPE_NUMERIC;
127  else if (values instanceof Date[])
128  return UUType.U_TYPE_DATE;

131  else if (values instanceof Timestamp[]) {
132  for (int i = 0; i < ((Object[]) values).length; i++) {
133  if (!CUBRIDTimestamp

135  return UUType.U_TYPE_DATETIME;

137  }
138  return UUType.U_TYPE_TIMESTAMP;
139  } else if (values instanceof CUBRIDOID[])
140  return UUType.U_TYPE_OBJECT;
141  else if (values instanceof CUBRIDBlob[])
142  return UUType.U_TYPE_BLOB;
143  else if (values instanceof CUBRIDClob[])

149  static byte getObjectDBtype(Object value) {
150  if (value == null)
151  return UUType.U_TYPE_NULL;
152  else if (value instanceof String)
153  return UUType.U_TYPE_STRING;
154  else if (value instanceof Byte)
155  return UUType.U_TYPE_SHORT;
156  else if (value instanceof byte[])
157  return UUType.U_TYPE_VARBIT;
158  else if (value instanceof Boolean)
159  return UUType.U_TYPE_BIT;
160  else if (value instanceof Short)
161  return UUType.U_TYPE_SHORT;
162  else if (value instanceof Integer)
163  return UUType.U_TYPE_INT;
164  else if (value instanceof Long)
165  return UUType.U_TYPE_BIGINT;
166  else if (value instanceof Double)
167  return UUType.U_TYPE_DOUBLE;
168  else if (value instanceof Float)
169  return UUType.U_TYPE_FLOAT;
170  else if (value instanceof BigDecimal || value instanceof Long)
171  return UUType.U_TYPE_NUMERIC;
172  else if (value instanceof Date)
173  return UUType.U_TYPE_DATE;
174  else if (value instanceof Time)
175  return UUType.U_TYPE_TIME;
176  else if (value instanceof CUBRIDTimestamptz) {
177  if (CUBRIDTimestamp.isTimestampType((Timestamp) value)) {
178  return UUType.U_TYPE_TIMESTAMPTZ;
179  }
180  return UUType.U_TYPE_DATETIMETZ;
181  } else if (value instanceof Timestamp) {
182  if (CUBRIDTimestamp.isTimestampType((Timestamp) value)) {
183  return UUType.U_TYPE_TIMESTAMP;
184  }
185  return UUType.U_TYPE_DATETIME;
186  } else if (value instanceof CUBRIDOID)
187  return UUType.U_TYPE_OBJECT;
188  else if (value instanceof CUBRIDBlob)
189  return UUType.U_TYPE_BLOB;
190  else if (value instanceof CUBRIDClob)
191  return UUType.U_TYPE_CLOB;
192  else if (value instanceof Object[])
193  return UUType.U_TYPE_SEQUENCE;
194  else if (value instanceof CUBRIDBinaryString)
195  return UUType.U_TYPE_VARCHAR;
196  else
197  return UUType.U_TYPE_NULL;
198  }
