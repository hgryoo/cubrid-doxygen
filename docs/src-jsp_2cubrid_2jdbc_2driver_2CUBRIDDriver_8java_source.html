127  public Connection connect(String url, Properties info) throws SQLException {
128  if (!acceptsURL(url)) {
129  return null;
130  }
131 
132  if (url.toLowerCase().startsWith(JDBC_DEFAULT_CONNECTION)) {
133  return defaultConnection();
134  }
135 
136  Pattern pattern = Pattern.compile(URL_PATTERN, Pattern.CASE_INSENSITIVE);
137  Matcher matcher = pattern.matcher(url);
138  if (!matcher.find()) {
139  throw new CUBRIDException(CUBRIDJDBCErrorCode.invalid_url, url, null);
140  }
141 
142  String match = matcher.group();
143  if (!match.equals(url)) {
144  throw new CUBRIDException(CUBRIDJDBCErrorCode.invalid_url, url, null);
145  }
146 
147  String dummy;
148  String host = matcher.group(2);
149  String portString = matcher.group(3);
150  String db = matcher.group(4);
151  String user;
152  String pass;
153  String prop = matcher.group(7);
154  int port = default_port;
155 
156  UConnection u_con;
157  String resolvedUrl;
158  ConnectionProperties connProperties;
159 
160  if (host == null || host.length() == 0) {
161  host = default_hostname;
162  }
163 
164  if (portString == null || portString.length() == 0) {
165  port = default_port;
166  } else {
167  port = Integer.parseInt(portString);
168  }
169 
170  connProperties = new ConnectionProperties();
171  connProperties.setProperties(prop);
172 
173  user = info.getProperty("user");
174  if (user == null) {
175  user = matcher.group(5);
176  }
177 
178  pass = info.getProperty("password");
179  if (pass == null) {
180  pass = matcher.group(6);
181  }
182 
183  resolvedUrl = "jdbc:cubrid:" + host + ":" + port + ":" + db + ":" + user + ":********:";
184  if (prop != null) {
185  resolvedUrl += prop;
186  }
187 
188  connProperties.setProperties(info);
189 
190  dummy = connProperties.getAltHosts();
191  if (dummy != null) {
192  ArrayList<String> altHostList = new ArrayList<String>();
193  altHostList.add(host + ":" + port);
194 
195  StringTokenizer st = new StringTokenizer(dummy, ",", false);
196  while (st.hasMoreTokens()) {
197  altHostList.add(st.nextToken());
198  }
199 
200  if (connProperties.getConnLoadBal()) {
201  Collections.shuffle(altHostList);
202  }
203  try {
204  u_con = UJCIManager.connect(altHostList, db, user, pass, resolvedUrl);
205  } catch (CUBRIDException e) {
206  throw e;
207  }
208  } else {
209  try {
210  u_con = UJCIManager.connect(host, port, db, user, pass, resolvedUrl);
211  } catch (CUBRIDException e) {
212  throw e;
213  }
214  }
215 
216  u_con.setCharset(connProperties.getCharSet());
217  u_con.setZeroDateTimeBehavior(connProperties.getZeroDateTimeBehavior());
218  u_con.setResultWithCUBRIDTypes(connProperties.getResultWithCUBRIDTypes());
219 
220  u_con.setConnectionProperties(connProperties);
221  u_con.tryConnect();
222  return new CUBRIDConnection(u_con, url, user);
223  }

247  public boolean acceptsURL(String url) throws SQLException {
248  if (url == null) {
249  return false;
250  }
251 
252  String urlHeader = CUBRID_JDBC_URL_HEADER;
253  String className = CUBRIDDriver.class.getName();
254  if (className.matches(".*mysql.*")) {
255  urlHeader += "-mysql:";
256  } else if (className.matches(".*oracle.*")) {
257  urlHeader += "-oracle:";
258  }
259  else {
260  urlHeader += ":";
261  }
262 
263 
264  if (url.toLowerCase().startsWith(urlHeader)) {
265  return true;
266  }
267 
268  if (url.toLowerCase().startsWith(JDBC_DEFAULT_CONNECTION)) {
269  return true;
270  }
271 
272  return false;
273  }
