296  public boolean getMoreResults() throws SQLException {
297  try {
298  checkIsOpen();
299 
300  synchronized (con) {
301  synchronized (this) {
302  checkIsOpen();
303 
304  if (current_result_set != null) {
305  current_result_set.close();
306  current_result_set = null;
307  }
308 
309  if (completed) {
310  update_count = -1;
311  return false;
312  }
313 
314  if (result_index == result_info.length) {
315  if (u_stmt.getCommandType() != CUBRIDCommandType.CUBRID_STMT_CALL_SP)
316  complete();
317  update_count = -1;
318  return false;
319  }
320 
321  if (result_index != 0) {
322  u_stmt.nextResult();
323  error = u_stmt.getRecentError();
324  switch (error.getErrorCode()) {
325  case UErrorCode.ER_NO_ERROR:
326  break;
327  default:
328  throw con.createCUBRIDException(error);
329  }
330  }
331 
332  boolean result_type = result_info[result_index]
333  .isResultSet();
334 
335  if (result_type) {
336  int rs_type = type;
337  int rs_concurrency = concurrency;
338  if (type == ResultSet.TYPE_SCROLL_SENSITIVE
339  && u_stmt.isOIDIncluded() == false)
340  rs_type = ResultSet.TYPE_SCROLL_INSENSITIVE;
341  if (concurrency == ResultSet.CONCUR_UPDATABLE
342  && u_stmt.isOIDIncluded() == false)
343  rs_concurrency = ResultSet.CONCUR_READ_ONLY;
344  current_result_set = new CUBRIDResultSet(con, this,
345  rs_type, rs_concurrency, is_holdable);
346  update_count = -1;
347  } else {
348  update_count = result_info[result_index]
349  .getResultCount();
350  }
351 
352  result_index++;
353  return result_type;
354  }
355  }
356  } catch (NullPointerException e) {
357  throw con.createCUBRIDException(CUBRIDJDBCErrorCode.statement_closed, e);
358  }

432  complete();
433  }

437  batch_querys[i] = (String) batchs.get(i);

439  batch_results = u_con.batchExecute(batch_querys, query_timeout);
440  setShardId(u_con.getShardId());
441 
442  error = u_con.getRecentError();
443 
444  batchs.clear();

446  case UErrorCode.ER_NO_ERROR:
447  break;
448  default:
449  throw con.createCUBRIDException(error);

451 
452  con.autoCommit();

455  }
