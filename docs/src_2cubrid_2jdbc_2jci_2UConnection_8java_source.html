441  synchronized public UBatchResult batchExecute(String batchSqlStmt[], int queryTimeout) {
442  errorHandler = new UError(this);
443  setShardId(UShardInfo.SHARD_ID_INVALID);
444 
445  if (isClosed == true) {
446  errorHandler.setErrorCode(UErrorCode.ER_IS_CLOSED);
447  return null;
448  }
449  if (batchSqlStmt == null) {
450  errorHandler.setErrorCode(UErrorCode.ER_INVALID_ARGUMENT);
451  return null;
452  }
453  try {
454  checkReconnect();
455  if (errorHandler.getErrorCode() != UErrorCode.ER_NO_ERROR)
456  return null;
457 
458  outBuffer.newRequest(output, UFunctionCode.EXECUTE_BATCH_STATEMENT);
459  outBuffer.addByte(getAutoCommit() ? (byte) 1 : (byte) 0);
460  if (protoVersionIsAbove(UConnection.PROTOCOL_V4)) {
461  long remainingTime = getRemainingTime(queryTimeout * 1000);
462  if (queryTimeout > 0 && remainingTime <= 0) {
463  throw createJciException(UErrorCode.ER_TIMEOUT);
464  }
465  outBuffer.addInt((int) remainingTime);
466  }
467 
468  for (int i = 0; i < batchSqlStmt.length; i++) {
469  if (batchSqlStmt[i] != null)
470  outBuffer.addStringWithNull(batchSqlStmt[i]);
471  else
472  outBuffer.addNull();
473  }
474 
475  UInputBuffer inBuffer;
476  inBuffer = send_recv_msg();
477 
478  int result;
479  UBatchResult batchResult = new UBatchResult(inBuffer.readInt());
480  for (int i = 0; i < batchResult.getResultNumber(); i++) {
481  batchResult.setStatementType(i, inBuffer.readByte());
482  result = inBuffer.readInt();
483  if (result < 0) {
484  int err_code = inBuffer.readInt();
485  batchResult.setResultError(i, err_code, inBuffer.readString(
486  inBuffer.readInt(), UJCIManager.sysCharsetName));
487  }
488  else {
489  batchResult.setResult(i, result);
490  // jci 3.0
491  inBuffer.readInt();
492  inBuffer.readShort();
493  inBuffer.readShort();
494  }
495  }
496 
497  if (protoVersionIsAbove(UConnection.PROTOCOL_V5)) {
498  setShardId(inBuffer.readInt());
499  }
500 
501  update_executed = true;
502  return batchResult;
503  } catch (UJciException e) {
504  logException(e);
505  e.toUError(errorHandler);
506  } catch (IOException e) {
507  logException(e);

591  synchronized public void endTransaction(boolean type) {
592  errorHandler = new UError(this);
593 
594  if (isClosed == true) {
595  errorHandler.setErrorCode(UErrorCode.ER_IS_CLOSED);
596  return;
597  }
598 
599  if (needReconnection == true)
600  return;
601 
602  try {
603  if (client != null
604  && getCASInfoStatus() != CAS_INFO_STATUS_INACTIVE) {
605  setBeginTime();
606  checkReconnect();
607  if (errorHandler.getErrorCode() != UErrorCode.ER_NO_ERROR)
608  return;
609 
610  if (getCASInfoStatus() == CAS_INFO_STATUS_ACTIVE) {
611  if (UJCIUtil.isConsoleDebug()) {
612  if (!lastAutoCommit || isAutoCommitBySelf
613  || type == false) {
614  // this is ok;
615  } else {
616  // we need check
617  throw new Exception("Check It Out!");
618  }
619  }
620  outBuffer.newRequest(output, UFunctionCode.END_TRANSACTION);
621  outBuffer.addByte((type == true) ? END_TRAN_COMMIT
622  : END_TRAN_ROLLBACK);
623 
624  send_recv_msg();
625  if (lastAutoCommit) {
626  turnOffAutoCommitBySelf();
627  }
628  }
629  }
630  } catch (UJciException e) {
631  logException(e);
632  e.toUError(errorHandler);
633  } catch (IOException e) {
634  logException(e);
635  errorHandler.setErrorCode(UErrorCode.ER_COMMUNICATION);
636  } catch (Exception e) {
637  logException(e);
638  errorHandler.setErrorMessage(UErrorCode.ER_UNKNOWN, e.getMessage());

643  * errorHandler.clear();
644  */

648  int reconnectTime = connectionProperties.getReconnectTime();
649  UUnreachableHostList unreachableHosts = UUnreachableHostList.getInstance();
650

652  && currentTime - lastFailureTime > reconnectTime) {

654  keepConnection = false;
655  lastFailureTime = 0;
656  }
