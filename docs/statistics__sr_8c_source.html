74 /*
75  * xstats_update_statistics () - Updates the statistics for the objects
76  * of a given class
77  * return:
78  * class_id(in): Identifier of the class
79  * with_fullscan(in): true iff WITH FULLSCAN
80  *
81  * Note: It first retrieves the whole catalog information about this class,
82  * including all possible forms of disk representations for the instance
83  * objects. Then, it performs a complete pass on the heap file of the
84  * class, reading in all of the instance objects one by one and
85  * calculating the ranges of numeric attribute values (ie. min. & max.
86  * values for each numeric attribute).
87  *
88  * During this pass on the heap file, these values are maintained
89  * separately for objects with the same representation. Each minimum and
90  * maximum value is initialized when the first instance of the class
91  * with the corresponding representation is encountered. These values are
92  * continually updated as attribute values exceeding the known range are
93  * encountered. At the end of this pass, these individual ranges for
94  * each representation are uniformed in the last (the current)
95  * representation, building the global range values for the attributes
96  * of the class. Then, the btree statistical information is obtained for
97  * each attribute that is indexed and stored in this final representation
98  * structure. Finally, a new timestamp is obtained for these class
99  * statistics and they are stored to disk within the catalog structure
100  * for the last class representation.
101  */

105  CLS_INFO *cls_info_p = NULL;
106  REPR_ID repr_id;
107  DISK_REPR *disk_repr_p = NULL;

120 
121  OID_SET_NULL (&dir_oid);

144  goto error;
145  }
146 
147  catalog_access_info.class_oid = class_id_p;
148  catalog_access_info.dir_oid = &dir_oid;
149  catalog_access_info.class_name = class_name;
150  error_code = catalog_start_access_with_dir_oid (thread_p, &catalog_access_info, S_LOCK);
151  if (error_code != NO_ERROR)
152  {
153  goto error;
154  }
155 
156  cls_info_p = catalog_get_class_info (thread_p, class_id_p, &catalog_access_info);
157  if (cls_info_p == NULL)
158  {
159  goto error;
160  }
161 
162  (void) catalog_end_access_with_dir_oid (thread_p, &catalog_access_info, NO_ERROR);
163 
164  er_set (ER_NOTIFICATION_SEVERITY, ARG_FILE_LINE, ER_LOG_STARTED_TO_UPDATE_STATISTICS, 4,
165  class_name ? class_name : "*UNKNOWN-CLASS*", class_id_p->volid, class_id_p->pageid, class_id_p->slotid);
166 
167  /* if class information was not obtained */
168  if (cls_info_p->ci_hfid.vfid.fileid < 0 || cls_info_p->ci_hfid.vfid.volid < 0)
169  {
170  /* The class does not have a heap file (i.e. it has no instances); so no statistics can be obtained for this
171  * class; just set to 0 and return. */
172 
173  cls_info_p->ci_tot_pages = 0;
174  cls_info_p->ci_tot_objects = 0;
175 
176  error_code = catalog_start_access_with_dir_oid (thread_p, &catalog_access_info, X_LOCK);
177  if (error_code != NO_ERROR)
178  {
179  goto error;
180  }
181 
182  error_code = catalog_add_class_info (thread_p, class_id_p, cls_info_p, &catalog_access_info);
183  if (error_code != NO_ERROR)
184  {
185  goto error;
186  }
187 
188  goto end;
189  }

192  if (error_code != NO_ERROR)
193  {
194  goto error;
195  }
196 
197  if (count != 0)
198  {
199  /* Update statistics for all partitions and the partitioned class */
200  assert (partitions != NULL);
201  catalog_free_class_info_and_init (cls_info_p);
202  error_code = stats_update_partitioned_statistics (thread_p, class_id_p, partitions, count, with_fullscan);
203  db_private_free (thread_p, partitions);
204  if (error_code != NO_ERROR)
205  {
206  goto error;
207  }
208 
209  goto end;
210  }
211 
212  error_code = catalog_start_access_with_dir_oid (thread_p, &catalog_access_info, S_LOCK);
213  if (error_code != NO_ERROR)
214  {
215  goto error;
216  }
217 
218  if (catalog_get_last_representation_id (thread_p, class_id_p, &repr_id) != NO_ERROR)
219  {
220  goto error;
221  }
222 
223  disk_repr_p = catalog_get_representation (thread_p, class_id_p, repr_id, &catalog_access_info);
224  if (disk_repr_p == NULL)
225  {
226  goto error;
227  }

229 
230  npages = estimated_nobjs = 0;
231 
232  /* do not use estimated npages, get correct info */
233  if (file_get_num_user_pages (thread_p, &(cls_info_p->ci_hfid.vfid), &npages) != NO_ERROR)
234  {
235  goto error;
236  }
237  assert (npages > 0);
238  cls_info_p->ci_tot_pages = MAX (npages, 0);
239 
240  estimated_nobjs = heap_estimate_num_objects (thread_p, &(cls_info_p->ci_hfid));
241  if (estimated_nobjs == -1)
242  {
243  /* cannot get estimates from the heap, use old info */
244  assert (cls_info_p->ci_tot_objects >= 0);
245  }
246  else
247  {
248  cls_info_p->ci_tot_objects = estimated_nobjs;
249  }
250 
251  /* update the index statistics for each attribute */
252 
253  for (i = 0; i < disk_repr_p->n_fixed + disk_repr_p->n_variable; i++)
254  {
255  if (i < disk_repr_p->n_fixed)
256  {
257  disk_attr_p = disk_repr_p->fixed + i;
258  }
259  else
260  {
261  disk_attr_p = disk_repr_p->variable + (i - disk_repr_p->n_fixed);
262  }
263 
264  for (j = 0, btree_stats_p = disk_attr_p->bt_stats; j < disk_attr_p->n_btstats; j++, btree_stats_p++)
265  {
266  assert_release (!BTID_IS_NULL (&btree_stats_p->btid));
267  assert_release (btree_stats_p->pkeys_size > 0);
268  assert_release (btree_stats_p->pkeys_size <= BTREE_STATS_PKEYS_NUM);
269 
270  if (btree_get_stats (thread_p, btree_stats_p, with_fullscan) != NO_ERROR)
271  {
272  goto error;
273  }
274 
275  assert_release (btree_stats_p->keys >= 0);
276  } /* for (j = 0; ...) */
277  } /* for (i = 0; ...) */
278 
279  error_code = catalog_start_access_with_dir_oid (thread_p, &catalog_access_info, X_LOCK);
280  if (error_code != NO_ERROR)
281  {
282  goto error;
283  }
284 
285  /* replace the current disk representation structure/information in the catalog with the newly computed statistics */
286  assert (!OID_ISNULL (&(cls_info_p->ci_rep_dir)));
287  error_code =
288  catalog_add_representation (thread_p, class_id_p, repr_id, disk_repr_p, &(cls_info_p->ci_rep_dir),
289  &catalog_access_info);
290  if (error_code != NO_ERROR)
291  {
292  goto error;
293  }
294 
295  cls_info_p->ci_time_stamp = stats_get_time_stamp ();
296 
297  error_code = catalog_add_class_info (thread_p, class_id_p, cls_info_p, &catalog_access_info);
298  if (error_code != NO_ERROR)
299  {
300  goto error;
301  }
302 
303 end:
304 
305  (void) catalog_end_access_with_dir_oid (thread_p, &catalog_access_info, error_code);
306 
307  lock_unlock_object (thread_p, class_id_p, oid_Root_class_oid, SCH_S_LOCK, false);
308 
309  if (disk_repr_p)
310  {
311  catalog_free_representation_and_init (disk_repr_p);
312  }
313 
314  if (cls_info_p)
315  {
316  catalog_free_class_info_and_init (cls_info_p);
317  }
318 
319  er_set (ER_NOTIFICATION_SEVERITY, ARG_FILE_LINE, ER_LOG_FINISHED_TO_UPDATE_STATISTICS, 5,
320  class_name ? class_name : "*UNKNOWN-CLASS*", class_id_p->volid, class_id_p->pageid, class_id_p->slotid,
321  error_code);
322 
323  if (class_name)
324  {
325  free_and_init (class_name);
326  }
327 
328  thread_p->pop_resource_tracks ();
329 
330  return error_code;
331 
332 error:
333 
334  if (error_code == NO_ERROR && (error_code = er_errid ()) == NO_ERROR)
335  {
336  error_code = ER_FAILED;
337  }
338  goto end;
339 }

341 /*
342  * xstats_update_all_statistics () - Updates the statistics
343  * for all the classes of the database
344  * return:
345  * with_fullscan(in): true iff WITH FULLSCAN
346  *
347  * Note: It performs this by getting the list of all classes existing in the
348  * database and their OID's from the catalog's class collection
349  * (maintained in an extendible hashing structure) and calling the
350  * "xstats_update_statistics" function for each one of the elements
351  * of this list one by one.
352  */
353 int
354 xstats_update_all_statistics (THREAD_ENTRY * thread_p, bool with_fullscan)
355 {
356  int error = NO_ERROR;
357  RECDES recdes = RECDES_INITIALIZER; /* Record descriptor for peeking object */
358  HFID root_hfid;
359  OID class_oid;
360 #if !defined(NDEBUG)
361  char *classname = NULL;
362 #endif
363  HEAP_SCANCACHE scan_cache;
364  MVCC_SNAPSHOT *mvcc_snapshot = NULL;
365 
366  mvcc_snapshot = logtb_get_mvcc_snapshot (thread_p);
367  if (mvcc_snapshot == NULL)
368  {
369  return DISK_ERROR;
370  }
371 
372  /* Find every single class */
373 
374  if (boot_find_root_heap (&root_hfid) != NO_ERROR || HFID_IS_NULL (&root_hfid))
375  {
376  goto exit_on_error;
377  }
378 
379  if (heap_scancache_start (thread_p, &scan_cache, &root_hfid, oid_Root_class_oid, false, false, mvcc_snapshot) !=
380  NO_ERROR)
381  {
382  goto exit_on_error;
383  }
384 
385  class_oid.volid = root_hfid.vfid.volid;
386  class_oid.pageid = NULL_PAGEID;
387  class_oid.slotid = NULL_SLOTID;
388 
389  recdes.data = NULL;
390  while (heap_next (thread_p, &root_hfid, oid_Root_class_oid, &class_oid, &recdes, &scan_cache, COPY) == S_SUCCESS)
391  {
392 #if !defined(NDEBUG)
393  classname = or_class_name (&recdes);
394  assert (classname != NULL);
395  assert (strlen (classname) < 255);
396 #endif
397 
398  error = xstats_update_statistics (thread_p, &class_oid, with_fullscan);
399  if (error == ER_UPDATE_STAT_CANNOT_GET_LOCK || error == ER_SP_UNKNOWN_SLOTID)
400  {
401  /* continue with other classes */
402  er_clear ();
403  error = NO_ERROR;
404  }
405  else if (error != NO_ERROR)
406  {
407  break;
408  }
409  recdes.data = NULL;
410  } /* while (...) */
411 
412  /* End the scan cursor */
413  if (heap_scancache_end (thread_p, &scan_cache) != NO_ERROR)
414  {
415  goto exit_on_error;
416  }
417 
418  return error;
419 
420 exit_on_error:
421 
422  return DISK_ERROR;
423 }

425 /*
426  * xstats_get_statistics_from_server () - Retrieves the class statistics
427  * return: buffer contaning class statistics, or NULL on error
428  * class_id(in): Identifier of the class
429  * timestamp(in):
430  * length(in): Length of the buffer
431  *
432  * Note: This function retrieves the statistics for the given class from the
433  * catalog manager and stores them into a buffer. Note that since the
434  * statistics are kept on the current (last) representation structure of
435  * the catalog, only this structure is retrieved. Note further that
436  * since the statistics are used only on the client side they are not
437  * put into a structure here on the server side (not even temporarily),
438  * but stored into a buffer area to be transmitted to the client side.
439  */
440 char *
441 xstats_get_statistics_from_server (THREAD_ENTRY * thread_p, OID * class_id_p, unsigned int time_stamp, int *length_p)
442 {
443  CLS_INFO *cls_info_p;
444  REPR_ID repr_id;
445  DISK_REPR *disk_repr_p;
446  DISK_ATTR *disk_attr_p;
447  BTREE_STATS *btree_stats_p;
448  OID dir_oid;
449  int npages, estimated_nobjs, max_unique_keys;
450  int i, j, k, size, n_attrs, tot_n_btstats, tot_key_info_size;
451  char *buf_p, *start_p;
452  int key_size;
453  int lk_grant_code;
454  CATALOG_ACCESS_INFO catalog_access_info = CATALOG_ACCESS_INFO_INITIALIZER;
455 
456  /* init */
457  cls_info_p = NULL;
458  disk_repr_p = NULL;
459 
460  thread_p->push_resource_tracks ();
461 
462  *length_p = -1;
463 
464  /* the lock on class to avoid changes of representation from rollbacked UPDATE statistics */
465  lk_grant_code = lock_object (thread_p, class_id_p, oid_Root_class_oid, SCH_S_LOCK, LK_UNCOND_LOCK);
466  if (lk_grant_code != LK_GRANTED)
467  {
468  char *class_name = NULL;
469 
470  if (heap_get_class_name (thread_p, class_id_p, &class_name) != NO_ERROR)
471  {
472  /* ignore */
473  er_clear ();
474  }
475 
476  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_UPDATE_STAT_CANNOT_GET_LOCK, 1,
477  class_name ? class_name : "*UNKNOWN-CLASS*");
478 
479  if (class_name != NULL)
480  {
481  free_and_init (class_name);
482  }
483  goto exit_on_error;
484  }
485 
486  if (catalog_get_dir_oid_from_cache (thread_p, class_id_p, &dir_oid) != NO_ERROR)
487  {
488  goto exit_on_error;
489  }
490 
491  catalog_access_info.class_oid = class_id_p;
492  catalog_access_info.dir_oid = &dir_oid;
493  if (catalog_start_access_with_dir_oid (thread_p, &catalog_access_info, S_LOCK) != NO_ERROR)
494  {
495  goto exit_on_error;
496  }
497 
498  cls_info_p = catalog_get_class_info (thread_p, class_id_p, &catalog_access_info);
499  if (cls_info_p == NULL)
500  {
501  goto exit_on_error;
502  }
503 
504  if (time_stamp > 0 && time_stamp >= cls_info_p->ci_time_stamp)
505  {
506  *length_p = 0;
507  goto exit_on_error;
508  }
509 
510  if (catalog_get_last_representation_id (thread_p, class_id_p, &repr_id) != NO_ERROR)
511  {
512  goto exit_on_error;
513  }
514 
515  disk_repr_p = catalog_get_representation (thread_p, class_id_p, repr_id, &catalog_access_info);
516  if (disk_repr_p == NULL)
517  {
518  goto exit_on_error;
519  }
520 
521  (void) catalog_end_access_with_dir_oid (thread_p, &catalog_access_info, NO_ERROR);
522 
523  lock_unlock_object (thread_p, class_id_p, oid_Root_class_oid, SCH_S_LOCK, false);
524 
525  n_attrs = disk_repr_p->n_fixed + disk_repr_p->n_variable;
526 
527  tot_n_btstats = tot_key_info_size = 0;
528  for (i = 0; i < n_attrs; i++)
529  {
530  if (i < disk_repr_p->n_fixed)
531  {
532  disk_attr_p = disk_repr_p->fixed + i;
533  }
534  else
535  {
536  disk_attr_p = disk_repr_p->variable + (i - disk_repr_p->n_fixed);
537  }
538 
539  tot_n_btstats += disk_attr_p->n_btstats;
540  for (j = 0, btree_stats_p = disk_attr_p->bt_stats; j < disk_attr_p->n_btstats; j++, btree_stats_p++)
541  {
542  tot_key_info_size += or_packed_domain_size (btree_stats_p->key_type, 0);
543  assert (btree_stats_p->pkeys_size <= BTREE_STATS_PKEYS_NUM);
544  tot_key_info_size += (btree_stats_p->pkeys_size * OR_INT_SIZE); /* pkeys[] */
545  }
546  }
547 
548  size = (OR_INT_SIZE /* time_stamp of CLS_INFO */
549  + OR_INT_SIZE /* tot_objects of CLS_INFO */
550  + OR_INT_SIZE /* tot_pages of CLS_INFO */
551  + OR_INT_SIZE /* n_attrs from DISK_REPR */
552  + (OR_INT_SIZE /* id of DISK_ATTR */
553  + OR_INT_SIZE /* type of DISK_ATTR */
554  + OR_INT_SIZE /* n_btstats of DISK_ATTR */
555  ) * n_attrs); /* number of attributes */
556 
557  size += ((OR_BTID_ALIGNED_SIZE /* btid of BTREE_STATS */
558  + OR_INT_SIZE /* leafs of BTREE_STATS */
559  + OR_INT_SIZE /* pages of BTREE_STATS */
560  + OR_INT_SIZE /* height of BTREE_STATS */
561  + OR_INT_SIZE /* keys of BTREE_STATS */
562  + OR_INT_SIZE /* does the BTREE_STATS correspond to a function index */
563  ) * tot_n_btstats); /* total number of indexes */
564 
565  size += tot_key_info_size; /* key_type, pkeys[] of BTREE_STATS */
566 
567  size += OR_INT_SIZE; /* max_unique_keys */
568 
569  start_p = buf_p = (char *) malloc (size);
570  if (buf_p == NULL)
571  {
572  goto exit_on_error;
573  }
574  memset (start_p, 0, size);
575 
576  OR_PUT_INT (buf_p, cls_info_p->ci_time_stamp);
577  buf_p += OR_INT_SIZE;
578 
579  npages = estimated_nobjs = max_unique_keys = -1;
580 
581  assert (cls_info_p->ci_tot_objects >= 0);
582  assert (cls_info_p->ci_tot_pages >= 0);
583 
584  if (HFID_IS_NULL (&cls_info_p->ci_hfid))
585  {
586  /* The class does not have a heap file (i.e. it has no instances); so no statistics can be obtained for this
587  * class */
588  OR_PUT_INT (buf_p, cls_info_p->ci_tot_objects); /* #objects */
589  buf_p += OR_INT_SIZE;
590 
591  OR_PUT_INT (buf_p, cls_info_p->ci_tot_pages); /* #pages */
592  buf_p += OR_INT_SIZE;
593  }
594  else
595  {
596  /* use estimates from the heap since it is likely that its estimates are more accurate than the ones gathered at
597  * update statistics time */
598  estimated_nobjs = heap_estimate_num_objects (thread_p, &(cls_info_p->ci_hfid));
599  if (estimated_nobjs < 0)
600  {
601  /* cannot get estimates from the heap, use ones from the catalog */
602  estimated_nobjs = cls_info_p->ci_tot_objects;
603  }
604  else
605  {
606  /* heuristic is that big nobjs is better than small */
607  estimated_nobjs = MAX (estimated_nobjs, cls_info_p->ci_tot_objects);
608  }
609 
610  OR_PUT_INT (buf_p, estimated_nobjs); /* #objects */
611  buf_p += OR_INT_SIZE;
612 
613  /* do not use estimated npages, get correct info */
614  assert (!VFID_ISNULL (&cls_info_p->ci_hfid.vfid));
615  if (file_get_num_user_pages (thread_p, &cls_info_p->ci_hfid.vfid, &npages) != NO_ERROR)
616  {
617  int err;
618 
619  ASSERT_ERROR_AND_SET (err);
620  /* cannot get #pages from the heap, use ones from the catalog */
621  npages = cls_info_p->ci_tot_pages;
622  assert (0 < npages || err == ER_INTERRUPTED);
623  npages = MAX (npages, 1); /* safe-guard */
624  }
625  assert (npages > 0);
626 
627  OR_PUT_INT (buf_p, npages); /* #pages */
628  buf_p += OR_INT_SIZE;
629  }
630 
631  OR_PUT_INT (buf_p, n_attrs);
632  buf_p += OR_INT_SIZE;
633 
634  /* put the statistics information of each attribute to the buffer */
635  for (i = 0; i < n_attrs; i++)
636  {
637  if (i < disk_repr_p->n_fixed)
638  {
639  disk_attr_p = disk_repr_p->fixed + i;
640  }
641  else
642  {
643  disk_attr_p = disk_repr_p->variable + (i - disk_repr_p->n_fixed);
644  }
645 
646  OR_PUT_INT (buf_p, disk_attr_p->id);
647  buf_p += OR_INT_SIZE;
648 
649  OR_PUT_INT (buf_p, disk_attr_p->type);
650  buf_p += OR_INT_SIZE;
651 
652  OR_PUT_INT (buf_p, disk_attr_p->n_btstats);
653  buf_p += OR_INT_SIZE;
654 
655  for (j = 0, btree_stats_p = disk_attr_p->bt_stats; j < disk_attr_p->n_btstats; j++, btree_stats_p++)
656  {
657  /* collect maximum unique keys info */
658  if (xbtree_get_unique_pk (thread_p, &btree_stats_p->btid))
659  {
660  max_unique_keys = MAX (max_unique_keys, btree_stats_p->keys);
661  }
662 
663  OR_PUT_BTID (buf_p, &btree_stats_p->btid);
664  buf_p += OR_BTID_ALIGNED_SIZE;
665 
666  /* defense for not gathered statistics */
667  btree_stats_p->leafs = MAX (1, btree_stats_p->leafs);
668  btree_stats_p->pages = MAX (1, btree_stats_p->pages);
669  btree_stats_p->height = MAX (1, btree_stats_p->height);
670 
671  /* If the btree file has currently more pages than when we gathered statistics, assume that all growth happen
672  * at the leaf level. If the btree is smaller, we use the gathered statistics since the btree may have an
673  * external file (unknown at this level) to keep overflow keys. */
674  if (file_get_num_user_pages (thread_p, &btree_stats_p->btid.vfid, &npages) != NO_ERROR)
675  {
676  int err;
677 
678  /* what to do here? */
679  ASSERT_ERROR_AND_SET (err);
680  npages = btree_stats_p->pages;
681  assert (0 < npages || err == ER_INTERRUPTED);
682  npages = MAX (npages, 1); /* safe-guard */
683  }
684  assert (npages > 0);
685  if (npages > btree_stats_p->pages)
686  {
687  OR_PUT_INT (buf_p, (btree_stats_p->leafs + (npages - btree_stats_p->pages)));
688  buf_p += OR_INT_SIZE;
689 
690  OR_PUT_INT (buf_p, npages);
691  buf_p += OR_INT_SIZE;
692  }
693  else
694  {
695  OR_PUT_INT (buf_p, btree_stats_p->leafs);
696  buf_p += OR_INT_SIZE;
697 
698  OR_PUT_INT (buf_p, btree_stats_p->pages);
699  buf_p += OR_INT_SIZE;
700  }
701 
702  OR_PUT_INT (buf_p, btree_stats_p->height);
703  buf_p += OR_INT_SIZE;
704 
705  OR_PUT_INT (buf_p, btree_stats_p->has_function);
706  buf_p += OR_INT_SIZE;
707 
708  /* check and handle with estimation, since pkeys[] is not gathered before update stats */
709  if (estimated_nobjs > 0)
710  {
711  /* is non-empty index */
712  btree_stats_p->keys = MAX (btree_stats_p->keys, 1);
713 
714  /* If the estimated objects from heap manager is greater than the estimate when the statistics were
715  * gathered, assume that the difference is in distinct keys. */
716  if (cls_info_p->ci_tot_objects > 0 && estimated_nobjs > cls_info_p->ci_tot_objects)
717  {
718  btree_stats_p->keys += (estimated_nobjs - cls_info_p->ci_tot_objects);
719  }
720  }
721 
722  OR_PUT_INT (buf_p, btree_stats_p->keys);
723  buf_p += OR_INT_SIZE;
724 
725  assert_release (btree_stats_p->leafs >= 1);
726  assert_release (btree_stats_p->pages >= 1);
727  assert_release (btree_stats_p->height >= 1);
728  assert_release (btree_stats_p->keys >= 0);
729 
730  buf_p = or_pack_domain (buf_p, btree_stats_p->key_type, 0, 0);
731 
732  /* get full key size */
733  if (TP_DOMAIN_TYPE (btree_stats_p->key_type) == DB_TYPE_MIDXKEY)
734  {
735  key_size = tp_domain_size (btree_stats_p->key_type->setdomain);
736  }
737  else
738  {
739  key_size = 1;
740  }
741  assert (key_size >= btree_stats_p->pkeys_size);
742 
743  assert (btree_stats_p->pkeys_size <= BTREE_STATS_PKEYS_NUM);
744  for (k = 0; k < btree_stats_p->pkeys_size; k++)
745  {
746  /* check and handle with estimation, since pkeys[] is not gathered before update stats */
747  if (estimated_nobjs > 0)
748  {
749  /* is non-empty index */
750  btree_stats_p->pkeys[k] = MAX (btree_stats_p->pkeys[k], 1);
751  }
752 
753  if (k + 1 == key_size)
754  {
755  /* this last pkeys must be equal to keys */
756  btree_stats_p->pkeys[k] = btree_stats_p->keys;
757  }
758 
759  OR_PUT_INT (buf_p, btree_stats_p->pkeys[k]);
760  buf_p += OR_INT_SIZE;
761  }
762  } /* for (j = 0, ...) */
763  }
764 
765  OR_PUT_INT (buf_p, max_unique_keys);
766  buf_p += OR_INT_SIZE;
767 
768  catalog_free_representation_and_init (disk_repr_p);
769  catalog_free_class_info_and_init (cls_info_p);
770 
771  *length_p = CAST_STRLEN (buf_p - start_p);
772 
773  thread_p->pop_resource_tracks ();
774 
775  return start_p;
776 
777 exit_on_error:
778 
779  if (catalog_access_info.access_started)
780  {
781  (void) catalog_end_access_with_dir_oid (thread_p, &catalog_access_info, ER_FAILED);
782  }
783 
784  if (disk_repr_p)
785  {
786  catalog_free_representation_and_init (disk_repr_p);
787  }
788  if (cls_info_p)
789  {
790  catalog_free_class_info_and_init (cls_info_p);
791  }
792 
793  thread_p->pop_resource_tracks ();
794 
795  return NULL;
796 }
