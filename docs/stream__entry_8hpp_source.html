37  /*
38  * entry of a stream: this is an abstract class which encapsulates a collection of objects packable into a stream
39  *
40  * it has a mandatory header structure to be implemented by derived class along with associated methods:
41  * - get_packed_header_size,
42  * - set_header_data_size,
43  * - pack_stream_entry_header
44  * - unpack_stream_entry_header
45  * - get_packable_entry_count_from_header
46  * The header size must be aligned at MAX_ALIGNMENT.
47  * The objects are of type PO (template argument).
48  * The concrete class (of entry) must provide the factory get_builder method to build objects
49  *
50  * As a "protocol", each object has an integer identifier which is packed before actual object packaging:
51  * at unpack, this identifier is used to instantiate an object (using the integrated factory), before calling
52  * object.unpack
53  *
54  * get_packer method needs to be implemented (it should be per instance object for multi-threaded usage)
55  *
56  * is_equal methods is mainly needed for unit testing (is expected to validate that the entire trees of two entries
57  * are equivalent)
58  */
59  template <typename PO>
60  class entry
61  {
62  protected:
63  std::vector <PO *> m_packable_entries;
64 
65  multi_thread_stream *m_stream;
66 
67  stream_position m_data_start_position;
68 
69  stream::write_func_t m_packing_func;
70  stream::read_prepare_func_t m_prepare_func;
71  stream::read_func_t m_unpack_func;

73  /*
74  * packing_function :
75  * 1. init packer (set start pointer and amount of buffer)
76  * 2. pack header of entry
77  * 3. pack each object of entry
78  * Header is aligned at MAX_ALIGNMENT.
79  * Each packed object is aligned at MAX_ALIGNMENT.
80  */
81  int packing_func (const stream_position &pos, char *ptr, const size_t reserved_amount)
82  {
83  unsigned int i;
84  cubpacking::packer *serializator = get_packer ();
85 
86  size_t aligned_amount = DB_ALIGN (reserved_amount, MAX_ALIGNMENT);
87  serializator->init (ptr, aligned_amount);
88 
89  pack_stream_entry_header ();
90 
91  for (i = 0; i < m_packable_entries.size (); i++)
92  {
93  serializator->align (MAX_ALIGNMENT);
94 #if !defined (NDEBUG)
95  const char *start_ptr = serializator->get_curr_ptr ();
96  const char *curr_ptr;
97  size_t entry_size = m_packable_entries[i]->get_packed_size (serializator);
98 #endif
99 
100  m_packable_entries[i]->pack (serializator);
101 
102 #if !defined (NDEBUG)
103  curr_ptr = serializator->get_curr_ptr ();
104  assert (curr_ptr - start_ptr == (int) entry_size);
105 #endif
106  }
107  serializator->align (MAX_ALIGNMENT);
108 
109  int packed_amount = (int) (serializator->get_curr_ptr () - serializator->get_packer_buffer ());
110 
111  return packed_amount;
112  };
113 
114  /* callback header-read function for entry (called with multi_thread_stream::read_serial)
115  * 1. init packer
116  * 2. unpack entry header
117  * 3. saves start of data payload logical position in entry object (to be retrieved at unpack of entry)
118  * 4. returns size of payload (value unpacked in header)
119  */
120  int prepare_func (const stream_position &data_start_pos, char *ptr, const size_t header_size,
121  size_t &payload_size)
122  {
123  cubpacking::packer *serializator = get_packer ();
124  int error_code;
125 
126  assert (header_size == get_packed_header_size ());
127 
128  serializator->init (ptr, header_size);
129 
130  error_code = unpack_stream_entry_header ();
131  if (error_code != NO_ERROR)
132  {
133  return error_code;
134  }
135 
136  m_data_start_position = data_start_pos;
137  payload_size = get_data_packed_size ();
138 
139  return error_code;
140  };

143  /* unpack function for entry;
144  * this unpacks only the payload (objects) of the the entry, the header was unpacked with entry::prepare_func
145  * 1. init (un-)packer
146  * 2. reads count of packed objects to expect from entry header (previously decoded at prepare_func)
147  * 3. peeks the id of next object (the current pointer of packer is not advanced)
148  * 4. creates an object using the id
149  * 5. unpacks the object
150  * 6. adds the object to entry
151  */
152  int unpack_func (char *ptr, const size_t data_size)
153  {
154  unsigned int i;
155  int error_code = NO_ERROR;
156  int object_id;
157  cubpacking::packer *serializator = get_packer ();
158  size_t count_packable_entries = get_packable_entry_count_from_header ();
159 
160  serializator->init (ptr, data_size);
161 
162  for (i = 0 ; i < count_packable_entries; i++)
163  {
164  serializator->align (MAX_ALIGNMENT);
165  /* peek type of object : it will be consumed by object's unpack */
166  serializator->peek_unpack_int (&object_id);
167 
168  PO *packable_entry = get_builder ()->create_object (object_id);
169  if (packable_entry == NULL)
170  {
171  error_code = ER_STREAM_UNPACKING_INV_OBJ_ID;
172  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_STREAM_UNPACKING_INV_OBJ_ID, 1, object_id);
173  return error_code;
174  }
175 
176  assert (packable_entry != NULL);
177 
178  /* unpack one replication entry */
179  add_packable_entry (packable_entry);
180  error_code = packable_entry->unpack (serializator);
181  if (error_code != NO_ERROR)
182  {
183  return error_code;
184  }
185  }
186  serializator->align (MAX_ALIGNMENT);
187 
188  assert (serializator->get_curr_ptr () - ptr == (int) data_size);
189 
190  return error_code;
191  };

228  /*
229  * pack method:
230  * 1. compute header and data size
231  * 2. set data size value in header field (before packing)
232  * 3. stream.write using packing_function as argument
233  */
234  int pack (void)
235  {
236  size_t total_stream_entry_size;
237  size_t data_size;
238  int err;
239  static size_t header_size = get_packed_header_size ();
240 
241  assert (DB_WASTED_ALIGN (header_size, MAX_ALIGNMENT) == 0);
242 
243  data_size = (m_packable_entries.size () > 0) ? get_entries_size () : 0;
244  data_size = DB_ALIGN (data_size, MAX_ALIGNMENT);
245  total_stream_entry_size = header_size + data_size;
246 
247  set_header_data_size (data_size);
248  assert (DB_WASTED_ALIGN (total_stream_entry_size, MAX_ALIGNMENT) == 0);
249 
250  err = m_stream->write (total_stream_entry_size, m_packing_func);
251 
252  return (err < 0) ? err : NO_ERROR;
253  };

255  /*
256  * this is pre-unpack method : it fetches enough data to unpack stream header contents,
257  * then fetches (receive from socket) the actual amount of data without unpacking it.
258  */
259  int prepare (void)
260  {
261  static size_t stream_entry_header_size = get_packed_header_size ();
262  int err;
263 
264  err = m_stream->read_serial (stream_entry_header_size, m_prepare_func);
265 
266  return (err < 0) ? err : NO_ERROR;
267  }

269  int unpack (void)
270  {
271  int err = NO_ERROR;
272  size_t data_size = get_data_packed_size ();
273 
274  if (data_size > 0)
275  {
276  /* read the stream starting from data contents */
277  err = m_stream->read (m_data_start_position, data_size, m_unpack_func);
278  }
279 
280  return (err < 0) ? err : NO_ERROR;
281  };

288  size_t get_entries_size (void)
289  {
290  size_t entry_size, total_size = 0;
291  unsigned int i;
292 
293  cubpacking::packer *serializator = get_packer ();
294  for (i = 0; i < m_packable_entries.size (); i++)
295  {
296  total_size = DB_ALIGN (total_size, MAX_ALIGNMENT);
297  entry_size = m_packable_entries[i]->get_packed_size (serializator);
298  total_size += entry_size;
299  }
300 
301  return total_size;
302  };
