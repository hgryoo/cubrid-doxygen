20 /*
21  * stream_transfer_[sender/receiver].cpp
22  *
23  * - sends or receives chunks(usually MTU bytes) from cubstream::stream
24  * - the sender interogates the respective stream and tries to send the minimum between MTU and what is left of
25  * the stream
26  * - it contains a daemon that is started automatically at creation, that does the task of sending/receiving bytes
27  * - usage can be found in transfer_channel/test_main.cpp;
28  * on one machine will exist a transfer_sender instance and on the other a transfer_receiver.
29  * the instances are created using consumer/producer streams and connected communication channels.
30  * the stream will be sent automatically until m_last_committed_pos
31  */

42  class transfer_sender_task : public cubthread::task_without_context
43  {
44  public:
45  transfer_sender_task (cubstream::transfer_sender &producer_channel)
46  : this_producer_channel (producer_channel),
47  m_first_loop (true)
48  {
49  }

51  void execute () override
52  {
53  css_error_code rc = NO_ERRORS;
54  stream_position last_reported_ready_pos = this_producer_channel.m_stream.get_last_committed_pos ();
55 
56  if (m_first_loop)
57  {
58  UINT64 last_sent_position = 0;
59  std::size_t max_len = sizeof (UINT64);
60 
61  assert (this_producer_channel.m_channel.is_connection_alive ());
62  assert (sizeof (stream_position) == sizeof (UINT64));
63 
64  rc = this_producer_channel.m_channel.recv ((char *) &last_sent_position, max_len);
65  this_producer_channel.m_last_sent_position = last_sent_position;
66 
67  assert (max_len == sizeof (UINT64));
68 
69  if (rc != NO_ERRORS)
70  {
71  this_producer_channel.m_channel.close_connection ();
72  return;
73  }
74 
75  m_first_loop = false;
76  }
77 
78  while (this_producer_channel.m_last_sent_position < last_reported_ready_pos)
79  {
80  std::size_t byte_count = std::min ((stream_position) cubcomm::MTU,
81  last_reported_ready_pos - this_producer_channel.m_last_sent_position);
82  int error_code = NO_ERROR;
83 
84  error_code = this_producer_channel.m_stream.read (this_producer_channel.m_last_sent_position, byte_count,
85  this_producer_channel.m_read_action_function);
86 
87  if (error_code != NO_ERROR)
88  {
89  this_producer_channel.m_channel.close_connection ();
90  break;
91  }
92  }
93  }
94 
95  private:
96  cubstream::transfer_sender &this_producer_channel;
97  bool m_first_loop;
98  };
