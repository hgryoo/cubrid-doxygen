225  unsigned step = 0;

231  {

236 
237  /* step 1: test allocate / allocate and so on*/
238  T **pointers = new T * [alloc_count];
239  for (size_t i = 0; i < alloc_count; i++)
240  {
241  pointers[i] = alloc->allocate (1);
242  }
243  results.register_time (timer, alloc_index, step++);
244 
245  /* step 2: test deallocate / deallocate and so on*/
246  for (size_t i = 0; i < alloc_count; i++)
247  {
248  alloc->deallocate (pointers[i], 1);
249  }
250  results.register_time (timer, alloc_index, step++);
251 
252  test_common::custom_assert (step == results.get_step_count ());
253 
254  delete pointers;
255  delete alloc;
256 
257  return 0;

325  /* function tests performance with allocating a series of random-sized pointers.
326  *
327  * First a pool of random-sized pointers is created. Then a loop that deallocates one random pointer and replaces it
328  * with another random-sized pointer. At the end deallocates everything.
329  *
330  * Template:
331  *
332  * Alloc - tested allocator. its specialization should be char.
333  */
334  template <typename Alloc>
335  static int
336  test_performance_random (test_common::perf_compare &result, size_t ptr_pool_size, unsigned alloc_count,
337  const random_values &actions)
338  {
339  /* local definition to allocate a pointer and save it in pool at ptr_index_ */
340 #define PTR_ALLOC(ptr_index_) \
341  do { \
342  /* get random index in pointer size array. */ \
343  size_t ptr_size_index = actions.get_at (random_value_cursor++, PTR_MEMSIZE_COUNT); \
344  /* get pointer size */ \
345  size_t ptr_size = ptr_sizes[ptr_size_index]; \
346  /* allocate pointer */ \
347  char *ptr = alloc->allocate (ptr_size); \
348  /* save pointer and its size */; \
349  pointers_pool[ptr_index_] = std::make_pair (ptr, ptr_size); \
350  } while (false)
351 
352  typedef std::pair<char *, size_t> ptr_with_size;
353 
354  const unsigned PTR_MEMSIZE_COUNT = 14;
355  static const std::array<size_t, PTR_MEMSIZE_COUNT> ptr_sizes =
356  {{ 1, 64, 64, 64, 64, 64, 256, 1024, 1024, 1024, 4096, 8192, 16384, 65536 }};
357 
358  custom_thread_entry cte; /* thread entry wrapper */
359 
360  /* function header */
361  static std::string function_header = std::string (" random_perf") + "<" + typeid (Alloc).name () + ">\n";
362  test_common::sync_cout (function_header);
363 
364  /* instantiate allocator */
365  Alloc *alloc = NULL;
366  test_allocator_type alloc_type;
367  init_allocator<char> (cte, alloc, alloc_type);
368  size_t alloc_index = static_cast <size_t> (alloc_type);
369 
370  /* */
371  ptr_with_size *pointers_pool = new ptr_with_size [ptr_pool_size];
372  size_t ptr_index;
373 
374  /* init step and timer */
375  unsigned step = 0;
376  unsigned random_value_cursor = 0;
377  test_common::us_timer timer;
378 
379  /* first step: fill pool */
380  for (ptr_index = 0; ptr_index < ptr_pool_size; ptr_index++)
381  {
382  PTR_ALLOC (ptr_index);
383  }
384 
385  /* second step: deallocate and allocate a new pointer */
386  for (size_t count = 0; count < alloc_count; count++)
387  {
388  ptr_index = actions.get_at (random_value_cursor++, static_cast<unsigned> (ptr_pool_size));
389  /* deallocate current pointer */
390  alloc->deallocate (pointers_pool[ptr_index].first, pointers_pool[ptr_index].second);
391  /* replace with new pointer */
392  PTR_ALLOC (ptr_index);
393  }
394 
395  /* free pool */
396  for (ptr_index = 0; ptr_index < ptr_pool_size; ptr_index++)
397  {
398  alloc->deallocate (pointers_pool[ptr_index].first, pointers_pool[ptr_index].second);
399  }
400 
401  result.register_time (timer, alloc_index, step++);
402 
403  test_common::custom_assert (random_value_cursor == actions.get_size ());
404  test_common::custom_assert (step == result.get_step_count ());
405 
406  delete pointers_pool;
407  delete alloc;
408 
409  return 0;
410 
411 #undef PTR_ALLOC
412  }
