78  /* test_append_strings - append append_count strings of size append_size into given buffer.
79  *
80  *
81  * Template:
82  *
83  * Buffer having member function append (const char *str, size_t len);
84  *
85  *
86  * How it works:
87  *
88  * Loops and executes repeated appends (strings are filled with blank spaces) and times the duration of all append
89  * operations.
90  */
91  template <typename Buf>
92  static int
93  test_append_strings (test_common::perf_compare &result, Buf &buf, test_string_buffer_types buf_type,
94  size_t append_size, unsigned append_count)
95  {
96  static std::string log_string = std::string (4,' ') + PORTABLE_FUNC_NAME + "<" + typeid (Buf).name () + ">\n";
97  test_common::sync_cout (log_string);
98 
99  char *str = new char [append_size];
100  memset (str, ' ', append_size);
101 
102  /* start timing */
103  unsigned step = 0;
104  test_common::us_timer timer;
105 
106  for (unsigned count = 0; count < append_count; count++)
107  {
108  buf.append (str, append_size);
109  }
110  result.register_time (timer, static_cast <size_t> (buf_type), step++);
111 
112  test_common::custom_assert (step == result.get_step_count ());
113 
114  delete str;
115 
116  return 0;
117  }

119  /* test_compare_append_strings_performance -
120  *
121  * Run the same number of same size string appends with three types of buffers: cubrid extensible array, std::string
122  * and C-Style static allocated array. The purpose is to have a performance as close as possible to C-Style array.
123  *
124  * Templates:
125  *
126  * AppendSize - size of one string
127  * AppendCount - number of append operations
128  *
129  *
130  * How it works:
131  *
132  * Instantiates the timer collector which is passed to specialized test_append_strings functions. Results are
133  * printed after.
134  *
135  *
136  * How to call:
137  *
138  * Just specialize the size and count of append operations.
139  */
140  template <size_t AppendSize, unsigned AppendCount>
141  static void

144  test_common::perf_compare compare_result (string_buffer_names, append_step_names);
