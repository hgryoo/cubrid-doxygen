73  {
74  stream_mover test_stream_mover;
75  cubstream::stream_position curr_pos;
76  cubstream::stream_position last_pos;
77  last_pos = stream1->get_last_committed_pos ();
78 
79  size_t copy_chunk_size = test_stream_mover.get_buf_size ();
80  int read_bytes, written_bytes;
81 
82  for (curr_pos = 0; curr_pos < last_pos;)
83  {
84  copy_chunk_size = MIN (copy_chunk_size, last_pos - curr_pos);
85 
86  read_bytes = stream1->read (curr_pos, copy_chunk_size, test_stream_mover.m_reader_func);
87  if (read_bytes <= 0)
88  {
89  break;
90  }
91 
92  written_bytes = stream2->write (read_bytes, test_stream_mover.m_writer_func);
93  assert (read_bytes == written_bytes);
94  if (written_bytes <= 0)
95  {
96  break;
97  }
98 
99  curr_pos += read_bytes;
100  }
101 
102  return NO_ERROR;
