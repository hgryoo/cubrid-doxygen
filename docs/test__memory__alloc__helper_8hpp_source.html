166  std::thread *workers = new std::thread [worker_count];
167 
168  for (unsigned int i = 0; i < worker_count; i++)
169  {
170  workers[i] = std::thread (std::forward<Func> (f), std::forward<Args> (args)...);
171  }

173  {
174  workers[i].join ();
175  }
