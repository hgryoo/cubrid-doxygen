48 perf_compare::perf_compare (const string_collection &scenarios,
49  const string_collection &steps) :
50  m_scenario_names (scenarios),
51  m_step_names (steps)
52 {
53 
54  /* init all values */
55  m_values.resize (m_scenario_names.get_count ());
56  for (size_t scenario_index = 0; scenario_index < m_scenario_names.get_count (); scenario_index++)
57  {
58  m_values[scenario_index].reserve (m_step_names.get_count ());
59  for (size_t step_index = 0; step_index < m_step_names.get_count (); step_index++)

62  }
63  }
64 
65  /* make sure leftmost_column_length is big enough */
66  m_leftmost_column_length = m_step_names.get_max_length () + 1;
67  if (m_leftmost_column_length < TEST_RESULT_NAME_PRINT_LENGTH)
68  {
69  m_leftmost_column_length = TEST_RESULT_NAME_PRINT_LENGTH;
70  }

183 inline void
184 perf_compare::check_row_and_print_warning (size_t row, bool &no_warnings, std::ostream &output)
185 {
186  bool found_better = false;
187 
188  for (size_t iter = 1; iter < m_scenario_names.get_count (); iter++)
189  {
190  if (m_values[iter][row] < m_values[0][row])
191  {
192  /* first scenario is worse */
193  print_warning_header (no_warnings, output);
194  if (!found_better)
195  {
196  output << " ";
197  output << m_scenario_names.get_name (0);
198  output << " is slower than ";
199  found_better = true;
200  }
201  else
202  {
203  output << ", ";
204  }
205  output << m_scenario_names.get_name (iter);
206  }
207  }
208  output << std::endl;
