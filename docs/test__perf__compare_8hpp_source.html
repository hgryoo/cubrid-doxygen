33 /* Collect runtime timer statistics with various scenarios and compare results.
34  *
35  *
36  * How it works:
37  *
38  * Stores timer values for each step of each scenario (it is assumed that each scenarios goes through same steps).
39  * This data structure is designed to be used by multiple threads concurrently (timers are incremented using
40  * atomic operations).
41  *
42  *
43  * How to use:
44  *
45  * Instantiate using scenario names and step names (results will be printed using these names).
46  * Pass same instance to functions running all scenarios. Scenarios can run in parallel.
47  * The running function should time each step using register_time function.
48  * After running all scenarios, results can be printed.
49  * If first scenario is expected to be best, it can be checked against the other scenarios and warnings will be
50  * printed for each slower step.
51  * print_results_and_warnings can be used to print both.
52  */
53 class perf_compare
54 {
55  public:
56 
57  /* instantiate with an vector containing step names. */
58  perf_compare (const string_collection &scenarios, const string_collection &steps);
59 
60  /* register the time for step_index step of scenario_index scenario. multiple timers can stack on the same value
61  * concurrently. */
62  inline void register_time (us_timer &timer, size_t scenario_index, size_t step_index)
63  {
64  custom_assert (scenario_index < m_scenario_names.get_count ());
65  custom_assert (step_index < m_step_names.get_count ());
66 
67  value_type time = timer.time_and_reset ().count ();
68 
69  /* can be called concurrently, so use atomic inc */
70  (void) ATOMIC_INC_64 (&m_values[scenario_index][step_index], time);
71  }
72 
73  /* print formatted results */
74  void print_results (std::ostream &output);
