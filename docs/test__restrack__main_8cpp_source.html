214 test_abort (void)
215 {
216  // test not aborted; leak one resource
217  {
218  test_restrack rt (DEFAULT_CT_ARGS);
219  rt.push_track ();
220  for (unsigned idx = 0; idx < DEFAULT_MAX_ITEMS; idx++)
221  {
222  rt.increment (ARG_FILE_LINE, idx);
223  }
224  for (unsigned idx = 0; idx < DEFAULT_MAX_ITEMS - 1; idx++)
225  {
226  rt.decrement (idx);
227  }
228  rt.pop_track ();
229  }
230  check_has_error ();
