169  template<std::size_t Count>
170  looper::looper (const std::array<delta_time, Count> periods)
171  : looper ()
172  {
173  static_assert (Count <= MAX_LOOPER_PERIODS, "Count template cannot exceed MAX_LOOPER_PERIODS=3");
174  m_periods_count = std::min (Count, MAX_LOOPER_PERIODS);
175 
176  // wait increasing period on timeouts
177  for (std::size_t i = 0; i < m_periods_count; i++)
178  {
179  m_periods[i] = periods[i];
180  // check increasing periods
181  assert (i == 0 || m_periods[i - 1] < m_periods[i]);
182  }
183 
184  m_setup_period = std::bind (&looper::setup_increasing_waits, std::ref (*this), std::placeholders::_1,
185  std::placeholders::_2);
