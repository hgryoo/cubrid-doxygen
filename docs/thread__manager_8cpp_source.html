121  template<typename Res, typename ... CtArgs>
122  inline Res *manager::create_and_track_resource (std::vector<Res *> &tracker, size_t entries_count, CtArgs &&... args)
123  {
124  check_not_single_thread ();
125 
126  std::unique_lock<std::mutex> lock (m_entries_mutex); // safe-guard
127 
128  if (m_available_entries_count < entries_count)
129  {
130  return NULL;
131  }
132  m_available_entries_count -= entries_count;
133 
134  Res *new_res = new Res (std::forward<CtArgs> (args)...);
135 
136  tracker.push_back (new_res);
137 
138  return new_res;
139  }

211  template<typename Res>
212  inline void
213  manager::destroy_and_untrack_resource (std::vector<Res *> &tracker, Res *&res, std::size_t entries_count)
214  {
215  std::unique_lock<std::mutex> lock (m_entries_mutex); // safe-guard
216  check_not_single_thread ();
217 
218  for (auto iter = tracker.begin (); iter != tracker.end (); ++iter)
219  {
220  if (res == *iter)
221  {
222  // remove resource from tracker
223  (void) tracker.erase (iter);
224 
225  // stop resource and delete
226  res->stop_execution ();
227  delete res;
228  res = NULL;
229 
230  // update available entries
231  m_available_entries_count += entries_count;
232 
233  return;
234  }
235  }
236  // resource not found
237  assert (false);
238  }

396  void
397  manager::set_max_thread_count_from_config (void)
398  {
399  // todo: is there a better way to decide on the maximum number of thread entries?
400  std::size_t max_active_workers = NUM_NON_SYSTEM_TRANS; // one per each connection
401  std::size_t max_conn_workers = NUM_NON_SYSTEM_TRANS; // one per each connection
402  std::size_t max_vacuum_workers = prm_get_integer_value (PRM_ID_VACUUM_WORKER_COUNT);
403  std::size_t max_daemons = 128; // magic number to cover predictable requirements; not cool
404 
405  // note: thread entry initialization is slow, that is why we keep a static pool initialized from the beginning to
406  // quickly claim entries. in my opinion, it would be better to have thread contexts that can be quickly
407  // generated at "runtime" (after thread starts its task). however, with current thread entry design, that is
408  // rather unlikely.
409 
410  m_max_threads = max_active_workers + max_conn_workers + max_vacuum_workers + max_daemons;
411  }
