60  // cubthread::manager
61  //
62  // description:
63  // thread and thread context (entry) manager
64  // CUBRID interface for using daemons and worker pools with thread entries
65  //
66  // daemons -
67  // creates, destroys and tracks all daemons
68  // provides thread entries to daemons
69  // available in SERVER_MODE only
70  // see more details in thread_daemon.hpp
71  //
72  // worker pools -
73  // create, destroys and tracks all worker pools
74  // provides thread entries to daemons
75  // available in both SERVER_MODE and SA_MODE; SA_MODE however does not actually create worker pools, but instead
76  // execute required tasks immediately (on current thread)
77  // see more details in thread_worker_pool.hpp
78  //
79  // entries -
80  // creates a pool of entries; pool cannot be extended
81  // reserves entries for daemons and worker pools; if entry resources are depleted, it will refuse to create
82  // additional daemons and worker pools
83  // dispatches entries when worker/daemon threads start execution and manages entry retirement/reuse
84  // note -
85  // thread entries should be seen as thread local variables. however, they are bulky structures that may take
86  // a long time to initialize/finalize, so they are pooled by manager; expensive initialize/finalize are
87  // replaced by claim from pool and retire to pool. note that claim also saves the entry to thread local
88  // pointer to thread_entry (see claim_entry/retire_entry)
89  //
90  // how to use:
91  // 1. daemon -
92  // daemon *my_daemon = cubthread::get_manager ()->create_daemon (daemon_looper, daemon_task_p);
93  // // daemon loops and execute task on each iteration
94  // cubthread::get_manager ()->destroy_daemon (my_daemon);
95  //
96  // 2. entry_workpool -
97  // entry_workpool *my_workpool = cubthread::get_manager ()->create_worker_pool (MAX_THREADS, MAX_JOBS);
98  // cubthread::get_manager ()->push_task (entry_workpool, entry_task_p);
99  // cubthread::get_manager ()->destroy_worker_pool (my_workpool);
100  //
