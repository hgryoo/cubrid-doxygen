53  // cubtread::worker_pool<Context>
54  //
55  // templates
56  // Context - thread context; a class to cache helpful information for task execution
57  //
58  // description
59  // a pool of threads to execute tasks in parallel
60  // for high-loads (more tasks than threads), stores tasks in queues to be executed when a thread is freed.
61  // for low-loads (fewer tasks than threads), retires thread when no new tasks are available and creates new
62  // threads when tasks are added again
63  // in high-loads, thread context is shared between task
64  //
65  // how to use
66  // // note that worker_pool must be specialized with a thread context
67  //
68  // // define the thread context; for CUBRID, that is usually cubthread::entry
69  // class custom_context { ... };
70  //
71  // // then define the task
72  // class custom_task : public task<custom_context>
73  // {
74  // void execute (Context &) override { ... }
75  // void create_context (void) override { ... }
76  // void retire_context (Context &) override { ... }
77  // };
78  //
79  // // create worker pool
80  // cubthread::worker_pool<custom_context> thread_pool (THREAD_COUNT, MAX_TASKS);
81  //
82  // // push tasks
83  // for (std::size_t i = 0; i < task_count; i++)
84  // {
85  // thread_pool.execute (new custom_task ()); // tasks are deallocated after execution
86  //
87  // // if you push more than worker pool can handle, assert is hit; release will wait for task to be pushed
88  // }
89  //
90  // // on destroy, worker pools stops execution (jobs in queue are not executed) and joins any running threads
91  //
92  //
93  // implementation
94  //
95  // the worker pool can be partitioned into cores - a middle layer above a group of workers. this is an
96  // optimization for high-contention systems and only one core can be set if that's not the case.
97  //
98  // core manages a number of workers, tracks available resource - free active workers and inactive workers and
99  // queues tasks that could not be executed immediately.
100  //
101  // a worker starts by being inactive (does not have a thread running). it spawns a thread on its first task,
102  // becoming active, and stays active as long as it finds more tasks to execute.
103  //
104  // This is how a new task is processed:
105  //
106  // 1. Worker pool assigns the task to a core via round robin method*. The core can then accept the task (if an
107  // available worker is found) or reject it. If task is rejected, then it is stored in a queue to be processed
108  // later.
109  // *round robin scheduling behavior may be overwritten by using execute_on_core instead of execute. it is
110  // recommended to understand how cores and workers work before trying it.
111  // sometimes however, if current tasks may be blocked until other incoming tasks are finished, a more careful
112  // core management is required.
113  //
114  // 2. Core checks if a thread is available
115  // - first checks free active list
116  // - if no free active worker is found, it checks inactive list
117  // - if a worker is found, then it is assigned the task
118  // - if no worker is found, task is saved in queue
119  //
120  // 3. Task is executed by worker in one of three ways:
121  // 3.1. worker was inactive and starts a new thread to execute the task
122  // after it finishes its first task, it tries to find new ones:
123  // 3.2. gets a queued task on its parent core
124  // 3.3. if there is no queue task, notifies core of its status (free and active) and waits for new task.
125  // note: 3.2. and 3.3. together is an atomic operation (protected by mutex)
126  // Worker stops if waiting for new task times out (and becomes inactive).
127  //
128  // NOTE: core class is private nested to worker pool and cannot be instantiated outside it.
129  // worker class is private nested to core class.
130  //
131  // todo:
132  // [Optional] Define a way to stop worker pool, but to finish executing everything it has in queue.
133  //

340  // worker_pool<Context>::worker
341  //
342  // description
343  // the worker is a worker pool nested class and represents one instance of execution. its purpose is to store the
344  // context, manage multiple task executions of a single thread and collect statistics.
345  //
346  // how it works
347  // the worker is assigned a task and a new thread is started. when task is finished, the worker tries to execute
348  // more tasks, either by consuming one from task queue or by waiting for one. if it waits too long and it is given
349  // no task, the thread stops
350  //
351  // there are two types of workers in regard with the thread status:
352  //
353  // 1. inactive worker (initial state), thread is not running and must be started before executing task
354  // 2. active worker, either executing task or waiting for a new task.
355  //
356  // there are three ways task is executed:
357  //
358  // 1. by an inactive worker; it goes through next phases:
359  // - claiming from inactive list of workers
360  // - starting thread
361  // - claiming context
362  // - executing task
363  //
364  // 2. by an active worker (thread is running); it goes through next phases:
365  // - claiming from active list of workers
366  // - notifying and waking thread
367  // - executing task
368  //
369  // 3. by being claimed from task queue; if no worker (active or inactive) is available, task is queued on core
370  // to be executed when a worker finishes its current task; it goes through next phases
371  // - adding task to queue
372  // - claiming task from queue
373  // - executing task
374  //
375  // a more sensible part of task execution is pushing task to running thread, due to limit cases. there are up to
376  // three threads implicated into this process:
377  //
378  // 1. task pusher
379  // - whenever worker is claimed from free active list, the task is directly assigned (m_task_p is set)!
380  // (as a consequence, waiting thread cannot reject the task)
381  //
382  // 2. worker pool stopper
383  // - thread requests worker pool to stop. the signal is passed to all cores and workers; including workers
384  // that are waiting for tasks. that is signaled using m_stop field
385  //
386  // 3. task waiter
387  // the thread "lingers" waiting for new arriving tasks. the worker first adds itself to core's free active
388  // list and then waits until one of next conditions happen
389  // - task is assigned (m_task_p is not nil). it executes the task (regardless of m_stop).
390  // - thread is stopped (m_stop is true).
391  // - wait times out
392  // after waking up, if no task was assigned up to this point, the thread will attempt to remove its worker
393  // from free active list. a task may yet be assigned until the worker is removed from this list; if worker is
394  // not found to be removed, it means the worker was claimed and a task was pushed or is being pushed. thread
395  // is forced to wait for assigned task.
396  // if worker is removed successfully from free active list, its thread will stop and it will be added to
397  // inactive list.
398  // note that after wake up, m_stop does not affect the course of action in any way, only m_task_p. In most
399  // cases, m_stop being true will be equivalent to m_task_p being nil. in the very limited case when m_stop is
400  // true and a task is also assigned, we let the worker execute the task.
401  //
402  // note
403  // class is nested to worker pool and cannot be used outside it
404  //

533  template <typename Context>
534  worker_pool<Context>::worker_pool (std::size_t pool_size, std::size_t task_max_count,
535  context_manager_type &context_mgr, const char *name, std::size_t core_count,
536  bool debug_log, bool pool_threads, wait_seconds wait_for_task_time)
537  : m_max_workers (pool_size)
538  , m_task_max_count (task_max_count)
539  , m_task_count (0)
540  , m_context_manager (context_mgr)
541  , m_core_array (NULL)
542  , m_core_count (core_count)
543  , m_round_robin_counter (0)
544  , m_stopped (false)
545  , m_log (debug_log)
546  , m_pool_threads (pool_threads)
547  , m_wait_for_task_time (wait_for_task_time)
548  , m_name (name == NULL ? "" : name)
549  {
550  // initialize cores; we'll try to distribute pool evenly to all cores. if core count is not fully contained in
551  // pool size, some cores will have one additional worker
552 
553  if (m_core_count == 0)
554  {
555  assert (false);
556  m_core_count = 1;
557  }
558 
559  if (m_core_count > pool_size)
560  {
561  m_core_count = pool_size;
562  }
563 
564  m_core_array = new core[m_core_count];
565 
566  std::size_t quotient = m_max_workers / m_core_count;
567  std::size_t remainder = m_max_workers % m_core_count;
568  std::size_t it = 0;
569 
570  for (; it < remainder; it++)
571  {
572  m_core_array[it].init_pool_and_workers (*this, quotient + 1);
573  }
574  for (; it < m_core_count; it++)
575  {
576  m_core_array[it].init_pool_and_workers (*this, quotient);
577  }
578  }

621  template <typename Context>
622  void
623  worker_pool<Context>::stop_execution (void)
624  {
625  if (m_stopped.exchange (true))
626  {
627  // already stopped
628  return;
629  }
630  else
631  {
632  // I am responsible with stopping threads
633  }
634 
635  const std::chrono::seconds time_wait_to_thread_stop (30); // timeout duration = 30 secs
636  const std::chrono::milliseconds time_spin_sleep (10); // sleep between spins for 10 milliseconds
637 
638  // loop until all workers are stopped or until timeout expires
639  std::size_t stop_count = 0;
640  auto timeout = std::chrono::system_clock::now () + time_wait_to_thread_stop;
641 
642  bool is_not_stopped;
643  while (true)
644  {
645  // notify all cores to stop
646  is_not_stopped = false; // assume all are stopped
647  for (std::size_t it = 0; it < m_core_count; it++)
648  {
649  // notify all workers to stop. if any worker is still running, is_not_stopped = true is output
650  m_core_array[it].notify_stop (is_not_stopped);
651  }
652 
653  if (!is_not_stopped)
654  {
655  // all stopped
656  break;
657  }
658 
659  if (std::chrono::system_clock::now () > timeout)
660  {
661  // timed out
662  assert (false);
663  break;

777  template <typename Context>
778  std::size_t
779  worker_pool<Context>::get_round_robin_core_hash (void)
780  {
781  // cores are not necessarily equal, so we try to preserve the assignments proportional to their size.
782  // if the worker pool size is 15 and there are four cores, three of them will have four workers and one only three.
783  // task are dispatched in this order:
784  //
785  // core 1 | core 2 | core 3 | core 4
786  // 1 | 2 | 3 | 4
787  // 5 | 6 | 7 | 8
788  // 9 | 10 | 11 | 12
789  // 13 | 14 | 15 // last one is skipped this round to keep proportions
790  // 16 | 17 | 18 | 19
791  // ...
792  //
793 
794  // get a core index atomically
795  std::size_t index;
796  std::size_t next_index;
797 
798  while (true)
799  {
800  index = m_round_robin_counter;
801 
802  next_index = index + 1;
803  if (next_index == m_max_workers)
804  {
805  next_index = 0;
806  }
807 
808  if (m_round_robin_counter.compare_exchange_strong (index, next_index))
809  {
810  // my index is found
811  break;
812  }
813  }
814 
815  return index;
816  }

1004  template <typename Context>
1005  void
1006  worker_pool<Context>::core::start_all_workers (void)
1007  {
1008  worker *refp = NULL;
1009  const std::size_t AVAILABLE_STACK_DEFAULT_SIZE = 1024;
1010  extensible_array<worker *, AVAILABLE_STACK_DEFAULT_SIZE> available_stack;
1011 
1012  // how this works:
1013  //
1014  // we need to start all workers, but we need to consider the fact that some may be already running. what we need
1015  // to do is process the available workers and start threads for all those that don't have a thread started.
1016  //
1017  // workers that already have threads are saved and added back after processing all available workers.
1018  //
1019  // NOTE: this function does not guarantee that at the end all workers have threads. workers that stop their threads
1020  // during processing available workers are not restarted. however, we end up starting all or almost all
1021  // threads, which is good enough.
1022  //
1023 
1024  while (true)
1025  {
1026  // processing is done in two steps:
1027  //
1028  // 1. retrieve worker from available workers holding workers mutex
1029  // 2. verify if worker has a thread started
1030  // 2.1. if it doesn't have a thread, start one
1031  // 2.2. if it does have thread, save it to available_stack.
1032  //
1033 
1034  std::unique_lock<std::mutex> core_lock (m_workers_mutex);
1035  if (m_available_count == 0)
1036  {
1037  break;
1038  }
1039  refp = m_available_workers[--m_available_count];
1040  core_lock.unlock ();
1041 
1042  if (refp->has_thread ())
1043  {
1044  // stack to make available at the end
1045  available_stack.append (&refp, 1);
1046 
1047  // note: this worker's thread may stop soon or may have stopped already. this case is accepted.
1048  }
1049  else
1050  {
1051  // this thread is already stopped and we can start its thread
1052  refp->set_push_time_now ();
1053  refp->set_has_thread ();
1054  refp->start_thread ();
1055  }
1056  }
1057 
1058  // copy all workers having threads back to available array.
1059  if (available_stack.get_size () > 0)
1060  {
1061  std::unique_lock<std::mutex> core_lock (m_workers_mutex);
1062  if (m_available_count > 0)
1063  {
1064  // move current available to make room for older ones
1065  std::memmove (m_available_workers + available_stack.get_size (), m_available_workers,
1066  m_available_count * sizeof (worker *));
1067  }
1068 
1069  // copy from stack at the beginning of m_available_workers
1070  std::memcpy (m_available_workers, available_stack.get_array (), available_stack.get_memsize ());
1071 
1072  // update available count
1073  m_available_count += available_stack.get_size ();
1074  }
1075  }

1301  template <typename Context>
1302  bool
1303  worker_pool<Context>::core::worker::get_new_task (void)
1304  {
1305  assert (m_task_p == NULL);
1306 
1307  std::unique_lock<std::mutex> ulock (m_task_mutex, std::defer_lock);
1308 
1309  // check stop condition
1310  if (!m_stop)
1311  {
1312  // get a queued task or wait for one to come
1313 
1314  // either get a queued task or add to free active list
1315  // note: returned task cannot be saved directly to m_task_p. if worker is added to wait queue and NULL is returned,
1316  // current thread may be preempted. worker is then claimed from free active list and worker is assigned
1317  // a task. this changes expected behavior and can have unwanted consequences.
1318  task_type *task_p = m_parent_core->get_task_or_become_available (*this);
1319  if (task_p != NULL)
1320  {
1321  wp_worker_statset_time_and_increment (m_statistics, Wpstat_found_in_queue);
1322 
1323  // it is safe to set here
1324  m_task_p = task_p;
1325  return true;
1326  }
1327 
1328  // wait for task
1329  ulock.lock ();
1330  if (m_task_p == NULL && !m_stop)
1331  {
1332  // wait until a task is received or stopped ...
1333  // ... or time out
1334  condvar_wait (m_task_cv, ulock, m_parent_core->get_parent_pool ()->get_wait_for_task_time (),
1335  [this] () -> bool { return m_task_p != NULL || m_stop; });
1336  }
1337  else
1338  {
1339  // no need to wait
1340  }
1341  }
1342  else
1343  {
1344  // we need to add to available list
1345  m_parent_core->become_available (*this);
1346 
1347  ulock.lock ();
1348  }
1349 
1350  // did I get a task?
1351  if (m_task_p == NULL)
1352  {
1353  // no; this thread will stop. from this point forward, if a new task is assigned, a new thread must be spawned
1354  m_has_thread = false;
1355 
1356  // finish_run; we neet to retire context before another thread uses this worker
1357  m_statistics.m_timept = cubperf::clock::now ();
1358  finish_run ();
1359 
1360  return false;
1361  }
1362  else
1363  {
1364  // unlock mutex
1365  ulock.unlock ();
1366 
1367  // safe-guard - threads should no longer be available
1368  m_parent_core->check_worker_not_available (*this);
1369 
1370  // found task
1371  m_statistics.m_timept = m_push_time;
1372  wp_worker_statset_time_and_increment (m_statistics, Wpstat_wakeup_with_task);
1373  return true;
1374  }
1375  }

1380  {

1383  init_run (); // do stuff at the beginning like creating context
1384 
1385  if (m_task_p == NULL)
1386  {
1387  // started without task; get one
1388  if (get_new_task ())
1389  {
1390  assert (m_task_p != NULL);
1391  }
1392  }
1393
