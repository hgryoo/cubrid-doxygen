2  * tr_tree.c
3  *
4  * Implementation of treap.
5  * Copyright (C) 2001 Farooq Mela.
6  *
7  * $Id: tr_tree.c,v 1.14 2001/12/17 08:12:40 farooq Exp $
8  *
9  * cf. [Aragon and Seidel, 1996], [Knuth 1998]
10  *
11  * A treap is a randomized data structure in which each node of tree has an
12  * associated key and priority. The priority is chosen at random when the node
13  * is inserted into the tree. Each node is inserted so that the lexicographic
14  * order of the keys is preserved, and the priority of any node is less than
15  * the priority of either of its child nodes; in this way the treap is a
16  * combination of a tree and a min-heap. In this implementation, this is
17  * accomplished by first inserting the node according to lexigraphical order of
18  * keys as in a normal binary tree, and then, if needed, sifting the node
19  * upwards using a series of rotations until the heap property of the tree is
20  * restored.
21  */

173 int
174 tr_tree_insert(tr_tree *tree, void *key, void *dat, int overwrite)
175 {
176  int rv = 0;
177  tr_node *node, *parent = NULL;
178  priority_t prio;
179 
180  ASSERT(tree != NULL);
181 
182  node = tree->root;
183  while (node) {
184  rv = tree->key_cmp(key, node->key);
185  if (rv < 0)
186  parent = node, node = node->llink;
187  else if (rv > 0)
188  parent = node, node = node->rlink;
189  else {
190  if (overwrite == 0)
191  return 1;
192  if (tree->key_del)
193  tree->key_del(node->key);
194  if (tree->dat_del)
195  tree->dat_del(node->dat);
196  node->key = key;
197  node->dat = dat;
198  return 0;
199  }
200  }
201 
202  if ((node = node_new(key, dat)) == NULL)
203  return -1;
204  prio = (tree->randgen * RGEN_A + RGEN_M) & RGEN_MASK;
205  node->prio = tree->randgen = prio;
206 
207  if ((node->parent = parent) == NULL) {
208  tree->root = node;
209  ASSERT(tree->count == 0);
210  tree->count = 1;
211  return 0;
212  }
213  if (rv < 0)
214  parent->llink = node;
215  else
216  parent->rlink = node;
217 
218  do {
219  if (parent->prio <= node->prio)
220  break;
221  if (parent->llink == node)
222  rot_right(tree, parent);
223  else
224  rot_left(tree, parent);
225  parent = node->parent;
226  } while (parent);
227 
228  tree->count++;
229  return 0;
230 }

284 int
285 tr_tree_remove(tr_tree *tree, const void *key, int del)
286 {
287  int rv;
288  tr_node *node, *out, *parent = NULL;
289 
290  ASSERT(tree != NULL);
291 
292  node = tree->root;
293  while (node) {
294  rv = tree->key_cmp(key, node->key);
295  if (rv < 0)
296  parent = node, node = node->llink;
297  else if (rv > 0)
298  parent = node, node = node->rlink;
299  else
300  break;
301  }
302 
303  if (node == NULL)
304  return -1;
305 
306  while (node->llink && node->rlink) {
307  if (node->llink->prio < node->rlink->prio)
308  rot_right(tree, node);
309  else
310  rot_left(tree, node);
311  }
312  parent = node->parent;
313  out = node->llink ? node->llink : node->rlink;
314  if (out)
315  out->parent = parent;
316  if (parent) {
317  if (parent->llink == node)
318  parent->llink = out;
319  else
320  parent->rlink = out;
321  } else {
322  tree->root = out;
323  }
324 
325  if (del) {
326  if (tree->key_del)
327  tree->key_del(node->key);
328  if (tree->dat_del)
329  tree->dat_del(node->dat);
330  }
331  FREE(node);
332 
333  tree->count--;
334  return 0;
335 }
