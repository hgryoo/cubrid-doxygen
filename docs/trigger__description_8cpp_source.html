65 int trigger_description::init (struct db_object *trobj)
66 {
67  assert (trobj != NULL);
68 
69  char *condition = NULL, *action = NULL, *classname;
70  TR_TRIGGER *trigger;
71 
72  trigger = tr_map_trigger (trobj, 1);
73  if (trigger == NULL)
74  {
75  return ER_FAILED;
76  }
77 
78  /* even though we have the trigger, use these to get the expressions translated into a simple string */
79  if (db_trigger_condition (trobj, &condition) != NO_ERROR)
80  {
81  ws_free_string (condition);
82  return ER_FAILED;
83  }
84  if (db_trigger_action (trobj, &action) != NO_ERROR)
85  {
86  ws_free_string (action);
87  return ER_FAILED;
88  }
89 
90  /* copy these */
91  this->name = object_print::copy_string (trigger->name);
92  this->attribute = object_print::copy_string (trigger->attribute);
93  this->comment = object_print::copy_string (trigger->comment);
94 
95  /* these are already copies */
96  this->condition = condition;
97  this->action = action;
98 
99  /* these are constant strings that don't need to ever change */
100  this->event = tr_event_as_string (trigger->event);
101  this->condition_time = object_printer::describe_trigger_condition_time (*trigger);
102  this->action_time = object_printer::describe_trigger_action_time (*trigger);
103 
104  /* only show status if its inactive */
105  if (trigger->status != TR_STATUS_ACTIVE)
106  {
107  this->status = tr_status_as_string (trigger->status);
108  }
109 
110  /* if its 0, leave it out */
111  if (trigger->priority != 0.0)
112  {
113  char temp_buffer[64];
114 
115  sprintf (temp_buffer, "%f", trigger->priority);
116  this->priority = object_print::copy_string (temp_buffer);
117  }
118 
119  if (trigger->class_mop != NULL)
120  {
121  classname = (char *) sm_get_ch_name (trigger->class_mop);
122  if (classname != NULL)
123  {
124  this->class_name = object_print::copy_string ((char *) classname);
125  }
126  else
127  {
128  this->class_name = object_print::copy_string ("*** deleted class ***");
129  }
130 
131  /* format the full event specification so csql can display it without being dependent on syntax */
132 
133  char buffer[ (SM_MAX_IDENTIFIER_LENGTH * 2) + 32];
134 
135  if (this->attribute != NULL)
136  {
137  sprintf (buffer, "%s ON %s(%s)", this->event, this->class_name, this->attribute);
138  }
139  else
140  {
141  sprintf (buffer, "%s ON %s", this->event, this->class_name);
142  }
143 
144  this->full_event = object_print::copy_string (buffer);
145  }
146  else
147  {
148  /* just make a copy of this so csql can simply use it without thinking */
149  this->full_event = object_print::copy_string ((char *) this->event);
150  }
151 
152  return NO_ERROR;
153 }
