184 /* TR_RECURSION_DECISION */
185 /* When analyzing the stack searching for recursion (which is forbidden),
186  * there are three scenarios:
187  * 1. no recursive activity found: continue
188  * 2. we found a recursive trigger and we must return an error
189  * 3. we found a recursive STATEMENT trigger. This is supposed to happen, and
190  * we should not continue the trigger firing chain, but we should allow
191  * the transaction to go on.
192  */
193 typedef enum
194 {
195  TR_DECISION_CONTINUE,
196  TR_DECISION_HALT_WITH_ERROR,
197  TR_DECISION_DO_NOT_CONTINUE
198 } TR_RECURSION_DECISION;
