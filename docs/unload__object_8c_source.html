201  * set_referenced_subclasses - set class as referenced
202  * return: NO_ERROR, if successful, error number, if not successful.
203  * class(in): root class
204  * Note:
205  * CURRENTLY, ALWAYS RETURN NO_ERROR
206  */
207 static int
208 set_referenced_subclasses (DB_OBJECT * class_)
209 {
210  int error = NO_ERROR;
211  int *cls_no_ptr;
212  SM_CLASS *class_ptr;

217

219  if (error_code == NO_ERROR && cls_no_ptr != NULL)

226  {

228  }
229  MARK_CLASS_REFERENCED (*cls_no_ptr);
230  }
231  }
232  else
233  {
234  if (!IS_CLASS_REFERENCED (*cls_no_ptr) && !IS_CLASS_REQUESTED (*cls_no_ptr))
235  {
236  check_reference_chain = true;
237  }
238  MARK_CLASS_REFERENCED (*cls_no_ptr);
239  }
240  }
241  else
242  {
243 #if defined(CUBRID_DEBUG)
244  fprintf (stdout, "cls_no_ptr is NULL\n");
245 #endif /* CUBRID_DEBUG */
246  }
247 
248  ws_find (class_, (MOBJ *) (&class_ptr));

274 /*
275  * check_referenced_domain - check for OBJECT domain as referenced
276  * return: true, if found referened OBJECT domain. false, if not
277  * found referenced OBJECT domain.
278  * dom_list(in): domain list of class attributes
279  * set_cls_ref(in): for true, do checking. for false, do marking
280  * num_cls_refp(out): number of referenced classes.
281  * Note:
282  * for referenced CLASS domain, mark the CLASS and set the number of
283  * referenced classes.
284  */
285 static bool
286 check_referenced_domain (DB_DOMAIN * dom_list, bool set_cls_ref, int *num_cls_refp)
287 {
288  bool found_object_dom;
289  DB_DOMAIN *dom;
290  DB_TYPE type;
291  DB_OBJECT *class_;
292 
293  found_object_dom = false; /* init */
294 
295  for (dom = dom_list; dom && !found_object_dom; dom = db_domain_next (dom))
296  {
297  type = TP_DOMAIN_TYPE (dom);
298  switch (type)
299  {
300  case DB_TYPE_OBJECT:
301  class_ = db_domain_class (dom);
302  if (class_ == NULL)
303  {
304  return true; /* found object domain */
305  }
306 
307  *num_cls_refp += 1; /* increase number of reference to class */
308 
309  if (set_cls_ref)
310  {
311  if (set_referenced_subclasses (class_) != NO_ERROR)
312  {
313  /* cause error - currently, not happened */
314  return true;
315  }
316  }
317  break;
318  case DB_TYPE_SET:
319  case DB_TYPE_MULTISET:
320  case DB_TYPE_SEQUENCE:
321  found_object_dom = check_referenced_domain (db_domain_set (dom), set_cls_ref, num_cls_refp);
322  break;
323  default:
324  break;
325  }
326  }
327  return found_object_dom;
328 }

377 /*
378  * mark_referenced_domain - mark given SM_CLASS closure
379  * return: true if no error, false otherwise
380  * class_ptr(in): SM_CLASS
381  * num_set(out): amortized marking number of SM_CLASS closure
382  */
383 static bool
384 mark_referenced_domain (SM_CLASS * class_ptr, int *num_set)
385 {
386  SM_ATTRIBUTE *attribute;
387 
388  if (class_ptr == NULL)
389  return true;
390 
391  for (attribute = class_ptr->shared; attribute != NULL; attribute = (SM_ATTRIBUTE *) attribute->header.next)
392  {
393  if (check_referenced_domain (attribute->domain, true /* do marking */ ,
394  num_set) != false)
395  {
396  return false;
397  }
398  }
399 
400  for (attribute = class_ptr->class_attributes; attribute != NULL; attribute = (SM_ATTRIBUTE *) attribute->header.next)
401  {
402  if (check_referenced_domain (attribute->domain, true /* do marking */ ,
403  num_set) != false)
404  {
405  return false;
406  }
407  }
408 
409  for (attribute = class_ptr->ordered_attributes; attribute; attribute = attribute->order_link)
410  {
411  if (attribute->header.name_space != ID_ATTRIBUTE)
412  {
413  continue;
414  }
415  if (check_referenced_domain (attribute->domain, true /* do marking */ ,
416  num_set) != false)
417  {
418  return false;
419  }
420  }
421  return true;
422 }

427  * return: 0 for success. 1 for error
428  * exec_name(in): utility name
429  */
430 int
431 extractobjects (const char *exec_name)
432 {
433  int i, error;
434  HFID *hfid;
435  int est_objects = 0;
436  int cache_size;
437  SM_CLASS *class_ptr;
438  const char **cptr;
439  int status = 0;
440  int num_unload_classes = 0;
441  DB_OBJECT **unload_class_table = NULL;
442  bool has_obj_ref;
443  int num_cls_ref;
444  SM_ATTRIBUTE *attribute;
445  void (*prev_intr_handler) (int sig);
446  void (*prev_term_handler) (int sig);
447 #if !defined (WINDOWS)
448  void (*prev_quit_handler) (int sig);
449 #endif
450  LOG_LSA lsa;
451  char unloadlog_filename[PATH_MAX];
452 
453  /* register new signal handlers */
454  prev_intr_handler = os_set_signal_handler (SIGINT, extractobjects_term_handler);
455  prev_term_handler = os_set_signal_handler (SIGTERM, extractobjects_term_handler);
456 #if !defined(WINDOWS)
457  prev_quit_handler = os_set_signal_handler (SIGQUIT, extractobjects_term_handler);
458 #endif
459 
460  if (cached_pages <= 0)
461  {
462  fprintf (stderr,
463  msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_UNLOADDB, UNLOADDB_MSG_INVALID_CACHED_PAGES));
464  return 1;
465  }
466  if (page_size < (ssize_t) (sizeof (OID) + sizeof (int)))
467  {
468  fprintf (stderr,
469  msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_UNLOADDB, UNLOADDB_MSG_INVALID_CACHED_PAGE_SIZE));
470  return 1;
471  }
472 
473  /*
474  * Open output file
475  */
476  if (output_dirname == NULL)
477  output_dirname = ".";
478  if (strlen (output_dirname) > PATH_MAX - 8)
479  {
480  fprintf (stderr, msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_UNLOADDB, UNLOADDB_MSG_INVALID_DIR_NAME));
481  return 1;
482  }
483 
484  if (!datafile_per_class)
485  {
486  output_filename = (char *) malloc (PATH_MAX);
487 
488  if (output_filename == NULL)
489  {
490  return 1;
491  }
492  snprintf (output_filename, PATH_MAX - 1, "%s/%s%s", output_dirname, output_prefix, OBJECT_SUFFIX);
493 
494  obj_out->fp = fopen_ex (output_filename, "wb");
495  if (obj_out->fp == NULL)
496  {
497  fprintf (stderr, "%s: %s.\n\n", exec_name, strerror (errno));
498  free_and_init (output_filename);
499  return errno;
500  }
501  }
502 
503  {
504  struct stat stbuf;
505  int blksize;
506 
507  blksize = 4096; /* init */
508  if (stat (output_dirname, &stbuf) == -1)
509  {
510  ; /* nop */
511  }
512  else
513  {
514 #if defined (WINDOWS)
515  blksize = 4096;
516 #else /* !WINDOWS */
517  blksize = stbuf.st_blksize;
518 #endif /* WINDOWS */
519  }
520 
521  /*
522  * Determine the IO buffer size by specifying a multiple of the
523  * natural block size for the device.
524  * NEED FUTURE OPTIMIZATION
525  */
526  obj_out->iosize = 1024 * 1024; /* 1 Mbyte */
527  obj_out->iosize -= (obj_out->iosize % blksize);
528 
529  obj_out->buffer = (char *) malloc (obj_out->iosize);
530 
531  obj_out->ptr = obj_out->buffer; /* init */
532  obj_out->count = 0; /* init */
533  }
534 
535  /*
536  * The user indicates which classes are to be processed by
537  * using -i with a file that contains a list of classes.
538  * If the -i option is not used, it means process all classes.
539  * Thus if input_filename is null, it means process all classes.
540  * Three bit arrays are allocated to indicate whether a class
541  * is requested, is referenced or is processed. The index
542  * into these arrays is the same as the index into class_table->mops.
543  */
544  if ((unload_class_table = (DB_OBJECT **) malloc (DB_SIZEOF (void *) * class_table->num)) == NULL)
545  {
546  status = 1;
547  goto end;
548  }
549  for (i = 0; i < class_table->num; ++i)
550  {
551  unload_class_table[i] = NULL;
552  }
553  if ((class_requested = (char *) malloc ((class_table->num + 7) / 8)) == NULL)
554  {
555  status = 1;
556  goto end;
557  }
558  if ((class_referenced = (char *) malloc ((class_table->num + 7) / 8)) == NULL)
559  {
560  status = 1;
561  goto end;
562  }
563  if ((class_processed = (char *) malloc ((class_table->num + 7) / 8)) == NULL)
564  {
565  status = 1;
566  goto end;
567  }
568 
569  memset (class_requested, 0, (class_table->num + 7) / 8);
570  memset (class_referenced, 0, (class_table->num + 7) / 8);
571  memset (class_processed, 0, (class_table->num + 7) / 8);
572 
573  /*
574  * Create the class hash table
575  * Its purpose is to hash a class OID to the index into the
576  * class_table->mops array.
577  */
578  cl_table = fh_create ("class hash", 4096, 1024, 4, NULL, FH_OID_KEY, DB_SIZEOF (int), oid_hash, oid_compare_equals);
579  if (cl_table == NULL)
580  {
581  status = 1;
582  goto end;
583  }
584 
585  has_obj_ref = false; /* init */
586  num_cls_ref = 0; /* init */
587 
588  /*
589  * Total the number of objects & mark requested classes.
590  */
591 #if defined(CUBRID_DEBUG)
592  fprintf (stdout, "----- all class dump -----\n");
593 #endif /* CUBRID_DEBUG */
594  for (i = 0; i < class_table->num; i++)
595  {
596  if (WS_IS_DELETED (class_table->mops[i]) || class_table->mops[i] == sm_Root_class_mop)
597  {
598  continue;
599  }
600 
601  error = au_fetch_class (class_table->mops[i], NULL, AU_FETCH_READ, AU_SELECT);
602  if (error != NO_ERROR)
603  {
604  continue;
605  }
606 
607  ws_find (class_table->mops[i], (MOBJ *) (&class_ptr));
608  if (class_ptr == NULL)
609  {
610  status = 1;
611  goto end;
612  }
613 
614  for (cptr = prohibited_classes; *cptr; ++cptr)
615  {
616  if (strcmp (*cptr, sm_ch_name ((MOBJ) class_ptr)) == 0)
617  {
618  break;
619  }
620  }
621  if (*cptr == NULL)
622  {
623 #if defined(CUBRID_DEBUG)
624  fprintf (stdout, "%s%s%s\n", PRINT_IDENTIFIER (sm_ch_name ((MOBJ) class_ptr)));
625 #endif /* CUBRID_DEBUG */
626 
627  fh_put (cl_table, ws_oid (class_table->mops[i]), &i);
628  if (input_filename)
629  {
630  if (is_req_class (class_table->mops[i]))
631  {
632  MARK_CLASS_REQUESTED (i);
633  }
634  else if (!required_class_only)
635  {
636  error = sm_is_system_class (class_table->mops[i]);
637  if (error < 0)
638  {
639  status = 1;
640  goto end;
641  }
642  if (error > 0)
643  {
644  MARK_CLASS_REQUESTED (i);
645  error = NO_ERROR;
646  }
647  }
648  }
649  else
650  MARK_CLASS_REQUESTED (i);
651 
652  if (!datafile_per_class && (!required_class_only || IS_CLASS_REQUESTED (i)))
653  {
654  if (text_print
655  (obj_out, NULL, 0, "%cid %s%s%s %d\n", '%', PRINT_IDENTIFIER (sm_ch_name ((MOBJ) class_ptr)),
656  i) != NO_ERROR)
657  {
658  status = 1;
659  goto end;
660  }
661  }
662 
663  if (IS_CLASS_REQUESTED (i))
664  {
665  if (!datafile_per_class)
666  {
667  if (!has_obj_ref)
668  { /* not found object domain */
669  for (attribute = class_ptr->shared; attribute != NULL;
670  attribute = (SM_ATTRIBUTE *) attribute->header.next)
671  {
672  /* false -> don't set */
673  if ((has_obj_ref = check_referenced_domain (attribute->domain, false, &num_cls_ref)) == true)
674  {
675 #if defined(CUBRID_DEBUG)
676  fprintf (stdout, "found OBJECT domain: %s%s%s->%s\n",
677  PRINT_IDENTIFIER (class_ptr->header.name), db_attribute_name (attribute));
678 #endif /* CUBRID_DEBUG */
679  break;
680  }
681  }
682  }
683 
684  if (!has_obj_ref)
685  { /* not found object domain */
686  for (attribute = class_ptr->class_attributes; attribute != NULL;
687  attribute = (SM_ATTRIBUTE *) attribute->header.next)
688  {
689  /* false -> don't set */
690  if ((has_obj_ref = check_referenced_domain (attribute->domain, false, &num_cls_ref)) == true)
691  {
692 #if defined(CUBRID_DEBUG)
693  fprintf (stdout, "found OBJECT domain: %s%s%s->%s\n",
694  PRINT_IDENTIFIER (class_ptr->header.name), db_attribute_name (attribute));
695 #endif /* CUBRID_DEBUG */
696  break;
697  }
698  }
699  }
700 
701  if (!has_obj_ref)
702  { /* not found object domain */
703  for (attribute = class_ptr->ordered_attributes; attribute; attribute = attribute->order_link)
704  {
705  if (attribute->header.name_space != ID_ATTRIBUTE)
706  {
707  continue;
708  }
709  has_obj_ref = check_referenced_domain (attribute->domain, false
