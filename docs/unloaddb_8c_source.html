84 /*
85  * unloaddb - main function
86  * return: 0 if successful, non zero if error.
87  * argc(in): number of command line arguments
88  * argv(in): array containing command line arguments
89  */
90 int
91 unloaddb (UTIL_FUNCTION_ARG * arg)
92 {
93  UTIL_ARG_MAP *arg_map = arg->arg_map;
94  const char *exec_name = arg->command_name;
95  char er_msg_file[PATH_MAX];
96  int error;
97  int status = 0;
98  int i;
99  char *user, *password;
100  int au_save;
101  EMIT_STORAGE_ORDER order;
102 
103  if (utility_get_option_string_table_size (arg_map) != 1)
104  {
105  unload_usage (arg->argv0);
106  return -1;
107  }
108 
109  input_filename = utility_get_option_string_value (arg_map, UNLOAD_INPUT_CLASS_FILE_S, 0);
110  include_references = utility_get_option_bool_value (arg_map, UNLOAD_INCLUDE_REFERENCE_S);
111  required_class_only = utility_get_option_bool_value (arg_map, UNLOAD_INPUT_CLASS_ONLY_S);
112  datafile_per_class = utility_get_option_bool_value (arg_map, UNLOAD_DATAFILE_PER_CLASS_S);
113  lo_count = utility_get_option_int_value (arg_map, UNLOAD_LO_COUNT_S);
114  est_size = utility_get_option_int_value (arg_map, UNLOAD_ESTIMATED_SIZE_S);
115  cached_pages = utility_get_option_int_value (arg_map, UNLOAD_CACHED_PAGES_S);
116  output_dirname = utility_get_option_string_value (arg_map, UNLOAD_OUTPUT_PATH_S, 0);
117  do_schema = utility_get_option_bool_value (arg_map, UNLOAD_SCHEMA_ONLY_S);
118  do_objects = utility_get_option_bool_value (arg_map, UNLOAD_DATA_ONLY_S);
119  output_prefix = utility_get_option_string_value (arg_map, UNLOAD_OUTPUT_PREFIX_S, 0);
120  hash_filename = utility_get_option_string_value (arg_map, UNLOAD_HASH_FILE_S, 0);
121  verbose_flag = utility_get_option_bool_value (arg_map, UNLOAD_VERBOSE_S);
122  database_name = utility_get_option_string_value (arg_map, OPTION_STRING_TABLE, 0);
123  user = utility_get_option_string_value (arg_map, UNLOAD_USER_S, 0);
124  password = utility_get_option_string_value (arg_map, UNLOAD_PASSWORD_S, 0);
125  if (utility_get_option_bool_value (arg_map, UNLOAD_KEEP_STORAGE_ORDER_S))
126  {
127  order = FOLLOW_STORAGE_ORDER;
128  }
129  else
130  {
131  order = FOLLOW_ATTRIBUTE_ORDER;
132  }
133 
134  /* depreciated */
135  utility_get_option_bool_value (arg_map, UNLOAD_USE_DELIMITER_S);
136 
137 
138  if (database_name == NULL)
139  {
140  status = 1;
141  /* TODO: print usage */
142  util_log_write_errid (MSGCAT_UTIL_GENERIC_INVALID_ARGUMENT);
143  goto end;
144  }
145 
146  if (!output_prefix)
147  {
148  output_prefix = database_name;
149  }
150 
151  /* error message log file */
152  snprintf (er_msg_file, sizeof (er_msg_file) - 1, "%s_%s.err", database_name, exec_name);
153  er_init (er_msg_file, ER_NEVER_EXIT);
154 
155  sysprm_set_force (prm_get_name (PRM_ID_JAVA_STORED_PROCEDURE), "no");
156 
157  /*
158  * Open db
159  */
160  if (user == NULL || user[0] == '\0')
161  {
162  user = (char *) "DBA";
163  }
164 
165  error = db_restart_ex (exec_name, database_name, user, password, NULL, DB_CLIENT_TYPE_ADMIN_UTILITY);
166  if (error == NO_ERROR)
167  {
168  /* pass */
169  }
170  else if (password == NULL && db_error_code () == ER_AU_INVALID_PASSWORD)
171  {
172  /* console input a password */
173  password =
174  getpass (msgcat_message (MSGCAT_CATALOG_UTILS, MSGCAT_UTIL_SET_UNLOADDB, UNLOADDB_MSG_PASSWORD_PROMPT));
175  error = db_restart_ex (exec_name, database_name, user, password, NULL, DB_CLIENT_TYPE_ADMIN_UTILITY);
176  if (error != NO_ERROR)
177  {
178  PRINT_AND_LOG_ERR_MSG ("%s: %s\n", exec_name, db_error_string (3));
179  status = error;
180  goto end;
181  }
182  }
183  else if (error != NO_ERROR)
184  {
185  /* error */
186  PRINT_AND_LOG_ERR_MSG ("%s: %s\n", exec_name, db_error_string (3));
187  status = error;
188  goto end;
189  }
190 
191  ignore_err_flag = prm_get_bool_value (PRM_ID_UNLOADDB_IGNORE_ERROR);
192 
193  if (!status)
194  {
195  db_set_lock_timeout (prm_get_integer_value (PRM_ID_UNLOADDB_LOCK_TIMEOUT));
196  }
197 
198  if (!input_filename)
199  {
200  required_class_only = false;
201  }
202  if (required_class_only && include_references)
203  {
204  include_references = false;
205  fprintf (stdout, "warning: '-ir' option is ignored.\n");
206  fflush (stdout);
207  }
208 
209  class_table = locator_get_all_mops (sm_Root_class_mop, DB_FETCH_READ, NULL);
210  if (input_filename)
211  {
212  /* It may not be needed */
213  if (locator_decache_all_lock_instances (sm_Root_class_mop) != NO_ERROR)
214  {
215  util_log_write_errstr ("%s\n", db_error_string (3));
216  status = 1;
217  goto end;
218  }
219  }
220 
221  if (class_table == NULL)
222  {
223  util_log_write_errstr ("%s\n", db_error_string (3));
224  status = 1;
225  goto end;
226  }
227 
228  req_class_table = (DB_OBJECT **) malloc (DB_SIZEOF (void *) * class_table->num);
229  if (req_class_table == NULL)
230  {
231  util_log_write_errid (MSGCAT_UTIL_GENERIC_NO_MEM);
232  status = 1;
233  goto end;
234  }
235 
236  for (i = 0; i < class_table->num; ++i)
237  {
238  req_class_table[i] = NULL;
239  }
240 
241  if (get_requested_classes (input_filename, req_class_table) != 0)
242  {
243  util_log_write_errstr ("%s\n", db_error_string (3));
244  status = 1;
245  goto end;
246  }
247 
248  if (input_filename)
249  {
250  int error;
251 
252  for (i = 0; req_class_table[i]; i++)
253  {
254  error = au_fetch_class (req_class_table[i], NULL, AU_FETCH_READ, AU_SELECT);
255  if (error != NO_ERROR)
256  {
257  /* A required class is not granted. */
258  MOBJ object = NULL;
259 
260  ws_find (req_class_table[i], &object);
261  if (object != NULL)
262  {
263  PRINT_AND_LOG_ERR_MSG ("%s: %s\n", sm_ch_name (object), db_error_string (3));
264  }
265  status = 1;
266  }
267  }
268 
269  if (status != 0)
270  {
271  goto end;
272  }
273  }
274 
275  if (!status && (do_schema || !do_objects))
276  {
277  /* do authorization as well in extractschema() */
278  if (extractschema (exec_name, 1, order))
279  {
280  status = 1;
281  }
282  }
283 
284  AU_SAVE_AND_ENABLE (au_save);
285  if (!status && (do_objects || !do_schema))
286  {
287  if (extractobjects (exec_name))
288  {
289  status = 1;
290  }
291  }
292  AU_RESTORE (au_save);
293 
294  /* if an error occur, print error message */
295  if (status)
296  {
297  if (db_error_code () != NO_ERROR)
298  {
299  PRINT_AND_LOG_ERR_MSG ("%s: %s\n", exec_name, db_error_string (3));
300  }
301  }
302 
303  /*
304  * Shutdown db
305  */
306  error = db_shutdown ();
307  if (error != NO_ERROR)
308  {
309  PRINT_AND_LOG_ERR_MSG ("%s: %s\n", exec_name, db_error_string (3));
310  status = error;
311  }
312 
313 end:
314  if (class_table)
315  {
316  locator_free_list_mops (class_table);
317  }
318  if (req_class_table)
319  {
320  free_and_init (req_class_table);
321  }
322 
323  return status;
324 }
