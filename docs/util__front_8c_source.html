299 /*
300  * get_long_arg_by_old_arg() - get new argument with long format
301  * return: argument string with long format
302  * old_arg(in): old argument string
303  * match_table(in): a mapping table of arguments
304  */
305 static char *
306 get_long_arg_by_old_arg (char *old_arg, ARG_MAP_TABLE * match_table)
307 {
308  int i;
309  for (i = 0; match_table[i].old_arg != 0; i++)
310  {
311  if (strcmp (old_arg, match_table[i].old_arg) == 0)
312  {
313  return (char *) match_table[i].long_arg;
314  }
315  }
316  return NULL;
317 }
318 
319 /*
320  * make_argv() - convert old arguments to new arguments with long format
321  * return:
322  * argc(in): size of the array
323  * argv(out): an array of new arguments
324  * match_table(in): a mapping table of arguments
325  */
326 static int
327 convert_argv (int argc, char **argv, ARG_MAP_TABLE * match_table)
328 {
329  int i, idx;
330  char *new_arg;
331 
332  for (i = 0, idx = 0; i < argc && argv[i] != NULL; i++)
333  {
334  new_arg = get_long_arg_by_old_arg (argv[i], match_table);
335  if (new_arg == NULL && argv[i][0] == '-')
336  {
337  fprintf (stderr, "invalid option: %s\n", argv[i]);
338  return ER_FAILED;
339  }
340  else if (new_arg == (char *) -1)
341  {
342  continue;
343  }
344  else if (new_arg != NULL)
345  {
346  argv[idx++] = new_arg;
347  }
348  else
349  {
350  argv[idx++] = argv[i];
351  }
352  }
353 
354  argv[idx] = 0;
355  return NO_ERROR;
356 }

358 /*
359  * main() - csql main module.
360  * return: EXIT_SUCCESS or EXIT_FAILURE
361  */
362 int
363 main (int argc, char **argv)
364 {
365  char *program_name = basename (argv[0]);
366  int i, argc_index = 0;
367  char **admin_argv = NULL;
368 
369  /* validate the number of arguments to avoid klocwork's error message */
370  if (argc < 0 || argc > 1024)
371  {
372  goto failure;
373  }
374 
375  if (strcmp (program_name, UTIL_SQLX_NAME) == 0)
376  {
377  admin_argv = (char **) malloc ((argc + 1) * sizeof (char *));
378  if (admin_argv == NULL)
379  {
380  goto failure;
381  }
382 
383  memcpy (admin_argv, argv, argc * sizeof (char *));
384  if (convert_argv (argc, &admin_argv[1], ua_Sqlx_map) != NO_ERROR)
385  {
386  goto failure;
387  }
388 
389  if (execvp (UTIL_CSQL_NAME, admin_argv) == -1)
390  {
391  perror ("execvp");
392  }
393  }
394  else if (strcmp (program_name, UTIL_OLD_COMMDB_NAME) == 0)
395  {
396  admin_argv = (char **) malloc ((argc + 1) * sizeof (char *));
397  if (admin_argv == NULL)
398  {
399  goto failure;
400  }
401 
402  memcpy (admin_argv, argv, argc * sizeof (char *));
403  if (convert_argv (argc, &admin_argv[1], us_Commdb_map) != NO_ERROR)
404  {
405  goto failure;
406  }
407 
408  if (execvp (UTIL_COMMDB_NAME, argv) == -1)
409  {
410  perror ("execvp");
411  }
412  }
413  else
414  {
415  for (i = 0; ua_Util_table[i].app_name != 0; i++)
416  {
417  if (strcmp (ua_Util_table[i].app_name, program_name) == 0)
418  {
419  admin_argv = (char **) malloc ((argc + 2) * sizeof (char *));
420  if (admin_argv == NULL)
421  {
422  goto failure;
423  }
424 
425  admin_argv[argc_index++] = (char *) UTIL_CUBRID;
426  admin_argv[argc_index++] = (char *) ua_Util_table[i].util_name;
427  memcpy (&admin_argv[argc_index], &argv[1], (argc - 1) * sizeof (char *));
428  if (convert_argv (argc, &admin_argv[argc_index], ua_Util_table[i].match_table) != NO_ERROR)
429  {
430  goto failure;
431  }
432  break;
433  }
434  }
435 
436  if (argc_index == 0)
437  {
438  goto failure;
439  }
440 
441  if (execvp (UTIL_ADMIN_NAME, admin_argv) == -1)
442  {
443  perror ("execvp");
444  }
445  }
446 
447  if (admin_argv)
448  {
449  free (admin_argv);
450  }
451  return EXIT_SUCCESS;
452 
453 failure:
454  fprintf (stderr, "This utility is deprecated. Use 'cubrid' utility.\n");
455  if (admin_argv)
456  {
457  free (admin_argv);
458  }
459  return EXIT_FAILURE;
460 }
