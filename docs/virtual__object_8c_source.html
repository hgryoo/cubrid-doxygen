63  * vid_get_class_object() - get class object given its class mop
64  * return: NO_ERROR if all OK, a negative error code otherwise
65  * class_p(in): a class mop
66  * obj(out): class_p' class object if all OK, NULL otherwise

68  * Note:
69  * modifies: er state, obj
70  * effects : set obj to class_p' class object

74 {

78  }

86  oid = &oid_Null_oid;

96  }
97  return NO_ERROR;

185 /*
186  * vid_fetch_instance() - FETCH A VIRTUAL INSTANCE
187  * return: MOBJ
188  * mop(in): Memory Object Pointer of virtual instance to fetch
189  * purpose(in): Fetch purpose: Valid ones:
190  * DB_FETCH_READ
191  * DB_FETCH_WRITE
192  * DB_FETCH_DIRTY
193  * read_fetch_version_type(in): Fetch version type in case of read
194  *
195  * Note:
196  * Fetch the instance associated with the given mop for the given purpose.
197  * Locks are handled by the underlying database.
198  */
199 MOBJ
200 vid_fetch_instance (MOP mop, DB_FETCH_MODE purpose, LC_FETCH_VERSION_TYPE read_fetch_version_type)
201 {
202  MOBJ inst;
203  MOP class_mop, base_mop;
204  SM_CLASS *class_p;
205  LC_FETCH_VERSION_TYPE fetch_version_type;
206 
207  if (!mop->is_vid)
208  {
209  return (MOBJ) 0;
210  }
211 
212  class_mop = ws_class_mop (mop);
213 
214  if (vid_get_class_object (class_mop, &class_p) < 0)
215  {
216  return (MOBJ) 0;
217  }
218 
219  ws_find (mop, &inst);
220 
221  /* check for out-of-date instance */
222  if (inst && (mop->lock == NULL_LOCK))
223  {
224  ws_decache (mop);
225  inst = (MOBJ) 0;
226  }
227  if (!inst)
228  {
229  /* fetch the object pointed to by the vclass */
230  base_mop = vid_get_referenced_mop (mop);
231  if (base_mop)
232  {
233  AU_FETCHMODE fetch_mode;
234 
235  if (purpose == DB_FETCH_WRITE)
236  {
237  fetch_mode = AU_FETCH_WRITE;
238  fetch_version_type = LC_FETCH_MVCC_VERSION;
239  }
240  else
241  {
242  fetch_mode = AU_FETCH_READ;
243  fetch_version_type = read_fetch_version_type;
244  }
245  if (au_fetch_instance_force (base_mop, &inst, fetch_mode, fetch_version_type) != NO_ERROR)
246  {
247  inst = (MOBJ) 0;
248  }
249  }
250  else
251  {
252  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_VID_LOST_NON_UPDATABLE_OBJECT, 0);
253  }
254  }
255  if ((purpose == DB_FETCH_WRITE) && (inst))
256  {
257  ws_dirty (mop);
258  }
259 
260  return inst;
261 }

264 /*
265  * vid_convert_object_attr_value() - Convert a object attribute value.
266  * return:
267  * attribute_p(in): attribute
268  * source_value(in): source value
269  * destination_value(in): destination value
270  * has_obj(out): nonzero iff source_value is (or has) a DB_TYPE_OBJECT (element)
271  * disguised into destination_value as a string value
272  *
273  * Note:
274  * Convert appropriate values from DB_TYPE_STRING to DB_TYPE_OBJECT &
275  * from DB_TYPE_OBJECT to DB_TYPE_STRING.
276  */
277 static int
278 vid_convert_object_attr_value (SM_ATTRIBUTE * attribute_p, DB_VALUE * source_value, DB_VALUE * destination_value,
279  int *has_object)
280 {
281  int error = NO_ERROR;
282  VID_INFO *ref_vid_info;
283  DB_SET *set;
284  DB_SET *new_set;
285  int set_size;
286  int set_index;
287  DB_VALUE set_value;
288  DB_VALUE new_set_value;
289  MOP ref_mop, proxy_class_mop;
290  DB_OBJECT *temp_object = NULL;
291 
292  *has_object = 0; /* assume no disguised objects until proven otherwise */
293 
294  switch (DB_VALUE_TYPE (source_value))
295  {
296  case DB_TYPE_OBJECT:
297  {
298  db_value_domain_init (destination_value, DB_TYPE_STRING, DB_DEFAULT_PRECISION, 0);
299  temp_object = db_get_object (source_value);
300  if (temp_object != NULL)
301  {
302  ref_vid_info = temp_object->oid_info.vid_info;
303  if (ref_vid_info)
304  {
305  pr_clone_value (&ref_vid_info->keys, destination_value);
306  *has_object = 1; /* yes, it's an object disguised as a string */
307  }
308  }
309  }
310  break;
311 
312  case DB_TYPE_VOBJ:
313  {
314  DB_TYPE type_id;
315 
316  db_make_null (destination_value);
317  if (attribute_p->domain == NULL)
318  {
319  return error;
320  }
321 
322  type_id = TP_DOMAIN_TYPE (attribute_p->domain);
323 
324  if (type_id == DB_TYPE_OBJECT || TP_IS_SET_TYPE (type_id))
325  {
326  DB_VALUE setval;
327 
328  set = db_get_set (source_value);
329  (void) db_seq_get (set, 1, &setval);
330  proxy_class_mop = db_get_object (&setval);
331 
332  if (proxy_class_mop == NULL)
333  {
334  return error;
335  }
336  (void) db_seq_get (set, 2, &setval);
337  ref_mop = ws_vmop (proxy_class_mop, VID_UPDATABLE | VID_BASE, &setval);
338  if (ref_mop)
339  {
340  db_make_object (destination_value, ref_mop);
341  }
342  pr_clear_value (&setval);
343  }
344  }
345  break;
346 
347  case DB_TYPE_SET:
348  case DB_TYPE_MULTISET:
349  {
350  set = db_get_set (source_value);
351  new_set = db_get_set (destination_value);
352  set_size = db_set_cardinality (set);
353  for (set_index = 0; set_index < set_size; ++set_index)
354  {
355  error = db_set_get (set, set_index, &set_value);
356  if (error != NO_ERROR)
357  {
358  continue;
359  }
360 
361  error = vid_convert_object_attr_value (attribute_p, &set_value, &new_set_value, has_object);
362  if (error == NO_ERROR)
363  {
364  error = db_set_add (new_set, &new_set_value);
365  pr_clear_value (&new_set_value);
366  }
367  }
368  }
369  break;
370 
371  case DB_TYPE_SEQUENCE:
372  {
373  set = db_get_set (source_value);
374  new_set = db_get_set (destination_value);
375  set_size = db_seq_size (set);
376  for (set_index = 0; set_index < set_size; ++set_index)
377  {
378  error = db_seq_get (set, set_index, &set_value);
379  if (error != NO_ERROR)
380  {
381  continue;
382  }
383  error = vid_convert_object_attr_value (attribute_p, &set_value, &new_set_value, has_object);
384  if (error == NO_ERROR)
385  {
386  error = db_seq_put (new_set, set_index, &new_set_value);
387  pr_clear_value (&new_set_value);
388  }
389  }
390  }
391  break;
392 
393  default:
394  pr_clone_value (source_value, destination_value);
395  break;
396  }
397  return error;
398 }

401 /*
402  * vid_upd_instance() - PREPARE A VIRTUAL INSTANCE FOR UPDATE
403  * return: MOBJ
404  * mop(in): Mop of object that it is going to be updated
405  *
406  * Note:
407  * Prepare an instance for update. The instance is fetched for exclusive
408  * mode and it is set dirty. Note that it is very important the
409  * the instance is set dirty before it is actually updated, otherwise,
410  * the workspace may remain with a corrupted instance if a failure happens
411  *
412  * This function should be called before the instance is actually updated.
413  */
414 MOBJ
415 vid_upd_instance (MOP mop)
416 {
417  MOBJ object; /* The instance object */
418 
419  if (vid_is_new_oobj (mop))
420  {
421  /*
422  * don't fetch new (embryonic) OO instances because they
423  * are not there yet. we are buffering OO inserts.
424  */
425  ws_find (mop, &object);
426  }
427  else
428  {
429  object = vid_fetch_instance (mop, DB_FETCH_WRITE, LC_FETCH_MVCC_VERSION);
430  /* The base instance is marked dirty by vid_fetch_instance */
431  }
432  return object;
433 }

464 /*
465  * vid_flush_and_rehash() - flush and rehash one proxy object instance
466  * return: NO_ERROR if all OK, an ER code otherwise
467  * mop(in): a dirty proxy object to be flushed
468  *
469  * Note:
470  * requires: mop points to a dirty proxy object to be flushed and rehashed
471  * modifies: mop
472  * effects : flush the object associated with the given mop and rehash it
473  * into its (possibly new) workspace hashtable address.
474  */
475 int
476 vid_flush_and_rehash (MOP mop)
477 {
478  int return_code, isvid, isnew_oo;
479  bool isbase;
480  SM_CLASS *class_p;
481 
482  isvid = mop->is_vid;
483  isbase = vid_is_base_instance (mop);
484  isnew_oo = vid_is_new_oobj (mop);
485 
486  /* flush the proxy instance */
487  return_code = vid_flush_instance (mop, NULL);
488  if (return_code == WS_MAP_FAIL)
489  {
490  /* make sure we return an error */
491  assert (er_errid () != NO_ERROR);
492  return_code = er_errid ();
493  if (return_code >= NO_ERROR)
494  {
495  return_code = ER_GENERIC_ERROR;
496  }
497  ws_decache (mop);
498  }
499  else if (isvid && isbase && !isnew_oo)
500  {
501  /*
502  * rehash relational proxy mop into its new ws hashtable address.
503  * we must not rehash a new OO proxy mop because vid_flush_instance
504  * rehashes a new OO proxy mop (please see vid_store_oid_instance).
505  */
506  return_code = vid_get_class_object (WS_CLASS_MOP (mop), &class_p);
507  if (return_code == NO_ERROR && !ws_rehash_vmop (mop, (MOBJ) class_p, NULL))
508  {
509  er_set (ER_ERROR_SEVERITY, ARG_FILE_LINE, ER_WS_REHASH_VMOP_ERROR, 0);
510  return_code = ER_WS_REHASH_VMOP_ERROR;
511  }
512  }
513  return return_code;
514 }

516 /*
517  * vid_flush_and_rehash_lbl() - flush and rehash a proxy object instance label
518  *
519  * return: val if all OK, NULL otherwise
520  * val(in): an interpreter parameter value container
521  *
522  * Note:
523  * requires: val is an interpreter parameter value container that may
524  * point to a dirty proxy object to be flushed and rehashed
525  * modifies: val's dirty proxy object
526  * effects : if val holds anything other than a new dirty proxy object then
527  * simply return val. Otherwise, flush and rehash val's new dirty
528  * proxy object and return val.
529  */
530 DB_VALUE *
531 vid_flush_and_rehash_lbl (DB_VALUE * value)
532 {
533  DB_OBJECT *mop;
534 
535  if (!value)
536  {
537  return value;
538  }
539 
540  if (DB_VALUE_TYPE (value) != DB_TYPE_OBJECT)
541  {
542  return value;
543  }
544 
545  mop = db_get_object (value);
546 
547  /* if val has anything other than a new dirty proxy object then do nothing */
548  if (mop == NULL || !vid_is_new_pobj (mop))
549  {
550  return value;
551  }
552 
553  /* flush and rehash the new proxy object instance */
554  if (vid_flush_and_rehash (mop) != NO_ERROR)
555  {
556  return NULL; /* an error */
557  }
558 
559  return value; /* all OK */
560 }

562 /*
563  * vid_allflush() - flush all dirty vclass objects
564  * return: NO_ERROR or ER_FAILED
565  *
566  * Note:
567  * modifies: all dirty vclass objects
568  * effects : flush all dirty vclass objects
569  */
570 int
571 vid_allflush (void)
572 {
573  DB_OBJLIST *cl;
574  bool return_code;
575  int isvirt;
576 
577  /*
578  * traverse the resident class list and
579  * for each proxy/vclass that has dirty instances
580  * call vid_flush_all to flush those dirty instances
581  */
582  for (cl = ws_Resident_classes, return_code = NO_ERROR; cl != NULL && return_code == NO_ERROR; cl = cl->next)
583  {
584  if (ws_has_dirty_objects (cl->op, &isvirt) && isvirt)
585  {
586  return_code = vid_flush_all_instances (cl->op, DONT_DECACHE);
587  }
588  }
589  return return_code;
590 }

1883  * classify the object into one of:
1884  * - instance of a class
1885  * - instance of a proxy
1886  * - instance of a vclass of a class
1887  * - instance of a vclass of a proxy
1888  * - nonupdatable object
1889  */
1890

1893  {

2046  }
