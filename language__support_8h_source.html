63 #define LANG_RT_COMMON_COLL(c1, c2, coll) \
64  do { \
65  coll = -1; \
66  if ((c1) == (c2)) \
67  { \
68  coll = (c1); \
69  } \
70  else if (LANG_IS_COERCIBLE_COLL (c1)) \
71  { \
72  if (!LANG_IS_COERCIBLE_COLL (c2)) \

74  coll = (c2); \
75  } \
76  else \
77  { \

83  } \
