206 static void
207 crash_handler (int signo, siginfo_t * siginfo, void *dummyp)
208 {
209  int pid;
210 
211  if (signo != SIGABRT && siginfo != NULL && siginfo->si_code <= 0)
212  {
213  register_fatal_signal_handler (signo);
214  return;
215  }
216 
217  if (os_set_signal_handler (signo, SIG_DFL) == SIG_ERR)
218  {
219  return;
220  }
221 
222  if (!BO_IS_SERVER_RESTARTED () || !prm_get_bool_value (PRM_ID_AUTO_RESTART_SERVER))
223  {
224  return;
225  }
226 
227  pid = fork ();
228  if (pid == 0)
229  {
230  char err_log[PATH_MAX];
231  int ppid;
232  int fd, fd_max;
233 
234  fd_max = css_get_max_socket_fds ();
235 
236  for (fd = 3; fd < fd_max; fd++)
237  {
238  close (fd);
239  }
240 
241  ppid = getppid ();
242  while (1)
243  {
244  if (kill (ppid, 0) < 0)
245  {
246  break;
247  }
248  sleep (1);
249  }
250 
251  unmask_signal (signo);
252 
253  if (prm_get_string_value (PRM_ID_ER_LOG_FILE) != NULL)
254  {
255  snprintf (err_log, PATH_MAX, "%s.%d", prm_get_string_value (PRM_ID_ER_LOG_FILE), ppid);
256  rename (prm_get_string_value (PRM_ID_ER_LOG_FILE), err_log);
257  }
258 
259  execl (executable_path, executable_path, database_name, NULL);
260  exit (0);
261  }
262 }
