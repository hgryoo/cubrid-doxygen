468  }
469  }

471  }
472 
473  if (!keep_src_mount)
474  {
475  fileio_dismount (thread_p, from_vdes);
476  }
477  if (!keep_dest_mount)
478  {
479  fileio_dismount (thread_p, to_vdes);
480  }
481  return err;
482 }

1057  * cipher_buffer (out) : Encrypted data
1058  *

1104  {

1106  }
