30 static int
31 count_bits (const unsigned char *mem, int nbits)
32 {
33  int byte, bit;
34  int count = 0;
35 
36  for (byte = 0; byte < nbits / 8; byte++)
37  {
38  for (bit = 0; bit < 8; bit++)
39  {
40  if ((mem[byte] & (((char) 1) << bit)) != 0)
41  {
42  count++;
43  }
44  }
45  }
46  return count;
47 }

112  {

114  {
115  bitset[count++] = ((mem[byte] & (((char) 1) << bit)) != 0) ? '1' : '0';

117  bitset[count++] = ' ';
