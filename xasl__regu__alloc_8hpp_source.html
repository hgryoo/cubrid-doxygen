112 regu_alloc (T *&ptr)
113 {
114  ptr = reinterpret_cast<T *> (pt_alloc_packing_buf ((int) sizeof (T)));
115  if (ptr == NULL)
116  {
117  regu_set_error_with_zero_args (ER_REGU_NO_SPACE);
118  return;
119  }
120  regu_init (*ptr);

123 template <typename T>
124 void
125 regu_array_alloc (T **ptr, size_t size)
126 {
127  if (size == 0)
128  {
129  *ptr = NULL;
130  return;
131  }
132  *ptr = reinterpret_cast<T *> (pt_alloc_packing_buf ((int) (sizeof (T) * size)));
133  if (*ptr == NULL)
134  {
135  regu_set_error_with_zero_args (ER_REGU_NO_SPACE);
136  return;
137  }
138  for (size_t idx = 0; idx < size; idx++)
139  {
140  regu_init ((*ptr)[idx]);
141  }
142 }
